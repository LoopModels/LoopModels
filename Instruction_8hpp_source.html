452  auto createCondition(BumpAlloc<> &alloc, Predicate::Intersection pred,
453  bool swap) -> Instruction * {
454  size_t popCount = pred.popCount();
455  if (popCount == 0) {
456  // everything is true
457  return getConstant(alloc, predicates[0]->getType(), 1);
458  } else if (popCount == 1) {
459  size_t ind = pred.getFirstIndex();
460  Instruction *I = predicates[ind];
461  return swap ? I->negate(alloc, *this) : I;
462  }
463  // we have more than one instruction
464  auto And = Intrinsic(Intrinsic::OpCode{llvm::Instruction::And});
465  size_t ind = pred.getFirstIndex();
466  Instruction *I = predicates[ind];
467  ind = pred.getNextIndex(ind);
468  // we keep I &= predicates[ind] until ind is invalid
469  // ind will be >= 32 when it is invalid
470  // getNextIndex will return a valid answer at least once, because
471  // popCount > 1
472  // there may be a better order than folding from the left
473  // e.g. a binary tree could allow for more out of order execution
474  // but I think a later pass should handle that sort of associativity
475  do {
476  I = getInstruction(alloc, And, I, predicates[ind], I->getType());
477  ind = pred.getNextIndex(ind);
478  } while (ind < 32);
479  return I;
480  }
