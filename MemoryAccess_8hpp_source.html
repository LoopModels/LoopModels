248  void peelLoops(size_t numToPeel) {
249  assert(numToPeel > 0 && "Shouldn't be peeling 0 loops");
250  assert(numToPeel <= getNumLoops() && "Cannot peel more loops than exist");
251  // we're dropping the outer-most `numToPeel` loops
252  // current memory layout:
253  // - indexMatrix (getNumLoops() x getArrayDim())
254  // - offsetMatrix (getArrayDim() x getNumSymbols())
255  // - fusionOmegas (getNumLoops()+1)
256  //
257  // indexMatrix rows are in innermost -> outermost order
258  // fusionOmegas are in outer<->inner order
259  // so we copy `offsetMatrix` `numToPeel * getArrayDim()` elements earlier
260  int64_t *p = data();
261  // When copying overlapping ranges, std::copy is appropriate when copying
262  // to the left (beginning of the destination range is outside the source
263  // range) while std::copy_backward is appropriate when copying to the
264  // right (end of the destination range is outside the source range).
265  // https://en.cppreference.com/w/cpp/algorithm/copy
266  int64_t *offOld = p + getArrayDim() * getNumLoops();
267  int64_t *fusOld = offOld + getArrayDim() * getNumSymbols();
268  int64_t *offNew = p + getArrayDim() * (getNumLoops() - numToPeel);
269  int64_t *fusNew = offNew + getArrayDim() * getNumSymbols();
270  std::copy(offOld, fusOld, offNew);
271  std::copy(fusOld + numToPeel, fusOld + getNumLoops() + 1, fusNew);
272  }
