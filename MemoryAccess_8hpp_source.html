291  void peelLoops(size_t numToPeel) {
292  assert(numToPeel > 0 && "Shouldn't be peeling 0 loops");
293  assert(numToPeel <= getNumLoops() && "Cannot peel more loops than exist");
294  // we're dropping the outer-most `numToPeel` loops
295  // current memory layout:
296  // - indexMatrix (getNumLoops() x getArrayDim())
297  // - offsetMatrix (getArrayDim() x getNumSymbols())
298  // - fusionOmegas (getNumLoops()+1)
299  //
300  // indexMatrix rows are in innermost -> outermost order
301  // fusionOmegas are in outer<->inner order
302  // so we copy `offsetMatrix` `numToPeel * getArrayDim()` elements earlier
303  int64_t *p = data();
304  // When copying overlapping ranges, std::copy is appropriate when copying
305  // to the left (beginning of the destination range is outside the source
306  // range) while std::copy_backward is appropriate when copying to the
307  // right (end of the destination range is outside the source range).
308  // https://en.cppreference.com/w/cpp/algorithm/copy
309  int64_t *offOld = p + getArrayDim() * getNumLoops();
310  int64_t *fusOld = offOld + getArrayDim() * getNumSymbols();
311  int64_t *offNew = p + getArrayDim() * (getNumLoops() - numToPeel);
312  int64_t *fusNew = offNew + getArrayDim() * getNumSymbols();
313  std::copy(offOld, fusOld, offNew);
314  std::copy(fusOld + numToPeel, fusOld + getNumLoops() + 1, fusNew);
315  }
