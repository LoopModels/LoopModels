165  auto addScheduledNode(IR::Cache &cache, IR::Stow stow) -> ScheduledNode * {
166  // how are we going to handle load duplication?
167  // we also need to duplicate the instruction graph leading to the node
168  // implying we need to track that tree.
169  // w = a[i]
170  // x = log(w)
171  // y = 2*x
172  // z = 3*x
173  // p = z / 5
174  // q = 5 / z
175  // s = p - q
176  // b[i] = y
177  // c[i] = s
178  // if adding c[i] after b[i], we must duplicate `w` and `x`
179  // but duplicating `z, `p`, `q`, or `s` is unnecessary.
180  // We don't need to duplicate those instructions where
181  // all uses only lead to `c[i]`.
182  // The trick we use is to mark each instruction with
183  // the store that visited it.
184  // If one has already been visited, duplicate and
185  // mark the new one.
186  auto [storedVal, maxLoop] =
187  searchOperandsForLoads(stow, stow.getStoredVal());
188  maxLoop = deeperLoop(maxLoop, stow.getLoop());
189  stow.setVal(storedVal); // in case it changed
190  return addScheduledNode(cache.getAllocator(), stow, maxLoop);
191  }
192  static constexpr auto deeperLoop(poly::Loop *a, poly::Loop *b)
193  -> poly::Loop * {
194  if (!a) return b;
195  if (!b) return a;
196  return (a->getNumLoops() > b->getNumLoops()) ? a : b;

199  auto searchOperandsForLoads(IR::Cache &cache, IR::Stow stow, Value *val)
200  -> std::pair<Value *, poly::Loop *> {
201  Instruction *inst = dyn_cast<Instruction>(val);
202  if (!inst) return {val, nullptr};
203  // we use parent/child relationships here instead of next/prev
204  if (Load load = IR::Load(inst)) {
205  // TODO: check we don't have mutually exclusive predicates
206  // we found a load; first we check if it has already been added
207  if (load.getParent() != nullptr) {
208  Arena<> *allocator = cache.getAllocator();
209  IR::Addr *reload = ((Addr *)load)->reload(allocator);
210  Dependence::copyDependencies(allocator, load, reload);
211  invariant(reload->isLoad());
212  load = reload;
213  }
214  stow.insertChild(load);
215  return {load, load.getLoop()};
216  // it has been, therefore we need to copy the load
217  }
218  // if not a load, check if it is stored, so we reload
219  Addr *store{nullptr};
220  for (Value *use : inst->getUsers()) {
221  if (IR::Stow other = IR::Stow(use)) {
222  store = other;
223  if (other == stow) break; // scan all users
224  }
225  }
226  if (store && (store != stow)) {
227  Addr *load = Dependence::reload(allocator, store);
228  stow.insertChild(load); // insert load after stow
229  return {load, load.getLoop()};
230  }
231  Compute *C = llvm::cast<Compute>(inst);
232  // could not find a load, so now we recurse, searching operands
233  poly::Loop *maxLoop = nullptr;
234  auto s = allocator.scope(); // create temporary
235  unsigned numOps = C->getNumOperands();
236  MutPtrVector<Value *> newOperands{allocator, numOps};
237  bool opsChanged = false;
238  for (ptrdiff_t i = 0; i < numOps; ++i) {
239  Value *op = C->getOperand(i);
240  auto [updatedOp, loop] = searchOperandsForLoads(cache, stow, op);
241  maxLoop = deeperLoop(maxLoop, loop);
242  if (op != updatedOp) opsChanged = true;
243  newOperands[i] = updatedOp;
244  }
245  if (opsChanged) val = cache.similarCompute(C, newOperands);
246  return {val, maxLoop};
247  }

249  void shiftOmega(ScheduledNode *node) {
250  unsigned nLoops = node->getNumLoops();
251  if (nLoops == 0) return;
252  auto p0 = allocator.checkpoint();
253  MutPtrVector<int64_t> offs = vector<int64_t>(allocator, nLoops);
254  auto p1 = allocator.checkpoint();
255  MutSquarePtrMatrix<int64_t> A = matrix<int64_t>(allocator, nLoops + 1);
256  // MutPtrVector<BumpPtrVector<int64_t>> offsets{
257  // vector<BumpPtrVector<int64_t>>(allocator, nLoops)};
258  // for (size_t i = 0; i < nLoops; ++i)
259  // offsets[i] = BumpPtrVector<int64_t>(allocator);
260  // BumpPtrVector<std::pair<BitSet64, int64_t>> omegaOffsets{allocator};
261  // // we check all memory accesses in the node, to see if applying the same
262  // omega offsets can zero dependence offsets. If so, we apply the shift.
263  // we look for offsets, then try and validate that the shift
264  // if not valid, we drop it from the potential candidates.
265  bool foundNonZeroOffset = false;
266  unsigned rank = 0, L = nLoops - 1;
267  for (ScheduledNode *n = node; n; n = n->getNext()) {
268  for (Addr *m = n->getStore(); m;
269  m = llvm::cast_or_null<Addr>(m->getChild())) {
270  for (Dependence *d = addr->getEdgeIn(); d; d = d->getNextInput()) {
271  const DepPoly *depPoly = d->getDepPoly();
272  unsigned numSyms = depPoly->getNumSymbols(),
273  dep0 = depPoly->getDim0(), dep1 = depPoly->getDim1();
274  PtrMatrix<int64_t> E = depPoly->getE();
275  if (dep.input()->getNode() == n) {
276  // dep within node
277  unsigned depCommon = std::min(dep0, dep1),
278  depMax = std::max(dep0, dep1);
279  invariant(nLoops >= depMax);
280  // input and output, no relative shift of shared loops possible
281  // but indices may of course differ.
282  for (ptrdiff_t d = 0; d < E.numRow(); ++d) {
283  MutPtrVector<int64_t> x = A(rank, _);
284  x[last] = E(d, 0);
285  foundNonZeroOffset |= x[last] != 0;
286  ptrdiff_t j = 0;
287  for (; j < depCommon; ++j)
288  x[L - j] = E(d, j + numSyms) + E(d, j + numSyms + dep0);
289  if (dep0 != dep1) {
290  ptrdiff_t offset = dep0 > dep1 ? numSyms : numSyms + dep0;
291  for (; j < depMax; ++j) x[L - j] = E(d, j + offset);
292  }
293  for (; j < nLoops; ++j) x[L - j] = 0;
294  rank = NormalForm::updateForNewRow(A(_(0, rank + 1), _));
295  }
296  } else {
297  // dep between nodes
298  // is forward means other -> mem, else mem <- other
299  unsigned offset = dep.isForward() ? numSyms + dep0 : numSyms,
300  numDep = dep.isForward() ? dep1 : dep0;
301  for (ptrdiff_t d = 0; d < E.numRow(); ++d) {
302  MutPtrVector<int64_t> x = A(rank, _);
303  x[last] = E(d, 0);
304  foundNonZeroOffset |= x[last] != 0;
305  ptrdiff_t j = 0;
306  for (; j < numDep; ++j) x[L - j] = E(d, j + offset);
307  for (; j < nLoops; ++j) x[L - j] = 0;
308  rank = NormalForm::updateForNewRow(A(_(0, rank + 1), _));
309  }
310  }
311  }
312  for (Dependence *d = addr->getEdgeOut(); d; d = d->getNextOutput()) {
313  if (d->output()->getNode() == n) continue;
314  const DepPoly *depPoly = d->getDepPoly();
315  unsigned numSyms = depPoly->getNumSymbols(),
316  dep0 = depPoly->getDim0(), dep1 = depPoly->getDim1();
317  PtrMatrix<int64_t> E = depPoly->getE();
318  // is forward means mem -> other, else other <- mem
319  unsigned offset = dep.isForward() ? numSyms : numSyms + dep0,
320  numDep = dep.isForward() ? dep0 : dep1;
321  for (ptrdiff_t d = 0; d < E.numRow(); ++d) {
322  MutPtrVector<int64_t> x = A(rank, _);
323  x[last] = E(d, 0);
324  foundNonZeroOffset |= x[last] != 0;
325  ptrdiff_t j = 0;
326  for (; j < numDep; ++j) x[L - j] = E(d, j + offset);
327  for (; j < nLoops; ++j) x[L - j] = 0;
328  rank = NormalForm::updateForNewRow(A(_(0, rank + 1), _));
329  }
330  }
331  }
332  }
333  if (!foundNonZeroOffset) return allocator.rollback(p0);
334  bool nonZero = false;
335  // matrix A is reasonably diagonalized, should indicate
336  ptrdiff_t c = 0;
337  for (ptrdiff_t r = 0; r < rank; ++r) {
338  int64_t off = A(r, last);
339  if (off == 0) continue;
340  for (; c < nLoops; ++c) {
341  if (A(r, c) != 0) break;
342  offs[L - c] = 0;
343  }
344  if (c == nLoops) return;
345  int64_t Arc = A(r, c), x = off / Arc;
346  if (x * Arc != off) continue;
347  offs[L - c++] = x; // decrement loop `L-c` by `x`
348  nonZero = true;
349  }
350  if (!nonZero) return allocator.rollback(p0);
351  allocator.rollback(p1);
352  for (; c < nLoops; ++c) offs[L - c] = 0;
353  node.setOffsets(offs.data());
354  // now we iterate over the edges again
355  // perhaps this should be abstracted into higher order functions that
356  // iterate over the edges?
357  for (ScheduledNode *n = node; n; n = n->getNext()) {
358  for (Addr *m = n->getStore(); m;
359  m = llvm::cast_or_null<Addr>(m->getChild())) {
360  for (Dependence *d = addr->getEdgeIn(); d; d = d->getNextInput()) {
361  d->copySimplices(allocator); // in case it is aliased
362  DepPoly *depPoly = d->getDepPoly();
363  unsigned numSyms = depPoly->getNumSymbols(),
364  dep0 = depPoly->getDim0(), dep1 = depPoly->getDim1();
365  MutPtrMatrix<int64_t> satL = d->getSatLambda();
366  MutPtrMatrix<int64_t> bndL = d->getBndLambda();
367  bool pick = d->isForward(), repeat = d->input()->getNode() == nIdx;
368  while (true) {
369  unsigned offset = pick ? numSyms + dep0 : numSyms,
370  numDep = pick ? dep1 : dep0;
371  for (ptrdiff_t l = 0; l < numDep; ++l) {
372  int64_t mlt = offs[l];
373  if (mlt == 0) continue;
374  satL(0, _) -= mlt * satL(offset + l, _);
375  bndL(0, _) -= mlt * bndL(offset + l, _);
376  }
377  if (!repeat) break;
378  repeat = false;
379  pick = !pick;
380  }
381  }
382  }
383  for (Dependence *d = addr->getEdgeOut(); d; d = d->getNextOutput()) {
384  if (d->output()->getNode() == n) continue; // handled above
385  d->copySimplices(allocator); // we don't want to copy twice
386  DepPoly *depPoly = d->getDepPoly();
387  unsigned numSyms = depPoly->getNumSymbols(), dep0 = depPoly->getDim0(),
388  dep1 = depPoly->getDim1();
389  MutPtrMatrix<int64_t> satL = d->getSatLambda();
390  MutPtrMatrix<int64_t> bndL = d->getBndLambda();
391  unsigned offset = d->isForward() ? numSyms : numSyms + dep0,
392  numDep = d->isForward() ? dep0 : dep1;
393  for (size_t l = 0; l < numDep; ++l) {
394  int64_t mlt = offs[l];
395  if (mlt == 0) continue;
396  satL(0, _) -= mlt * satL(offset + l, _);
397  bndL(0, _) -= mlt * bndL(offset + l, _);
398  }
399  }
400  }
401  }

403  auto optOrth(ScheduledNode *nodes, unsigned maxDepth) -> bool {
404  // check for orthogonalization opportunities
405  bool tryOrth = false;
406  for (ScheduledNode *node : nodes->nodesRange()) {
407  for (Dependence *edge : node->inputEdges()) {
408  // this edge's output is `node`
409  // we want edges whose input is also `node`,
410  // i.e. edges that are within the node
411  if (edge->input()->getNode() != node) continue;
412  DensePtrMatrix<int64_t> indMat = edge->getInIndMat();
413  // check that we haven't already scheduled on an earlier
414  // iteration of this loop, and that the indmats are the same
415  if (node->phiIsScheduled(0) || (indMat != edge->getOutIndMat()))
416  continue;
417  size_t r = NormalForm::rank(indMat);
418  if (r == edge->getInNumLoops()) continue;
419  // TODO handle linearly dependent acceses, filtering them out
420  if (r != size_t(indMat.numRow())) continue;
421  node->schedulePhi(indMat, r);
422  tryOrth = true;
423  }
424  }
425  if (tryOrth) {
426  if (optimize(g, 0, maxDepth)) return true;
427  for (ScheduledNode *n : nodes->nodesRange()) n->unschedulePhi();
428  }
429  return optimize(g, 0, maxDepth);
430  }

432  [[nodiscard]] auto optimize(ScheduledNode *nodes, unsigned d,
433  unsigned maxDepth) -> bool {
434  if (d >= maxDepth) return true;
435  countAuxParamsAndConstraints(g, d);
436  setScheduleMemoryOffsets(g, d);
437  // if we fail on this level, break the graph
438  BitSet activeEdgesBackup = g.activeEdges;
439  if (std::optional<BitSet> depSat = solveGraph(g, d, false)) {
440  const size_t dp1 = d + 1;
441  if (dp1 == maxDepth) return *depSat;
442  if (std::optional<BitSet> depSatNest = optimize(g, dp1, maxDepth)) {
443  bool depSatEmpty = depSat->empty();
444  *depSat |= *depSatNest;
445  if (!(depSatEmpty || depSatNest->empty())) // try and sat all this level
446  return optimizeSatDep(g, d, maxDepth, *depSat, activeEdgesBackup);
447  return *depSat;
448  }
449  }
450  return breakGraph(g, d);

871  auto instantiateOmniSimplex(const Graph &g, size_t d, bool satisfyDeps)
872  -> std::unique_ptr<Simplex> {
873  auto omniSimplex = Simplex::create(
874  numConstraints + numSlack, numBounding + numActiveEdges + numPhiCoefs +
875  numOmegaCoefs + numSlack + numLambda);
876  auto C{omniSimplex->getConstraints()};
877  C << 0;
878  // layout of omniSimplex:
879  // Order: C, then rev-priority to minimize
880  // C, lambdas, slack, omegas, Phis, w, u
881  // rows give constraints; each edge gets its own
882  // numBounding = num u
883  // numActiveEdges = num w
884  Row c = 0;
885  Col l = 1, o = 1 + numLambda + numSlack, p = o + numOmegaCoefs,
886  w = p + numPhiCoefs, u = w + numActiveEdges;
887  for (size_t e = 0; e < edges.size(); ++e) {
888  Dependence &edge = edges[e];
889  if (g.isInactive(e, d)) continue;
890  size_t outNodeIndex = edge.nodeOut(), inNodeIndex = edge.nodeIn();
891  const auto [satC, satL, satPp, satPc, satO, satW] =
892  edge.splitSatisfaction();
893  const auto [bndC, bndL, bndPp, bndPc, bndO, bndWU] = edge.splitBounding();
894  const size_t numSatConstraints = satC.size(),
895  numBndConstraints = bndC.size();
896  const Col nPc = satPc.numCol(), nPp = satPp.numCol();
897  invariant(nPc, bndPc.numCol());
898  invariant(nPp, bndPp.numCol());
899  const ScheduledNode &outNode = nodes[outNodeIndex];
900  const ScheduledNode &inNode = nodes[inNodeIndex];
901 
902  Row cc = c + numSatConstraints;
903  Row ccc = cc + numBndConstraints;
904 
905  Col ll = l + satL.numCol();
906  Col lll = ll + bndL.numCol();
907  C(_(c, cc), _(l, ll)) << satL;
908  C(_(cc, ccc), _(ll, lll)) << bndL;
909  l = lll;
910  // bounding
911  C(_(cc, ccc), w++) << bndWU(_, 0);
912  Col uu = u + bndWU.numCol() - 1;
913  C(_(cc, ccc), _(u, uu)) << bndWU(_, _(1, end));
914  u = uu;
915  if (satisfyDeps) C(_(c, cc), 0) << satC + satW;
916  else C(_(c, cc), 0) << satC;
917  C(_(cc, ccc), 0) << bndC;
918  // now, handle Phi and Omega
919  // phis are not constrained to be 0
920  if (outNodeIndex == inNodeIndex) {
921  if (d < outNode.getNumLoops()) {
922  if (nPc == nPp) {
923  if (outNode.phiIsScheduled(d)) {
924  // add it constants
925  auto sch = outNode.getSchedule(d);
926  C(_(c, cc), 0) -= satPc * sch[_(0, nPc)] + satPp * sch[_(0, nPp)];
927  C(_(cc, ccc), 0) -=
928  bndPc * sch[_(0, nPc)] + bndPp * sch[_(0, nPp)];
929  } else {
930  // FIXME: phiChild = [14:18), 4 cols
931  // while Dependence seems to indicate 2
932  // loops why the disagreement?
933  auto po = outNode.getPhiOffset() + p;
934  C(_(c, cc), _(po, po + nPc)) << satPc + satPp;
935  C(_(cc, ccc), _(po, po + nPc)) << bndPc + bndPp;
936  }
937  } else if (outNode.phiIsScheduled(d)) {
938  // add it constants
939  // note that loop order in schedule goes
940  // inner -> outer
941  // so we need to drop inner most if one has less
942  auto sch = outNode.getSchedule(d);
943  auto schP = sch[_(0, nPp)];
944  auto schC = sch[_(0, nPc)];
945  C(_(c, cc), 0) -= satPc * schC + satPp * schP;
946  C(_(cc, ccc), 0) -= bndPc * schC + bndPp * schP;
947  } else if (nPc < nPp) {
948  // Pp has more cols, so outer/leftmost overlap
949  auto po = outNode.getPhiOffset() + p, poc = po + nPc,
950  pop = po + nPp;
951  C(_(c, cc), _(po, poc)) << satPc + satPp(_, _(0, nPc));
952  C(_(cc, ccc), _(po, poc)) << bndPc + bndPp(_, _(0, nPc));
953  C(_(c, cc), _(poc, pop)) << satPp(_, _(nPc, end));
954  C(_(cc, ccc), _(poc, pop)) << bndPp(_, _(nPc, end));
955  } else /* if (nPc > nPp) */ {
956  auto po = outNode.getPhiOffset() + p, poc = po + nPc,
957  pop = po + nPp;
958  C(_(c, cc), _(po, pop)) << satPc(_, _(0, nPp)) + satPp;
959  C(_(cc, ccc), _(po, pop)) << bndPc(_, _(0, nPp)) + bndPp;
960  C(_(c, cc), _(pop, poc)) << satPc(_, _(nPp, end));
961  C(_(cc, ccc), _(pop, poc)) << bndPc(_, _(nPp, end));
962  }
963  C(_(c, cc), outNode.getOmegaOffset() + o) << satO(_, 0) + satO(_, 1);
964  C(_(cc, ccc), outNode.getOmegaOffset() + o)
965  << bndO(_, 0) + bndO(_, 1);
966  }
967  } else {
968  if (d < edge.getOutNumLoops())
969  updateConstraints(C, outNode, satPc, bndPc, d, c, cc, ccc, p);
970  if (d < edge.getInNumLoops()) {
971  if (d < edge.getOutNumLoops() && !inNode.phiIsScheduled(d) &&
972  !outNode.phiIsScheduled(d)) {
973  invariant(inNode.getPhiOffset() != outNode.getPhiOffset());
974  }
975  updateConstraints(C, inNode, satPp, bndPp, d, c, cc, ccc, p);
976  }
977  // Omegas are included regardless of rotation
978  if (d < edge.getOutNumLoops()) {
979  if (d < edge.getInNumLoops())
980  invariant(inNode.getOmegaOffset() != outNode.getOmegaOffset());
981  C(_(c, cc), outNode.getOmegaOffset() + o)
982  << satO(_, edge.isForward());
983  C(_(cc, ccc), outNode.getOmegaOffset() + o)
984  << bndO(_, edge.isForward());
985  }
986  if (d < edge.getInNumLoops()) {
987  C(_(c, cc), inNode.getOmegaOffset() + o)
988  << satO(_, !edge.isForward());
989  C(_(cc, ccc), inNode.getOmegaOffset() + o)
990  << bndO(_, !edge.isForward());
991  }
992  }
993  c = ccc;
994  }
995  invariant(size_t(l), size_t(1 + numLambda));
996  invariant(size_t(c), size_t(numConstraints));
997  addIndependentSolutionConstraints(omniSimplex.get(), g, d);
998  return omniSimplex;
999  }
1000  static void updateConstraints(MutPtrMatrix<int64_t> C,
1001  const ScheduledNode &node,
1002  PtrMatrix<int64_t> sat, PtrMatrix<int64_t> bnd,
1003  size_t d, Row c, Row cc, Row ccc, Col p) {
1004  invariant(sat.numCol(), bnd.numCol());
1005  if (node.phiIsScheduled(d)) {
1006  // add it constants
1007  auto sch = node.getSchedule(d)[_(0, sat.numCol())];
1008  // order is inner <-> outer
1009  // so we need the end of schedule if it is larger
1010  C(_(c, cc), 0) -= sat * sch;
1011  C(_(cc, ccc), 0) -= bnd * sch;
1012  } else {
1013  // add it to C
1014  auto po = node.getPhiOffset() + p;
1015  C(_(c, cc), _(po, po + sat.numCol())) << sat;
1016  C(_(cc, ccc), _(po, po + bnd.numCol())) << bnd;
1017  }
1018  }
1019  [[nodiscard]] auto solveGraph(Graph &g, size_t depth, bool satisfyDeps)
1020  -> std::optional<BitSet> {
1021  if (numLambda == 0) {
1022  setSchedulesIndependent(g, depth);
1023  return checkEmptySatEdges(g, depth);
1024  }
1025  auto omniSimplex = instantiateOmniSimplex(g, depth, satisfyDeps);
1026  if (omniSimplex->initiateFeasible()) return std::nullopt;
1027  auto sol = omniSimplex->rLexMinStop(numLambda + numSlack);
1028  assert(sol.size() ==
1029  numBounding + numActiveEdges + numPhiCoefs + numOmegaCoefs);
1030  updateSchedules(g, depth, sol);
1031  return deactivateSatisfiedEdges(g, depth,
1032  sol[_(numPhiCoefs + numOmegaCoefs, end)]);
1033  }

1041  outPhi = outNode.getPhi()(_(0, depth + 1), _);
1042  if (edge.checkEmptySat(allocator, inNode.getLoopNest(),
1043  inNode.getOffset(), inPhi, outNode.getLoopNest(),

1051  Simplex::Solution sol) -> BitSet {

1053  return checkEmptySatEdges(g, depth);

1058  if (g.isInactive(e, depth)) continue;

1069  outPhi = outNode.getPhi()(_(0, depth + 1), _);

1071  inNode.getOffset(), inPhi, outNode.getLoopNest(),
1072  outNode.getOffset(), outPhi)) {
1073  g.activeEdges.remove(e);
1074  }
1075  }

1077  }

1080  static void setDepFreeSchedule(PtrVector<MemoryAccess *> mem,
1081  ScheduledNode &node, size_t depth) {
1082  node.getOffsetOmega(depth) = 0;
1083  if (node.phiIsScheduled(depth)) return;
1084  // we'll check the null space of the phi's so far
1085  // and then search for array indices
1086  if (depth == 0) {
1087  // for now, if depth == 0, we just set last active
1088  MutPtrVector<int64_t> phiv{node.getSchedule(0)};
1089  phiv[_(0, last)] << 0;
1090  phiv[last] = 1;
1091  return;
1092  }
1093  DenseMatrix<int64_t> nullSpace; // d x lfull
1094  DenseMatrix<int64_t> A{node.getPhi()(_(0, depth), _).transpose()};
1095  NormalForm::nullSpace11(nullSpace, A);
1096  invariant(size_t(nullSpace.numRow()), node.getNumLoops() - depth);
1097  // now, we search index matrices for schedules not in the null space of
1098  // existing phi.
1099  // Here, we collect candidates for the next schedule
1100  DenseMatrix<int64_t> candidates{DenseDims{0, node.getNumLoops() + 1}};
1101  Vector<int64_t> indv;
1102  indv.resizeForOverwrite(node.getNumLoops());
1103  for (size_t ind : node.getMemory()) {
1104  PtrMatrix<int64_t> indMat = mem[ind]->indexMatrix(); // lsub x d
1105  A.resizeForOverwrite(DenseDims{nullSpace.numRow(), indMat.numCol()});
1106  A = nullSpace(_, _(0, indMat.numRow())) * indMat;
1107  // we search A for rows that aren't all zero
1108  for (size_t d = 0; d < A.numCol(); ++d) {
1109  if (allZero(A(_, d))) continue;
1110  indv << indMat(_, d);
1111  bool found = false;
1112  for (size_t j = 0; j < candidates.numRow(); ++j) {
1113  if (candidates(j, _(0, last)) != indv) continue;
1114  found = true;
1115  ++candidates(j, 0);
1116  break;
1117  }
1118  if (!found) {
1119  candidates.resize(candidates.numRow() + 1);
1120  assert(candidates(last, 0) == 0);
1121  candidates(last, _(1, end)) << indv;
1122  }
1123  }
1124  }
1125  if (Row R = candidates.numRow()) {
1126  // >= 1 candidates, pick the one with with greatest lex, favoring
1127  // number of repetitions (which were placed in first index)
1128  size_t i = 0;
1129  for (size_t j = 1; j < candidates.numRow(); ++j)
1130  if (candidates(j, _) > candidates(i, _)) i = j;
1131  node.getSchedule(depth) << candidates(i, _(1, end));
1132  return;
1133  }
1134  // do we want to pick the outermost original loop,
1135  // or do we want to pick the outermost lex null space?
1136  node.getSchedule(depth) << 0;
1137  for (size_t c = 0; c < nullSpace.numCol(); ++c) {
1138  if (allZero(nullSpace(_, c))) continue;
1139  node.getSchedule(depth)[c] = 1;
1140  return;
1141  }
1142  invariant(false);
1143  }
1144  void updateSchedules(const Graph &g, size_t depth, Simplex::Solution sol) {
1145 #ifndef NDEBUG
1146  if (numPhiCoefs > 0)
1147  assert(
1148  std::ranges::any_of(sol, [](Rational s) -> bool { return s != 0; }));
1149 #endif
1150  size_t o = numOmegaCoefs;
1151  for (auto &&node : nodes) {
1152  if (depth >= node.getNumLoops()) continue;
1153  if (!hasActiveEdges(g, node)) {
1154  setDepFreeSchedule(memory, node, depth);
1155  continue;
1156  }
1157  Rational sOmega = sol[node.getOmegaOffset()];
1158  // TODO: handle s.denominator != 1
1159  if (!node.phiIsScheduled(depth)) {
1160  auto phi = node.getSchedule(depth);
1161  auto s = sol[node.getPhiOffsetRange() + o];
1162  int64_t baseDenom = sOmega.denominator;
1163  int64_t l = lcm(s.denomLCM(), baseDenom);
1164 #ifndef NDEBUG
1165  for (size_t i = 0; i < phi.size(); ++i)
1166  assert(((s[i].numerator * l) / (s[i].denominator)) >= 0);
1167 #endif
1168  if (l == 1) {
1169  node.getOffsetOmega(depth) = sOmega.numerator;
