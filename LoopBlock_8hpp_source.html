184  constexpr void schedulePhi(PtrMatrix<int64_t> indMat, size_t r) {
185  // indMat indvars are indexed from inside<->outside
186  // phi indvars are indexed from inside<->outside
187  // so, indMat is indvars[inside<->outside] x array dim
188  // phi is loop[inside<->outside] x
189  // indvars[inside<->outside]
190  MutSquarePtrMatrix<int64_t> phi = getPhi();
191  const size_t indR = size_t(indMat.numRow());
192  const size_t phiOff = size_t(phi.numCol()) - indR;
193  for (size_t i = 0; i < r; ++i) {
194  phi(last - i, _(begin, phiOff)) << 0;
195  phi(last - i, _(phiOff, phiOff + indR)) << indMat(_, i);
196  }
197  rank = r;
198  }

425  [[nodiscard]] auto searchValueForStores(aset<llvm::User *> &visited,
426  ScheduledNode &node, llvm::User *user,
427  unsigned nodeIndex) -> bool {
428  for (llvm::User *use : user->users()) {
429  if (visited.contains(use)) continue;

432  if (memAccess == userToMemory.end())

435  MemoryAccess *store = memory[memId];

820  // the plan is to generally avoid instantiating the omni-simplex
821  // first, we solve individual problems
822  auto instantiateOmniSimplex(const Graph &g, size_t d, bool satisfyDeps)
823  -> Optional<Simplex *> {
824  auto omniSimplex =
825  Simplex::create(allocator, numConstraints + numOmegaCoefs,
826  1 + numBounding + numActiveEdges + numPhiCoefs +
827  2 * numOmegaCoefs + numLambda);
828  auto C{omniSimplex->getConstraints()};
829  C << 0;
830  // layout of omniSimplex:
831  // Order: C, then rev-priority to minimize
832  // C, lambdas, omegas, Phis, w, u
833  // rows give constraints; each edge gets its own
834  // numBounding = num u
835  // numActiveEdges = num w
836  Row c = 0;
837  Col l = 1, o = l + numLambda, w = o + numOmegaCoefs + numPhiCoefs,
838  u = w + numActiveEdges;
839  for (size_t e = 0; e < edges.size(); ++e) {
840  Dependence &edge = *edges[e];
841  if (g.isInactive(e, d)) continue;
842  const BitSet &outNodeIndexSet = edge.nodesOut();
843  const BitSet &inNodeIndexSet = edge.nodesIn();
844  const auto [satC, satL, satPp, satPc, satO, satW] =
845  edge.splitSatisfaction();
846  const auto [bndC, bndL, bndPp, bndPc, bndO, bndWU] = edge.splitBounding();
847  const size_t numSatConstraints = satC.size();
848  const size_t numBndConstraints = bndC.size();
849  for (auto outNodeIndex : outNodeIndexSet) {
850  const ScheduledNode &outNode = nodes[outNodeIndex];
851  for (auto inNodeIndex : inNodeIndexSet) {
852  const ScheduledNode &inNode = nodes[inNodeIndex];
853 
854  Row cc = c + numSatConstraints;
855  Row ccc = cc + numBndConstraints;
856 
857  Col ll = l + satL.numCol();
858  Col lll = ll + bndL.numCol();
859  C(_(c, cc), _(l, ll)) = satL;
860  C(_(cc, ccc), _(ll, lll)) = bndL;
861  l = lll;
862  // bounding
863  C(_(cc, ccc), w++) = bndWU(_, 0);
864  Col uu = u + bndWU.numCol() - 1;
865  C(_(cc, ccc), _(u, uu)) = bndWU(_, _(1, end));
866  u = uu;
867  if (satisfyDeps) C(_(c, cc), 0) << satC + satW;
868  else C(_(c, cc), 0) = satC;
869  C(_(cc, ccc), 0) = bndC;
870  // now, handle Phi and Omega
871  // phis are not constrained to be 0
872  if (outNodeIndex == inNodeIndex) {
873  if (d < outNode.getNumLoops()) {
874  if (satPc.numCol() == satPp.numCol()) {
875  if (outNode.phiIsScheduled(d)) {
876  // add it constants
877  auto sch = outNode.getSchedule(d);
878  C(_(c, cc), 0) -= satPc * sch[_(end - satPc.numCol(), end)] +
879  satPp * sch[_(end - satPp.numCol(), end)];
880  C(_(cc, ccc), 0) -=
881  bndPc * sch[_(end - bndPc.numCol(), end)] +
882  bndPp * sch[_(end - satPp.numCol(), end)];
883  } else {
884  // FIXME: phiChild = [14:18), 4 cols
885  // while Dependence seems to indicate 2
886  // loops why the disagreement?
887  auto phiChild = outNode.getPhiOffset();
888  C(_(c, cc), _(phiChild - satPc.numCol(), phiChild))
889  << satPc + satPp;
890  C(_(cc, ccc), _(phiChild - bndPc.numCol(), phiChild))
891  << bndPc + bndPp;
892  }
893  } else if (outNode.phiIsScheduled(d)) {
894  // add it constants
895  // note that loop order in schedule goes
896  // inner -> outer
897  // so we need to drop inner most if one has less
898  auto sch = outNode.getSchedule(d);
899  auto schP = sch[_(end - satPp.numCol(), end)];
900  auto schC = sch[_(end - satPc.numCol(), end)];
901  C(_(c, cc), 0) -= satPc * schC + satPp * schP;
902  C(_(cc, ccc), 0) -= bndPc * schC + bndPp * schP;
903  } else if (satPc.numCol() < satPp.numCol()) {
904  auto phiChild = outNode.getPhiOffset();
905  Col P = satPc.numCol();
906  auto m = phiChild - P;
907  C(_(c, cc), _(phiChild - satPp.numCol(), m))
908  << satPp(_, _(begin, end - P));
909  C(_(cc, ccc), _(phiChild - bndPp.numCol(), m))
910  << bndPp(_, _(begin, end - P));
911  C(_(c, cc), _(m, phiChild))
912  << satPc + satPp(_, _(end - P, end));
913  C(_(cc, ccc), _(m, phiChild))
914  << bndPc + bndPp(_, _(end - P, end));
915  } else /* if (satPc.numCol() > satPp.numCol()) */ {
916  auto phiChild = outNode.getPhiOffset();
917  Col P = satPp.numCol();
918  auto m = phiChild - P;
919  C(_(c, cc), _(phiChild - satPc.numCol(), m))
920  << satPc(_, _(begin, end - P));
921  C(_(cc, ccc), _(phiChild - bndPc.numCol(), m))
922  << bndPc(_, _(begin, end - P));
923  C(_(c, cc), _(m, phiChild))
924  << satPc(_, _(end - P, end)) + satPp;
925  C(_(cc, ccc), _(m, phiChild))
926  << bndPc(_, _(end - P, end)) + bndPp;
927  }
928  C(_(c, cc), outNode.getOmegaOffset() + o)
929  << satO(_, 0) + satO(_, 1);
930  C(_(cc, ccc), outNode.getOmegaOffset() + o)
931  << bndO(_, 0) + bndO(_, 1);
932  }
933  } else {
934  if (d < edge.getOutNumLoops())
935  updateConstraints(C, outNode, satPc, bndPc, d, c, cc, ccc);
936  if (d < edge.getInNumLoops())
937  updateConstraints(C, inNode, satPp, bndPp, d, c, cc, ccc);
938  // Omegas are included regardless of rotation
939  if (d < edge.getOutNumLoops()) {
940  C(_(c, cc), outNode.getOmegaOffset() + o)
941  << satO(_, !edge.isForward());
942  C(_(cc, ccc), outNode.getOmegaOffset() + o)
943  << bndO(_, !edge.isForward());
944  }
945  if (d < edge.getInNumLoops()) {
946  C(_(c, cc), inNode.getOmegaOffset() + o)
947  << satO(_, edge.isForward());
948  C(_(cc, ccc), inNode.getOmegaOffset() + o)
949  << bndO(_, edge.isForward());
950  }
951  }
952  c = ccc;
953  }
954  }
955  }
956  addIndependentSolutionConstraints(omniSimplex, g, d);
957  // earlier, we only had this check in optimizeLevel, when satisfyDeps=false
958  // what is this check doing?
959  assert(!allZero(omniSimplex->getConstraints()(last, _)));
960  return omniSimplex->initiateFeasible() ? nullptr : omniSimplex;
961  }
962  static void updateConstraints(MutPtrMatrix<int64_t> C,
963  const ScheduledNode &node,
964  PtrMatrix<int64_t> sat, PtrMatrix<int64_t> bnd,
965  size_t d, Row c, Row cc, Row ccc) {
966  if (node.phiIsScheduled(d)) {
967  // add it constants
968  auto sch = node.getSchedule(d);
969  // order is inner <-> outer
970  // so we need the end of schedule if it is larger
971  C(_(c, cc), 0) -= sat * sch[_(end - sat.numCol(), end)];
972  C(_(cc, ccc), 0) -= bnd * sch[_(end - bnd.numCol(), end)];
973  } else {
974  assert(sat.numCol() == bnd.numCol());
975  // add it to C
976  auto phiChild = node.getPhiOffset();
977  C(_(c, cc), _(phiChild - sat.numCol(), phiChild)) = sat;
978  C(_(cc, ccc), _(phiChild - bnd.numCol(), phiChild)) = bnd;
979  }
980  }

996  [[nodiscard]] auto deactivateSatisfiedEdges(Graph &g, size_t depth,
997  Simplex::Solution sol) -> BitSet {
998  if (allZero(sol[_(begin, numBounding + numActiveEdges)])) return {};
999  size_t u = 0, w = numBounding;
1000  BitSet deactivated;
1001  for (size_t e = 0; e < edges.size(); ++e) {
1002  if (g.isInactive(e, depth)) continue;
1003  const Dependence &edge = *edges[e];
1004  Col uu = u + edge.getNumDynamicBoundingVar();
1005  if ((sol[w++] != 0) || (!(allZero(sol[_(u, uu)])))) {
1006  g.activeEdges.remove(e);
1007  deactivated.insert(e);
1008  for (size_t inIndex : edge.nodesIn())
1009  carriedDeps[inIndex].setCarriedDependency(depth);
1010  for (size_t outIndex : edge.nodesOut())
1011  carriedDeps[outIndex].setCarriedDependency(depth);
1012  }
1013  u = size_t(uu);
1014  }
1015  return deactivated;
1016  }
1017  void updateSchedules(const Graph &g, size_t depth, Simplex::Solution sol) {
1018 #ifndef NDEBUG
1019  if (depth & 1) {
1020  bool allZero = true;
1021  for (auto s : sol) allZero &= (s == 0);
1022  // if (allZero) llvm::errs() << "omniSimplex = " << omniSimplex << "\n";
1023  assert(!allZero);
1024  }
1025 #endif
1026  for (auto &&node : nodes) {
1027  if (depth >= node.getNumLoops()) continue;
1028  if (!hasActiveEdges(g, node)) {
1029  node.getOffsetOmega()[depth] = std::numeric_limits<int64_t>::min();
1030  if (!node.phiIsScheduled(depth))
1031  node.getSchedule(depth) << std::numeric_limits<int64_t>::min();
1032  continue;
1033  }
1034  Rational sOmega = sol[node.getOmegaOffset() - 1];
1035  // TODO: handle s.denominator != 1
1036  if (!node.phiIsScheduled(depth)) {
1037  auto phi = node.getSchedule(depth);
1038  auto s = sol[node.getPhiOffsetRange() - 1];
1039  int64_t baseDenom = sOmega.denominator;
1040  int64_t l = lcm(s.denomLCM(), baseDenom);
1041  for (size_t i = 0; i < phi.size(); ++i)
1042  assert(((s[i].numerator * l) / (s[i].denominator)) >= 0);
1043  if (l == 1) {
1044  node.getOffsetOmega(depth) = sOmega.numerator;
1045  for (size_t i = 0; i < phi.size(); ++i) phi[i] = s[i].numerator;
1046  } else {
1047  node.getOffsetOmega(depth) = (sOmega.numerator * l) / baseDenom;
1048  for (size_t i = 0; i < phi.size(); ++i)
1049  phi[i] = (s[i].numerator * l) / (s[i].denominator);
1050  }
1051  assert(!(allZero(phi)));
1052  // node.schedule.getPhi()(depth, _) =
1053  // sol(node.getPhiOffset() - 1) *
1054  // denomLCM(sol(node.getPhiOffset() - 1));
1055  } else {
1056  node.getOffsetOmega(depth) = sOmega.numerator;
1057  }
1058 #ifndef NDEBUG
1059  if (!node.phiIsScheduled(depth)) {
1060  int64_t l = sol[node.getPhiOffsetRange() - 1].denomLCM();
1061  for (size_t i = 0; i < node.getPhi().numCol(); ++i)
1062  assert(node.getPhi()(last - depth, i) ==
1063  sol[node.getPhiOffsetRange() - 1][i] * l);
1064  }
1065 #endif
1066  }
1067  }
1068  [[nodiscard]] static auto lexSign(PtrVector<int64_t> x) -> int64_t {
1069  // TODO: `std::ranges::reverse_view(x)` is broken; update when
1070  // upgrading to a compiler that supports it.
1071  // NOLINTNEXTLINE(modernize-loop-convert)
1072  for (auto b = x.rbegin(), e = x.rend(); b != e; ++b)
1073  if (*b) return 2 * ((*b) > 0) - 1;
1074  return 0;

1076  void addIndependentSolutionConstraints(NotNull<Simplex> omniSimplex,
1077  const Graph &g, size_t depth) {
1078  // omniSimplex->setNumCons(omniSimplex->getNumCons() +
1079  // memory.size());
1080  // omniSimplex->reserveExtraRows(memory.size());
1081  auto C{omniSimplex->getConstraints()};
1082  size_t i = size_t{C.numRow()} - numOmegaCoefs;
1083  if (depth == 0) {
1084  // add ones >= 0
1085  for (auto &&node : nodes) {
1086  if (node.phiIsScheduled(depth) || (!hasActiveEdges(g, node))) continue;
1087  C(i, 0) = 1;
1088  C(i, node.getPhiOffsetRange()) << 1;
1089  C(i++, last) = -1; // for >=
1090  }
1091  return;
1092  }
1093  DenseMatrix<int64_t> A, N;
1094  for (auto &&node : nodes) {
1095  if (node.phiIsScheduled(depth) || (depth >= node.getNumLoops()) ||
1096  (!hasActiveEdges(g, node)))
1097  continue;
1098  A << node.getPhi()(_(end - depth, end), _).transpose();
1099  NormalForm::nullSpace11(N, A);
1100  C(i, 0) = 1;
1101  MutPtrVector<int64_t> cc{C(i, node.getPhiOffsetRange())};
1102  // sum(N,dims=1) >= 1 after flipping row signs to be lex > 0
1103  for (size_t m = 0; m < N.numRow(); ++m) cc += N(m, _) * lexSign(N(m, _));
1104  C(i++, last) = -1; // for >=
1105  }
1106  assert(i == C.numRow());
1107  assert(!allZero(omniSimplex->getConstraints()(last, _)));
1108  }

1174  [[nodiscard]] auto breakGraph(Graph g, size_t d) -> std::optional<BitSet> {
1175  llvm::SmallVector<BitSet> components;
1176  Graphs::stronglyConnectedComponents(components, g);
1177  if (components.size() <= 1) return {};
1178  // components are sorted in topological order.
1179  // We split all of them, solve independently,
1180  // and then try to fuse again after if/where optimal schedules
1181  // allow it.
1182  auto graphs = g.split(components);
1183  assert(graphs.size() == components.size());
