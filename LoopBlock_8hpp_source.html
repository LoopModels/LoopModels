569  llvm::Optional<BitSet> optOrth(Graph g) {
570 
571  const size_t maxDepth = calcMaxDepth();
572  // check for orthogonalization opportunities
573  bool tryOrth = false;
574  for (size_t e = 0; e < edges.size(); ++e) {
575  Dependence &edge = edges[e];
576  if (edge.in->isLoad == edge.out->isLoad)
577  continue;
578  llvm::Optional<size_t> maybeIndex = getOverlapIndex(edge);
579  if (!maybeIndex)
580  continue;
581  size_t index = *maybeIndex;
582  ScheduledNode &node = nodes[index];
583  if (node.phiIsScheduled(0) ||
584  (edge.in->indexMatrix() != edge.out->indexMatrix()))
585  continue;
586  PtrMatrix<int64_t> indMat = edge.in->indexMatrix();
587  size_t r = NormalForm::rank(indMat);
588  if (r == edge.in->getNumLoops())
589  continue;
590  // TODO handle linearly dependent acceses, filtering them out
591  if (r == edge.in->ref.getArrayDim()) {
592  // indMat indvars are indexed from outside<->inside
593  // phi indvars are indexed from inside<->outside
594  // so, indMat is indvars[outside<->inside] x array dim
595  // phi is loop[outside<->inside] x
596  // indvars[inside<->outside]
597  MutPtrMatrix<int64_t> phi = node.schedule.getPhi();
598  const size_t indR = indMat.numRow();
599  const size_t phiOffset = phi.numCol() - indR;
600  for (size_t rr = 0; rr < r; ++rr) {
601  phi(rr, _(begin, phiOffset)) = 0;
602  phi(rr, _(phiOffset, phiOffset + indR)) = indMat(_, rr);
603  }
604  // node.schedule.getPhi()(_(0, r), _) =
605  indMat.transpose();
606  node.rank = r;
607  tryOrth = true;
608  }
609  }
610  if (tryOrth) {
611  if (llvm::Optional<BitSet> opt = optimize(g, 0, maxDepth)) {
612  llvm::errs() << "orth opt succeeded!\n";
613  return opt;
614  }
615  for (auto &&n : nodes)
616  n.rank = 0;
617  }
618  return optimize(std::move(g), 0, maxDepth);
619  }

745  void instantiateOmniSimplex(const Graph &g, size_t d,
746  bool satisfyDeps = false) {
747  // defines numScheduleCoefs, numLambda, numBounding, and
748  // numConstraints
749  omniSimplex.reserve(numConstraints + numOmegaCoefs,
750  1 + numBounding + numActiveEdges + numPhiCoefs +
751  2 * numOmegaCoefs + numLambda);
752  omniSimplex.resizeForOverwrite(
753  numConstraints, 1 + numBounding + numActiveEdges + numPhiCoefs +
754  numOmegaCoefs + numLambda);
755  auto C{omniSimplex.getConstraints()};
756  C = 0;
757  // layout of omniSimplex:
758  // Order: C, then priority to minimize
759  // all : C, u, w, Phis, omegas, lambdas
760  // rows give constraints; each edge gets its own
761  // constexpr size_t numOmega =
762  // DependencePolyhedra::getNumOmegaCoefficients();
763  size_t u = 1, w = 1 + numBounding;
764  size_t c = 0, l = getLambdaOffset();
765  for (size_t e = 0; e < edges.size(); ++e) {
766  Dependence &edge = edges[e];
767  if (g.isInactive(e, d))
768  continue;
769  BitSet &outNodeIndexSet = edge.out->nodeIndex;
770  BitSet &inNodeIndexSet = edge.in->nodeIndex;
771  const auto [satC, satL, satPp, satPc, satO, satW] =
772  edge.splitSatisfaction();
773  const auto [bndC, bndL, bndPp, bndPc, bndO, bndWU] =
774  edge.splitBounding();
775  const size_t numSatConstraints = satC.size();
776  const size_t numBndConstraints = bndC.size();
777  for (auto outNodeIndex : outNodeIndexSet) {
778  const ScheduledNode &outNode = nodes[outNodeIndex];
779  for (auto inNodeIndex : inNodeIndexSet) {
780  const ScheduledNode &inNode = nodes[inNodeIndex];
781 
782  size_t cc = c + numSatConstraints;
783  size_t ccc = cc + numBndConstraints;
784 
785  size_t ll = l + satL.numCol();
786  size_t lll = ll + bndL.numCol();
787  C(_(c, cc), _(l, ll)) = satL;
788  C(_(cc, ccc), _(ll, lll)) = bndL;
789  l = lll;
790 
791  // bounding
792  C(_(cc, ccc), w++) = bndWU(_, 0);
793  size_t uu = u + bndWU.numCol() - 1;
794  C(_(cc, ccc), _(u, uu)) = bndWU(_, _(1, end));
795  u = uu;
796  if (satisfyDeps)
797  C(_(c, cc), 0) = satC + satW;
798  else
799  C(_(c, cc), 0) = satC;
800  C(_(cc, ccc), 0) = bndC;
801  // now, handle Phi and Omega
802  // phis are not constrained to be 0
803  if (outNodeIndex == inNodeIndex) {
804  // llvm::errs() << "outNodeIndex == inNodeIndex\n";
805  if (d < outNode.getNumLoops()) {
806  if (satPc.numCol() == satPp.numCol()) {
807  if (outNode.phiIsScheduled(d)) {
808  // add it constants
809  auto sch = outNode.getSchedule(d);
810  C(_(c, cc), 0) -=
811  satPc *
812  sch(_(end - satPc.numCol(), end)) +
813  satPp *
814  sch(_(end - satPp.numCol(), end));
815  C(_(cc, ccc), 0) -=
816  bndPc *
817  sch(_(end - bndPc.numCol(), end)) +
818  bndPp *
819  sch(_(end - satPp.numCol(), end));
820  } else {
821  // FIXME: phiChild = [14:18), 4 cols
822  // while Dependence seems to indicate 2
823  // loops why the disagreement?
824  auto phiChild = outNode.getPhiOffset();
825  C(_(c, cc), _(phiChild - satPc.numCol(),
826  phiChild)) = satPc + satPp;
827  C(_(cc, ccc), _(phiChild - bndPc.numCol(),
828  phiChild)) = bndPc + bndPp;
829  }
830  } else if (outNode.phiIsScheduled(d)) {
831  // add it constants
832  // note that loop order in schedule goes
833  // inner -> outer
834  // so we need to drop inner most if one has less
835  auto sch = outNode.getSchedule(d);
836  auto schP = sch(_(end - satPp.numCol(), end));
837  auto schC = sch(_(end - satPc.numCol(), end));
838  C(_(c, cc), 0) -= satPc * schC + satPp * schP;
839  C(_(cc, ccc), 0) -= bndPc * schC + bndPp * schP;
840  } else if (satPc.numCol() < satPp.numCol()) {
841  auto phiChild = outNode.getPhiOffset();
842  size_t P = satPc.numCol();
843  auto m = phiChild - P;
844  C(_(c, cc), _(phiChild - satPp.numCol(), m)) =
845  satPp(_, _(begin, end - P));
846  C(_(cc, ccc), _(phiChild - bndPp.numCol(), m)) =
847  bndPp(_, _(begin, end - P));
848  C(_(c, cc), _(m, phiChild)) =
849  satPc + satPp(_, _(end - P, end));
850  C(_(cc, ccc), _(m, phiChild)) =
851  bndPc + bndPp(_, _(end - P, end));
852  } else /* if (satPc.numCol() > satPp.numCol()) */ {
853  auto phiChild = outNode.getPhiOffset();
854  size_t P = satPp.numCol();
855  auto m = phiChild - P;
856  C(_(c, cc), _(phiChild - satPc.numCol(), m)) =
857  satPc(_, _(begin, end - P));
858  C(_(cc, ccc), _(phiChild - bndPc.numCol(), m)) =
859  bndPc(_, _(begin, end - P));
860  C(_(c, cc), _(m, phiChild)) =
861  satPc(_, _(end - P, end)) + satPp;
862  C(_(cc, ccc), _(m, phiChild)) =
863  bndPc(_, _(end - P, end)) + bndPp;
864  }
865  C(_(c, cc), outNode.omegaOffset) =
866  satO(_, 0) + satO(_, 1);
867  C(_(cc, ccc), outNode.omegaOffset) =
868  bndO(_, 0) + bndO(_, 1);
869  }
870  } else {
871  // llvm::errs() << "outNodeIndex != inNodeIndex\n";
872  if (d < edge.out->getNumLoops())
873  updateConstraints(C, outNode, satPc, bndPc, d, c,
874  cc, ccc);
875  if (d < edge.in->getNumLoops())
876  updateConstraints(C, inNode, satPp, bndPp, d, c, cc,
877  ccc);
878  // Omegas are included regardless of rotation
879  if (d < edge.out->getNumLoops()) {
880  C(_(c, cc), outNode.omegaOffset) =
881  satO(_, !edge.forward);
882  C(_(cc, ccc), outNode.omegaOffset) =
883  bndO(_, !edge.forward);
884  }
885  if (d < edge.in->getNumLoops()) {
886  C(_(c, cc), inNode.omegaOffset) =
887  satO(_, edge.forward);
888  C(_(cc, ccc), inNode.omegaOffset) =
889  bndO(_, edge.forward);
890  }
891  }
892  c = ccc;
893  }
894  }
895  }
896  }
897  void updateConstraints(MutPtrMatrix<int64_t> C, const ScheduledNode &node,
898  PtrMatrix<int64_t> sat, PtrMatrix<int64_t> bnd,
899  size_t d, size_t c, size_t cc, size_t ccc) {
900  if (node.phiIsScheduled(d)) {
901  // add it constants
902  auto sch = node.getSchedule(d);
903  // order is inner <-> outer
904  // so we need the end of schedule if it is larger
905  C(_(c, cc), 0) -= sat * sch(_(end - sat.numCol(), end));
906  C(_(cc, ccc), 0) -= bnd * sch(_(end - bnd.numCol(), end));
907  } else {
908  assert(sat.numCol() == bnd.numCol());
909  // add it to C
910  auto phiChild = node.getPhiOffset();
911  C(_(c, cc), _(phiChild - sat.numCol(), phiChild)) = sat;
912  C(_(cc, ccc), _(phiChild - bnd.numCol(), phiChild)) = bnd;
913  }
914  }
