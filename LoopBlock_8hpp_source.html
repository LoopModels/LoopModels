154  auto optimize(IR::Cache &cache, IR::TreeResult tr) -> Result {
155  // first, we peel loops for which affine repr failed
156  if (unsigned numReject = tr.rejectDepth) {
157  auto *SE = cache.getScalarEvolution();
158  for (Addr *stow = tr.stow; stow;
159  stow = llvm::cast_or_null<Addr>(stow->getNext()))
160  stow->peelLoops(cache.getAllocator(), numReject, SE);
161  for (Addr *load = tr.load; load;
162  load = llvm::cast_or_null<Addr>(load->getNext()))
163  load->peelLoops(cache.getAllocator(), numReject, SE);
164  }
165  // fill the dependence edges between memory accesses
166  for (Addr *stow = tr.stow; stow;) {
167  // TODO: check we don't have mutually exclusive predicates
168  Addr *next = llvm::cast_or_null<Addr>(stow->getNext());
169  for (Addr *other = next; other;
170  other = llvm::cast_or_null<Addr>(other->getNext()))
171  Dependence::check(&allocator, stow, other);
172  for (Addr *other = tr.load; other;
173  other = llvm::cast_or_null<Addr>(other->getNext()))
174  Dependence::check(&allocator, stow, other);
175  stow = next;
176  }
177  // link stores with loads connected through registers
178  ScheduledNode *nodes{nullptr};
179  for (Addr *stow = tr.stow; stow;
180  stow = llvm::cast_or_null<Addr>(stow->getNext()))
181  nodes = addScheduledNode(cache, stow)->addNext(nodes);
182  unsigned maxDepth = 0;
183  for (ScheduledNode *node : nodes->getVertices()) {
184  maxDepth = std::max(maxDepth, node->getNumLoops());
185  shiftOmega(node);
186  }
187  return optOrth(nodes, maxDepth);

195  auto addScheduledNode(IR::Cache &cache, IR::Stow stow) -> ScheduledNode * {
196  // how are we going to handle load duplication?
197  // we also need to duplicate the instruction graph leading to the node
198  // implying we need to track that tree.
199  // w = a[i]
200  // x = log(w)
201  // y = 2*x
202  // z = 3*x
203  // p = z / 5
204  // q = 5 / z
205  // s = p - q
206  // b[i] = y
207  // c[i] = s
208  // if adding c[i] after b[i], we must duplicate `w` and `x`
209  // but duplicating `z, `p`, `q`, or `s` is unnecessary.
210  // We don't need to duplicate those instructions where
211  // all uses only lead to `c[i]`.
212  // The trick we use is to mark each instruction with
213  // the store that visited it.
214  // If one has already been visited, duplicate and
215  // mark the new one.
216  auto [storedVal, maxLoop] =
217  searchOperandsForLoads(cache, stow, stow.getStoredVal());
218  maxLoop = deeperLoop(maxLoop, stow.getLoop());
219  stow.setVal(storedVal); // in case it changed
220  return ScheduledNode::construct(cache.getAllocator(), stow, maxLoop);
221  }
222  static constexpr auto deeperLoop(poly::Loop *a, poly::Loop *b)
223  -> poly::Loop * {
224  if (!a) return b;
225  if (!b) return a;
226  return (a->getNumLoops() > b->getNumLoops()) ? a : b;
227  }

253  auto searchOperandsForLoads(IR::Cache &cache, IR::Stow stow, Value *val)
254  -> std::pair<Value *, poly::Loop *> {
255  auto *inst = llvm::dyn_cast<Instruction>(val);
256  if (!inst) return {val, nullptr};
257  // we use parent/child relationships here instead of next/prev
258  if (Load load = IR::Load(inst)) {
259  // TODO: check we don't have mutually exclusive predicates
260  // we found a load; first we check if it has already been added
261  if (load.getParent() != nullptr) {
262  Arena<> *alloc = cache.getAllocator();
263  IR::Addr *reload = ((Addr *)load)->reload(alloc);
264  Dependence::copyDependencies(alloc, load, reload);
265  invariant(reload->isLoad());
266  load = reload;
267  }
268  stow.insertChild(load);
269  return {load, load.getLoop()};
270  // it has been, therefore we need to copy the load
271  }
272  // if not a load, check if it is stored, so we reload
273  Addr *store{nullptr};
274  for (Value *use : inst->getUsers()) {
275  if (auto other = IR::Stow(use)) {
276  store = other;
277  if (other == stow) break; // scan all users
278  }
279  }
280  if (store && (store != (Addr *)stow)) {
281  Addr *load = Dependence::reload(&allocator, store);
282  stow.insertChild(load); // insert load after stow
283  return {load, load->getLoop()};
284  }
285  auto *C = llvm::cast<IR::Compute>(inst);
286  // could not find a load, so now we recurse, searching operands
287  poly::Loop *maxLoop = nullptr;
288  auto s = allocator.scope(); // create temporary
289  unsigned numOps = C->getNumOperands();
290  MutPtrVector<Value *> newOperands{
291  math::vector<Value *>(&allocator, numOps)};
292  bool opsChanged = false;
293  for (ptrdiff_t i = 0; i < numOps; ++i) {
294  Value *op = C->getOperand(i);
295  auto [updatedOp, loop] = searchOperandsForLoads(cache, stow, op);
296  maxLoop = deeperLoop(maxLoop, loop);
297  if (op != updatedOp) opsChanged = true;
298  newOperands[i] = updatedOp;
299  }
300  if (opsChanged) val = cache.similarCompute(C, newOperands);
301  return {val, maxLoop};
302  }

304  void shiftOmega(ScheduledNode *node) {
305  unsigned nLoops = node->getNumLoops();
306  if (nLoops == 0) return;
307  auto p0 = allocator.checkpoint();
308  MutPtrVector<int64_t> offs = math::vector<int64_t>(&allocator, nLoops);
309  auto p1 = allocator.checkpoint();
310  MutSquarePtrMatrix<int64_t> A =
311  math::matrix<int64_t>(&allocator, nLoops + 1);
312  // BumpPtrVector<std::pair<BitSet64, int64_t>> omegaOffsets{allocator};
313  // // we check all memory accesses in the node, to see if applying the same
314  // omega offsets can zero dependence offsets. If so, we apply the shift.
315  // we look for offsets, then try and validate that the shift
316  // if not valid, we drop it from the potential candidates.
317  bool foundNonZeroOffset = false;
318  unsigned rank = 0, L = nLoops - 1;
319  for (ScheduledNode *n = node; n; n = n->getNext()) {
320  for (Addr *m = n->getStore(); m;
321  m = llvm::cast_or_null<Addr>(m->getChild())) {
322  for (Dependence *dep = m->getEdgeIn(); dep; dep = dep->getNextInput()) {
323  const DepPoly *depPoly = dep->getDepPoly();
324  unsigned numSyms = depPoly->getNumSymbols(),
325  dep0 = depPoly->getDim0(), dep1 = depPoly->getDim1();
326  PtrMatrix<int64_t> E = depPoly->getE();
327  if (dep->input()->getNode() == n) {
328  // dep within node
329  unsigned depCommon = std::min(dep0, dep1),
330  depMax = std::max(dep0, dep1);
331  invariant(nLoops >= depMax);
332  // input and output, no relative shift of shared loops possible
333  // but indices may of course differ.
334  for (ptrdiff_t d = 0; d < E.numRow(); ++d) {
335  MutPtrVector<int64_t> x = A(rank, _);
336  x[last] = E(d, 0);
337  foundNonZeroOffset |= x[last] != 0;
338  ptrdiff_t j = 0;
339  for (; j < depCommon; ++j)
340  x[L - j] = E(d, j + numSyms) + E(d, j + numSyms + dep0);
341  if (dep0 != dep1) {
342  ptrdiff_t offset = dep0 > dep1 ? numSyms : numSyms + dep0;
343  for (; j < depMax; ++j) x[L - j] = E(d, j + offset);
344  }
345  for (; j < nLoops; ++j) x[L - j] = 0;
346  rank = math::NormalForm::updateForNewRow(A(_(0, rank + 1), _));
347  }
348  } else {
349  // dep between nodes
350  // is forward means other -> mem, else mem <- other
351  unsigned offset = dep->isForward() ? numSyms + dep0 : numSyms,
352  numDep = dep->isForward() ? dep1 : dep0;
353  for (ptrdiff_t d = 0; d < E.numRow(); ++d) {
354  MutPtrVector<int64_t> x = A(rank, _);
355  x[last] = E(d, 0);
356  foundNonZeroOffset |= x[last] != 0;
357  ptrdiff_t j = 0;
358  for (; j < numDep; ++j) x[L - j] = E(d, j + offset);
359  for (; j < nLoops; ++j) x[L - j] = 0;
360  rank = math::NormalForm::updateForNewRow(A(_(0, rank + 1), _));
361  }
362  }
363  }
364  for (Dependence *dep = m->getEdgeOut(); dep;
365  dep = dep->getNextOutput()) {
366  if (dep->output()->getNode() == n) continue;
367  const DepPoly *depPoly = dep->getDepPoly();
368  unsigned numSyms = depPoly->getNumSymbols(),
369  dep0 = depPoly->getDim0(), dep1 = depPoly->getDim1();
370  PtrMatrix<int64_t> E = depPoly->getE();
371  // is forward means mem -> other, else other <- mem
372  unsigned offset = dep->isForward() ? numSyms : numSyms + dep0,
373  numDep = dep->isForward() ? dep0 : dep1;
374  for (ptrdiff_t d = 0; d < E.numRow(); ++d) {
375  MutPtrVector<int64_t> x = A(rank, _);
376  x[last] = E(d, 0);
377  foundNonZeroOffset |= x[last] != 0;
378  ptrdiff_t j = 0;
379  for (; j < numDep; ++j) x[L - j] = E(d, j + offset);
380  for (; j < nLoops; ++j) x[L - j] = 0;
381  rank = math::NormalForm::updateForNewRow(A(_(0, rank + 1), _));
382  }
383  }
384  }
385  }
386  if (!foundNonZeroOffset) return allocator.rollback(p0);
387  bool nonZero = false;
388  // matrix A is reasonably diagonalized, should indicate
389  ptrdiff_t c = 0;
390  for (ptrdiff_t r = 0; r < rank; ++r) {
391  int64_t off = A(r, last);
392  if (off == 0) continue;
393  for (; c < nLoops; ++c) {
394  if (A(r, c) != 0) break;
395  offs[L - c] = 0;
396  }
397  if (c == nLoops) return;
398  int64_t Arc = A(r, c), x = off / Arc;
399  if (x * Arc != off) continue;
400  offs[L - c++] = x; // decrement loop `L-c` by `x`
401  nonZero = true;
402  }
403  if (!nonZero) return allocator.rollback(p0);
404  allocator.rollback(p1);
405  for (; c < nLoops; ++c) offs[L - c] = 0;
406  node->setOffsets(offs.data());
407  // now we iterate over the edges again
408  // perhaps this should be abstracted into higher order functions that
409  // iterate over the edges?
410  for (ScheduledNode *n = node; n; n = n->getNext()) {
411  for (Addr *m = n->getStore(); m;
412  m = llvm::cast_or_null<Addr>(m->getChild())) {
413  for (Dependence *d = m->getEdgeIn(); d; d = d->getNextInput()) {
414  d->copySimplices(&allocator); // in case it is aliased
415  DepPoly *depPoly = d->getDepPoly();
416  unsigned numSyms = depPoly->getNumSymbols(),
417  dep0 = depPoly->getDim0(), dep1 = depPoly->getDim1();
418  MutPtrMatrix<int64_t> satL = d->getSatLambda();
419  MutPtrMatrix<int64_t> bndL = d->getBndLambda();
420  bool pick = d->isForward(), repeat = d->input()->getNode() == n;
421  while (true) {
422  unsigned offset = pick ? numSyms + dep0 : numSyms,
423  numDep = pick ? dep1 : dep0;
424  for (ptrdiff_t l = 0; l < numDep; ++l) {
425  int64_t mlt = offs[l];
426  if (mlt == 0) continue;
427  satL(0, _) -= mlt * satL(offset + l, _);
428  bndL(0, _) -= mlt * bndL(offset + l, _);
429  }
430  if (!repeat) break;
431  repeat = false;
432  pick = !pick;
433  }
434  }
435  for (Dependence *d = m->getEdgeOut(); d; d = d->getNextOutput()) {
436  if (d->output()->getNode() == n) continue; // handled above
437  d->copySimplices(&allocator); // we don't want to copy twice
438  DepPoly *depPoly = d->getDepPoly();
439  unsigned numSyms = depPoly->getNumSymbols(),
440  dep0 = depPoly->getDim0(), dep1 = depPoly->getDim1();
441  MutPtrMatrix<int64_t> satL = d->getSatLambda();
442  MutPtrMatrix<int64_t> bndL = d->getBndLambda();
443  unsigned offset = d->isForward() ? numSyms : numSyms + dep0,
444  numDep = d->isForward() ? dep0 : dep1;
445  for (size_t l = 0; l < numDep; ++l) {
446  int64_t mlt = offs[l];
447  if (mlt == 0) continue;
448  satL(0, _) -= mlt * satL(offset + l, _);
449  bndL(0, _) -= mlt * bndL(offset + l, _);
450  }
451  }
452  }
453  }
454  }
455  // returns a `1` for each level containing a dependency
456  auto optOrth(ScheduledNode *nodes, unsigned maxDepth) -> Result {
457  // check for orthogonalization opportunities
458  bool tryOrth = false;
459  for (ScheduledNode *node : nodes->getVertices()) {
460  for (Dependence *edge : node->inputEdges()) {
461  // this edge's output is `node`
462  // we want edges whose input is also `node`,
463  // i.e. edges that are within the node
464  if (edge->input()->getNode() != node) continue;
465  DensePtrMatrix<int64_t> indMat = edge->getInIndMat();
466  // check that we haven't already scheduled on an earlier
467  // iteration of this loop, and that the indmats are the same
468  if (node->phiIsScheduled(0) || (indMat != edge->getOutIndMat()))
469  continue;
470  size_t r = math::NormalForm::rank(indMat);
471  if (r == edge->getInNumLoops()) continue;
472  // TODO handle linearly dependent acceses, filtering them out
473  if (r != size_t(indMat.numRow())) continue;
474  node->schedulePhi(indMat, r);
475  tryOrth = true;
476  }
477  }
478  if (tryOrth) {
479  if (Result r = optimize(nodes, 0, maxDepth)) return r;
480  for (ScheduledNode *n : nodes->getVertices()) n->unschedulePhi();
481  }
482  return optimize(nodes, 0, maxDepth);
483  }

510  static constexpr auto setScheduleMemoryOffsets(ScheduledNode *nodes,
511  unsigned d)
512  -> std::array<unsigned, 3> {
513  // C, lambdas, omegas, Phis
514  unsigned numOmegaCoefs = 0, numPhiCoefs = 0, numSlack = 0;
515  for (ScheduledNode *node : nodes->getVertices()) {
516  // note, we had d > node->getNumLoops() for omegas earlier; why?
517  // TODO: audit edge-sat and skip checks; e.g. are edges being deactivated
518  // when depth exceeds the number of loops?
519  if ((d >= node->getNumLoops()) || (!node->hasActiveEdges(d))) continue;
520  numOmegaCoefs = node->updateOmegaOffset(numOmegaCoefs);
521  if (node->phiIsScheduled(d)) continue;
522  numPhiCoefs = node->updatePhiOffset(numPhiCoefs);
523  ++numSlack;
524  }
525  return {numOmegaCoefs, numPhiCoefs, numSlack};

527  static constexpr auto calcCoefs(ScheduledNode *nodes, unsigned d)
528  -> CoefCounts {

530  setScheduleMemoryOffsets(nodes, d);
531  auto [numLambda, numBounding, numConstraints, numActiveEdges] =
532  countAuxParamsAndConstraints(nodes, d);
533  return {numOmegaCoefs, numPhiCoefs, numSlack, numLambda,
534  numBounding, numConstraints, numActiveEdges};

546  [[nodiscard]] auto optimize(ScheduledNode *nodes, unsigned d,
547  unsigned maxDepth) -> Result {
548  if (d >= maxDepth) return Result::independent();
549  if (Result r = solveGraph(nodes, maxDepth, false)) {
550  unsigned descend = d + 1;
551  if (descend == maxDepth) return r;
552  if (Result n = optimize(nodes, descend, maxDepth)) {
553  if ((r == Result::dependent()) &&
554  (n == Result::dependent())) // try and sat all this level
555  return optimizeSatDep(nodes, d, maxDepth, r & n);
556  return r & n;
557  }
558  }
559  return breakGraph(nodes, d);
560  }
566  [[nodiscard]] auto solveGraph(ScheduledNode *nodes, unsigned depth,
567  bool satisfyDeps) -> Result {
568  CoefCounts counts{calcCoefs(nodes, depth)};
569  return solveGraph(nodes, depth, satisfyDeps, counts);

571  [[nodiscard]] auto solveGraph(ScheduledNode *nodes, unsigned depth,
572  bool satisfyDeps, CoefCounts counts) -> Result {
573  if (counts.numLambda == 0) {
574  setSchedulesIndependent(nodes, depth);
575  return checkEmptySatEdges(nodes, depth);
576  }
577  // TODO: sat Deps should check which stashed ones to satisfy
578  // use `edge->isCondIndep()`/`edge->preventsReodering()` to check
579  // which edges should be satisfied on this level if `satisfyDeps`
580  auto omniSimplex =
581  instantiateOmniSimplex(nodes, depth, satisfyDeps, counts);
582  if (omniSimplex->initiateFeasible()) return {};
583  auto sol = omniSimplex->rLexMinStop(counts.numLambda + counts.numSlack);
584  assert(sol.size() == counts.numBounding + counts.numActiveEdges +
585  counts.numPhiCoefs + counts.numOmegaCoefs);
586  updateSchedules(nodes, depth, counts, sol);
587  return deactivateSatisfiedEdges(
588  nodes, depth, counts,
589  sol[_(counts.numPhiCoefs + counts.numOmegaCoefs, end)]);
590  }

611  }

613  DenseMatrix<int64_t> nullSpace; // d x lfull
614  DenseMatrix<int64_t> A{node->getPhi()(_(0, depth), _).transpose()};
615  math::NormalForm::nullSpace11(nullSpace, A);
616  invariant(unsigned(nullSpace.numRow()), node->getNumLoops() - depth);
617  // Now, we search index matrices for schedules not in the null space of
618  // existing phi. This is because we're looking to orthogonalize a
619  // memory access if possible, rather than setting a schedule
620  // arbitrarily.
621  // Here, we collect candidates for the next schedule
622  DenseMatrix<int64_t> candidates{
623  math::DenseDims{0, node->getNumLoops() + 1}};
624  Vector<int64_t> indv;
625  indv.resizeForOverwrite(node->getNumLoops());
626  for (Addr *mem : node->localAddr()) {
627  PtrMatrix<int64_t> indMat = mem->indexMatrix(); // lsub x d
628  A.resizeForOverwrite(
629  math::DenseDims{nullSpace.numRow(), indMat.numCol()});
630  A = nullSpace(_, _(0, indMat.numRow())) * indMat;
631  // we search A for rows that aren't all zero
632  for (ptrdiff_t d = 0; d < A.numCol(); ++d) {
633  if (allZero(A(_, d))) continue;
634  indv << indMat(_, d);
635  bool found = false;
636  for (ptrdiff_t j = 0; j < candidates.numRow(); ++j) {
637  if (candidates(j, _(0, last)) != indv) continue;
638  found = true;
639  ++candidates(j, 0);
640  break;
641  }
642  if (!found) {
643  candidates.resize(candidates.numRow() + 1);
644  assert(candidates(last, 0) == 0);
645  candidates(last, _(1, end)) << indv;
646  }
647  }
648  }
649  if (Row R = candidates.numRow()) {
650  // >= 1 candidates, pick the one with with greatest lex, favoring
651  // number of repetitions (which were placed in first index)
652  ptrdiff_t i = 0;
653  for (ptrdiff_t j = 1; j < candidates.numRow(); ++j)
654  if (candidates(j, _) > candidates(i, _)) i = j;
655  node->getSchedule(depth) << candidates(i, _(1, end));
656  return;
657  }
658  // do we want to pick the outermost original loop,
659  // or do we want to pick the outermost lex null space?
660  node->getSchedule(depth) << 0;
661  for (ptrdiff_t c = 0; c < nullSpace.numCol(); ++c) {
662  if (allZero(nullSpace(_, c))) continue;
663  node->getSchedule(depth)[c] = 1;
664  return;
665  }
666  invariant(false);
667  }
668  static void updateSchedules(ScheduledNode *nodes, unsigned depth,
669  CoefCounts counts, Simplex::Solution sol) {
670 #ifndef NDEBUG
671  if (counts.numPhiCoefs > 0)
672  assert(std::ranges::any_of(
673  sol, [](math::Rational s) -> bool { return s != 0; }));
674 #endif
675  unsigned o = counts.numOmegaCoefs;
676  for (ScheduledNode *node : nodes->getVertices()) {
677  if (depth >= node->getNumLoops()) continue;
678  if (!node->hasActiveEdges(depth)) {
679  setDepFreeSchedule(node, depth);
680  continue;
681  }
682  math::Rational sOmega = sol[node->getOmegaOffset()];
683  // TODO: handle s.denominator != 1
684  if (!node->phiIsScheduled(depth)) {
685  auto phi = node->getSchedule(depth);
686  auto s = sol[node->getPhiOffsetRange() + o];
687  int64_t baseDenom = sOmega.denominator;
688  int64_t l = math::lcm(s.denomLCM(), baseDenom);
689 #ifndef NDEBUG
690  for (ptrdiff_t i = 0; i < phi.size(); ++i)
691  assert(((s[i].numerator * l) / (s[i].denominator)) >= 0);
692 #endif
693  if (l == 1) {
694  node->getOffsetOmega(depth) = sOmega.numerator;
695  for (ptrdiff_t i = 0; i < phi.size(); ++i) phi[i] = s[i].numerator;
696  } else {
697  node->getOffsetOmega(depth) = (sOmega.numerator * l) / baseDenom;
698  for (ptrdiff_t i = 0; i < phi.size(); ++i)
699  phi[i] = (s[i].numerator * l) / (s[i].denominator);
700  }
701  assert(!(allZero(phi)));
702  } else {
703  node->getOffsetOmega(depth) = sOmega.numerator;
704  }
705 #ifndef NDEBUG
706  if (!node->phiIsScheduled(depth)) {
707  int64_t l = sol[node->getPhiOffsetRange() + o].denomLCM();
708  for (ptrdiff_t i = 0; i < node->getPhi().numCol(); ++i)
709  assert(node->getPhi()(depth, i) ==
710  sol[node->getPhiOffsetRange() + o][i] * l);
711  }
712 #endif
713  }
714  }
715  [[nodiscard]] auto deactivateSatisfiedEdges(ScheduledNode *nodes,
716  unsigned depth, CoefCounts counts,
717  Simplex::Solution sol) -> Result {
718  if (allZero(sol[_(begin, counts.numBounding + counts.numActiveEdges)]))
719  return checkEmptySatEdges(nodes, depth);
720  ptrdiff_t w = 0, u = counts.numActiveEdges;
721  // TODO: update the deactivated edge handling
722  // must consider it w/ respect to `optimize`, `breakGraph`, and
723  // `optimizeSatDep`
724  // We want an indicator of which edges to try and eagerly satisfy
725  // `optimizeSatDep`; this ought to just be the `edge->satLevel()`
726  // so the flags we return here only need to be an indicator
727  // of whether we had to deactivate an edge on level `depth`.
728  //
729  // We don't set `deactivated=1 for `checkEmptySat` as we still have `w == 0`
730  // and `u == 0`, meaning it is still a parallizable loop -- so we haven't
731  // lost anything! The idea of trying to consolidate dependencies into one
732  // loop is, if we must already execute this loop in order, we should try and
733  // cover as many dependencies at that time as possible.
734  Result result{Result::Independent};
735  for (ScheduledNode *outNode : nodes->getVertices()) {
736  for (Dependence *edge : outNode->inputEdges()) {
737  if (edge->isInactive(depth)) continue;
738  Col uu = u + edge->getNumDynamicBoundingVar();
739  if ((sol[w++] != 0) || (anyNEZero(sol[_(u, uu)]))) {
740  edge->setSatLevelLP(depth);
741  result = Result::dependent();

743  ScheduledNode *inNode = edge->input()->getNode();
744  DensePtrMatrix<int64_t> inPhi = inNode->getPhi()(_(0, depth + 1), _),
745  outPhi =
746  outNode->getPhi()(_(0, depth + 1), _);
747  edge->checkEmptySat(
748  &allocator, inNode->getLoopNest(), inNode->getOffset(), inPhi,
749  outNode->getLoopNest(), outNode->getOffset(), outPhi);
750  }
751  u = ptrdiff_t(uu);
752  }
753  }
754  return result;

784  static auto popStash(
785  math::ResizeableView<std::pair<poly::AffineSchedule, ScheduledNode *>,
786  unsigned>
787  old) -> void {
788  // reconnect nodes, in case they became disconnected in breakGraph
789  // because we go in reverse order, connections should be the same
790  // so the original `nodes` should be restored.
791  ScheduledNode *n = nullptr;
792  for (auto &it : std::ranges::reverse_view(old)) {
793  n = it.second->addNext(n);
794  n->getSchedule() << it.first; // copy over
795  }
796  for (ScheduledNode *node : n->getVertices())
797  for (Dependence *d : node->inputEdges()) d->popSatLevel();
798  }

800  [[nodiscard]] auto optimizeSatDep(ScheduledNode *nodes, unsigned depth,
801  unsigned maxDepth, Result backupResult)
802  -> Result {
803  // if we're here, there are satisfied deps in both
804  // depSatLevel and depSatNest
805  // what we want to know is, can we satisfy all the deps
806  // in depSatNest?
807  // backup in case we fail
808  // activeEdges was the old original; swap it in
809  // we don't create long lasting allocations
810  auto scope = allocator.scope();
811  auto [old, counts] = stashFit(nodes, depth);
812  if (Result depSat = solveGraph(nodes, depth, true, counts))
813  if (Result depSatN = optimize(nodes, depth + 1, maxDepth))
814  return depSat & depSatN;
815  popStash(old);
816  return backupResult;
817  }

819  auto tryFuse(ScheduledNode *n0, ScheduledNode *n1, unsigned depth) -> Result {
820  auto s = allocator.scope();

832  auto s = allocator.scope();

853  [[nodiscard]] auto breakGraph(ScheduledNode *node, unsigned d) -> Result {
854  // Get a top sorting of SCC's; because we couldn't solve the graph
855  // with these dependencies fused, we'll try splitting them.
856  ScheduledNode *components =
857  graph::stronglyConnectedComponents(ScheduleGraph(d), node);
858  if (components->getNextComponent() == nullptr) return {};
859  // components are sorted in topological order.
860  // We split all of them, solve independently,
861  // and then try to fuse again after if/where optimal schedules
862  // allow it.
863  Result res{Result::Independent};
864  for (auto *g : components->getComponents())
865  if (Result sat = solveSplitGraph(g, d)) res &= sat;
866  else return Result::failure();
867  // We find we can successfully solve by splitting all legal splits.
868  // We could try and implement a better algorithm in the future, but for now
869  // we take a single greedy pass over the components.
870  // On each iteration, we either fuse our seed with the current component, or
871  // swap them. Thus, on each iteration, we're trying to merge each component
872  // with the topologically previous one (and those that one has fused with).
873  auto range = components->getComponents();
874  auto it = range.begin();
875  ScheduledNode *seed = *it;
876  int64_t unfusedOffset = 0;
877  for (auto e = decltype(range)::end(); ++it != e;) {
878  if (auto opt = tryFuse(seed, *it, d)) res &= opt;
879  else {
880  for (ScheduledNode *v : seed->getVertices())
881  v->getFusionOmega(d) = unfusedOffset;
882  ++unfusedOffset;
883  }
884  seed = *it; // if fused, seed was appended to `*it`
885  }
886  for (ScheduledNode *v : seed->getVertices())
887  v->getFusionOmega(d) = unfusedOffset;
888  return res;
889  }
905  static auto instantiateOmniSimplex(ScheduledNode *nodes, unsigned d,
906  bool satisfyDeps, CoefCounts counts)
907  -> std::unique_ptr<Simplex> {
908  auto [numOmegaCoefs, numPhiCoefs, numSlack, numLambda, numBounding,
909  numConstraints, numActiveEdges] = counts;
910  auto omniSimplex = Simplex::create(
911  numConstraints + numSlack, numBounding + numActiveEdges + numPhiCoefs +
912  numOmegaCoefs + numSlack + numLambda);
913  auto C{omniSimplex->getConstraints()};
914  C << 0;
915  // layout of omniSimplex:
916  // Order: C, then rev-priority to minimize
917  // C, lambdas, slack, omegas, Phis, w, u
918  // rows give constraints; each edge gets its own
919  // numBounding = num u
920  // numActiveEdges = num w
921  Row c = 0;
922  Col l = 1, o = 1 + numLambda + numSlack, p = o + numOmegaCoefs,
923  w = p + numPhiCoefs, u = w + numActiveEdges;
924  for (ScheduledNode *inNode : nodes->getVertices()) {
925  for (Dependence *edge : inNode->outputEdges(d)) {
926  ScheduledNode *outNode = edge->output()->getNode();
927  const auto [satPp, satPc] = edge->satPhiCoefs();
928  const auto [bndPp, bndPc] = edge->bndPhiCoefs();
929  math::StridedVector<int64_t> satC{edge->getSatConstants()},
930  satW{edge->getSatW()}, bndC{edge->getBndConstants()};
931  math::PtrMatrix<int64_t> satL{edge->getSatLambda()},
932  bndL{edge->getBndLambda()}, satO{edge->getSatOmegaCoefs()},
933  bndO{edge->getBndOmegaCoefs()}, bndWU{edge->getBndCoefs()};
934  const ptrdiff_t numSatConstraints = satC.size(),
935  numBndConstraints = bndC.size();
936  const Col nPc = satPc.numCol(), nPp = satPp.numCol();
937  invariant(nPc, bndPc.numCol());
938  invariant(nPp, bndPp.numCol());
939  Row cc = c + numSatConstraints;
940  Row ccc = cc + numBndConstraints;
941 
942  Col ll = l + satL.numCol();
943  Col lll = ll + bndL.numCol();
944  C(_(c, cc), _(l, ll)) << satL;
945  C(_(cc, ccc), _(ll, lll)) << bndL;
946  l = lll;
947  // bounding
948  C(_(cc, ccc), w++) << bndWU(_, 0);
949  Col uu = u + bndWU.numCol() - 1;
950  C(_(cc, ccc), _(u, uu)) << bndWU(_, _(1, end));
951  u = uu;
952  if (!satisfyDeps || !edge->stashedPreventsReordering(d))
953  C(_(c, cc), 0) << satC;
954  else C(_(c, cc), 0) << satC + satW;
955  C(_(cc, ccc), 0) << bndC;
956  // now, handle Phi and Omega
957  // phis are not constrained to be 0
958  if (outNode == inNode) {
959  if (d < outNode->getNumLoops()) {
960  if (nPc == nPp) {
961  if (outNode->phiIsScheduled(d)) {
962  // add it constants
963  auto sch = outNode->getSchedule(d);
964  C(_(c, cc), 0) -=
965  satPc * sch[_(0, nPc)] + satPp * sch[_(0, nPp)];
966  C(_(cc, ccc), 0) -=
967  bndPc * sch[_(0, nPc)] + bndPp * sch[_(0, nPp)];
968  } else {
969  // FIXME: phiChild = [14:18), 4 cols
970  // while Dependence seems to indicate 2
971  // loops why the disagreement?
972  auto po = outNode->getPhiOffset() + p;
973  C(_(c, cc), _(po, po + nPc)) << satPc + satPp;
974  C(_(cc, ccc), _(po, po + nPc)) << bndPc + bndPp;
975  }
976  } else if (outNode->phiIsScheduled(d)) {
977  // add it constants
978  // note that loop order in schedule goes
979  // inner -> outer
980  // so we need to drop inner most if one has less
981  auto sch = outNode->getSchedule(d);
982  auto schP = sch[_(0, nPp)];
983  auto schC = sch[_(0, nPc)];
984  C(_(c, cc), 0) -= satPc * schC + satPp * schP;
985  C(_(cc, ccc), 0) -= bndPc * schC + bndPp * schP;
986  } else if (nPc < nPp) {
987  // Pp has more cols, so outer/leftmost overlap
988  auto po = outNode->getPhiOffset() + p, poc = po + nPc,
989  pop = po + nPp;
990  C(_(c, cc), _(po, poc)) << satPc + satPp(_, _(0, nPc));
991  C(_(cc, ccc), _(po, poc)) << bndPc + bndPp(_, _(0, nPc));
992  C(_(c, cc), _(poc, pop)) << satPp(_, _(nPc, end));
993  C(_(cc, ccc), _(poc, pop)) << bndPp(_, _(nPc, end));
994  } else /* if (nPc > nPp) */ {
995  auto po = outNode->getPhiOffset() + p, poc = po + nPc,
996  pop = po + nPp;
997  C(_(c, cc), _(po, pop)) << satPc(_, _(0, nPp)) + satPp;
998  C(_(cc, ccc), _(po, pop)) << bndPc(_, _(0, nPp)) + bndPp;
999  C(_(c, cc), _(pop, poc)) << satPc(_, _(nPp, end));
1000  C(_(cc, ccc), _(pop, poc)) << bndPc(_, _(nPp, end));
1001  }
1002  C(_(c, cc), outNode->getOmegaOffset() + o)
1003  << satO(_, 0) + satO(_, 1);
1004  C(_(cc, ccc), outNode->getOmegaOffset() + o)
1005  << bndO(_, 0) + bndO(_, 1);
1006  }
1007  } else {
1008  if (d < edge->getOutNumLoops())
1009  updateConstraints(C, outNode, satPc, bndPc, d, c, cc, ccc, p);
1010  if (d < edge->getInNumLoops()) {
1011  if (d < edge->getOutNumLoops() && !inNode->phiIsScheduled(d) &&
1012  !outNode->phiIsScheduled(d)) {
1013  invariant(inNode->getPhiOffset() != outNode->getPhiOffset());
1014  }
1015  updateConstraints(C, inNode, satPp, bndPp, d, c, cc, ccc, p);
1016  }
1017  // Omegas are included regardless of rotation
1018  if (d < edge->getOutNumLoops()) {
1019  if (d < edge->getInNumLoops())
1020  invariant(inNode->getOmegaOffset() != outNode->getOmegaOffset());
1021  C(_(c, cc), outNode->getOmegaOffset() + o)
1022  << satO(_, edge->isForward());
1023  C(_(cc, ccc), outNode->getOmegaOffset() + o)
1024  << bndO(_, edge->isForward());
1025  }
1026  if (d < edge->getInNumLoops()) {
1027  C(_(c, cc), inNode->getOmegaOffset() + o)
1028  << satO(_, !edge->isForward());
1029  C(_(cc, ccc), inNode->getOmegaOffset() + o)
1030  << bndO(_, !edge->isForward());
1031  }
1032  }
1033  c = ccc;
1034  }
1035  }
1036  invariant(size_t(l), size_t(1 + numLambda));
1037  invariant(size_t(c), size_t(numConstraints));
1038  addIndependentSolutionConstraints(omniSimplex.get(), nodes, d, counts);
1039  return omniSimplex;
1040  }
1041  static void updateConstraints(MutPtrMatrix<int64_t> C,
1042  const ScheduledNode *node,
1043  PtrMatrix<int64_t> sat, PtrMatrix<int64_t> bnd,
1044  unsigned d, Row c, Row cc, Row ccc, Col p) {
1045  invariant(sat.numCol(), bnd.numCol());
1046  if (node->phiIsScheduled(d)) {
1047  // add it constants
1048  auto sch = node->getSchedule(d)[_(0, sat.numCol())];
1049  // order is inner <-> outer
1050  // so we need the end of schedule if it is larger
1051  C(_(c, cc), 0) -= sat * sch;
1052  C(_(cc, ccc), 0) -= bnd * sch;
1053  } else {
1054  // add it to C
1055  auto po = node->getPhiOffset() + p;
1056  C(_(c, cc), _(po, po + sat.numCol())) << sat;
1057  C(_(cc, ccc), _(po, po + bnd.numCol())) << bnd;
1058  }
1059  }
1060  static void addIndependentSolutionConstraints(NotNull<Simplex> omniSimplex,
1061  const ScheduledNode *nodes,
1062  unsigned d, CoefCounts counts) {
1063  // omniSimplex->setNumCons(omniSimplex->getNumCons() +
1064  // memory.size());
1065  // omniSimplex->reserveExtraRows(memory.size());
1066  auto C{omniSimplex->getConstraints()};
1067  ptrdiff_t i = ptrdiff_t{C.numRow()} - counts.numSlack, s = counts.numLambda,
1068  o = 1 + counts.numSlack + counts.numLambda + counts.numOmegaCoefs;
1069  if (d == 0) {
1070  // add ones >= 0
1071  for (const ScheduledNode *node : nodes->getVertices()) {
1072  if (node->phiIsScheduled(d) || (!node->hasActiveEdges(d))) continue;
1073  C(i, 0) = 1;
1074  C(i, node->getPhiOffsetRange() + o) << 1;
1075  C(i++, ++s) = -1; // for >=
1076  }
1077  } else {
1078  DenseMatrix<int64_t> A, N;
1079  for (const ScheduledNode *node : nodes->getVertices()) {
1080  if (node->phiIsScheduled(d) || (d >= node->getNumLoops()) ||
1081  (!node->hasActiveEdges(d)))
1082  continue;
1083  A.resizeForOverwrite(Row{ptrdiff_t(node->getPhi().numCol())}, Col{d});
1084  A << node->getPhi()(_(0, d), _).transpose();
1085  math::NormalForm::nullSpace11(N, A);
1086  // we add sum(NullSpace,dims=1) >= 1
1087  // via 1 = sum(NullSpace,dims=1) - s, s >= 0
1088  C(i, 0) = 1;
1089  MutPtrVector<int64_t> cc{C(i, node->getPhiOffsetRange() + o)};
1090  // sum(N,dims=1) >= 1 after flipping row signs to be lex > 0
1091  for (ptrdiff_t m = 0; m < N.numRow(); ++m)
1092  cc += N(m, _) * lexSign(N(m, _));
1093  C(i++, ++s) = -1; // for >=
1094  }
1095  }
1096  invariant(ptrdiff_t(omniSimplex->getNumCons()), i);
1097  assert(!allZero(omniSimplex->getConstraints()(last, _)));
1098  }
