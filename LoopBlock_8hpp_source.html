439  searchValueForStores(aset<llvm::User *> &visited, ScheduledNode &node,
440  amap<llvm::User *, unsigned> &userToMemory,
441  llvm::User *user, unsigned nodeIndex) -> bool {
442  for (llvm::User *use : user->users()) {
443  if (visited.contains(use)) continue;
444  if (llvm::isa<llvm::StoreInst>(use)) {
445  auto *ma = userToMemory.find(use);
446  if (ma == userToMemory.end()) continue;
447  // we want to reload a store
448  // this store will be treated as a load
449  NotNull<MemoryAccess> store = memory[ma->second];
450  auto [load, d] = Dependence::reload(allocator, store);
451  // for every store->store, we also want a load->store
452  for (auto o : store->outputEdges()) {
453  Dependence edge = edges[o];

455  pushToEdgeVector(edges, edge.replaceInput(load));
456  }

458  unsigned memId = memory.size();
459  memory.push_back(load);

879  auto instantiateOmniSimplex(const Graph &g, size_t d, bool satisfyDeps)
880  -> Optional<Simplex *> {
881  auto omniSimplex =
882  Simplex::create(allocator, numConstraints + numSlack,
883  numBounding + numActiveEdges + numPhiCoefs +
884  numOmegaCoefs + numSlack + numLambda);
885  auto C{omniSimplex->getConstraints()};
886  C << 0;
887  // layout of omniSimplex:
888  // Order: C, then rev-priority to minimize
889  // C, lambdas, slack, omegas, Phis, w, u
890  // rows give constraints; each edge gets its own
891  // numBounding = num u
892  // numActiveEdges = num w
893  Row c = 0;
894  Col l = 1, o = 1 + numLambda + numSlack, p = o + numOmegaCoefs,
895  w = p + numPhiCoefs, u = w + numActiveEdges;
896  for (size_t e = 0; e < edges.size(); ++e) {
897  Dependence &edge = edges[e];
898  if (g.isInactive(e, d)) continue;
899  const BitSet &outNodeIndexSet = edge.nodesOut();
900  const BitSet &inNodeIndexSet = edge.nodesIn();
901  const auto [satC, satL, satPp, satPc, satO, satW] =
902  edge.splitSatisfaction();
903  const auto [bndC, bndL, bndPp, bndPc, bndO, bndWU] = edge.splitBounding();
904  const size_t numSatConstraints = satC.size(),
905  numBndConstraints = bndC.size();
906  const Col nPc = satPc.numCol(), nPp = satPp.numCol();
907  invariant(nPc, bndPc.numCol());
908  invariant(nPp, bndPp.numCol());
909  for (auto outNodeIndex : outNodeIndexSet) {
910  const ScheduledNode &outNode = nodes[outNodeIndex];
911  for (auto inNodeIndex : inNodeIndexSet) {
912  const ScheduledNode &inNode = nodes[inNodeIndex];
913 
914  Row cc = c + numSatConstraints;
915  Row ccc = cc + numBndConstraints;
916 
917  Col ll = l + satL.numCol();
918  Col lll = ll + bndL.numCol();
919  C(_(c, cc), _(l, ll)) << satL;
920  C(_(cc, ccc), _(ll, lll)) << bndL;
921  l = lll;
922  // bounding
923  C(_(cc, ccc), w++) << bndWU(_, 0);
924  Col uu = u + bndWU.numCol() - 1;
925  C(_(cc, ccc), _(u, uu)) << bndWU(_, _(1, end));
926  u = uu;
927  if (satisfyDeps) C(_(c, cc), 0) << satC + satW;
928  else C(_(c, cc), 0) << satC;
929  C(_(cc, ccc), 0) << bndC;
930  // now, handle Phi and Omega
931  // phis are not constrained to be 0
932  if (outNodeIndex == inNodeIndex) {
933  if (d < outNode.getNumLoops()) {
934  if (nPc == nPp) {
935  if (outNode.phiIsScheduled(d)) {
936  // add it constants
937  auto sch = outNode.getSchedule(d);
938  C(_(c, cc), 0) -=
939  satPc * sch[_(0, nPc)] + satPp * sch[_(0, nPp)];
940  C(_(cc, ccc), 0) -=
941  bndPc * sch[_(0, nPc)] + bndPp * sch[_(0, nPp)];
942  } else {
943  // FIXME: phiChild = [14:18), 4 cols
944  // while Dependence seems to indicate 2
945  // loops why the disagreement?
946  auto po = outNode.getPhiOffset() + p;
947  C(_(c, cc), _(po, po + nPc)) << satPc + satPp;
948  C(_(cc, ccc), _(po, po + nPc)) << bndPc + bndPp;
949  }
950  } else if (outNode.phiIsScheduled(d)) {
951  // add it constants
952  // note that loop order in schedule goes
953  // inner -> outer
954  // so we need to drop inner most if one has less
955  auto sch = outNode.getSchedule(d);
956  auto schP = sch[_(0, nPp)];
957  auto schC = sch[_(0, nPc)];
958  C(_(c, cc), 0) -= satPc * schC + satPp * schP;
959  C(_(cc, ccc), 0) -= bndPc * schC + bndPp * schP;
960  } else if (nPc < nPp) {
961  // Pp has more cols, so outer/leftmost overlap
962  auto po = outNode.getPhiOffset() + p, poc = po + nPc,
963  pop = po + nPp;
964  C(_(c, cc), _(po, poc)) << satPc + satPp(_, _(0, nPc));
965  C(_(cc, ccc), _(po, poc)) << bndPc + bndPp(_, _(0, nPc));
966  C(_(c, cc), _(poc, pop)) << satPp(_, _(nPc, end));
967  C(_(cc, ccc), _(poc, pop)) << bndPp(_, _(nPc, end));
968  } else /* if (nPc > nPp) */ {
969  auto po = outNode.getPhiOffset() + p, poc = po + nPc,
970  pop = po + nPp;
971  C(_(c, cc), _(po, pop)) << satPc(_, _(0, nPp)) + satPp;
972  C(_(cc, ccc), _(po, pop)) << bndPc(_, _(0, nPp)) + bndPp;
973  C(_(c, cc), _(pop, poc)) << satPc(_, _(nPp, end));
974  C(_(cc, ccc), _(pop, poc)) << bndPc(_, _(nPp, end));
975  }
976  C(_(c, cc), outNode.getOmegaOffset() + o)
977  << satO(_, 0) + satO(_, 1);
978  C(_(cc, ccc), outNode.getOmegaOffset() + o)
979  << bndO(_, 0) + bndO(_, 1);
980  }
981  } else {
982  if (d < edge.getOutNumLoops())
983  updateConstraints(C, outNode, satPc, bndPc, d, c, cc, ccc, p);
984  if (d < edge.getInNumLoops()) {
985  if (d < edge.getOutNumLoops() && !inNode.phiIsScheduled(d) &&
986  !outNode.phiIsScheduled(d)) {
987  invariant(inNode.getPhiOffset() != outNode.getPhiOffset());
988  }
989  updateConstraints(C, inNode, satPp, bndPp, d, c, cc, ccc, p);
990  }
991  // Omegas are included regardless of rotation
992  if (d < edge.getOutNumLoops()) {
993  if (d < edge.getInNumLoops())
994  invariant(inNode.getOmegaOffset() != outNode.getOmegaOffset());
995  C(_(c, cc), outNode.getOmegaOffset() + o)
996  << satO(_, edge.isForward());
997  C(_(cc, ccc), outNode.getOmegaOffset() + o)
998  << bndO(_, edge.isForward());
999  }
1000  if (d < edge.getInNumLoops()) {
1001  C(_(c, cc), inNode.getOmegaOffset() + o)
1002  << satO(_, !edge.isForward());
1003  C(_(cc, ccc), inNode.getOmegaOffset() + o)
1004  << bndO(_, !edge.isForward());
1005  }
1006  }
1007  c = ccc;
1008  }
1009  }
1010  }
1011  invariant(size_t(l), size_t(1 + numLambda));
1012  invariant(size_t(c), size_t(numConstraints));
1013  addIndependentSolutionConstraints(omniSimplex, g, d);
1014  return omniSimplex->initiateFeasible() ? nullptr : (Simplex *)omniSimplex;
1015  }
