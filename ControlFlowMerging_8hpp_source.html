201  // merge the operands of `A` and `B`
202  // An abstraction that runs an algorithm to look for merging opportunities,
203  // either counting the number of selects needed, or allocating selects
204  // and returning the new operand vector.
205  // We generally aim to have analysis/cost modeling and code generation
206  // take the same code paths, to both avoid code duplication and to
207  // make sure the cost modeling reflects the actual code we're generating.
208  template <typename S>
209  auto mergeOperands(Instruction *A, Instruction *B, S selects) {
210  // TODO: does this update the predicates?
211  // now, we need to check everything connected to O and I in the mergeMap
212  // to see if any of them need to be updated.
213  // TODO: we want to estimate select cost
214  // worst case scenario is 1 select per operand (p is pred):
215  // select(p, f(a,b), f(c,d)) => f(select(p, a, c), select(p, b, d))
216  // but we can often do better, e.g. we may have
217  // select(p, f(a,b), f(c,b)) => f(select(p, a, c), b)
218  // additionally, we can check `I->associativeOperandsFlag()`
219  // select(p, f(a,b), f(c,a)) => f(a, select(p, b, c))
220  // we need to figure out which operands we're merging with which,
221  //
222  // We need to give special consideration to the case where
223  // arguments are merged, as this may be common when two
224  // control flow branches have relatively similar pieces.
225  // E.g., if b and c are already merged,
226  // and if `f`'s ops are associative, then we'd get
227  // select(p, f(a,b), f(c,a)) => f(a, b)
228  // so we need to check if any operand pairs are merged with each other.
229  // note `isMerged(a,a) == true`, so that's the one query we need to use.
230  auto selector = init(selects, A, B);
231  MutPtrVector<Value *> operandsA = A->getOperands();
232  MutPtrVector<Value *> operandsB = B->getOperands();
233  size_t numOperands = operandsA.size();
234  assert(numOperands == operandsB.size());
236  uint8_t associativeOpsFlag = B->associativeOperandsFlag();
237  // For example,
238  // we keep track of which operands we've already merged,
239  // f(a, b), f(b, b)
240  // we can't merge b twice!
241  for (size_t i = 0; i < numOperands; ++i) {
242  auto *opA = A->getOperand(i);
243  auto *opB = B->getOperand(i);
244  auto [assoc, assocFlag] = popBit(associativeOpsFlag);
245  associativeOpsFlag = assocFlag;
246  if (opA == opB) continue;
247  // if both operands were merged, we can ignore it's associativity
248  if (isMerged(opB, opA)) {
249  // we cast, because isMerged confirms they're Instructions, given
250  // that opA != opB, which we checked above
251  continue;
252  }
253  if (!((assoc) && (assocFlag))) {
254  // this op isn't associative with any remaining
255  selector.select(i, opA, opB);
256  continue;
257  }
258  // we look forward
259  size_t j = i;
260  bool merged = false;
261  while (assocFlag) {
262  auto shift = std::countr_zero(assocFlag);
263  j += ++shift;
264  assocFlag >>= shift;
265  auto *opjA = A->getOperand(j);
266  auto *opjB = B->getOperand(j);
267  // if elements in these pairs weren't already used
268  // to drop a select, and they're merged with each other
269  // we'll use them now to drop a select.
270  if (isMerged(opB, opjA)) {
271  std::swap(operandsA[i], operandsA[j]);
272  merged = true;
273  break;
274  }
275  if (isMerged(opjB, opA)) {
276  std::swap(operandsB[i], operandsB[j]);
277  merged = true;
278  break;
279  }
280  }
281  // we couldn't find any candidates
282  if (!merged) selector.select(i, opA, opB);
283  }
284  return unsigned(selector);
285  }

287  void merge(Arena<> *alloc, llvm::TargetTransformInfo &TTI,
288  unsigned int vectorBits, Instruction *A, Instruction *B) {
289  mergeList.emplace_back(A, B);
290  auto *aA = ancestorMap.find(B);
291  auto *aB = ancestorMap.find(A);
292  assert(aA != ancestorMap.end());
293  assert(aB != ancestorMap.end());
294  // in the old MergingCost where they're separate instructions,
295  // we leave their ancestor PtrMaps intact.
296  // in the new MergingCost where they're the same instruction,
297  // we assign them the same ancestors.
298  auto *merged = new (alloc) aset<Instruction *>(*aA->second);
299  merged->insert(aB->second->begin(), aB->second->end());
300  aA->second = merged;
301  aB->second = merged;
302  unsigned numSelects = mergeOperands(A, B, Count{});
303  // TODO:
304  // increase cost by numSelects, decrease cost by `I`'s cost
305  unsigned int W = vectorBits / B->getNumScalarBits();
306  if (numSelects)
307  cost += numSelects * Operation::selectCost(TTI, B->getType(W));
308  cost -= B->getCost(TTI, VectorWidth(W)).recipThroughput;
309  auto *mB = findMerge(B);
310  if (mB) cycleUpdateMerged(merged, B, mB);
311  // fuse the merge map cycles
312  auto &mMA = mergeMap[A], &mMB = mergeMap[B];
313  if (auto *mA = findMerge(A)) {
314  cycleUpdateMerged(merged, A, mA);
315  if (mB) {
316  mMB = mA;
317  mMA = mB;
318  } else {
319  mMB = mA;
320  mMA = B;
321  }
322  } else if (mB) {
323  mMA = mB;
324  mMB = A;
325  } else {
326  mMB = A;
327  mMA = B;
328  }
329  }
330  auto operator<(const MergingCost &other) const -> bool {
331  return cost < other.cost;
332  }
333  auto operator>(const MergingCost &other) const -> bool {
334  return cost > other.cost;
335  }

337  inline void mergeInstructions(IR::Cache &cache, Arena<> *tAlloc,
338  Instruction *A, Instruction *B,
339  amap<Instruction *, Predicate::Set> &valToPred,
340  ReMapper &reMap, UList<Value *> *pred) {
341  A = reMap[A];
342  B = reMap[B];
343  if (A == B) return; // is this possible?
344  invariant(A->getNumOperands(), B->getNumOperands());
345  if (auto *I = llvm::dyn_cast<Instruction>(A)) {
346  auto *J = llvm::cast<Instruction>(B);
347  // could be stores
348  if (auto *C = llvm::dyn_cast<Compute>(I)) {
349  Compute *D = cache.copyOperation(C);
350  MergingCost::Allocate allocInst{tAlloc, cache, reMap,
351  valToPred, pred, D->getOperands()};
352  mergeOperands(I, J, allocInst);
353  D = cache.cse(D);
354  cache.replaceAllUsesWith(A, D);
355  reMap.remapFromTo(A, D);
356  A = D; // D replaces `A` as new op
357  } else {
358  invariant(Node::VK_Stow == I->getKind());
359  MergingCost::Allocate allocInst{tAlloc, cache, reMap,
360  valToPred, pred, I->getOperands()};
361  mergeOperands(I, J, allocInst);
362  }
363  }
364  cache.replaceAllUsesWith(B, A);
365  reMap.remapFromTo(B, A);
366  }

370 inline void mergeInstructions(
371  Arena<> *alloc, IR::Cache &cache, Predicate::Map &predMap,
372  llvm::TargetTransformInfo &TTI, unsigned int vectorBits,
373  amap<Instruction::Identifier, math::ResizeableView<Instruction *, unsigned>>
374  opMap,
375  amap<Instruction *, Predicate::Set> &valToPred,
376  llvm::SmallVectorImpl<MergingCost *> &mergingCosts, Instruction *J,
377  llvm::BasicBlock *BB, Predicate::Set &preds) {
378  // have we already visited?
379  if (mergingCosts.front()->visited(J)) return;
380  for (auto *C : mergingCosts) {
381  if (C->visited(J)) return;
382  C->initAncestors(alloc, J);
383  }
384  auto op = J->getIdentifier();
385  // TODO: confirm that `vec` doesn't get moved if `opMap` is resized
386  auto &vec = opMap[op];
387  // consider merging with every instruction sharing an opcode
388  for (Instruction *other : vec) {
389  // check legality
390  // illegal if:
391  // 1. pred intersection not empty
392  if (!preds.intersectionIsEmpty(valToPred[other])) continue;
393  // 2. one op descends from another
394  // because of our traversal pattern, this should not happen
395  // unless a fusion took place
396  // A -> B -> C
397  // -> D -> E
398  // if we merge B and E, it would be illegal to merge C and D
399  // because C is an ancestor of B-E, and D is a predecessor of B-E
400  size_t numMerges = mergingCosts.size();
401  // we may push into mergingCosts, so to avoid problems of iterator
402  // invalidation, we use an indexed loop
403  for (size_t i = 0; i < numMerges; ++i) {
404  MergingCost *C = mergingCosts[i];
405  if (C->getAncestors(J)->contains(other)) continue;
406  // we shouldn't have to check the opposite condition
407  // if (C->getAncestors(other)->contains(I))
408  // because we are traversing in topological order
409  // that is, we haven't visited any descendants of `I`
410  // so only an ancestor had a chance
411  auto *MC = alloc->construct<MergingCost>(*C);
412  // MC is a copy of C, except we're now merging
413  MC->merge(alloc, TTI, vectorBits, other, J);
414  }
415  }
416  // descendants aren't legal merge candidates, so check before merging
417  for (Instruction *U : *J->getUsers()) {
418  if (llvm::BasicBlock *BBU = U->getBasicBlock()) {
419  if (BBU == BB) // fast path, skip lookup
420  mergeInstructions(alloc, cache, predMap, TTI, vectorBits, opMap,
421  valToPred, mergingCosts, llvm::cast<Instruction>(U),
422  BB, preds);
423  else if (auto *f = predMap.find(BBU); f != predMap.end())
424  mergeInstructions(alloc, cache, predMap, TTI, vectorBits, opMap,
425  valToPred, mergingCosts, llvm::cast<Instruction>(U),
426  BBU, f->second);
427  }
428  }
429  // descendants aren't legal merge candidates, so push after merging
430  if (vec.getCapacity() <= vec.size())
431  vec.reserve(alloc, std::max(unsigned(8), 2 * vec.size()));
432  vec.push_back(J);
433  valToPred[J] = preds;
434  // TODO: prune bad candidates from mergingCosts
435 }

448 [[nodiscard]] inline auto mergeInstructions(IR::Cache &cache,
449  Predicate::Map &predMap,
450  llvm::TargetTransformInfo &TTI,
451  Arena<> tAlloc, unsigned vectorBits,
452  TreeResult tr) -> TreeResult {
453  auto [completed, trret] = cache.completeInstructions(&predMap, tr);
454  tr = trret;
455  if (!predMap.isDivergent()) return tr;
456  // there is a divergence in the control flow that we can ideally merge
457  amap<Instruction::Identifier, math::ResizeableView<Instruction *, unsigned>>
458  opMap{&tAlloc};
459  amap<Instruction *, Predicate::Set> valToPred{&tAlloc};
460  llvm::SmallVector<MergingCost *> mergingCosts;
461  mergingCosts.emplace_back(tAlloc);
462  // We search through incomplete instructions inside the predMap
463  // this should yield all merge candidates.L
464  for (auto *C = completed; C; C = static_cast<Compute *>(C->getNext())) {
465  auto *f = predMap.find(C->getLLVMInstruction());
466  invariant(f != predMap.end());
467  mergeInstructions(&tAlloc, cache, predMap, TTI, vectorBits, opMap,
468  valToPred, mergingCosts, C, f->first, f->second);
469  }
470  MergingCost *minCostStrategy = *std::ranges::min_element(
471  mergingCosts, [](MergingCost *a, MergingCost *b) { return *a < *b; });
472  // and then apply it to the instructions.
473  ReMapper reMap;
474  // we use `alloc` for objects intended to live on
475 
476  // merge pair through `select`ing the arguments that differ
477  for (auto [A, B] : *minCostStrategy)
478  minCostStrategy->mergeInstructions(cache, &tAlloc, A, B, valToPred, reMap,
479  predMap.getPredicates());
480  return tr;
481 }
