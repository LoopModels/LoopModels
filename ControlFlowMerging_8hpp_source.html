166  // An abstraction that runs an algorithm to look for merging opportunities,
167  // either counting the number of selects needed, or allocating selects
168  // and returning the new operand vector.
169  // We generally aim to have analysis/cost modeling and code generation
170  // take the same code paths, to both avoid code duplication and to
171  // make sure the cost modeling reflects the actual code we're generating.
172  template <typename S>
173  auto mergeOperands(Instruction *A, Instruction *B, S selects) {
174  // now, we need to check everything connected to O and I in the mergeMap
175  // to see if any of them need to be updated.
176  // TODO: we want to estimate select cost
177  // worst case scenario is 1 select per operand (p is pred):
178  // select(p, f(a,b), f(c,d)) => f(select(p, a, c), select(p, b, d))
179  // but we can often do better, e.g. we may have
180  // select(p, f(a,b), f(c,b)) => f(select(p, a, c), b)
181  // additionally, we can check `I->associativeOperandsFlag()`
182  // select(p, f(a,b), f(c,a)) => f(a, select(p, b, c))
183  // we need to figure out which operands we're merging with which,
184  //
185  // We need to give special consideration to the case where
186  // arguments are merged, as this may be common when two
187  // control flow branches have relatively similar pieces.
188  // E.g., if b and c are already merged,
189  // and if `f`'s ops are associative, then we'd get
190  // select(p, f(a,b), f(c,a)) => f(a, b)
191  // so we need to check if any operand pairs are merged with each other.
192  // note `isMerged(a,a) == true`, so that's the one query we need to use.
193  auto selector = init(selects, A);
194  MutPtrVector<Instruction *> operandsA = A->getOperands();
195  MutPtrVector<Instruction *> operandsB = B->getOperands();
196  size_t numOperands = operandsA.size();
197  assert(numOperands == operandsB.size());
198  uint8_t associativeOpsFlag = B->associativeOperandsFlag();
199  // For example,
200  // we keep track of which operands we've already merged,
201  // f(a, b), f(b, b)
202  // we can't merge b twice!
203  // size_t numSelects = numOperands;
204  for (size_t i = 0; i < numOperands; ++i) {
205  auto *opA = A->getOperand(i);
206  auto *opB = B->getOperand(i);
207  auto [assoc, assocFlag] = popBit(associativeOpsFlag);
208  associativeOpsFlag = assocFlag;
209  // if both operands were merged, we can ignore it's associativity
210  if (isMerged(opB, opA)) {
211  selector.merge(i, opA, opB);
212  // --numSelects;
213  continue;
214  }
215  if (!((assoc) && (assocFlag))) {
216  // this op isn't associative with any remaining
217  selector.select(i, opA, opB);
218  continue;
219  }
220  // we look forward
221  size_t j = i;
222  bool merged = false;
223  while (assocFlag) {
224  auto shift = std::countr_zero(assocFlag);
225  j += ++shift;
226  assocFlag >>= shift;
227  auto *opjA = A->getOperand(j);
228  auto *opjB = B->getOperand(j);
229  // if elements in these pairs weren't already used
230  // to drop a select, and they're merged with each other
231  // we'll use them now to drop a select.
232  if (isMerged(opB, opjA)) {
233  std::swap(operandsA[i], operandsA[j]);
234  selector.merge(i, opjA, opB);
235  merged = true;
236  break;
237  }
238  if (isMerged(opjB, opA)) {
239  std::swap(operandsB[i], operandsB[j]);
240  selector.merge(i, opA, opjB);
241  merged = true;
242  break;
243  }
244  }
245  // we couldn't find any candidates
246  if (!merged) selector.select(i, opA, opB);
247  }
248  return selector;
249  }

251  void merge(BumpAlloc<> &alloc, llvm::TargetTransformInfo &TTI,
252  unsigned int vectorBits, Instruction *A, Instruction *B) {
253  mergeList.emplace_back(A, B);
254  auto *aA = ancestorMap.find(B);
255  auto *aB = ancestorMap.find(A);
256  assert(aA != ancestorMap.end());
257  assert(aB != ancestorMap.end());
258  // in the old MergingCost where they're separate instructions,
259  // we leave their ancestor PtrMaps intact.
260  // in the new MergingCost where they're the same instruction,
261  // we assign them the same ancestor Claptrap.
262  auto *merged = new (alloc) aset<Instruction *>(*aA->second);
263  merged->insert(aB->second->begin(), aB->second->end());
264  aA->second = merged;
265  aB->second = merged;
266  unsigned numSelects = mergeOperands(A, B, Count{});
267  // TODO:
268  // increase cost by numSelects, decrease cost by `I`'s cost
269  unsigned int W = vectorBits / B->getNumScalarBits();
270  if (numSelects) cost += numSelects * B->selectCost(TTI, W);
271  cost -= B->getCost(TTI, W).recipThroughput;
272  auto *mB = findMerge(B);
273  if (mB) cycleUpdateMerged(merged, B, mB);
274  // fuse the merge map cycles
275  if (auto *mA = findMerge(A)) {
276  cycleUpdateMerged(merged, A, mA);
277  if (mB) {
278  mergeMap[B] = mA;
279  mergeMap[A] = mB;
280  } else {
281  mergeMap[B] = mA;
282  mergeMap[A] = B;
283  }
284  } else if (mB) {
285  mergeMap[A] = mB;
286  mergeMap[B] = A;
287  } else {
288  mergeMap[B] = A;
289  mergeMap[A] = B;
290  }
291  }
292  auto operator<(const MergingCost &other) const -> bool {
293  return cost < other.cost;
294  }
295  auto operator>(const MergingCost &other) const -> bool {
296  return cost > other.cost;
297  }

301 inline void mergeInstructions(
302  BumpAlloc<> &alloc, Instruction::Cache &cache, Predicate::Map &predMap,
303  llvm::TargetTransformInfo &TTI, unsigned int vectorBits,
304  amap<std::pair<Instruction::Intrinsic, llvm::Type *>,
305  BumpPtrVector<std::pair<Instruction *, Predicate::Set>>> &opMap,
306  llvm::SmallVectorImpl<MergingCost *> &mergingCosts, Instruction *J,
307  llvm::BasicBlock *BB, Predicate::Set &preds) {
308  // have we already visited?
309  if (mergingCosts.front()->visited(J)) return;
310  for (auto *C : mergingCosts) {
311  if (C->visited(J)) return;
312  C->initAncestors(alloc, J);
313  }
314  auto op = J->getOpType();
315  // TODO: confirm that `vec` doesn't get moved if `opMap` is resized
316  auto &vec = opMap[op];
317  // consider merging with every instruction sharing an opcode
318  for (auto &pair : vec) {
319  Instruction *other = pair.first;
320  // check legality
321  // illegal if:
322  // 1. pred intersection not empty
323  if (!preds.intersectionIsEmpty(pair.second)) continue;
324  // 2. one op descends from another
325  // because of our traversal pattern, this should not happen
326  // unless a fusion took place
327  // A -> B -> C
328  // -> D -> E
329  // if we merge B and E, it would be illegal to merge C and D
330  // because C is an ancestor of B-E, and D is a predecessor of B-E
331  size_t numMerges = mergingCosts.size();
332  // we may push into mergingCosts, so to avoid problems of iterator
333  // invalidation, we use an indexed loop
334  for (size_t i = 0; i < numMerges; ++i) {
335  MergingCost *C = mergingCosts[i];
336  if (C->getAncestors(J)->contains(other)) continue;
337  // we shouldn't have to check the opposite condition
338  // if (C->getAncestors(other)->contains(I))
339  // because we are traversing in topological order
340  // that is, we haven't visited any descendants of `I`
341  // so only an ancestor had a chance
342  auto *MC = alloc.construct<MergingCost>(*C);
343  // MC is a copy of C, except we're now merging
344  MC->merge(alloc, TTI, vectorBits, other, J);
345  }
346  }
347  // descendants aren't legal merge candidates, so check before merging
348  for (Instruction *U : J->getUsers()) {
349  if (llvm::BasicBlock *BBU = U->getBasicBlock()) {
350  if (BBU == BB) {
351  // fast path, skip lookup
352  mergeInstructions(alloc, cache, predMap, TTI, vectorBits, opMap,
353  mergingCosts, U, BB, preds);
354  } else if (auto *f = predMap.find(BBU); f != predMap.rend()) {
355  mergeInstructions(alloc, cache, predMap, TTI, vectorBits, opMap,
356  mergingCosts, U, BBU, f->second);
357  }
358  }
359  }
360  // descendants aren't legal merge candidates, so push after merging
361  vec.push_back({J, preds});
362  // TODO: prune bad candidates from mergingCosts
363 }

376 inline void mergeInstructions(BumpAlloc<> &alloc, Instruction::Cache &cache,
377  Predicate::Map &predMap,
378  llvm::TargetTransformInfo &TTI,
379  BumpAlloc<> &tAlloc, unsigned int vectorBits) {
380  if (!predMap.isDivergent()) return;
381  // there is a divergence in the control flow that we can ideally merge
382  amap<std::pair<Instruction::Intrinsic, llvm::Type *>,
383  BumpPtrVector<std::pair<Instruction *, Predicate::Set>>>
384  opMap{tAlloc};
385  llvm::SmallVector<MergingCost *> mergingCosts;
386  mergingCosts.emplace_back(alloc);
387  for (auto &pred : predMap) {
388  for (llvm::Instruction &lI : *pred.first) {
389  if (Instruction *J = cache[&lI]) {
390  mergeInstructions(tAlloc, cache, predMap, TTI, vectorBits, opMap,
391  mergingCosts, J, pred.first, pred.second);
392  }
393  }
394  }
395  // TODO:
396  // pick the minimum cost mergingCost
397  MergingCost *minCostStrategy = *std::ranges::min_element(
398  mergingCosts, [](auto *a, auto *b) { return *a < *b; });
399  // and then apply it to the instructions.
400  ReMapper reMap;
401  // we use `alloc` for objects intended to live on
402  for (auto &pair : *minCostStrategy) {
403  // merge pair through `select`ing the arguments that differ
404  auto [A, B] = pair;
405  A = reMap[A];
406  B = reMap[B];
407  auto operands = minCostStrategy->mergeOperands(
408  A, B, MergingCost::Allocate{alloc, cache, reMap});
409  A->replaceAllUsesOf(B)->setOperands(operands);
410  reMap.remapFromTo(B, A);
411  }
412  // free memory
413  tAlloc.reset();
414 }
