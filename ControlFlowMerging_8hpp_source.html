58  // mergeMap can be thought of as containing doubly linked lists/cycles,
59  // e.g. a -> b -> c -> a, where a, b, c are instructions.
60  // if we merge c and d, we have
61  // c -> a => c -> d
62  // d => d -> a
63  // yielding: c -> d -> a -> b -> c
64  // if instead we're merging c and d, but d is also paired d <-> e, then
65  // c -> a => c -> e
66  // d -> e => d -> a
67  // yielding c -> e -> d -> a -> b -> c
68  // that is, if we're fusing c and d, we can make each point toward
69  // what the other one was pointing to, in order to link the chains.

117  // note: this is not the same as `isMerged(I) && isMerged(J)`,
118  // as `I` and `J` may be merged with different Instructions

187  // An abstraction that runs an algorithm to look for merging opportunities,
188  // either counting the number of selects needed, or allocating selects
189  // and returning the new operand vector.
190  // We generally aim to have analysis/cost modeling and code generation
191  // take the same code paths, to both avoid code duplication and to
192  // make sure the cost modeling reflects the actual code we're generating.
193  template <typename S>
194  auto mergeOperands(Instruction *A, Instruction *B, S selects) {
195  // now, we need to check everything connected to O and I in the mergeMap
196  // to see if any of them need to be updated.
197  // TODO: we want to estimate select cost
198  // worst case scenario is 1 select per operand (p is pred):
199  // select(p, f(a,b), f(c,d)) => f(select(p, a, c), select(p, b, d))
200  // but we can often do better, e.g. we may have
201  // select(p, f(a,b), f(c,b)) => f(select(p, a, c), b)
202  // additionally, we can check `I->associativeOperandsFlag()`
203  // select(p, f(a,b), f(c,a)) => f(a, select(p, b, c))
204  // we need to figure out which operands we're merging with which,
205  //
206  // We need to give special consideration to the case where
207  // arguments are merged, as this may be common when two
208  // control flow branches have relatively similar pieces.
209  // E.g., if b and c are already merged,
210  // and if `f`'s ops are associative, then we'd get
211  // select(p, f(a,b), f(c,a)) => f(a, b)
212  // so we need to check if any operand pairs are merged with each other.
213  // note `isMerged(a,a) == true`, so that's the one query we need to use.
214  auto selector = init(selects, A);
215  llvm::MutableArrayRef<Instruction *> operandsA = A->getOperands();
216  llvm::MutableArrayRef<Instruction *> operandsB = B->getOperands();
217  size_t numOperands = operandsA.size();
218  assert(numOperands == operandsB.size());
219  uint8_t associativeOpsFlag = B->associativeOperandsFlag();
220  // For example,
221  // we keep track of which operands we've already merged,
222  // f(a, b), f(b, b)
223  // we can't merge b twice!
224  // size_t numSelects = numOperands;
225  for (size_t i = 0; i < numOperands; ++i) {
226  auto *opA = A->getOperand(i);
227  auto *opB = B->getOperand(i);
228  auto [assoc, assocFlag] = popBit(associativeOpsFlag);
229  associativeOpsFlag = assocFlag;
230  // if both operands were merged, we can ignore it's associativity
231  if (isMerged(opB, opA)) {
232  selector.merge(i, opA, opB);
233  // --numSelects;
234  continue;
235  } else if (!((assoc) && (assocFlag))) {
236  // this op isn't associative with any remaining
237  selector.select(i, opA, opB);
238  continue;
239  }
240  // we look forward
241  size_t j = i;
242  bool merged = false;
243  while (assocFlag) {
244  auto shift = std::countr_zero(assocFlag);
245  j += ++shift;
246  assocFlag >>= shift;
247  auto *opjA = A->getOperand(j);
248  auto *opjB = B->getOperand(j);
249  // if elements in these pairs weren't already used
250  // to drop a select, and they're merged with each other
251  // we'll use them now to drop a select.
252  if (isMerged(opB, opjA)) {
253  std::swap(operandsA[i], operandsA[j]);
254  selector.merge(i, opjA, opB);
255  merged = true;
256  break;
257  } else if (isMerged(opjB, opA)) {
258  std::swap(operandsB[i], operandsB[j]);
259  selector.merge(i, opA, opjB);
260  merged = true;
261  break;
262  }
263  }
264  // we couldn't find any candidates
265  if (!merged) selector.select(i, opA, opB);
266  }
267  return selector;
268  };

270  void merge(BumpAlloc<> &alloc, llvm::TargetTransformInfo &TTI,
271  unsigned int vectorBits, Instruction *A, Instruction *B) {
272  mergeList.emplace_back(A, B);
273  auto aA = ancestorMap.find(B);
274  auto aB = ancestorMap.find(A);
275  assert(aA != ancestorMap.end());
276  assert(aB != ancestorMap.end());
277  // in the old MergingCost where they're separate instructions,
278  // we leave their ancestor PtrMaps intact.
279  // in the new MergingCost where they're the same instruction,
280  // we assign them the same ancestor Claptrap.
281  auto *merged = new (alloc) aset<Instruction *>(*aA->second);
282  merged->insert(aB->second->begin(), aB->second->end());
283  aA->second = merged;
284  aB->second = merged;
285  unsigned numSelects = mergeOperands(A, B, Count{});
286  // TODO:
287  // increase cost by numSelects, decrease cost by `I`'s cost
288  unsigned int W = vectorBits / B->getNumScalarBits();
289  if (numSelects) cost += numSelects * B->selectCost(TTI, W);
290  cost -= B->getCost(TTI, W).recipThroughput;
291  auto mB = findMerge(B);
292  if (mB) cycleUpdateMerged(merged, B, mB);
293  // fuse the merge map cycles
294  if (auto mA = findMerge(A)) {
295  cycleUpdateMerged(merged, A, mA);
296  if (mB) {
297  mergeMap[B] = mA;
298  mergeMap[A] = mB;
299  } else {
300  mergeMap[B] = mA;
301  mergeMap[A] = B;
302  }
303  } else if (mB) {
304  mergeMap[A] = mB;
305  mergeMap[B] = A;
306  } else {
307  mergeMap[B] = A;
308  mergeMap[A] = B;
309  }
310  }
311  constexpr auto operator<(const MergingCost &other) const -> bool {
312  return cost < other.cost;
313  }
314  constexpr auto operator>(const MergingCost &other) const -> bool {
315  return cost > other.cost;
316  }

319 void mergeInstructions(
320  BumpAlloc<> &alloc, Instruction::Cache &cache, Predicate::Map &predMap,
321  llvm::TargetTransformInfo &TTI, unsigned int vectorBits,
322  amap<std::pair<Instruction::Intrinsic, llvm::Type *>,
323  llvm::SmallVector<std::pair<Instruction *, Predicate::Set>>> &opMap,
324  llvm::SmallVectorImpl<MergingCost *> &mergingCosts, Instruction *I,
325  llvm::BasicBlock *BB, Predicate::Set &preds) {
326  // have we already visited?
327  if (mergingCosts.front()->visited(I)) return;
328  for (auto C : mergingCosts) {
329  if (C->visited(I)) return;
330  C->initAncestors(alloc, I);
331  }
332  auto op = I->getOpType();
333  // TODO: confirm that `vec` doesn't get moved if `opMap` is resized
334  auto &vec = opMap[op];
335  // consider merging with every instruction sharing an opcode
336  for (auto &pair : vec) {
337  Instruction *other = pair.first;
338  // check legality
339  // illegal if:
340  // 1. pred intersection not empty
341  if (!preds.intersectionIsEmpty(pair.second)) continue;
342  // 2. one op descends from another
343  // because of our traversal pattern, this should not happen
344  // unless a fusion took place
345  // A -> B -> C
346  // -> D -> E
347  // if we merge B and E, it would be illegal to merge C and D
348  // because C is an ancestor of B-E, and D is a predecessor of B-E
349  size_t numMerges = mergingCosts.size();
350  // we may push into mergingCosts, so to avoid problems of iterator
351  // invalidation, we use an indexed loop
352  for (size_t i = 0; i < numMerges; ++i) {
353  MergingCost *C = mergingCosts[i];
354  if (C->getAncestors(I)->contains(other)) continue;
355  // we shouldn't have to check the opposite condition
356  // if (C->getAncestors(other)->contains(I))
357  // because we are traversing in topological order
358  // that is, we haven't visited any descendants of `I`
359  // so only an ancestor had a chance
360  auto *MC = alloc.construct<MergingCost>(*C);
361  // MC is a copy of C, except we're now merging
362  MC->merge(alloc, TTI, vectorBits, other, I);
363  }
364  }
365  // descendants aren't legal merge candidates, so check before merging
366  for (Instruction *U : I->getUsers()) {
367  if (llvm::BasicBlock *BBU = U->getBasicBlock()) {
368  if (BBU == BB) {
369  // fast path, skip lookup
370  mergeInstructions(alloc, cache, predMap, TTI, vectorBits, opMap,
371  mergingCosts, U, BB, preds);
372  } else if (auto f = predMap.find(BBU); f != predMap.rend()) {
373  mergeInstructions(alloc, cache, predMap, TTI, vectorBits, opMap,
374  mergingCosts, U, BBU, f->second);
375  }
376  }
377  }
378  // descendants aren't legal merge candidates, so push after merging
379  vec.push_back({I, preds});
380  // TODO: prune bad candidates from mergingCosts
381 }

395  Predicate::Map &predMap, llvm::TargetTransformInfo &TTI,
396  BumpAlloc<> &tAlloc, unsigned int vectorBits) {

398  // there is a divergence in the control flow that we can ideally merge
399  amap<std::pair<Instruction::Intrinsic, llvm::Type *>,
400  llvm::SmallVector<std::pair<Instruction *, Predicate::Set>>>
401  opMap{};
402  llvm::SmallVector<MergingCost *> mergingCosts;
403  mergingCosts.push_back(alloc.construct<MergingCost>());
404  for (auto &pred : predMap) {
405  for (llvm::Instruction &lI : *pred.first) {
406  if (Instruction *I = cache[&lI]) {
407  mergeInstructions(tAlloc, cache, predMap, TTI, vectorBits, opMap,
408  mergingCosts, I, pred.first, pred.second);
409  }
410  }
411  }
412  // TODO:
413  // pick the minimum cost mergingCost
414  MergingCost *minCostStrategy = *std::ranges::min_element(
415  mergingCosts, [](auto *a, auto *b) { return *a < *b; });
416  // and then apply it to the instructions.
417  ReMapper reMap;
418  // we use `alloc` for objects intended to live on
419  for (auto &pair : *minCostStrategy) {
420  // merge pair through `select`ing the arguments that differ
421  auto [A, B] = pair;
422  A = reMap[A];
423  B = reMap[B];

425  minCostStrategy->mergeOperands(

427  A->replaceAllUsesOf(B)->setOperands(operands);
428  reMap.remapFromTo(B, A);
429  }
