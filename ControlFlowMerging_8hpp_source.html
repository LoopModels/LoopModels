36 // represents the cost of merging key=>values; cost is hopefully negative.
37 // cost is measured in reciprocal throughput
38 struct MergingCost {
39  // mergeMap can be thought of as containing doubly linked lists/cycles,
40  // e.g. a -> b -> c -> a, where a, b, c are instructions.
41  // if we merge c and d, we have
42  // c -> a => c -> d
43  // d => d -> a
44  // yielding: c -> d -> a -> b -> c
45  // if instead we have d <-> e, then
46  // c -> a => c -> e
47  // d -> e => d -> a
48  // yielding c -> e -> d -> a -> b -> c
49  // that is, if we're fusing c and d, we can make each point toward
50  // what the other one was pointing to, in order to link the chains.
51  llvm::DenseMap<Instruction *, Instruction *> mergeMap;
52  llvm::DenseMap<Instruction *, llvm::SmallPtrSet<Instruction *, 16> *>
53  ancestorMap;
54  llvm::InstructionCost cost;
57  auto initAncestors(llvm::BumpPtrAllocator &alloc, Instruction *key)
58  -> llvm::SmallPtrSet<Instruction *, 16> * {
59  auto *set = alloc.Allocate<llvm::SmallPtrSet<Instruction *, 16>>();
61  set->insert(key);
62  for (auto *op : key->operands) {
63  if (auto f = ancestorMap.find(op); f != ancestorMap.end()) {
64  set->insert(f->second->begin(), f->second->end());
65  }
66  }
67  ancestorMap[key] = set;
68  return set;
69  }
70  [[nodiscard]] auto visited(Instruction *key) const -> bool {
71  return ancestorMap.count(key);
72  }
73  auto getAncestors(llvm::BumpPtrAllocator &alloc, Instruction *key)
74  -> llvm::SmallPtrSet<Instruction *, 16> * {
75  if (auto it = ancestorMap.find(key); it != ancestorMap.end())
76  return it->second;
77  return initAncestors(alloc, key);
78  }
79  auto getAncestors(Instruction *key)
80  -> llvm::SmallPtrSet<Instruction *, 16> * {
81  if (auto it = ancestorMap.find(key); it != ancestorMap.end())
82  return it->second;
83  return nullptr;
84  }
85  auto findMerge(Instruction *key) -> Instruction * {
86  if (auto it = mergeMap.find(key); it != mergeMap.end())
87  return it->second;
88  return nullptr;
89  }
90  // follows the cycle, traversing H -> mergeMap[H] -> mergeMap[mergeMap[H]]
91  // ... until it reaches E, updating the ancestorMap pointer at each level of
92  // the recursion.
93  void cycleUpdateMerged(llvm::SmallPtrSet<Instruction *, 16> *ancestors,
94  Instruction *E, Instruction *H) {
95  while (H != E) {
96  ancestorMap[H] = ancestors;
97  H = mergeMap[H];
98  }
99  }
100  void merge(llvm::BumpPtrAllocator &alloc, Instruction *O, Instruction *I) {
101  auto aI = ancestorMap.find(I);
102  auto aO = ancestorMap.find(O);
103  assert(aI != ancestorMap.end());
104  assert(aO != ancestorMap.end());
105  // in the old MergingCost where they're separate instructions,
106  // we leave their ancestor PtrMaps intact.
107  // in the new MergingCost where they're the same instruction,
108  // we assign them the same ancestor Claptrap.
109  auto *merged =
110  new (alloc) llvm::SmallPtrSet<Instruction *, 16>(*aI->second);
111  merged->insert(aO->second->begin(), aO->second->end());
112  aI->second = merged;
113  aO->second = merged;
114  // now, we need to check everything connected to O and I in the mergeMap
115  // to see if any of them need to be updated.
116  auto mI = findMerge(I);
117  if (mI)
118  cycleUpdateMerged(merged, I, mI);
119  // fuse the merge map cycles
120  if (auto mO = findMerge(O)) {
121  cycleUpdateMerged(merged, O, mO);
122  if (mI) {
123  mergeMap[I] = mO;
124  mergeMap[O] = mI;
125  } else {
126  mergeMap[I] = mO;
127  mergeMap[O] = I;
128  }
129  } else if (mI) {
130  mergeMap[O] = mI;
131  mergeMap[I] = O;
132  } else {
133  mergeMap[I] = O;
134  mergeMap[O] = I;
135  }
136  }
137 };

139 void mergeInstructions(
140  llvm::BumpPtrAllocator &alloc, Instruction::Cache &cache,
141  Predicate::Map &predMap,
142  llvm::DenseMap<std::pair<llvm::Intrinsic::ID, llvm::Intrinsic::ID>,
143  llvm::SmallVector<std::pair<Instruction *, Predicate::Set>>>
144  &opMap,
145  llvm::SmallVectorImpl<MergingCost *> &mergingCosts, Instruction *I,
146  llvm::BasicBlock *BB, Predicate::Set &preds) {
147  // have we already visited?
148  if (mergingCosts.front()->visited(I))
149  return;
150  for (auto C : mergingCosts) {
151  if (C->visited(I))
152  return;
153  C->initAncestors(alloc, I);
154  }
155  auto op = I->getOpPair();
156  // TODO: confirm that `vec` doesn't get moved if `opMap` is resized
157  auto &vec = opMap[op];
158  // consider merging with every instruction sharing an opcode
159  for (auto &pair : vec) {
160  Instruction *other = pair.first;
161  // check legality
162  // illegal if:
163  // 1. pred intersection not empty
164  if (!preds.emptyIntersection(pair.second))
165  continue;
166  // 2. one op descends from another
167  // because of our traversal pattern, this should not happen
168  // unless a fusion took place
169  // A -> B -> C
170  // -> D -> E
171  // if we merge B and E, it would be illegal to merge C and D
172  // because C is an ancestor of B-E, and D is a predecessor of B-E
173  size_t numMerges = mergingCosts.size();
174  // we may push into mergingCosts, so to avoid problems of iterator
175  // invalidation, we use an indexed loop
176  for (size_t i = 0; i < numMerges; ++i) {
177  MergingCost *C = mergingCosts[i];
178  if (C->getAncestors(I)->contains(other)) {
179  continue;
180  }
181  // we shouldn't have to check the opposite condition
182  // if (C->getAncestors(other)->contains(I))
183  // because we are traversing in topological order
184  // that is, we haven't visited any descendants of `I`
185  // so only an ancestor had a chance
186  auto *MC = new (alloc) MergingCost(*C);
187  // MC is a copy of C, except we're now merging
188  MC->merge(alloc, other, I);
189  }
190  }
191  // descendants aren't legal merge candidates, so check before merging
192  for (Instruction *U : I->getUsers()) {
193  if (llvm::BasicBlock *BBU = U->getBasicBlock()) {
194  if (BBU == BB) {
195  // fast path, skip lookup
196  mergeInstructions(alloc, cache, predMap, opMap, mergingCosts, U,
197  BB, preds);
198  } else if (auto f = predMap.find(BBU); f != predMap.rend()) {
199  mergeInstructions(alloc, cache, predMap, opMap, mergingCosts, U,
200  BBU, f->second);
201  }
202  }
203  }
204  // descendants aren't legal merge candidates, so push after merging
205  vec.push_back({I, preds});
206  // TODO: prune bad candidates from mergingCosts
207 }

218 void mergeInstructions(llvm::BumpPtrAllocator &alloc,
219  llvm::BumpPtrAllocator &tAlloc,
220  Instruction::Cache &cache, Predicate::Map &predMap) {
221  if (!predMap.isDivergent())
222  return;
223  // there is a divergence in the control flow that we can ideally merge
224  auto &opMap = *alloc.Allocate<llvm::DenseMap<
225  std::pair<llvm::Intrinsic::ID, llvm::Intrinsic::ID>,
226  llvm::SmallVector<std::pair<Instruction *, Predicate::Set>>>>();
227  llvm::SmallVector<MergingCost *> mergingCosts;
228  mergingCosts.push_back(alloc.Allocate<MergingCost>());
229  for (auto &pred : predMap) {
230  for (llvm::Instruction &lI : *pred.first) {
231  if (Instruction *I = cache[&lI]) {
232  mergeInstructions(tAlloc, cache, predMap, opMap, mergingCosts,
233  I, pred.first, pred.second);
234  }
235  }
236  }
237  tAlloc.Reset();
