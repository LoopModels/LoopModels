22 struct OrthogonalAxes {
23  uint32_t data;
24  [[nodiscard]] constexpr auto contigAxis() const -> uint32_t {
25  return data & 0xff;
26  }
27  // mask containing `0` for dependent axes, 1s for independent
28  // should contain `0` for all non-existent loops, e.g.
29  // for (i = I, j = J, k = K, l = L) {
30  // A[j,l]
31  // for (a = A, b = B){ .... }
32  // }
33  // The mask should equal (1<<0) | (1<<2) (for the i and k).
34  // Only loops it is nested in that it doesn't depend on count.
35  [[nodiscard]] constexpr auto indepAxes() const -> uint32_t {
36  return data >> 8;
37  };
38 };
39 // costs is an array of length two.
40 // memory costs, unnormalized by `prod(unrolls)`
41 constexpr auto cost(MemoryCosts mc, AbstractVector auto invunrolls,
42  VectorizationFactor vfi, OrthogonalAxes orth)
43  -> utils::eltype_t<decltype(invunrolls)> {
44 
45  utils::eltype_t<decltype(invunrolls)> c;
46  uint32_t da = orth.indepAxes();
47  if (da) {
48  uint32_t tz = std::countr_zero(da);
49  c = invunrolls[tz++];
50  for (uint32_t d = da >> tz, i = tz; d; d >>= tz, i += tz) {
51  tz = std::countr_zero(d);
52  c *= invunrolls[i + tz++];
53  }
54  } else c = 1;
55 
56  if ((vfi.index < 32) && !(da & (1 << vfi.index))) {
57  // depends vectorized index
58  if (vfi.index == orth.contigAxis()) {
59  c *= mc.contiguous;
60  } else if (orth.contigAxis() >= 32) {
61  c *= mc.discontiguous;
62  } else {
63  // Discontiguous vector load.
64  // We consider two alternatives:
65  // 1. gather/scatter (discontiguous)
66  // 2. contiguous load for each vectorization factor of length equal to
67  // unroll, followed by shuffles.
68  // E.g., unroll contig by 4, another dim is vectorized by 8:
69  // we'd have 8 vloads (max(4/8,1) * 8), followed by 4*log2(8) +
70  // log2(max(8/4,1))*4 shuffles.
71  // Or, if we unroll contig by 8, and another dim is vectorzeed by 2, we'd
72  // have 8 = (max(8/2,1) * 2) vloads, 8*log2(2) + log2(max(2/8,1))*8
73  // shuffles.
74  // We divide by `u[contig]`, as it is now accounted for
75  // So we have
76  // max(v/u, 1) + u*log2(v) + log2(max(v/u ,1))*u
77  utils::eltype_t<decltype(invunrolls)> iu{invunrolls[orth.contigAxis()]},
78  u{1 / iu}, mr{math::smax((1 << vfi.l2factor) * iu, 1)};
79 
80  c *= math::smin(mc.contiguous * mr + u * (vfi.l2factor + log2(mr)),
81  mc.discontiguous);
82  }
83  }
84  return c;
85 }
