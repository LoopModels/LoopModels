114 constexpr auto cost(const AbstractMatrix auto &invunrolls, uint32_t indepAxes)
115  -> utils::eltype_t<decltype(invunrolls)> {
116  if (!indepAxes) return 1;
117  uint32_t tz = std::countr_zero(indepAxes);
118  utils::eltype_t<decltype(invunrolls)> c{invunrolls[0, tz++]};
119  for (uint32_t d = indepAxes >> tz, i = tz; d; d >>= tz, i += tz) {
120  tz = std::countr_zero(d);
121  c *= invunrolls[0, i + tz++];
122  }
123  return c;
124 }
125 // costs is an array of length two.
126 // memory costs, unnormalized by `prod(unrolls)`
127 // `invunrolls` is a matrix, row-0 are the inverse unrolls, row-1 unrolls.
128 constexpr auto cost(const AbstractMatrix auto &invunrolls, MemCostSummary mcs,
129  VectorizationFactor vfi)
130  -> math::SVector<utils::eltype_t<decltype(invunrolls)>, 3> {
131  auto [mc, orth] = mcs;
132  using T = utils::eltype_t<decltype(invunrolls)>;
133  T c{cost(invunrolls, orth.indep)};
134  double l{1.0}, s{1.0};
135  if (!(orth.indep & vfi.indexMask)) {
136  // depends on vectorized index
137  if (vfi.indexMask & orth.contig) {
138  l = mc[0].contiguous;
139  s = mc[1].contiguous;
140  } else if (!orth.contig) {
141  l = mc[0].discontiguous;
142  s = mc[1].discontiguous;
143  } else {
144  // Discontiguous vector load.
145  // We consider two alternatives:
146  // 1. gather/scatter (discontiguous)
147  // 2. contiguous load for each vectorization factor of length equal to
148  // unroll, followed by shuffles.
149  // E.g., unroll contig by 4, another dim is vectorized by 8:
150  // we'd have 8 vloads (max(4/8,1) * 8), followed by 4*log2(8) shuffles.
151  // w_0 = [0, 8, 16, 24]
152  // w_1 = [1, 9, 17, 25]
153  // w_2 = [2, 10, 18, 26]
154  // w_3 = [3, 11, 19, 27]
155  // w_4 = [4, 12, 20, 28]
156  // w_5 = [5, 13, 21, 29]
157  // w_6 = [6, 14, 22, 30]
158  // w_7 = [7, 15, 23, 31]
159  //
160  // x_0 = [0, 8, 16, 24, 4, 12, 20, 28]
161  // x_1 = [1, 9, 17, 25, 5, 13, 21, 29]
162  // x_2 = [2, 10, 18, 26, 6, 14, 22, 30]
163  // x_3 = [3, 11, 19, 27, 7, 15, 23, 31]
164  //
165  // y_0 = [0, 1, 16, 17, 4, 5, 20, 21]
166  // y_1 = [8, 9, 24, 25, 12, 13, 28, 29]
167  // y_2 = [2, 3, 18, 19, 6, 7, 22, 23]
168  // y_3 = [10, 11, 26, 27, 14, 15, 30, 31]
169  //
170  // z_0 = [0, 1, 2, 3, 4, 5, 6, 7]
171  // z_1 = [8, 9, 10, 11, 12, 13, 14, 15]
172  // z_2 = [16, 17, 18, 19, 20, 21, 22, 23]
173  // z_3 = [24, 25, 26, 27, 28, 29, 30, 31]
174  //
175  // Or, if we unroll contig by 8, and another dim is vectorzeed by 2, we'd
176  // have 8 = (max(8/2,1) * 2) vloads, 8*log2(2)
177  // shuffles.
178  // w_0_0 = [0, 2]
179  // w_0_1 = [4, 6]
180  // w_0_2 = [8, 10]
181  // w_0_3 = [12, 14]
182  // w_1_0 = [1, 3]
183  // w_1_1 = [5, 7]
184  // w_1_2 = [9, 11]
185  // w_1_3 = [13, 15]
186  //
187  // z_0 = [don't count shuffle costs among loads and stores?
188  // Only counting them among compute makes it T
189  // z_1 = [2, 3c*]
190  // z_3[0] = [6, 7]
191  // z_s[0] = [8c*, 9]
192  // z_5[1] = [10, 11]
193  // z_6[1] = [12,return {lshuf,sshuf,lshuf+sshuf}; 13]
194  // z_7 = [14, 15]
195  // Earlier, I had another term, `4*log2(max(8/4,1)) `8*log2(max(2/8,1))`
196  // i.e. u*log2(max(v/u,1))
197  // but I think we can avoid this by always working with vectors that are
198  // the larger of `u` and `v`, inserting at the start or extracting at the
199  // end, whichever is necessary.
200  // We divide by `u[contig]`, as it is now accounted for
201  // So we have
202  // v*max(u/v, 1) + u*log2(v)
203  T iu{invunrolls[0, orth.contig]}, u{invunrolls[1, orth.contig]};
204  utils::invariant(iu == 1 / u);
205  // Currently using contig for load/store costs...
206  // Without the need for shuffles, this should be >= discontig
207  T ufactor = math::smax<>(u, (1 << vfi.l2factor) * iu);
208  double ld = mc[0].discontiguous, sd = mc[1].discontiguous,
209  lc = mc[0].contiguous * ufactor, sc = mc[1].contiguous * ufactor;
210  T lshuf = math::smin<>(lc + u * vfi.l2factor, ld);
211  T sshuf = math::smin<>(sc + u * vfi.l2factor, sd);
212  return {lc * c, sc * c, (lshuf + sshuf) * c};
213  }
214  } else {
215  l = mc[0].scalar;
216  s = mc[1].scalar;
217  }
218  T lc{l * c}, sc{s * c};
219  return {lc, sc, lc + sc};
220 }
227 constexpr auto cost(const AbstractMatrix auto &invunrolls, MemCostSummary orth,
228  VectorizationFactor vfi, DensePtrMatrix<int64_t> inds)
229  -> math::SVector<utils::eltype_t<decltype(invunrolls)>, 3> {
230  using T = utils::eltype_t<decltype(invunrolls)>;
231  T c{1};
232  auto [arrayDim, numLoops] = shape(inds);
233  utils::invariant(numLoops > 0);
234  utils::invariant(arrayDim > 0);
235  utils::invariant(arrayDim <= 64);
236  utils::invariant(invunrolls.numCol(), inds.numCol());
237  for (ptrdiff_t d = 0; d < arrayDim; ++d) {
238  int64_t g = 0;
239  containers::BitSet64 bs;
240  T uprod;
241  for (ptrdiff_t l = 0; l < numLoops; ++l) {
242  if ((uint32_t(1) << l) == vfi.indexMask) continue;
243  int64_t a = inds[d, l];
244  if (!a) continue;
245  bool docontinue{false};
246  // We only
247  for (ptrdiff_t k = 0; k < arrayDim; ++k) {
248  if ((k == d) || (!inds[k, l])) continue;
249  docontinue = (inds[d, _] != inds[k, _]) || (d > k);
250  if (docontinue) break;
251  }
252  if (docontinue) continue;
253  if (bs.empty()) {
254  g = a;
255  uprod = invunrolls[0, l];
256  } else {
257  g = math::gcd(g, a);
258  uprod *= invunrolls[0, l];
259  }
260  bs.insert(l);
261  };
262  if (bs.size() < 2) continue;
263  T prod{1};
264  for (ptrdiff_t l : bs) {
265  if ((uint32_t(1) << l) == vfi.indexMask) continue;
266  int64_t a = inds[d, l];
267  if (!a) continue;
268  prod *= (1 - (a / g) * (uprod / invunrolls[0, l]));
269  }
270  c *= (1 - prod);
271  }
272  // c is a scaling factor; now we proceed to calculate cost similaly to the
273  // orth-axis implementation above.
274  return c * cost(invunrolls, orth, vfi);
275 }

539  // should only have to `init` once per `root`, with `VectorizationFactor`
540  // being adjustable.
541  // Note: we are dependent upon scanning in top order, so that operands'
542  // `calcLoopDepFlag()` are calculated before we get.
543  // TODO: vec factor should be a tree-flag
544  // Iteration order:
545  // We fully iterate over a loop before descending
546  // for (i : I){
547  // // block 0
548  // for (j : J){
549  // // block 1
550  // }
551  // // block 2
552  // for (j : J){
553  // // block 3
554  // }
555  // // block 4
556  // }
557  // we'd iterate 0, 2, 4, 1, 3.
558  // This way we can store once we hit the end.
559  // If there are no subloops to iterate to after, then we store the exit count.
560  // If there are, then the exit-count is 0, forward '1+exit' count to the last
561  // sub-loop, and `1` to all previous sub-loops.
562  // It's thus natural to implement recursively.
563  // NOLINTNEXTLINE(misc-no-recursion)
564  void initLoop(Arena<> *alloc, LoopDepSatisfaction deps, IR::Loop *L,
565  unsigned maxl2VF, const llvm::TargetTransformInfo &TTI,
566  ptrdiff_t depth, unsigned exitCount) {
567  invariant(depth > 0);
568  ptrdiff_t compute = compute_independence.size(), omemory = orth_axes.size(),
569  cmemory = conv_axes.size();
570  unsigned maxVF = 1 << maxl2VF;
571  // Loop and push throughput costs
572  for (IR::Node *N = L->getChild(); N; N = N->getNext()) {
573  if (auto *A = llvm::dyn_cast<IR::Addr>(N)) {
574  OrthogonalAxes oa = A->calcOrthAxes(depth);
575  IR::Addr::Costs rtl = A->calcCostContigDiscontig(TTI, maxVF);
576  if (oa.indep_axes) {
577  // check for duplicate
578  bool found = false;
579  for (ptrdiff_t i = omemory; i < orth_axes.size(); ++i) {
580  MemCostSummary &oai = orth_axes[i];
581  if (oai.orth != oa) continue;
582  found = true;
583  oai.loadstowcost[A->isStore()] += rtl;
584  break;
585  }
586  if (!found) orth_axes.emplace_back(memCostArray(A, rtl), oa);
587  } else {
588  bool found = false;
589  for (ptrdiff_t i = cmemory; i < conv_axes.size(); ++i) {
590  auto &cai = conv_axes[i];
591  if (cai.first.orth != oa) continue;
592  if (cai.second != A->indexMatrix()) continue;
593  found = true;
594  cai.first.loadstowcost[A->isStore()] += rtl;
595  break;
596  }
597  if (!found)
598  conv_axes.emplace_back(MemCostSummary{memCostArray(A, rtl), oa},
599  A->indexMatrix());
600  }
601  } else if (auto *C = llvm::dyn_cast<IR::Compute>(N)) {
602  bool found = false;
603  uint32_t indep = C->calcLoopDepFlag(depth);
604  float cc{float(
605  C->getCost(TTI, IR::VectorWidth{maxVF, maxl2VF}).recipThroughput)};
606  for (ptrdiff_t i = compute; i < compute_independence.size(); ++i) {
607  if (compute_independence[i].second != indep) continue;
608  found = true;
609  compute_independence[i].first += cc;
610  break;
611  }
612  if (!found) compute_independence.emplace_back(cc, indep);
613  } // else if (auto *S = llvm::dyn_cast<IR::Loop>(N)) {
614  }
615  auto [known_trip, trip_count] = L->getAffineLoop()->tripCount(depth);
616  uint16_t compcnt = compute_independence.size() - compute,
617  omemcnt = orth_axes.size() - omemory,
618  cmemcnt = conv_axes.size() - cmemory;
619  IR::Loop *SL = L->getSubLoop();
620  cost_counts.emplace_back(known_trip, trip_count, compcnt, omemcnt, cmemcnt,
621  SL ? 0 : exitCount);
622  if (SL) iterLoopLevel(alloc, deps, SL, maxl2VF, TTI, ++depth, exitCount);
623  else leafCosts(alloc, deps, L, maxl2VF, TTI);
624  }
625  static constexpr auto memCostArray(IR::Addr *A, IR::Addr::Costs c)
626  -> std::array<IR::Addr::Costs, 2> {
627  std::array<IR::Addr::Costs, 2> ret{};
628  ret[A->isStore()] = c;
629  return ret;
630  }
631  void leafCosts(Arena<> *alloc, LoopDepSatisfaction deps, IR::Loop *L,
632  unsigned maxl2VF, const llvm::TargetTransformInfo &TTI) {
633  // TODO: if (!SL) we're in a leaf, and need compute latency
634  // We use the `IROptimizer::loopDepSats` to check the depencencies held at
635  // the loop. We check these for those that look like reductions that are
636  // legal to reassociate (we check this earlier and set
637  // `in->reassociableReductionPair()==out`), e.g. integer add chains or
638  // floating point with the reassociate FMF set.
639  // When we have reductions, we have src->dst chains stored through
640  // `linkReductionDst()` that can be used for accumulating latencies.
641  // FIXME (maybe): Current implementation only allows each instruction to be
642  // a part of 1 chain.
643  // for (j in J){ // arbitrary number of outer
644  // loops
645  // %w = %array[j...];
646  // %x = foo(%w);
647  // for (i in I){ // inner loop(s)
648  // %y = bar(%x);
649  // }
650  // %z = quz(%y);
651  // %array[j...] = %z;
652  // }
653  // Rather than using PhiNodes, we represent dependencies through addresses.
654  // we can get legality from the loop.
655  // The tricker thing to compute here is register pressure
656  llvm::InstructionCost::CostType latency{0};
657  for (poly::Dependence d : deps.depencencies(L)) {
658  // instruction latency can be a function of vector width
659  latency =
660  std::max(latency, d.input()->reductionLatency(TTI, maxVectorWidth));
661  }
662  CostModeling::Legality legality = L->getLegality();
663  uint16_t l = std::numeric_limits<uint16_t>::max();
664  if (l > latency) l = latency;
665  // for reg use, lets add register dep flag
666  // what kind of traversal would minimize width?
667  // breadth-first lets us retire early, but can increase
668  // live count?
669  // Note, every reduction must add register contribution.
670  leafs.push_back(Pair<RegisterUseByUnroll, Pair<uint16_t, uint16_t>>(
671  RegisterUseByUnroll(alloc, TTI, L),
672  Pair<uint16_t, uint16_t>{l, legality.numReductions()}));
673  // for (IR::Node *N = L->getChild(); N; N = N->getNext()) {}
674  return;
675  };

679  unsigned maxl2VF, const llvm::TargetTransformInfo &TTI,

689 public:
690  // this is a vector fun, where indexing may do non-trivial computation
691  // also, mapping from this vector to loop position isn't trivial either
692  // hence, we use a 2 x max_depth matrix that we copy into as we descend
693  // (and pop from as we ascend). Row `0` is for inverse values,
694  // and row `1` for direct values.
695  // Inverses are favored as our costs fns use them more often.
696  constexpr auto operator()(alloc::Arena<> alloc,
697  const AbstractVector auto &x) const {
698  using T = utils::eltype_t<decltype(x)>;
699  static constexpr double numLoadUnits = 2.0;
700  static constexpr double numStoreUnits = 2.0;
701  static constexpr double coreWidth = 4.0;
702  utils::invariant(max_depth < 16);
703  // row 0: inverse unrolls
704  // row 1: unrolls
705  // row 2: cumprod invunroll
706  math::MutArray<T, math::DenseDims<3>> invunrolls{
707  math::matrix<T>(alloc, math::Row<3>{}, math::Col<>{max_depth})};
708  ptrdiff_t i = 0, depth = 0, mi = 0, mc = 0, ci = 0, li = 0;
709  double tripcounts[16];
710  VectorizationFactor vf{};
711  // we evaluate every iteration
712  T c{};
713  for (auto [comptimetrip, trip_count, compute, omem, cmem, exit, l2vw] :
714  cost_counts) {
715  if (l2vw) {
716  invariant(vf.l2factor == 0);
717  invariant(vf.indexMask == 0);
718  vf.l2factor = l2vw;
719  vf.indexMask = uint32_t(1) << depth;
720  }
721  invunrolls[1, depth] = x[i++];
722  invunrolls[2, depth] = invunrolls[0, depth] = 1 / invunrolls[1, depth];
723  if (depth) invunrolls[2, depth] *= invunrolls[2, depth - 1];
724  tripcounts[depth] =
725  (depth ? tripcounts[depth - 1] * trip_count : trip_count);
726  // load, store, all instr, latency
727  math::SVector<T, 3> c3{
728  memcosts(invunrolls, vf, orth_axes[_(0, omem) + mi])};
729  mi += omem;
730  c3 += memcosts(invunrolls, vf, conv_axes[_(0, cmem) + mc]);
731  mc += cmem;
732  c3[2] += compcosts(invunrolls, compute_independence[_(0, compute) + ci]);
733  ci += compute;
734  if (exit) {
735  const auto &[reguse, lt] = leafs[li++];
736  auto [l, numreduct] = lt;
737  // we're now in a leaf, meaning we must consider register costs,
738  // as well as reduction costs and latency of reduction chains.
739  math::SVector<T, 4> c4;
740  c4[_(0, 3)] << c3;
741  c4[3] = l * invunrolls[2, depth];
742  T ru = reguse.registerPressure(invunrolls);
743  c4[0] += ru;
744  c4[1] += ru;
745  c4[2] += 2 * ru;
746  if (numreduct) {
747  c4[2] +=
748  compcosts(invunrolls, compute_independence[_(0, numreduct) + ci]) *
749  log2(invunrolls[1, depth]) / trip_count;
750  ci += numreduct;
751  }
752  c += tripcounts[depth] *
753  smax(c4 * math::SVector<double, 4>{1.0 / numLoadUnits,
754  1.0 / numStoreUnits,
755  1.0 / coreWidth, 1.0});
756  } else {
757  c += tripcounts[depth] *
758  smax(c3 * math::SVector<double, 3>{1.0 / numLoadUnits,
759  1.0 / numStoreUnits,
