93 constexpr auto cost(const AbstractMatrix auto &invunrolls, uint32_t indepAxes)
94  -> utils::eltype_t<decltype(invunrolls)> {
95  utils::eltype_t<decltype(invunrolls)> c{};
96  if (indepAxes) {
97  uint32_t tz = std::countr_zero(indepAxes);
98  c = invunrolls[0, tz++];
99  for (uint32_t d = indepAxes >> tz, i = tz; d; d >>= tz, i += tz) {
100  tz = std::countr_zero(d);
101  c *= invunrolls[0, i + tz++];
102  }
103  } else c = 1;
104  return c;
105 }

107 // memory costs, unnormalized by `prod(unrolls)`
108 // `invunrolls` is a matrix, row-0 are the inverse unrolls, row-1 unrolls.
109 constexpr auto cost(const AbstractMatrix auto &invunrolls, OrthogonalAxes orth,
110  VectorizationFactor vfi)
111  -> utils::eltype_t<decltype(invunrolls)> {
112 
113  utils::eltype_t<decltype(invunrolls)> c{cost(invunrolls, orth.indep)};
114  if ((vfi.index < 32) && !(orth.indep & (1 << vfi.index))) {
115  // depends vectorized index
116  if (vfi.index == orth.contig) {
117  c *= orth.memcost.contiguous;
118  } else if (orth.contig >= 32) {
119  c *= orth.memcost.discontiguous;
120  } else {
121  // Discontiguous vector load.
122  // We consider two alternatives:
123  // 1. gather/scatter (discontiguous)
124  // 2. contiguous load for each vectorization factor of length equal to
125  // unroll, followed by shuffles.
126  // E.g., unroll contig by 4, another dim is vectorized by 8:
127  // we'd have 8 vloads (max(4/8,1) * 8), followed by 4*log2(8) +
128  // log2(max(8/4,1))*4 shuffles.
129  // Or, if we unroll contig by 8, and another dim is vectorzeed by 2, we'd
130  // have 8 = (max(8/2,1) * 2) vloads, 8*log2(2) + log2(max(2/8,1))*8
131  // shuffles.
132  // We divide by `u[contig]`, as it is now accounted for
133  // So we have
134  // max(v/u, 1) + u*log2(v) + log2(max(v/u ,1))*u
135  utils::eltype_t<decltype(invunrolls)> iu{invunrolls[0, orth.contig]},
136  u{invunrolls[1, orth.contig]},
137  mr{math::smax((1 << vfi.l2factor) * iu, 1)};
138  utils::invariant(iu == 1 / u);
139  c *=
140  math::smin(orth.memcost.contiguous * mr + u * (vfi.l2factor + log2(mr)),
141  orth.memcost.discontiguous);
142  }
143  }
144  return c;
145 }
153 constexpr auto cost(const AbstractMatrix auto &invunrolls, OrthogonalAxes orth,
154  VectorizationFactor vfi, DensePtrMatrix<int64_t> inds)
155  -> utils::eltype_t<decltype(invunrolls)> {
156  utils::eltype_t<decltype(invunrolls)> c{1};
157  auto [arrayDim, numLoops] = inds.size();
158  utils::invariant(numLoops > 0);
159  utils::invariant(arrayDim > 0);
160  utils::invariant(arrayDim <= 64);
161  utils::invariant(invunrolls.numCol(), inds.numCol());
162  for (ptrdiff_t d = 0; d < arrayDim; ++d) {
163  int64_t g = 0;
164  containers::BitSet64 bs;
165  utils::eltype_t<decltype(invunrolls)> uprod;
166  for (ptrdiff_t l = 0; l < numLoops; ++l) {
167  if (l == vfi.index) continue;
168  int64_t a = inds[d, l];
169  if (!a) continue;
170  bool docontinue{false};
171  // We only
172  for (ptrdiff_t k = 0; k < arrayDim; ++k) {
173  if ((k == d) || (!inds[k, l])) continue;
174  docontinue = (inds[d, _] != inds[k, _]) || (d > k);
175  if (docontinue) break;
176  }
177  if (docontinue) continue;
178  if (bs.empty()) {
179  g = a;
180  uprod = invunrolls[0, l];
181  } else {
182  g = math::gcd(g, a);
183  uprod *= invunrolls[0, l];
184  }
185  bs.insert(l);
186  };
187  if (bs.size() < 2) continue;
188  utils::eltype_t<decltype(invunrolls)> prod{1};
189  for (ptrdiff_t l : bs) {
190  if (l == vfi.index) continue;
191  int64_t a = inds[d, l];
192  if (!a) continue;
193  prod *= (1 - (a / g) * (uprod / invunrolls[0, l]));
194  }
195  c *= (1 - prod);
196  }
197  // c is a scaling factor; now we proceed to calculate cost similaly to the
198  // orth-axis implementation above.
199  return c * cost(invunrolls, vfi, orth);
200 }

215 constexpr auto registerPressure(const AbstractMatrix auto &invunrolls,
216  const RegisterUseByUnroll &r)
217  -> utils::eltype_t<decltype(invunrolls)> {
218  utils::eltype_t<decltype(invunrolls)> acc{0};
219  for (auto [c, m] : r) {
220  utils::eltype_t<decltype(invunrolls)> t{1};
221  containers::BitSet64 bs{std::array<uint64_t, 1>{m}};
222  for (ptrdiff_t i : bs) t *= invunrolls[1, i];
223  acc += c * t;
224  }

464 class LoopTreeCostFn {
465  math::PtrVector<LoopCostCounts> cost_counts;
466  math::PtrVector<OrthogonalAxes> orth_axes;
467  math::PtrVector<Pair<OrthogonalAxes, DensePtrMatrix<int64_t>>> conv_axes;
468  math::PtrVector<std::array<uint32_t, 2>> compute_independence;
469  math::PtrVector<Pair<RegisterUseByUnroll, Pair<uint16_t, uint16_t>>> leafs;
470  VectorizationFactor vf;
471  ptrdiff_t max_depth;
472 
473 public:
474  // this is a vector fun, where indexing may do non-trivial computation
475  // also, mapping from this vector to loop position isn't trivial either
476  // hence, we use a 2 x max_depth matrix that we copy into as we descend
477  // (and pop from as we ascend). Row `0` is for inverse values,
478  // and row `1` for direct values.
479  // Inverses are favored as our costs fns use them more often.
480  constexpr auto operator()(alloc::Arena<> alloc,
481  const AbstractVector auto &x) const {
482  using T = utils::eltype_t<decltype(x)>;
483  utils::invariant(max_depth < 16);
484  math::MutArray<T, math::DenseDims<2>> invunrolls{
485  math::matrix<T>(alloc, math::Row<2>{}, math::Col<>{max_depth})};
486  ptrdiff_t i = 0, depth = 0, mi = 0, mc = 0, ci = 0, li = 0;
487  double tripcounts[16];
488  // we evaluate every iteration
489  T c{};
490  for (auto [comptimetrip, trip_count, omem, cmem, exit, compute] :
491  cost_counts) {
492  invunrolls[1, depth] = x[i++];
493  invunrolls[0, depth] = 1 / invunrolls[1, depth];
494  tripcounts[depth] =
495  (depth ? tripcounts[depth - 1] * trip_count : trip_count);
496  T cc{compcosts(invunrolls, compute_independence[_(0, compute) + ci])};
497  ci += compute;
498  if (exit) {
499  auto [reguse, lt] = leafs[li++];
500  auto [l, numreduct] = lt;
501  // we're now in a leaf, meaning we must consider register costs,
502  // as well as reduction costs and latency of reduction chains.
503  cc = smax(cc, l / invunrolls[depth]);
504  cc += registerPressure(invunrolls, reguse);
505  if (numreduct) {
506  cc +=
507  compcost(invunrolls, compute_independence[_(0, numreduct) + ci]) *
508  log2(invunrolls[1, depth]) / trip_count;
509  ci += numreduct;
510  }
511  }
512  cc += memcosts(invunrolls, vf, orth_axes[_(0, omem) + mi]);
513  mi += omem;
514  cc += memcosts(invunrolls, vf, conv_axes[_(0, cmem) + mc]);
515  mc += cmem;
516  c += tripcounts[depth] * cc;
517  // Decrement depth by `exit - 1`; the `-1` corresponds
518  // to descending into this header, while we exit `exit` loops afterwards.
519  depth -= exit - 1;
520  }
521  return c;
522  }
