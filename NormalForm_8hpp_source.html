31 constexpr void zeroSupDiagonal(MutPtrMatrix<int64_t> A,
32  MutSquarePtrMatrix<int64_t> K, size_t i, Row M,
33  Col N) {
34  size_t minMN = std::min(size_t(M), size_t(N));
35  for (size_t j = i + 1; j < M; ++j) {
36  int64_t Aii = A(i, i);
37  if (int64_t Aji = A(j, i)) {
38  const auto [p, q, Aiir, Aijr] = gcdxScale(Aii, Aji);
39  for (size_t k = 0; k < minMN; ++k) {
40  int64_t Aki = A(i, k);
41  int64_t Akj = A(j, k);
42  int64_t Kki = K(i, k);
43  int64_t Kkj = K(j, k);
44  // when k == i, then
45  // p * Aii + q * Akj == r, so we set A(i,i) = r
46  A(i, k) = p * Aki + q * Akj;
47  // Aii/r * Akj - Aij/r * Aki = 0
48  A(j, k) = Aiir * Akj - Aijr * Aki;
49  // Mirror for K
50  K(i, k) = p * Kki + q * Kkj;
51  K(j, k) = Aiir * Kkj - Aijr * Kki;
52  }
53 
54  for (auto k = size_t(N); k < M; ++k) {
55  int64_t Kki = K(i, k);
56  int64_t Kkj = K(j, k);
57  K(i, k) = p * Kki + q * Kkj;
58  K(j, k) = Aiir * Kkj - Aijr * Kki;
59  }
60 
61  for (auto k = size_t(M); k < N; ++k) {
62  int64_t Aki = A(i, k);
63  int64_t Akj = A(j, k);
64  A(i, k) = p * Aki + q * Akj;
65  A(j, k) = Aiir * Akj - Aijr * Aki;
66  }
67  }
68  }

72 constexpr void zeroSubDiagonal(MutPtrMatrix<int64_t> A,
73  MutSquarePtrMatrix<int64_t> K, size_t k, Row M,
74  Col N) {
75  int64_t Akk = A(k, k);
76  if (Akk == -1) {
77  for (size_t m = 0; m < N; ++m) A(k, m) *= -1;
78  for (size_t m = 0; m < M; ++m) K(k, m) *= -1;
79  } else {
80  assert(Akk == 1);
81  }
82  size_t minMN = std::min(size_t(M), size_t(N));
83  for (size_t z = 0; z < k; ++z) {
84  // eliminate `A(k,z)`
85  if (int64_t Akz = A(z, k)) {
86  // A(k, k) == 1, so A(k,z) -= Akz * 1;
87  // A(z,_) -= Akz * A(k,_);
88  // K(z,_) -= Akz * K(k,_);
89 
90  for (size_t i = 0; i < minMN; ++i) {
91  A(z, i) -= Akz * A(k, i);
92  K(z, i) -= Akz * K(k, i);
93  }
94 
95  for (auto i = size_t(N); i < M; ++i) K(z, i) -= Akz * K(k, i);
96 
97  for (auto i = size_t(M); i < N; ++i) A(z, i) -= Akz * A(k, i);
98  }
99  }

137 constexpr auto orthogonalizeBang(MutPtrMatrix<int64_t> &A)
138  -> std::pair<SquareMatrix<int64_t>, Vector<unsigned>> {
139  // we try to orthogonalize with respect to as many rows of `A` as we can
140  // prioritizing earlier rows.
141  auto [M, N] = A.size();
142  SquareMatrix<int64_t> K{identity(std::allocator<int64_t>{}, unsigned(M))};
143  Vector<unsigned> included;
144  included.reserve(std::min(size_t(M), size_t(N)));
145  for (size_t i = 0, j = 0; i < std::min(size_t(M), size_t(N)); ++j) {
146  // zero ith row
147  if (pivotRows(A, K, i, M)) {
148  // cannot pivot, this is a linear combination of previous
149  // therefore, we drop the row
150  dropCol(A, i, M, --N);
151  } else {
152  zeroSupDiagonal(A, K, i, M, N);
153  int64_t Aii = A(i, i);
154  if (constexpr_abs(Aii) != 1) {
155  // including this row renders the matrix not unimodular!
156  // therefore, we drop the row.
157  dropCol(A, i, M, --N);
158  } else {
159  // we zero the sub diagonal
160  zeroSubDiagonal(A, K, i++, M, N);
161  included.push_back(j);
162  }
163  }
164  }
165  return std::make_pair(std::move(K), std::move(included));
166 }

212 constexpr void reduceSubDiagonal(MutPtrMatrix<int64_t> A, Col r, Row c) {
213  int64_t Akk = A(c, r);
214  if (Akk < 0) {
215  Akk = -Akk;
216  A(c, _) *= -1;
217  }
218  for (size_t z = 0; z < c; ++z) {
219  // try to eliminate `A(k,z)`
220  // if Akk == 1, then this zeros out Akz
221  if (int64_t Azr = A(z, r)) {
222  // we want positive but smaller subdiagonals
223  // e.g., `Akz = 5, Akk = 2`, then in the loop below when `i=k`, we
224  // set A(k,z) = A(k,z) - (A(k,z)/Akk) * Akk
225  // = 5 - 2*2 = 1
226  // or if `Akz = -5, Akk = 2`, then in the loop below we get
227  // A(k,z) = A(k,z) - ((A(k,z)/Akk) - ((A(k,z) % Akk) != 0) * Akk
228  // = -5 - (-2 - 1)*2 = = 6 - 5 = 1
229  // if `Akk = 1`, then
230  // A(k,z) = A(k,z) - (A(k,z)/Akk) * Akk
231  // = A(k,z) - A(k,z) = 0
232  // or if `Akz = -7, Akk = 39`, then in the loop below we get
233  // A(k,z) = A(k,z) - ((A(k,z)/Akk) - ((A(k,z) % Akk) != 0) * Akk
234  // = -7 - ((-7/39) - 1)*39 = = 6 - 5 = 1
235  int64_t AzrOld = Azr;
236  Azr /= Akk;
237  if (AzrOld < 0) Azr -= (AzrOld != (Azr * Akk));
238  A(z, _) -= Azr * A(c, _);
239  }
240  }

377 constexpr auto zeroWithRowOp(MutPtrMatrix<int64_t> A, Row i, Row j, Col k,
378  int64_t f) -> int64_t {
379  int64_t Aik = A(i, k);
380  if (!Aik) return f;
381  int64_t Ajk = A(j, k);
382  assert(Ajk != 0);
383  int64_t g = gcd(Aik, Ajk);
384  Aik /= g;
385  Ajk /= g;
386  int64_t ret = f * Ajk;
387  g = ret;
388  for (size_t l = 0; l < A.numCol(); ++l) {
389  int64_t Ail = Ajk * A(i, l) - Aik * A(j, l);
390  A(i, l) = Ail;
391  g = gcd(Ail, g);
392  }
393  if (g > 1) {
394  for (size_t l = 0; l < A.numCol(); ++l)
395  if (int64_t Ail = A(i, l)) A(i, l) = Ail / g;
396  assert(ret % g == 0);
397  ret /= g;
398  }
399  return ret;
400 }

429 constexpr void zeroColumnPair(std::array<MutPtrMatrix<int64_t>, 2> AB, Col c,
430  Row r) {
431  auto [A, B] = AB;
432  const Col N = A.numCol();
433  const Col K = B.numCol();
434  const Row M = A.numRow();
435  assert(M == B.numRow());
436  for (size_t j = 0; j < r; ++j) {
437  int64_t Arc = A(r, c);
438  if (int64_t Ajc = A(j, c)) {
439  int64_t g = gcd(Arc, Ajc);
440  Arc /= g;
441  Ajc /= g;
442  A(j, _) << Arc * A(j, _) - Ajc * A(r, _);
443  B(j, _) << Arc * B(j, _) - Ajc * B(r, _);
444  }
445  }
446  // greater rows in previous columns have been zeroed out
447  // therefore it is safe to use them for row operations with this row
448  for (Row j = r + 1; j < M; ++j) {
449  int64_t Arc = A(r, c);
450  if (int64_t Ajc = A(j, c)) {
451  const auto [p, q, Arcr, Ajcr] = gcdxScale(Arc, Ajc);
452 
453  for (size_t k = 0; k < N; ++k) {
454  int64_t Ark = A(r, k);
455  int64_t Ajk = A(j, k);
456  A(r, k) = q * Ajk + p * Ark;
457  A(j, k) = Arcr * Ajk - Ajcr * Ark;
458  }
459 
460  for (size_t k = 0; k < K; ++k) {
461  int64_t Brk = B(r, k);
462  int64_t Bjk = B(j, k);
463  B(r, k) = q * Bjk + p * Brk;
464  B(j, k) = Arcr * Bjk - Ajcr * Brk;
465  }
466  }
467  }

510  int64_t prev = 1, pivInd = 0;
511  for (size_t r = 0, c = 0; c < N && r < M; ++c) {
512  if (auto piv = pivotRowsBareiss(A, c, M, r)) {
513  pivots[pivInd++] = *piv;
514  for (size_t k = r + 1; k < M; ++k) {
515  for (size_t j = c + 1; j < N; ++j) {
516  auto Akj_u = A(r, c) * A(k, j) - A(k, c) * A(r, j);
517  auto Akj = Akj_u / prev;
518  assert(Akj_u % prev == 0);
519  A(k, j) = Akj;
520  }
521  A(k, r) = 0;
522  }
523  prev = A(r++, c);
524  }
525  }
526 }

593 constexpr void nullSpace11(LinAlg::DenseMatrix<int64_t> &B,
594  LinAlg::DenseMatrix<int64_t> &A) {
595  const Row M = A.numRow();
596  B.resizeForOverwrite(LinAlg::SquareDims{M});
597  B << 0;
598  B.diag() << 1;
599  solveSystem(A, B);
600  Row R = numNonZeroRows(A);
601  // slice B[_(R,end), :]
602  if (!R) return;
603  // we keep last D columns
604  Row D = M - R;
605  size_t o = size_t(R * M);
606  // we keep `D` columns
607  // TODO: shift pointer instead?
608  // This seems like a bad idea given ManagedArrays that must
609  // free their own ptrs; we'd have to still store either the old pointer or the
610  // offset.

612  // that takes a `BumpAlloc<>` as input to allocate `B`, as
613  // then we don't need to track the pointer.
614  std::copy_n(B.data() + o, size_t(D * M), B.data());
615  B.truncate(D);
616 }
