31 constexpr void zeroSupDiagonal(MutPtrMatrix<int64_t> A,
32  MutSquarePtrMatrix<int64_t> K, size_t i, Row M,
33  Col N) {
34  size_t minMN = std::min(size_t(M), size_t(N));
35  for (size_t j = i + 1; j < M; ++j) {
36  int64_t Aii = A(i, i);
37  if (int64_t Aji = A(j, i)) {
38  const auto [p, q, Aiir, Aijr] = gcdxScale(Aii, Aji);
39  for (size_t k = 0; k < minMN; ++k) {
40  int64_t Aki = A(i, k);
41  int64_t Akj = A(j, k);
42  int64_t Kki = K(i, k);
43  int64_t Kkj = K(j, k);
44  // when k == i, then
45  // p * Aii + q * Akj == r, so we set A(i,i) = r
46  A(i, k) = p * Aki + q * Akj;
47  // Aii/r * Akj - Aij/r * Aki = 0
48  A(j, k) = Aiir * Akj - Aijr * Aki;
49  // Mirror for K
50  K(i, k) = p * Kki + q * Kkj;
51  K(j, k) = Aiir * Kkj - Aijr * Kki;
52  }
53  for (auto k = size_t(N); k < M; ++k) {
54  int64_t Kki = K(i, k);
55  int64_t Kkj = K(j, k);
56  K(i, k) = p * Kki + q * Kkj;
57  K(j, k) = Aiir * Kkj - Aijr * Kki;
58  }
59  for (auto k = size_t(M); k < N; ++k) {
60  int64_t Aki = A(i, k);
61  int64_t Akj = A(j, k);
62  A(i, k) = p * Aki + q * Akj;
63  A(j, k) = Aiir * Akj - Aijr * Aki;
64  }
65  }
66  }

70 constexpr void zeroSubDiagonal(MutPtrMatrix<int64_t> A,
71  MutSquarePtrMatrix<int64_t> K, size_t k, Row M,
72  Col N) {
73  int64_t Akk = A(k, k);
74  if (Akk == -1) {
75  for (size_t m = 0; m < N; ++m) A(k, m) *= -1;
76  for (size_t m = 0; m < M; ++m) K(k, m) *= -1;
77  } else {
78  assert(Akk == 1);
79  }
80  size_t minMN = std::min(size_t(M), size_t(N));
81  for (size_t z = 0; z < k; ++z) {
82  // eliminate `A(k,z)`
83  if (int64_t Akz = A(z, k)) {
84  // A(k, k) == 1, so A(k,z) -= Akz * 1;
85  // A(z,_) -= Akz * A(k,_);
86  // K(z,_) -= Akz * K(k,_);
87  for (size_t i = 0; i < minMN; ++i) {
88  A(z, i) -= Akz * A(k, i);
89  K(z, i) -= Akz * K(k, i);
90  }
91  for (auto i = size_t(N); i < M; ++i) K(z, i) -= Akz * K(k, i);
92  for (auto i = size_t(M); i < N; ++i) A(z, i) -= Akz * A(k, i);
93  }
94  }

132 constexpr auto orthogonalizeBang(MutPtrMatrix<int64_t> &A)
133  -> std::pair<SquareMatrix<int64_t>, Vector<unsigned>> {
134  // we try to orthogonalize with respect to as many rows of `A` as we can
135  // prioritizing earlier rows.
136  auto [M, N] = A.size();
137  SquareMatrix<int64_t> K{identity(std::allocator<int64_t>{}, unsigned(M))};
138  Vector<unsigned> included;
139  included.reserve(std::min(size_t(M), size_t(N)));
140  for (size_t i = 0, j = 0; i < std::min(size_t(M), size_t(N)); ++j) {
141  // zero ith row
142  if (pivotRows(A, K, i, M)) {
143  // cannot pivot, this is a linear combination of previous
144  // therefore, we drop the row
145  dropCol(A, i, M, --N);
146  } else {
147  zeroSupDiagonal(A, K, i, M, N);
148  int64_t Aii = A(i, i);
149  if (constexpr_abs(Aii) != 1) {
150  // including this row renders the matrix not unimodular!
151  // therefore, we drop the row.
152  dropCol(A, i, M, --N);
153  } else {
154  // we zero the sub diagonal
155  zeroSubDiagonal(A, K, i++, M, N);
156  included.push_back(j);
157  }
158  }
159  }
160  return std::make_pair(std::move(K), std::move(included));
161 }

205 constexpr void reduceSubDiagonal(MutPtrMatrix<int64_t> A, Col c, Row r) {
206  int64_t Akk = A(r, c);
207  if (Akk < 0) {
208  Akk = -Akk;
209  A(r, _) *= -1;
210  }
211  for (size_t z = 0; z < r; ++z) {
212  // try to eliminate `A(k,z)`
213  // if Akk == 1, then this zeros out Akz
214  if (int64_t Azr = A(z, c)) {
215  // we want positive but smaller subdiagonals
216  // e.g., `Akz = 5, Akk = 2`, then in the loop below when `i=k`, we
217  // set A(k,z) = A(k,z) - (A(k,z)/Akk) * Akk
218  // = 5 - 2*2 = 1
219  // or if `Akz = -5, Akk = 2`, then in the loop below we get
220  // A(k,z) = A(k,z) - ((A(k,z)/Akk) - ((A(k,z) % Akk) != 0) * Akk
221  // = -5 - (-2 - 1)*2 = = 6 - 5 = 1
222  // if `Akk = 1`, then
223  // A(k,z) = A(k,z) - (A(k,z)/Akk) * Akk
224  // = A(k,z) - A(k,z) = 0
225  // or if `Akz = -7, Akk = 39`, then in the loop below we get
226  // A(k,z) = A(k,z) - ((A(k,z)/Akk) - ((A(k,z) % Akk) != 0) * Akk
227  // = -7 - ((-7/39) - 1)*39 = = 6 - 5 = 1
228  int64_t oAzr = Azr;
229  Azr /= Akk;
230  if (oAzr < 0) Azr -= (oAzr != (Azr * Akk));
231  A(z, _) -= Azr * A(r, _);
232  }
233  }

369 constexpr auto zeroWithRowOp(MutPtrMatrix<int64_t> A, Row i, Row j, Col k,
370  int64_t f) -> int64_t {
371  int64_t Aik = A(i, k);
372  if (!Aik) return f;
373  int64_t Ajk = A(j, k);
374  invariant(Ajk != 0);
375  int64_t g = gcd(Aik, Ajk);
376  Aik /= g;
377  Ajk /= g;
378  int64_t ret = f * Ajk;
379  g = ret;
380  for (size_t l = 0; l < A.numCol(); ++l) {
381  int64_t Ail = Ajk * A(i, l) - Aik * A(j, l);
382  A(i, l) = Ail;
383  g = gcd(Ail, g);
384  }
385  if (g > 1) {
386  for (size_t l = 0; l < A.numCol(); ++l)
387  if (int64_t Ail = A(i, l)) A(i, l) = Ail / g;
388  int64_t r = ret / g;
389  invariant(r * g, ret);
390  ret = r;
391  }
392  return ret;
393 }

421 // use row `r` to zero the remaining rows of column `c`
422 constexpr void zeroColumnPair(std::array<MutPtrMatrix<int64_t>, 2> AB, Col c,
423  Row r) {
424  auto [A, B] = AB;
425  const Col N = A.numCol();
426  const Col K = B.numCol();
427  const Row M = A.numRow();
428  assert(M == B.numRow());
429  for (size_t j = 0; j < r; ++j) {
430  int64_t Arc = A(r, c);
431  if (int64_t Ajc = A(j, c)) {
432  int64_t g = gcd(Arc, Ajc);
433  Arc /= g;
434  Ajc /= g;
435  A(j, _) << Arc * A(j, _) - Ajc * A(r, _);
436  B(j, _) << Arc * B(j, _) - Ajc * B(r, _);
437  }
438  }
439  // greater rows in previous columns have been zeroed out
440  // therefore it is safe to use them for row operations with this row
441  for (Row j = r + 1; j < M; ++j) {
442  int64_t Arc = A(r, c);
443  if (int64_t Ajc = A(j, c)) {
444  const auto [p, q, Arcr, Ajcr] = gcdxScale(Arc, Ajc);
445  for (size_t k = 0; k < N; ++k) {
446  int64_t Ark = A(r, k), Ajk = A(j, k);
447  A(r, k) = q * Ajk + p * Ark;
448  A(j, k) = Arcr * Ajk - Ajcr * Ark;
449  }
450  for (size_t k = 0; k < K; ++k) {
451  int64_t Brk = B(r, k), Bjk = B(j, k);
452  B(r, k) = q * Bjk + p * Brk;
453  B(j, k) = Arcr * Bjk - Ajcr * Brk;
454  }
455  }
456  }

499  for (size_t k = r + 1; k < M; ++k) {
500  for (size_t j = c + 1; j < N; ++j) {
501  auto uAkj = A(r, c) * A(k, j) - A(k, c) * A(r, j), Akj = uAkj / prev;
502  invariant(uAkj, Akj * prev);
503  A(k, j) = Akj;
504  }
505  A(k, r) = 0;
506  }
507  prev = A(r++, c);
508  }
509  }
510 }

518 // update a reduced matrix for a new row
519 // doesn't reduce last row (assumes you're solving for it)
520 constexpr auto updateForNewRow(MutPtrMatrix<int64_t> A) -> size_t {
521  // use existing rows to reduce
522  size_t M = size_t(A.numRow()), N = size_t(A.numCol()), MM = M - 1, NN = N - 1,
523  n = 0, i, j = std::numeric_limits<size_t>::max();
524  for (size_t m = 0; m < MM; ++m) {
525  assert(allZero(A(m, _(0, n))));
526  while (A(m, n) == 0) {
527  if ((j > NN) && (A(MM, n) != 0)) {
528  i = m;
529  j = n;
530  }
531  invariant((++n) < NN);
532  }
533  if (int64_t Aln = A(MM, n)) {
534  // use this to reduce last row
535  auto [x, y] = divgcd(Aln, A(m, n));
536  A(MM, _) << A(MM, _) * y - A(m, _) * x;
537  invariant(A(MM, n) == 0);
538  }
539  ++n;
540  }
541  // we've reduced the new row, now to use it...
542  // swap A(i,_(j,end)) with A(MM,_(j,end))
543  if (j <= NN) { // we could do with a lot less copying...
544  for (size_t l = i; l < MM; ++l)
545  for (size_t k = j; k < N; ++k) std::swap(A(l, k), A(MM, k));
546  } else {
547  // maybe there is a non-zero value
548  j = n;
549  for (; j < NN; ++j)
550  if (A(MM, j) != 0) break;
551  if (j == NN) return MM;
552  i = MM;
553  }
554  // zero out A(_(0,i),j) using A(i,j)
555  for (size_t k = 0; k < i; ++k) {
556  if (int64_t Akj = A(k, j)) {
557  auto [x, y] = divgcd(Akj, A(i, j));
558  A(k, _) << A(k, _) * y - A(i, _) * x;
559  }
560  }
561  return M;
562 }

623 constexpr void nullSpace11(LinAlg::DenseMatrix<int64_t> &B,
624  LinAlg::DenseMatrix<int64_t> &A) {

627  B << 0;
628  B.diag() << 1;

632  if (!R) return;
