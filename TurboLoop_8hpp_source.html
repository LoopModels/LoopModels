102  auto revLI = llvm::reverse(*LI);
103  auto ib = revLI.begin();
104  auto ie = revLI.end();
105  if (ib == ie) return;
106  // pushLoopTree wants a direct path from the last loop's exit block to
107  // E; we drop loops until we find one for which this is trivial.
108  llvm::BasicBlock *E = (*--ie)->getExitBlock();
109  while (!E) {

151  llvm::Loop *L, MutPtrVector<int> omega,
152  NotNull<poly::Loop> AL, IR::TreeResult tr)
153  -> IR::TreeResult {
154  // TODO: need to be able to connect instructions as we move out
155  auto predMapAbridged =
156  IR::Predicate::Map::descend(shortAllocator(), instructions, H, E, L);
157  if (!predMapAbridged) return {};
158  // Now we need to create Addrs
159  size_t depth = omega.size() - 1;
160  for (auto &[BB, P] : *predMapAbridged) { // rev order
161  for (llvm::Instruction &J : llvm::reverse(*BB)) {
162  if (L) assert(L->contains(&J));
163  llvm::Value *ptr{nullptr};
164  if (J.mayReadFromMemory())
165  if (auto *load = llvm::dyn_cast<llvm::LoadInst>(&J))
166  ptr = load->getPointerOperand();
167  else return {};
168  else if (J.mayWriteToMemory())
169  if (auto *store = llvm::dyn_cast<llvm::StoreInst>(&J))
170  ptr = store->getPointerOperand();
171  else return {};
172  else continue;
173  if (ptr == nullptr) return {};
174  auto [N, trret] = instructions.getArrayRef(&J, L, ptr, tr);
175  tr = trret;
176  if (tr.reject(depth)) return tr;
177  IR::Addr *A = llvm::cast<IR::Addr>(A);
178  A->setLoopNest(AL);
179  // if we didn't reject, it must have been an `Addr`
180  A->setFusionOmega(omega);
181  instructions.addPredicate(A, P, &*predMapAbridged);
182  }
183  }
184  return mergeInstructions(instructions, *predMapAbridged, TTI,
185  shortAllocator(), registers.getNumVectorBits(),

335  auto pushLoopTree(llvm::Loop *L, llvm::ArrayRef<llvm::Loop *> subLoops,
336  llvm::BasicBlock *H, llvm::BasicBlock *E,
337  NoWrapRewriter &nwr) -> size_t {
338 
339  size_t numSubLoops = subLoops.size();
340  if (!numSubLoops) return initLoopTree(pForest, L, H, E, nwr);
341  // branches of this tree;
342  llvm::SmallVector<NotNull<LoopTree>> branches;
343  branches.reserve(numSubLoops);
344  llvm::SmallVector<Predicate::Map> branchBlocks;
345  branchBlocks.reserve(numSubLoops + 1);
346  bool anyFail = false;
347  size_t interiorDepth = 0;
348  for (size_t i = 0; i < numSubLoops; ++i) {
349  llvm::Loop *subLoop = subLoops[i];
350  if (Loop *lret =
351  pushLoopTree(subLoop, subLoop->getSubLoops(), subLoop->getHeader(),
352  subLoop->getExitingBlock(), nwr)) {
353  // pushLoopTree succeeded, and we have `depth` inner loops
354  // within `subLoop` (inclusive, i.e. `depth == 1` would
355  // indicate that `subLoop` doesn't have any subLoops itself,
356  // which we check with the following assertion:
357  assert((depth > 1) || (subLoop->getSubLoops().empty()));
358 
359  // Now we check if we can create a direct path from `H` to
360  // `subLoop->getLoopPreheader();`
361  llvm::BasicBlock *subLoopPreheader = subLoop->getLoopPreheader();
362  if (auto predMap = Predicate::Map::descend(allocator, instructions, H,
363  subLoopPreheader, L)) {
364  interiorDepth = std::max(interiorDepth, depth);
365  branchBlocks.push_back(std::move(*predMap));
366  H = subLoop->getExitBlock();
367  } else {
368  // oops, no direct path, we split
369  anyFail = true;
370  if (branches.size() > 1) {
371  // we need to split off the last tree
372  LoopTree *lastTree = branches.pop_back_val();
373  // so we can push previous branches as one set
374  // for the final branchBlocks, we'll push H->H
375  split(branches, branchBlocks, H, L);
376  // reinsert last tree
377  branches.push_back(lastTree);
378  }
379  if (i + 1 < numSubLoops) H = subLoops[i + 1]->getLoopPreheader();
380  }
381  // for the next loop, we'll want a path to its preheader
382  // from this loop's exit block.
383  } else {
384  // `depth == 0` indicates failure, therefore we need to
385  // split loops
386  anyFail = true;
387  if (!branches.empty()) split(branches, branchBlocks, H, L);
388  if (i + 1 < numSubLoops) H = subLoops[i + 1]->getLoopPreheader();
389  }
390  }
391  if (!anyFail) {
392  // branches.size() > 0 because we have numSubLoops > 0 and !anyFail
393  // !anyFail means we called pushLoopTree, and returned depth > 0
394  // which means it must have called pushLoopTree, pushing into branches
395  // (pushLoopTree pushes into first arg, pForest, whenever ret > 0)
396  invariant(!branches.empty());
397  if (auto predMapAbridged =
398  Predicate::Map::descend(allocator, instructions, H, E, L)) {
399  branchBlocks.push_back(std::move(*predMapAbridged));
400 
401  auto *newTree = new (allocator)
402  LoopTree{allocator, L,
403  branches.front()->affineLoop->removeInnerMost(allocator),
404  std::move(branches), std::move(branchBlocks)};
405  pForest.push_back(newTree);
406  return ++interiorDepth;
407  }
408  }
409  if (!branches.empty()) split(branches, branchBlocks, H, L);
410  return 0;
411  }
