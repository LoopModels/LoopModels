214  uint64_t
215  fillAffineIndices(MutPtrVector<int64_t> v, Vector<int64_t> &offsets,
216  llvm::SmallVector<const llvm::SCEV *, 3> &symbolicOffsets,
217  const llvm::SCEV *S, int64_t mlt, size_t numPeeled) {
218  uint64_t blackList{0};
219  if (const llvm::SCEVAddRecExpr *x =
220  llvm::dyn_cast<const llvm::SCEVAddRecExpr>(S)) {
221  const llvm::Loop *L = x->getLoop();
222  size_t depth = L->getLoopDepth();
223  if (depth <= numPeeled) {
224  // we effectively have an offset
225  // we'll add an
226  addSymbolic(offsets, symbolicOffsets, S, 1);
227  for (size_t i = 1; i < x->getNumOperands(); ++i)
228  blackList |= blackListAllDependentLoops(x->getOperand(i));
229 
230  return blackList;
231  }
232  // outermost loop has loopInd 0
233  ptrdiff_t loopInd = ptrdiff_t(depth) - ptrdiff_t(numPeeled + 1);
234  if (x->isAffine()) {
235  if (loopInd >= 0) {
236  if (auto c = getConstantInt(x->getOperand(1))) {
237  // we want the innermost loop to have index 0
238  v(end - loopInd) += *c;
239  return fillAffineIndices(v, offsets, symbolicOffsets,
240  x->getOperand(0), mlt,
241  numPeeled);
242  } else
243  blackList |= (uint64_t(1) << uint64_t(loopInd));
244  }
245  // we separate out the addition
246  // the multiplication was either peeled or involved non-const
247  // multiple
248  blackList |=
249  fillAffineIndices(v, offsets, symbolicOffsets,
250  x->getOperand(0), mlt, numPeeled);
251  // and then add just the multiple here as a symbolic offset
252  const llvm::SCEV *addRec = SE->getAddRecExpr(
253  SE->getZero(x->getOperand(0)->getType()), x->getOperand(1),
254  x->getLoop(), x->getNoWrapFlags());
255  addSymbolic(offsets, symbolicOffsets, addRec, mlt);
256  return blackList;
257  } else if (loopInd >= 0)
258  blackList |= (uint64_t(1) << uint64_t(loopInd));
259  } else if (llvm::Optional<int64_t> c = getConstantInt(S)) {
260  offsets[0] += *c;
261  return 0;
262  } else if (const llvm::SCEVAddExpr *ex =
263  llvm::dyn_cast<const llvm::SCEVAddExpr>(S)) {
264  return fillAffineIndices(v, offsets, symbolicOffsets,
265  ex->getOperand(0), mlt, numPeeled) |
266  fillAffineIndices(v, offsets, symbolicOffsets,
267  ex->getOperand(1), mlt, numPeeled);
268  } else if (const llvm::SCEVMulExpr *ex =
269  llvm::dyn_cast<const llvm::SCEVMulExpr>(S)) {
270  if (auto op = getConstantInt(ex->getOperand(0))) {
271  return fillAffineIndices(v, offsets, symbolicOffsets,
272  ex->getOperand(1), mlt * (*op),
273  numPeeled);
274 
275  } else if (auto op = getConstantInt(ex->getOperand(1))) {
276  return fillAffineIndices(v, offsets, symbolicOffsets,
277  ex->getOperand(0), mlt * (*op),
278  numPeeled);
279  }
280  } else if (const llvm::SCEVCastExpr *ex =
281  llvm::dyn_cast<llvm::SCEVCastExpr>(S))
282  return fillAffineIndices(v, offsets, symbolicOffsets,
283  ex->getOperand(0), mlt, numPeeled);
284  addSymbolic(offsets, symbolicOffsets, S, mlt);
285  return blackList | blackListAllDependentLoops(S, numPeeled);
286  }

640  unsigned PTID = LT.parentLoop;
641  if (PTID == std::numeric_limits<unsigned>::max())
642  return;
643  LoopTree &PT = loopTrees[PTID];
644  size_t numLoops = LT.getNumLoops();

646  peelOuterLoops(loopTrees[ST], numLoops);
647 
648  LT.parentLoop =
649  std::numeric_limits<unsigned>::max(); // LT is now top of the tree
650  loopForests.push_back(LTID);
651  llvm::SmallVector<unsigned> &friendLoops = PT.subLoops;
652  // SHOW(LTID);
653  for (auto id : friendLoops)
654  llvm::errs() << ", " << id;
655  llvm::errs() << "\n";
656  if (friendLoops.front() != LTID) {
657  // we're cutting off the front
658  size_t numFriendLoops = friendLoops.size();
659  assert(numFriendLoops);
660  size_t loopIndex = 0;
661  for (size_t i = 1; i < numFriendLoops; ++i) {
662  if (friendLoops[i] == LTID) {
663  loopIndex = i;
664  break;
665  }
666  }
667  assert(loopIndex);
668  size_t j = loopIndex + 1;
669  if (j != numFriendLoops) {
670  // we have some remaining paths we split off
671  llvm::SmallVector<unsigned> tmp;
672  tmp.reserve(numFriendLoops - j);
673  // for paths, we're dropping LT
674  // thus, our paths are paths(_(0,j)), paths(_(j,end))
675  llvm::SmallVector<PredicatedChain> paths;
676  paths.reserve(numFriendLoops - loopIndex);
677  for (size_t i = j; i < numFriendLoops; ++i) {
678  peelOuterLoops(loopTrees[friendLoops[i]], numLoops - 1);
679  tmp.push_back(friendLoops[i]);
680  paths.push_back(std::move(PT.paths[i]));
681  }
682  paths.push_back(std::move(PT.paths[numFriendLoops]));
683  loopForests.push_back(loopTrees.size());
684  // TODO: split paths
685  loopTrees.emplace_back(std::move(tmp), std::move(paths));
686  }
687  friendLoops.truncate(loopIndex);
688  PT.paths.truncate(j);
689  } else {
690  friendLoops.erase(friendLoops.begin());
691  PT.paths.erase(PT.paths.begin());
692  }
693  conditionOnLoop(PT, PTID);
