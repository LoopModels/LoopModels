295  // NOLINTNEXTLINE(misc-no-recursion)
296  auto pushLoopTree(llvm::Loop *L, llvm::ArrayRef<llvm::Loop *> subLoops,
297  llvm::BasicBlock *H, llvm::BasicBlock *E,
298  NoWrapRewriter &nwr) -> size_t {
299 
300  size_t numSubLoops = subLoops.size();
301  if (!numSubLoops) return initLoopTree(pForest, L, H, E, nwr);
302  // branches of this tree;
303  llvm::SmallVector<NotNull<LoopTree>> branches;
304  branches.reserve(numSubLoops);
305  llvm::SmallVector<Predicate::Map> branchBlocks;
306  branchBlocks.reserve(numSubLoops + 1);
307  bool anyFail = false;
308  size_t interiorDepth = 0;
309  for (size_t i = 0; i < numSubLoops; ++i) {
310  llvm::Loop *subLoop = subLoops[i];
311  if (Loop *lret =
312  pushLoopTree(subLoop, subLoop->getSubLoops(), subLoop->getHeader(),
313  subLoop->getExitingBlock(), nwr)) {
314  // pushLoopTree succeeded, and we have `depth` inner loops
315  // within `subLoop` (inclusive, i.e. `depth == 1` would
316  // indicate that `subLoop` doesn't have any subLoops itself,
317  // which we check with the following assertion:
318  assert((depth > 1) || (subLoop->getSubLoops().empty()));
319 
320  // Now we check if we can create a direct path from `H` to
321  // `subLoop->getLoopPreheader();`
322  llvm::BasicBlock *subLoopPreheader = subLoop->getLoopPreheader();
323  if (auto predMap = Predicate::Map::descend(allocator, instrCache, H,
324  subLoopPreheader, L)) {
325  interiorDepth = std::max(interiorDepth, depth);
326  branchBlocks.push_back(std::move(*predMap));
327  H = subLoop->getExitBlock();
328  } else {
329  // oops, no direct path, we split
330  anyFail = true;
331  if (branches.size() > 1) {
332  // we need to split off the last tree
333  LoopTree *lastTree = branches.pop_back_val();
334  // so we can push previous branches as one set
335  // for the final branchBlocks, we'll push H->H
336  split(branches, branchBlocks, H, L);
337  // reinsert last tree
338  branches.push_back(lastTree);
339  }
340  if (i + 1 < numSubLoops) H = subLoops[i + 1]->getLoopPreheader();
341  }
342  // for the next loop, we'll want a path to its preheader
343  // from this loop's exit block.
344  } else {
345  // `depth == 0` indicates failure, therefore we need to
346  // split loops
347  anyFail = true;
348  if (!branches.empty()) split(branches, branchBlocks, H, L);
349  if (i + 1 < numSubLoops) H = subLoops[i + 1]->getLoopPreheader();
350  }
351  }
352  if (!anyFail) {
353  // branches.size() > 0 because we have numSubLoops > 0 and !anyFail
354  // !anyFail means we called pushLoopTree, and returned depth > 0
355  // which means it must have called pushLoopTree, pushing into branches
356  // (pushLoopTree pushes into first arg, pForest, whenever ret > 0)
357  invariant(!branches.empty());
358  if (auto predMapAbridged =
359  Predicate::Map::descend(allocator, instrCache, H, E, L)) {
360  branchBlocks.push_back(std::move(*predMapAbridged));
361 
362  auto *newTree = new (allocator)
363  LoopTree{allocator, L,
364  branches.front()->affineLoop->removeInnerMost(allocator),
365  std::move(branches), std::move(branchBlocks)};
366  pForest.push_back(newTree);
367  return ++interiorDepth;
368  }
369  }
370  if (!branches.empty()) split(branches, branchBlocks, H, L);
371  return 0;
372  }

406  auto arrayRef(llvm::Loop *L, MutPtrVector<unsigned> omega,
407  NotNull<poly::Loop> aln, llvm::Instruction *loadOrStore,
408  llvm::Instruction *ptr, TreeResult tr, const llvm::SCEV *elSz)
409  -> TreeResult {
410  // code modified from
411  // https://llvm.org/doxygen/Delinearization_8cpp_source.html#l00582
412  const llvm::SCEV *accessFn = SE->getSCEVAtScope(ptr, L);
413 
414  const llvm::SCEV *pb = SE->getPointerBase(accessFn);
415  const auto *arrayPtr = llvm::dyn_cast<llvm::SCEVUnknown>(pb);
416  // Do not delinearize if we cannot find the base pointer.
417  if (!arrayPtr) {
418  if (ORE && L) [[unlikely]] {
419  remark("ArrayRefDeliniarize", L,
420  "ArrayReference failed because !basePointer\n", ptr);
421  }
422  // L is invalid, so we optimize the subloops and return `nullptr`
423  return {nullptr};
424  }
425  accessFn = SE->getMinusSCEV(accessFn, arrayPtr);
426  llvm::SmallVector<const llvm::SCEV *, 3> subscripts, sizes;
427  llvm::delinearize(*SE, accessFn, subscripts, sizes, elSz);
428  size_t numDims = subscripts.size();
429  invariant(numDims, sizes.size());
430  if (numDims == 0) return zeroDimRef(omega, aln, loadOrStore, arrayPtr, tr);
431  size_t numPeeled = tr.rejectDepth, numLoops = omega.size() - 1 - numPeeled;
432  IntMatrix Rt{StridedDims{numDims, numLoops}, 0};
433  llvm::SmallVector<const llvm::SCEV *, 3> symbolicOffsets;
434  uint64_t blackList{0};
435  Vector<int64_t> coffsets{unsigned(numDims), 0};
436  MutDensePtrMatrix<int64_t> offsMat{nullptr, DenseDims{numDims, 0}};
437  {
438  Vector<int64_t> offsets;
439  for (size_t i = 0; i < numDims; ++i) {
440  offsets << 0;
441  blackList |=
442  fillAffineIndices(Rt(i, _), &coffsets[i], offsets, symbolicOffsets,
443  subscripts[i], 1, numPeeled);
444  if (offsets.size() > offsMat.numCol())
445  extendDensePtrMatCols(allocator, offsMat, Row{i},
446  Col{offsets.size()});
447  offsMat(i, _) << offsets;
448  }
449  }
450  size_t numExtraLoopsToPeel = 64 - std::countl_zero(blackList);
451  Addr *op = Addr::construct(allocator, arrayPtr, *aln, loadOrStore,
452  Rt(_, _(numExtraLoopsToPeel, end)),
453  {std::move(sizes), std::move(symbolicOffsets)},
454  coffsets, offsMat.data(), omega);
455  ++omega.back();
456  if (ORE) [[unlikely]] {
457  llvm::SmallVector<char> x;
458  llvm::raw_svector_ostream os{x};
459  os << "Found ref: " << *op;
460  remark("AffineRef", L, os.str());
461  }
462  return {op, numExtraLoopsToPeel + numPeeled};
463  }
