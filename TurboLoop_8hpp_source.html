57 inline auto countNumLoopsPlusLeaves(const llvm::Loop *L) -> size_t {
58  const std::vector<llvm::Loop *> &subLoops = L->getSubLoops();
59  if (subLoops.empty()) return 1;
60  size_t numLoops = subLoops.size();
61  for (const auto &SL : subLoops) numLoops += countNumLoopsPlusLeaves(SL);
62  return numLoops;

133  [[nodiscard]] auto parseBlocks(llvm::BasicBlock *H, llvm::BasicBlock *E,
134  llvm::Loop *L, MutPtrVector<int> omega,
135  poly::Loop *AL, IR::TreeResult tr)
136  -> IR::TreeResult {
137  // TODO: need to be able to connect instructions as we move out
138  auto predMapAbridged =
139  IR::Predicate::Map::descend(shortAllocator(), instructions, H, E, L, tr);
140  if (!predMapAbridged) return {};
141  // Now we need to create Addrs
142  unsigned depth = omega.size() - 1;
143  tr.maxDepth = std::max(tr.maxDepth, depth);
144  for (auto &[BB, P] : *predMapAbridged) { // rev order
145  for (llvm::Instruction &J : llvm::reverse(*BB)) {
146  if (L) assert(L->contains(&J));
147  llvm::Value *ptr{nullptr};
148  if (J.mayReadFromMemory())
149  if (auto *load = llvm::dyn_cast<llvm::LoadInst>(&J))
150  ptr = load->getPointerOperand();
151  else return {};
152  else if (J.mayWriteToMemory())
153  if (auto *store = llvm::dyn_cast<llvm::StoreInst>(&J))
154  ptr = store->getPointerOperand();
155  else return {};
156  else continue;
157  if (ptr == nullptr) return {};
158  auto [V, trret] = instructions.getArrayRef(&J, L, ptr, tr);
159  tr = trret;
160  if (tr.reject(depth)) return tr;
161  auto *A = llvm::cast<IR::Addr>(V);
162  // if we didn't reject, it must have been an `Addr`
163  A->setFusionOmega(omega);
164  instructions.addPredicate(A, P, &(*predMapAbridged));
165  A->setLoopNest(AL);
166  }
167  loopBBs.insert(BB);
168  }
169  return IR::mergeInstructions(instructions, *predMapAbridged, *TTI,
170  *shortAllocator(),
171  registers.getNumVectorBits(), tr);
172  }

176  poly::NoWrapRewriter &nwr) -> IR::TreeResult {

178  const auto *BT = poly::getBackedgeTakenCount(*SE, L);
179  if (llvm::isa<llvm::SCEVCouldNotCompute>(BT)) return {};
180  Arena<> *salloc{shortAllocator()};
181  Arena<> *lalloc{longAllocator()};

188  IR::TreeResult tr = parseExitBlocks(L);

208  auto parseExitBlocks(llvm::Loop *L) -> IR::TreeResult {
209  IR::TreeResult tr;
210  for (auto &P : L->getExitBlock()->phis()) {
211  for (unsigned i = 0, N = P.getNumIncomingValues(); i < N; ++i) {
212  auto *J = llvm::dyn_cast<llvm::Instruction>(P.getIncomingValue(i));
213  if (!J || !L->contains(J)) continue;
214  tr = instructions.getValue(J, nullptr, tr).second;
215  }
216  }
217  return tr;
218  }

271  auto runOnLoop(llvm::Loop *L, llvm::ArrayRef<llvm::Loop *> subLoops,
272  math::Vector<int, 8> &omega, poly::NoWrapRewriter &nwr)
273  -> IR::TreeResult {
274  unsigned numSubLoops = subLoops.size();
275  // This is a special case, as it is when we build poly::Loop
276  if (!numSubLoops) return initLoopTree(L, omega, nwr);
277  unsigned depth = omega.size();
278  bool failed = false;
279  IR::TreeResult tr = parseExitBlocks(L);
280  omega.push_back(0); // we start with 0 at the end, walking backwards
281  poly::Loop *AL = nullptr;
282  llvm::BasicBlock *E = L->getLoopLatch();
283  for (size_t i = numSubLoops; i--; --omega.back()) {
284  llvm::Loop *subLoop = subLoops[i];
285  // we need to parse backwards, so we first evaluate
286  // TODO: support having multiple exit blocks?
287  IR::TreeResult trec =
288  runOnLoop(subLoop, subLoop->getSubLoops(), omega, nwr);
289 
290  if (trec.accept(depth)) {
291  if (!AL) AL = trec.getLoop();
292  if (AL) {
293  // recursion succeeded; see if we can connect the path
294  llvm::BasicBlock *subLoopExit = subLoop->getExitBlock();
295  // for fusion, we need to build a path from subLoopExit to E
296  // where E is the preheader of the preceding loopnest
297  IR::TreeResult trblock =
298  parseBlocks(subLoopExit, E, L, omega, AL, tr);
299  if (trblock.accept(depth)) {
300  tr = trblock;
301  tr *= trec;
302  } else {
303  failed = true;
304  if (tr.accept(depth)) optimize(tr);
305  // we start now with trec
306  tr = trec;
307  }
308  E = subLoop->getLoopPreheader(); // want to draw a path from trec
309  continue;
310  }
311  }
312  // we reject, because we failed to build a trec with a LoopNest
313  failed = true;
314  optimize(tr);
315  tr = {};
316  // we don't need to draw a path from anything, so only exit needed
317  if (i) E = subLoops[i - 1]->getExitBlock();
318  }
319  if (failed) {
320  if (tr.accept(depth)) optimize(tr);
321  omega.pop_back();
322  return {};
323  }
324  // now we try to go from E to H
325  IR::TreeResult trblock = parseBlocks(L->getHeader(), E, L, omega, AL, tr);
326  if (trblock.reject(depth)) {
327  optimize(tr); // optimize old tr
328  tr = {};
329  } else tr = trblock;
330  omega.pop_back();
331  return tr;
332  }

334  void optimize(IR::TreeResult tr) {
335  // now we build the LinearProgram
336  lp::LoopBlock::OptimizationResult lpor =
337  loopBlock.optimize(instructions, tr);
338  if (!lpor.nodes) return;

340  loopBBs.insert(addr->getBasicBlock());

342  eraseCandidates, loopBlock.getAllocator(), lpor);
343  loopBBs.clear();
344  }
