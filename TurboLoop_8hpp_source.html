98  // count the number of loops, and then reserve enough memory to avoid

100  size_t numLoops = 0;
101  for (auto &L : *LI)

105  // affineLoopNests.reserve(numLoops);

108  // NOTE: LoopInfo stores loops in reverse program order (opposite of
109  // loops)
110  std::vector<llvm::Loop *> revLI{llvm::reverse(*LI).begin(),
111  llvm::reverse(*LI).end()};
112  if (revLI.empty())
113  return;
114  llvm::BasicBlock *E = revLI.back()->getExitingBlock();
115  while (!E) {
116  revLI.pop_back();
117  if (revLI.empty())
118  return;
119  E = revLI.back()->getExitingBlock();
120  }
121  llvm::BasicBlock *H = revLI.front()->getLoopPreheader();
122  while (!H) {
123  revLI.erase(revLI.begin());
124  if (revLI.empty())
125  return;
126  H = revLI.front()->getLoopPreheader();
127  }
128 
129  LoopTree::pushBack(loopTrees, loopForests, forest, nullptr, *SE, revLI,
130  H, E, true);

213  uint64_t
214  fillAffineIndices(MutPtrVector<int64_t> v, Vector<int64_t> &offsets,
215  llvm::SmallVector<const llvm::SCEV *, 3> &symbolicOffsets,
216  const llvm::SCEV *S, int64_t mlt, size_t numPeeled) {
217  uint64_t blackList{0};
218  if (const llvm::SCEVAddRecExpr *x =
219  llvm::dyn_cast<const llvm::SCEVAddRecExpr>(S)) {
220  const llvm::Loop *L = x->getLoop();
221  size_t depth = L->getLoopDepth();
222  if (depth <= numPeeled) {
223  // we effectively have an offset
224  // we'll add an
225  addSymbolic(offsets, symbolicOffsets, S, 1);
226  for (size_t i = 1; i < x->getNumOperands(); ++i)
227  blackList |= blackListAllDependentLoops(x->getOperand(i));
228 
229  return blackList;
230  }
231  // outermost loop has loopInd 0
232  ptrdiff_t loopInd = ptrdiff_t(depth) - ptrdiff_t(numPeeled + 1);
233  if (x->isAffine()) {
234  if (loopInd >= 0) {
235  if (auto c = getConstantInt(x->getOperand(1))) {
236  // we want the innermost loop to have index 0
237  v(end - loopInd) += *c;
238  return fillAffineIndices(v, offsets, symbolicOffsets,
239  x->getOperand(0), mlt,
240  numPeeled);
241  } else
242  blackList |= (uint64_t(1) << uint64_t(loopInd));
243  }
244  // we separate out the addition
245  // the multiplication was either peeled or involved non-const
246  // multiple
247  blackList |=
248  fillAffineIndices(v, offsets, symbolicOffsets,
249  x->getOperand(0), mlt, numPeeled);
250  // and then add just the multiple here as a symbolic offset
251  const llvm::SCEV *addRec = SE->getAddRecExpr(
252  SE->getZero(x->getOperand(0)->getType()), x->getOperand(1),
253  x->getLoop(), x->getNoWrapFlags());
254  addSymbolic(offsets, symbolicOffsets, addRec, mlt);
255  return blackList;
256  } else if (loopInd >= 0)
257  blackList |= (uint64_t(1) << uint64_t(loopInd));
258  } else if (llvm::Optional<int64_t> c = getConstantInt(S)) {
259  offsets[0] += *c;
260  return 0;
261  } else if (const llvm::SCEVAddExpr *ex =
262  llvm::dyn_cast<const llvm::SCEVAddExpr>(S)) {
263  return fillAffineIndices(v, offsets, symbolicOffsets,
264  ex->getOperand(0), mlt, numPeeled) |
265  fillAffineIndices(v, offsets, symbolicOffsets,
266  ex->getOperand(1), mlt, numPeeled);
267  } else if (const llvm::SCEVMulExpr *ex =
268  llvm::dyn_cast<const llvm::SCEVMulExpr>(S)) {
269  if (auto op = getConstantInt(ex->getOperand(0))) {
270  return fillAffineIndices(v, offsets, symbolicOffsets,
271  ex->getOperand(1), mlt * (*op),
272  numPeeled);
273 
274  } else if (auto op = getConstantInt(ex->getOperand(1))) {
275  return fillAffineIndices(v, offsets, symbolicOffsets,
276  ex->getOperand(0), mlt * (*op),
277  numPeeled);
278  }
279  } else if (const llvm::SCEVCastExpr *ex =
280  llvm::dyn_cast<llvm::SCEVCastExpr>(S))
281  return fillAffineIndices(v, offsets, symbolicOffsets,
282  ex->getOperand(0), mlt, numPeeled);
283  addSymbolic(offsets, symbolicOffsets, S, mlt);
284  return blackList | blackListAllDependentLoops(S, numPeeled);
285  }

638  void conditionOnLoop(LoopTree &LT, unsigned LTID) {
639  unsigned PTID = LT.parentLoop;
640  if (PTID == std::numeric_limits<unsigned>::max())
641  return;
642  LoopTree &PT = loopTrees[PTID];
643  size_t numLoops = LT.getNumLoops();
644  for (auto ST : LT)
645  peelOuterLoops(loopTrees[ST], numLoops);
646 
647  LT.parentLoop =
648  std::numeric_limits<unsigned>::max(); // LT is now top of the tree
649  loopForests.push_back(LTID);
650  llvm::SmallVector<unsigned> &friendLoops = PT.subLoops;
651  // SHOW(LTID);
652  for (auto id : friendLoops)
653  llvm::errs() << ", " << id;
654  llvm::errs() << "\n";
655  if (friendLoops.front() != LTID) {
656  // we're cutting off the front
657  size_t numFriendLoops = friendLoops.size();
658  assert(numFriendLoops);
659  size_t loopIndex = 0;
660  for (size_t i = 1; i < numFriendLoops; ++i) {
661  if (friendLoops[i] == LTID) {
662  loopIndex = i;
663  break;
664  }
665  }
666  assert(loopIndex);
667  size_t j = loopIndex + 1;
668  if (j != numFriendLoops) {
669  // we have some remaining paths we split off
670  llvm::SmallVector<unsigned> tmp;
671  tmp.reserve(numFriendLoops - j);
672  // for paths, we're dropping LT
673  // thus, our paths are paths(_(0,j)), paths(_(j,end))
674  llvm::SmallVector<PredicatedChain> paths;
675  paths.reserve(numFriendLoops - loopIndex);
676  for (size_t i = j; i < numFriendLoops; ++i) {
677  peelOuterLoops(loopTrees[friendLoops[i]], numLoops - 1);
678  tmp.push_back(friendLoops[i]);
679  paths.push_back(std::move(PT.paths[i]));
680  }
681  paths.push_back(std::move(PT.paths[numFriendLoops]));
682  loopForests.push_back(loopTrees.size());
683  // TODO: split paths
684  loopTrees.emplace_back(std::move(tmp), std::move(paths));
685  }
686  friendLoops.truncate(loopIndex);
687  PT.paths.truncate(j);
688  } else {
689  friendLoops.erase(friendLoops.begin());
690  PT.paths.erase(PT.paths.begin());
691  }
692  conditionOnLoop(PT, PTID);
693  }
