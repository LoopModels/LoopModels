91  void initializeLoopForest() {
92  // NOTE: LoopInfo stores loops in reverse program order (opposite of
93  // loops)
94  auto RLI = llvm::reverse(*LI);
95  auto RLIB = RLI.begin();
96  auto RLIE = RLI.end();
97  if (RLIB == RLIE) return;
98  // pushLoopTree wants a direct path from the last loop's exit block to
99  // E; we drop loops until we find one for which this is trivial.
100  llvm::BasicBlock *E = (*--RLIE)->getExitBlock();
101  while (!E) {
102  if (RLIE == RLIB) return;
103  E = (*--RLIE)->getExitingBlock();
104  }
105  // pushLoopTree wants a direct path from H to the first loop's header;
106  // we drop loops until we find one for which this is trivial.
107  llvm::BasicBlock *H = (*RLIB)->getLoopPreheader();
108  while (!H) {
109  if (RLIE == RLIB) return;
110  H = (*++RLIB)->getLoopPreheader();
111  }
112  // should normally be stack allocated; we want to avoid different
113  // specializations for `llvm::reverse(*LoopInfo)` and
114  // `llvm::Loop->getSubLoops()`.
115  // But we could consider specializing on top level vs not.
116  llvm::SmallVector<llvm::Loop *> revLI{RLIB, RLIE + 1};
117  // Track position within the loop nest
118  llvm::SmallVector<unsigned> omega;
119  {
120  NoWrapRewriter nwr(*SE);
121  pushLoopTree(loopForests, nullptr, revLI, H, E, nwr);
122  }
123  for (auto &forest : loopForests) forest->addZeroLowerBounds(loopMap);
124  }
154  auto pushLoopTree(llvm::SmallVectorImpl<NotNull<LoopTree>> &pForest,
155  llvm::Loop *L, llvm::ArrayRef<llvm::Loop *> subLoops,
156  llvm::BasicBlock *H, llvm::BasicBlock *E,
157  NoWrapRewriter &nwr) -> size_t {
158 
159  if (size_t numSubLoops = subLoops.size()) {
160  // branches of this tree;
161  llvm::SmallVector<NotNull<LoopTree>> branches;
162  branches.reserve(numSubLoops);
163  llvm::SmallVector<Predicate::Map> branchBlocks;
164  branchBlocks.reserve(numSubLoops + 1);
165  bool anyFail = false;
166  size_t interiorDepth = 0;
167  for (size_t i = 0; i < numSubLoops; ++i) {
168  llvm::Loop *subLoop = subLoops[i];
169  if (size_t depth = pushLoopTree(
170  branches, subLoop, subLoop->getSubLoops(), subLoop->getHeader(),
171  subLoop->getExitingBlock(), nwr)) {
172  // pushLoopTree succeeded, and we have `depth` inner loops
173  // within `subLoop` (inclusive, i.e. `depth == 1` would
174  // indicate that `subLoop` doesn't have any subLoops itself,
175  // which we check with the following assertion:
176  assert((depth > 1) || (subLoop->getSubLoops().empty()));
177 
178  // Now we check if we can create a direct path from `H` to
179  // `subLoop->getLoopPreheader();`
180  llvm::BasicBlock *subLoopPreheader = subLoop->getLoopPreheader();
181  if (auto predMap = Predicate::Map::descend(allocator, instrCache, H,
182  subLoopPreheader, L)) {
183  interiorDepth = std::max(interiorDepth, depth);
184  branchBlocks.push_back(std::move(*predMap));
185  H = subLoop->getExitBlock();
186  } else {
187  // oops, no direct path, we split
188  anyFail = true;
189  if (branches.size() > 1) {
190  // we need to split off the last tree
191  LoopTree *lastTree = branches.pop_back_val();
192  // so we can push previous branches as one set
193  // for the final branchBlocks, we'll push H->H
194  split(branches, branchBlocks, H, L);
195  // reinsert last tree
196  branches.push_back(lastTree);
197  }
198  if (i + 1 < numSubLoops) H = subLoops[i + 1]->getLoopPreheader();
199  }
200  // for the next loop, we'll want a path to its preheader
201  // from this loop's exit block.
202  } else {
203  // `depth == 0` indicates failure, therefore we need to
204  // split loops
205  anyFail = true;
206  if (branches.size()) split(branches, branchBlocks, H, L);
207  if (i + 1 < numSubLoops) H = subLoops[i + 1]->getLoopPreheader();
208  }
209  }
210  if (!anyFail) {
211  if (auto predMapAbridged =
212  Predicate::Map::descend(allocator, instrCache, H, E, L)) {
213  branchBlocks.push_back(std::move(*predMapAbridged));
214  LoopTree::split(allocator, pForest, branchBlocks, branches);
215  return ++interiorDepth;
216  }
217  }
218  if (branches.size()) split(branches, branchBlocks, H, L);
219  return 0;
220  } else if (auto BT = getBackedgeTakenCount(*SE, L);
221  !llvm::isa<llvm::SCEVCouldNotCompute>(BT)) {
222  // we're at the bottom of the recursion
223  if (auto predMapAbridged =
224  Predicate::Map::descend(allocator, instrCache, H, E, L)) {
225  auto *newTree = new (allocator)
226  LoopTree{L, nwr.visit(BT), *SE, {std::move(*predMapAbridged)}};
227  pForest.push_back(newTree);
228  return 1;
229  }
230  }
231  // Finally, we need `H` to have a direct path to `E`.
232  return 0;
233  }

269  static auto blackListAllDependentLoops(const llvm::SCEV *S) -> uint64_t {
270  uint64_t flag{0};
271  if (const auto *x = llvm::dyn_cast<const llvm::SCEVNAryExpr>(S)) {
272  if (const auto *y = llvm::dyn_cast<const llvm::SCEVAddRecExpr>(x))
273  flag |= uint64_t(1) << y->getLoop()->getLoopDepth();
274  for (size_t i = 0; i < x->getNumOperands(); ++i)
275  flag |= blackListAllDependentLoops(x->getOperand(i));
276  } else if (const auto *c = llvm::dyn_cast<const llvm::SCEVCastExpr>(S)) {
277  for (size_t i = 0; i < c->getNumOperands(); ++i)
278  flag |= blackListAllDependentLoops(c->getOperand(i));
279  return flag;
280  } else if (const auto *d = llvm::dyn_cast<const llvm::SCEVUDivExpr>(S)) {
281  for (size_t i = 0; i < d->getNumOperands(); ++i)
282  flag |= blackListAllDependentLoops(d->getOperand(i));
283  return flag;
284  }
285  return flag;
286  }

293  fillAffineIndices(MutPtrVector<int64_t> v, Vector<int64_t> &offsets,
294  llvm::SmallVector<const llvm::SCEV *, 3> &symbolicOffsets,
295  const llvm::SCEV *S, int64_t mlt, size_t numPeeled)
296  -> uint64_t {
297  uint64_t blackList{0};
298  if (const auto *x = llvm::dyn_cast<const llvm::SCEVAddRecExpr>(S)) {
299  const llvm::Loop *L = x->getLoop();
300  size_t depth = L->getLoopDepth();
301  if (depth <= numPeeled) {
302  // we effectively have an offset
303  // we'll add an
304  addSymbolic(offsets, symbolicOffsets, S, 1);
305  for (size_t i = 1; i < x->getNumOperands(); ++i)
306  blackList |= blackListAllDependentLoops(x->getOperand(i));
307 
308  return blackList;
309  }
310  // outermost loop has loopInd 0
311  ptrdiff_t loopInd = ptrdiff_t(depth) - ptrdiff_t(numPeeled + 1);
312  if (x->isAffine()) {
313  if (loopInd >= 0) {
314  if (auto c = getConstantInt(x->getOperand(1))) {
315  // we want the innermost loop to have index 0
316  v[last - loopInd] += *c;
317  return fillAffineIndices(v, offsets, symbolicOffsets,
318  x->getOperand(0), mlt, numPeeled);
319  } else blackList |= (uint64_t(1) << uint64_t(loopInd));
320  }
321  // we separate out the addition
322  // the multiplication was either peeled or involved
323  // non-const multiple
324  blackList |= fillAffineIndices(v, offsets, symbolicOffsets,
325  x->getOperand(0), mlt, numPeeled);
326  // and then add just the multiple here as a symbolic offset
327  const llvm::SCEV *addRec = SE->getAddRecExpr(
328  SE->getZero(x->getOperand(0)->getType()), x->getOperand(1),
329  x->getLoop(), x->getNoWrapFlags());
330  addSymbolic(offsets, symbolicOffsets, addRec, mlt);
331  return blackList;
332  } else if (loopInd >= 0) blackList |= (uint64_t(1) << uint64_t(loopInd));
333  } else if (std::optional<int64_t> c = getConstantInt(S)) {
334  offsets[0] += *c;
335  return 0;
336  } else if (const auto *ar = llvm::dyn_cast<const llvm::SCEVAddExpr>(S)) {
337  return fillAffineIndices(v, offsets, symbolicOffsets, ar->getOperand(0),
338  mlt, numPeeled) |
339  fillAffineIndices(v, offsets, symbolicOffsets, ar->getOperand(1),
340  mlt, numPeeled);
341  } else if (const auto *m = llvm::dyn_cast<const llvm::SCEVMulExpr>(S)) {
342  if (auto op0 = getConstantInt(m->getOperand(0))) {
343  return fillAffineIndices(v, offsets, symbolicOffsets, m->getOperand(1),
344  mlt * (*op0), numPeeled);
345 
346  } else if (auto op1 = getConstantInt(m->getOperand(1))) {
347  return fillAffineIndices(v, offsets, symbolicOffsets, m->getOperand(0),
348  mlt * (*op1), numPeeled);
349  }
350  } else if (const auto *ca = llvm::dyn_cast<llvm::SCEVCastExpr>(S))
351  return fillAffineIndices(v, offsets, symbolicOffsets, ca->getOperand(0),
352  mlt, numPeeled);
353  addSymbolic(offsets, symbolicOffsets, S, mlt);
354  return blackList | blackListAllDependentLoops(S, numPeeled);
355  }
356  auto arrayRef(LoopTree &LT, llvm::Instruction *ptr, const llvm::SCEV *elSize,
357  llvm::Instruction *loadOrStore,
358  llvm::MutableArrayRef<unsigned> omegas) -> bool {
359  llvm::Loop *L = LT.loop;
360  // code modified from
361  // https://llvm.org/doxygen/Delinearization_8cpp_source.html#l00582
362  const llvm::SCEV *accessFn = SE->getSCEVAtScope(ptr, L);
363 
364  const llvm::SCEV *pb = SE->getPointerBase(accessFn);
365  const auto *basePointer = llvm::dyn_cast<llvm::SCEVUnknown>(pb);
366  // Do not delinearize if we cannot find the base pointer.
367  if (!basePointer) {
368  if (ORE && L) {
369  remark("ArrayRefDeliniarize", LT.loop,
370  "ArrayReference failed because !basePointer\n", ptr);
371  }
372  conditionOnLoop(L);
373  return true;
374  }
375  accessFn = SE->getMinusSCEV(accessFn, basePointer);
376  llvm::SmallVector<const llvm::SCEV *, 3> subscripts, sizes;
377  llvm::delinearize(*SE, accessFn, subscripts, sizes, elSize);
378  assert(subscripts.size() == sizes.size());
379  AffineLoopNest<true> &aln = loopMap[L]->affineLoop;
380  if (sizes.size() == 0) {
381  LT.memAccesses.push_back(MemoryAccess::construct(
382  allocator, basePointer, aln, loadOrStore, omegas));
383  ++omegas.back();
384  return false;
385  }
386  size_t numLoops{aln.getNumLoops()};
387  // numLoops x arrayDim
388  // IntMatrix R(numLoops, subscripts.size());
389  size_t numPeeled = L->getLoopDepth() - numLoops;
390  // numLoops x arrayDim
391  IntMatrix Rt(subscripts.size(), numLoops);
392  IntMatrix Bt;
393  llvm::SmallVector<const llvm::SCEV *, 3> symbolicOffsets;
394  uint64_t blackList{0};
395  {
396  Vector<int64_t> offsets;
397  for (size_t i = 0; i < subscripts.size(); ++i) {
398  offsets.clear();
399  offsets.pushBack(0);
400  blackList |= fillAffineIndices(Rt(i, _), offsets, symbolicOffsets,
401  subscripts[i], 1, numPeeled);
402  Bt.resize(subscripts.size(), offsets.size());
403  Bt(i, _) = offsets;
404  }
405  }
406  if (blackList) {
407  // blacklist: inner - outer
408  uint64_t leadingZeros = std::countl_zero(blackList);
409  uint64_t numExtraLoopsToPeel = 64 - leadingZeros;
410  // need to condition on loop
411  // remove the numExtraLoopsToPeel from Rt
412  // that is, we want to move Rt(_,_(end-numExtraLoopsToPeel,end))
413  // to would this code below actually be expected to boost
414  // performance? if
415  // (Bt.numCol()+numExtraLoopsToPeel>Bt.rowStride())
416  // Bt.resize(Bt.numRow(),Bt.numCol(),Bt.numCol()+numExtraLoopsToPeel);
417  // order of loops in Rt is innermost -> outermost
418  size_t remainingLoops = numLoops - numExtraLoopsToPeel;
419  llvm::Loop *P = L;
420  for (size_t i = 1; i < remainingLoops; ++i) P = P->getParentLoop();
421  // remove
422  conditionOnLoop(P->getParentLoop());
423  for (size_t i = remainingLoops; i < numLoops; ++i) {
424  P = P->getParentLoop();
425  if (allZero(Rt(_, i))) continue;
426  // push the SCEV
427  auto IntType = P->getInductionVariable(*SE)->getType();
428  const llvm::SCEV *S = SE->getAddRecExpr(
429  SE->getZero(IntType), SE->getOne(IntType), P, llvm::SCEV::NoWrapMask);
430  if (size_t j = findSymbolicIndex(symbolicOffsets, S)) {
431  Bt(_, j) += Rt(_, i);
432  } else {
433  Col N = Bt.numCol();
434  Bt.resize(N + 1);
435  Bt(_, N) = Rt(_, i);
436  symbolicOffsets.push_back(S);
437  }
438  }
439  Rt.truncate(Col{numLoops - numExtraLoopsToPeel});
440  }
441  LT.memAccesses.push_back(MemoryAccess::construct(
442  allocator, basePointer, aln, loadOrStore, Rt,
443  {std::move(sizes), std::move(symbolicOffsets)}, Bt, omegas));
444  ++omegas.back();
445  return false;
446  }

580  void conditionOnLoop(LoopTree *LT) {
581  if (!LT->parentLoop) return;
582  LoopTree &PT = *LT->parentLoop;
583  size_t numLoops = LT->getNumLoops();
584  for (auto ST : *LT) peelOuterLoops(*ST, numLoops);
585 
586  LT->parentLoop = nullptr; // LT is now top of the tree
587  loopForests.push_back(LT);
588  llvm::SmallVector<NotNull<LoopTree>> &friendLoops = PT.subLoops;
589  for (auto id : friendLoops) llvm::errs() << ", " << id;
590  llvm::errs() << "\n";
591  if (friendLoops.front() != LT) {
592  // we're cutting off the front
593  size_t numFriendLoops = friendLoops.size();
594  assert(numFriendLoops);
595  size_t loopIndex = 0;
596  for (size_t i = 1; i < numFriendLoops; ++i) {
597  if (friendLoops[i] == LT) {
598  loopIndex = i;
599  break;
600  }
601  }
602  assert(loopIndex);
603  size_t j = loopIndex + 1;
604  if (j != numFriendLoops) {
605  // we have some remaining paths we split off
606  llvm::SmallVector<NotNull<LoopTree>> tmp;
607  tmp.reserve(numFriendLoops - j);
608  // for paths, we're dropping LT
609  // thus, our paths are paths(_(0,j)), paths(_(j,end))
610  llvm::SmallVector<Predicate::Map> paths;
611  paths.reserve(numFriendLoops - loopIndex);
612  for (size_t i = j; i < numFriendLoops; ++i) {
613  peelOuterLoops(*friendLoops[i], numLoops - 1);
614  tmp.push_back(friendLoops[i]);
615  paths.push_back(std::move(PT.paths[i]));
616  }
617  paths.push_back(std::move(PT.paths[numFriendLoops]));
618  auto *newTree =
619  new (allocator) LoopTree(std::move(tmp), std::move(paths));
620  loopForests.push_back(newTree);
621  // TODO: split paths
622  }
623  friendLoops.truncate(loopIndex);
624  PT.paths.truncate(j);
625  } else {
626  friendLoops.erase(friendLoops.begin());
627  PT.paths.erase(PT.paths.begin());
628  }
629  conditionOnLoop(&PT);
630  }
