135  [[nodiscard]] constexpr auto compactUnion(Intersection other) const
136  -> containers::TinyVector<Intersection, 2> {
137  if (empty()) return {other};
138  if (other.empty()) return {*this};
139  uint64_t x = predicates, y = other.predicates;
140  // 010000 = 010100 & 010000
141  uint64_t intersect = x & y;
142  if (x == intersect || y == intersect) return {Intersection{intersect}};
143  // 011100 = 010100 | 011000
144  // 010000 = 010100 & 011000
145  // we can't handle (a & b) | (a & !b & c) because
146  // (a & b) | (a & !b & c) = a & (b | c)) = (a & b) | (a & c)
147  // bit representation:
148  // 010000 = 010100 & 011001
149  // we thus check all bits equal after masking off `b`.
150  // We could consider returning a pair of options, so we can return the
151  // simplified expression.
152  uint64_t bitUnion = x | y;
153  uint64_t mask = emptyMask(bitUnion);
154  if (std::popcount(mask) == 1) { // a single b & !b case
155  uint64_t remUnionMask =
156  ~(mask | (mask << 1)); // 0s `b`, meaning b can be either.
157  uint64_t w = remUnionMask & x;
158  uint64_t z = remUnionMask & y;
159  if (w == z) return {Intersection{w}};
160  // if we now have
161  // a | a & c
162  // 010000 | 010001
163  uint64_t wz = w & z;
164  if (wz == w) return {*this, Intersection{z}};
165  if (wz == z) return {Intersection{w}, other};
166  }
167  return {};
168  }
