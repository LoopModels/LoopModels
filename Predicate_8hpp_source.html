139  [[nodiscard]] constexpr auto compactUnion(Intersection other) const
140  -> std::variant<std::monostate, Intersection,
141  std::pair<Intersection, Intersection>> {
142  if (isEmpty())
143  return other;
144  else if (other.isEmpty())
145  return *this;
146  uint64_t x = predicates, y = other.predicates;
147  // 010000 = 010100 & 010000
148  uint64_t intersect = x & y;
149  if (x == intersect || y == intersect) {
150  return Intersection{intersect};
151  }
152  // 011100 = 010100 | 011000
153  // 010000 = 010100 & 011000
154  // we can't handle (a & b) | (a & !b & c) because
155  // (a & b) | (a & !b & c) = a & (b | c)) = (a & b) | (a & c)
156  // bit representation:
157  // 010000 = 010100 & 011001
158  // we thus check all bits equal after masking off `b`.
159  // We could consider returning a pair of options, so we can return the
160  // simplified expression.
161  uint64_t bitUnion = x | y;
162  uint64_t mask = emptyMask(bitUnion);
163  if (std::popcount(mask) == 1) { // a single b & !b case
164  uint64_t remUnionMask =
165  ~(mask | (mask << 1)); // 0s `b`, meaning b can be either.
166  uint64_t w = remUnionMask & x;
167  uint64_t z = remUnionMask & y;
168  if (w == z)
169  return Intersection{w};
170  // if we now have
171  // a | a & c
172  // 010000 | 010001
173  uint64_t wz = w & z;
174  if (wz == w) {
175  return std::make_pair(*this, Intersection{z});
176  } else if (wz == z) {
177  return std::make_pair(Intersection{w}, other);
178  }
179  }
180  return {};
181  }
