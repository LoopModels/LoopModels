99  [[nodiscard]] constexpr auto compactUnion(Intersection other) const
100  -> std::variant<std::monostate, Intersection,
101  std::pair<Intersection, Intersection>> {
102  if (isEmpty())
103  return other;
104  else if (other.isEmpty())
105  return *this;
106  uint64_t x = predicates, y = other.predicates;
107  // 010000 = 010100 & 010000
108  uint64_t intersect = x & y;
109  if (x == intersect || y == intersect) {
110  return Intersection{intersect};
111  }
112  // 011100 = 010100 | 011000
113  // 010000 = 010100 & 011000
114  // we can't handle (a & b) | (a & !b & c) because
115  // (a & b) | (a & !b & c) = a & (b | c)) = (a & b) | (a & c)
116  // bit representation:
117  // 010000 = 010100 & 011001
118  // we thus check all bits equal after masking off `b`.
119  // We could consider returning a pair of options, so we can return the
120  // simplified expression.
121  uint64_t bitUnion = x | y;
122  uint64_t mask = emptyMask(bitUnion);
123  if (std::popcount(mask) == 1) { // a single b & !b case
124  uint64_t remUnionMask =
125  ~(mask | (mask << 1)); // 0s `b`, meaning b can be either.
126  uint64_t w = remUnionMask & x;
127  uint64_t z = remUnionMask & y;
128  if (w == z)
129  return Intersection{w};
130  // if we now have
131  // a | a & c
132  // 010000 | 010001
133  uint64_t wz = w & z;
134  if (wz == w) {
135  return std::make_pair(*this, Intersection{z});
136  } else if (wz == z) {
137  return std::make_pair(Intersection{w}, other);
138  }
139  }
140  return {};
141  }

211  bool simplifyPreds = false;

213  auto u = pred.compactUnion(other);
214  if (auto compact = std::get_if<Intersection>(&u)) {
215  pred = *compact;
216  return;
217  } else if (auto simplify =
218  std::get_if<std::pair<Intersection, Intersection>>(
219  &u)) {
220  pred = simplify->first;
221  other = simplify->second;
222  simplifyPreds = true;
223  }
224  }
225  intersectUnion.push_back(other);
226  while (simplifyPreds) {
227  simplifyPreds = false;
228  for (size_t i = 0; i < intersectUnion.size(); ++i) {
229  for (size_t j = i + 1; j < intersectUnion.size();) {
230  auto u = intersectUnion[i].compactUnion(intersectUnion[j]);
231  if (auto compact = std::get_if<Intersection>(&u)) {
232  // delete `j`, update `i`
233  intersectUnion[i] = *compact;
234  intersectUnion.erase(intersectUnion.begin() + j);
235  simplifyPreds = true;
236  } else {
237  if (auto simplify = std::get_if<
238  std::pair<Intersection, Intersection>>(&u)) {
239  // assert forward progress
240  assert(
241  (std::popcount(simplify->first.predicates) +
242  std::popcount(simplify->second.predicates)) <=
243  (std::popcount(intersectUnion[i].predicates) +
244  std::popcount(intersectUnion[j].predicates)));
245  intersectUnion[i] = simplify->first;
246  intersectUnion[j] = simplify->second;
247  simplifyPreds = true;
