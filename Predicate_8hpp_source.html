167  void predUnion(llvm::BumpPtrAllocator &alloc,
168  const PredicateRelations &other) {
169  if (other.isEmpty())
170  return;
171  else if (isEmpty()) {
172  relations = other.relations;
173  return;
174  }
175  size_t N = std::max(size(), other.size());
176  if (relations.size() < N)
177  relations.resize(N);
178  // `&` because `0` is `Any`
179  // and `Any` is the preferred default initialization
180  for (size_t i = 0; i < N; i++) {
181  uint64_t a = relations[i];
182  uint64_t b = other.relations[i];
183  if (a == b)
184  continue;
185  // we want to iterate to all pred relations where they are not equal
186  uint64_t c = a ^ b;
187  while (true) {
188  uint64_t tz = std::countr_zero(c);
189  uint64_t shift = tz & (~(uint64_t(1)));
190  // shift to non-zero relation
191  a >>= shift;
192  b >>= shift;
193  c >>= shift;
194  }
195  }
