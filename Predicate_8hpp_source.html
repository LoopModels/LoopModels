114  [[nodiscard]] constexpr auto compactUnion(Intersection other) const
115  -> std::variant<std::monostate, Intersection,
116  std::pair<Intersection, Intersection>> {
117  if (isEmpty())
118  return other;
119  else if (other.isEmpty())
120  return *this;
121  uint64_t x = predicates, y = other.predicates;
122  // 010000 = 010100 & 010000
123  uint64_t intersect = x & y;
124  if (x == intersect || y == intersect) {
125  return Intersection{intersect};
126  }
127  // 011100 = 010100 | 011000
128  // 010000 = 010100 & 011000
129  // we can't handle (a & b) | (a & !b & c) because
130  // (a & b) | (a & !b & c) = a & (b | c)) = (a & b) | (a & c)
131  // bit representation:
132  // 010000 = 010100 & 011001
133  // we thus check all bits equal after masking off `b`.
134  // We could consider returning a pair of options, so we can return the
135  // simplified expression.
136  uint64_t bitUnion = x | y;
137  uint64_t mask = emptyMask(bitUnion);
138  if (std::popcount(mask) == 1) { // a single b & !b case
139  uint64_t remUnionMask =
140  ~(mask | (mask << 1)); // 0s `b`, meaning b can be either.
141  uint64_t w = remUnionMask & x;
142  uint64_t z = remUnionMask & y;
143  if (w == z)
144  return Intersection{w};
145  // if we now have
146  // a | a & c
147  // 010000 | 010001
148  uint64_t wz = w & z;
149  if (wz == w) {
150  return std::make_pair(*this, Intersection{z});
151  } else if (wz == z) {
152  return std::make_pair(Intersection{w}, other);
153  }
154  }
155  return {};
