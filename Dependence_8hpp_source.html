15  // Plan here is...
16  // depPoly gives the constraints
17  // dependenceFwd gives forward constraints
18  // dependenceBwd gives forward constraints
19  // isBackward() indicates whether backward is non-empty
20  // bounding constraints, used for ILP solve, are reverse,
21  // i.e. fwd uses dependenceBwd and bwd uses dependenceFwd.
22  //
23  // Consider the following simple example dependencies:
24  // for (k = 0; k < K; ++k)
25  // for (i = 0; i < I; ++i)
26  // for (j = 0; j < J; ++j)
27  // for (l = 0; l < L; ++l)
28  // A(i, j) = f(A(i+1, j), A(i, j-1), A(j, j), A(j, i), A(i, j -
29  // k))
30  // label: 0 1 2 3 4 5
31  // We have...
33  // i_0 = i_1 + 1
34  // j_0 = j_1
35  // null spaces: [k_0, l_0], [k_1, l_1]
36  // forward: k_0 <= k_1 - 1
37  // l_0 <= l_1 - 1
38  // backward: k_0 >= k_1
39  // l_0 >= l_1
40  //
41  //
43  // i_0 = i_1
44  // j_0 = j_1 - 1
45  // null spaces: [k_0, l_0], [k_1, l_1]
46  // forward: k_0 <= k_1 - 1
47  // l_0 <= l_1 - 1
48  // backward: k_0 >= k_1
49  // l_0 >= l_1
50  //
52  // i_0 = j_1
53  // j_0 = j_1
54  // null spaces: [k_0, l_0], [i_1, k_1, l_1]
55  // forward: k_0 <= k_1 - 1
56  // l_0 <= l_1 - 1
57  // backward: k_0 >= k_1
58  // l_0 >= l_1
59  //
60  // i_0 = j_1, we essentially lose the `i` dimension.
61  // Thus, to get fwd/bwd, we take the intersection of nullspaces to get
62  // the time dimension?
63  // TODO: try and come up with counter examples where this will fail.
64  //
66  // i_0 = j_1
67  // j_0 = i_1
68  // null spaces: [k_0, l_0], [k_1, l_1]
69  // if j_0 > i_0) [store first]
70  // forward: k_0 >= k_1
71  // l_0 >= l_1
72  // backward: k_0 <= k_1 - 1
73  // l_0 <= l_1 - 1

88  // j_0 = j_1 - k_1

132  NotNull<MemoryAccess> x, NotNull<MemoryAccess> y,

137  pair[1]->copy(alloc)};

151  } else {
152  std::swap(in, out);
153  std::swap(pair[0], pair[1]);
154  }
155  pair[0]->truncateVars(1 + numLambda + numScheduleCoefs);
156  auto dep0 = Dependence{dxy->copy(alloc), pair, {in, out}, isFwd};
157  invariant(out->getNumLoops() + in->getNumLoops(),
158  dep0.getNumPhiCoefficients());
159  // pair is invalid
160  const size_t timeDim = dxy->getTimeDim(),
161  numVar = 1 + dxy->getNumVar() - timeDim;
162  invariant(timeDim > 0);
163  // 1 + because we're indexing into A and E, ignoring the constants
164  // remove the time dims from the deps
165  // dep0.depPoly->truncateVars(numVar);
166 
167  // dep0.depPoly->setTimeDim(0);
168  invariant(out->getNumLoops() + in->getNumLoops(),
169  dep0.getNumPhiCoefficients());
170  // now we need to check the time direction for all times
171  // anything approaching 16 time dimensions would be absolutely
172  // insane
173  Vector<bool, 16> timeDirection(timeDim);
174  size_t t = 0;
175  auto fE{farkasBackups[0]->getConstraints()(_, _(1, end))};
176  auto sE{farkasBackups[1]->getConstraints()(_, _(1, end))};
177  do {
178  // set `t`th timeDim to +1/-1
179  // basically, what we do here is set it to `step` and pretend it was
180  // a constant. so a value of c = a'x + t*step -> c - t*step = a'x so
181  // we update the constant `c` via `c -= t*step`.
182  // we have the problem that.

184  size_t v = numVar + t, i = 0;
185  while (true) {
186  for (size_t c = 0; c < numInequalityConstraintsOld; ++c) {
187  int64_t Acv = dxy->getA(c, v);
188  if (!Acv) continue;
189  Acv *= step;
190  fE(0, c + 1) -= Acv; // *1
191  sE(0, c + 1) -= Acv; // *1
192  }
193  for (size_t c = 0; c < numEqualityConstraintsOld; ++c) {
194  // each of these actually represents 2 inds
195  int64_t Ecv = dxy->getE(c, v);
196  if (!Ecv) continue;
197  Ecv *= step;
198  fE(0, c + ineqEnd) -= Ecv;
199  fE(0, c + posEqEnd) += Ecv;
200  sE(0, c + ineqEnd) -= Ecv;
201  sE(0, c + posEqEnd) += Ecv;
202  }
203  if (i++ != 0) break; // break after undoing
204  timeDirection[t] =
205  checkDirection(alloc, farkasBackups, *out, *in, numLambda,
206  dxy->getA().numCol() - dxy->getTimeDim());
207  step *= -1; // flip to undo, then break
208  }
209  } while (++t < timeDim);
210  t = 0;
211  do {
212  // checkDirection(farkasBackups, x, y, numLambda) == false
213  // correct time direction would make it return true
214  // thus sign = timeDirection[t] ? 1 : -1
215  int64_t step = (2 * timeDirection[t] - 1) * dxy->getNullStep(t);
216  size_t v = numVar + t;
217  for (size_t c = 0; c < numInequalityConstraintsOld; ++c) {
218  int64_t Acv = dxy->getA(c, v);
219  if (!Acv) continue;
220  Acv *= step;
221  dxy->getA(c, 0) -= Acv;
222  fE(0, c + 1) -= Acv; // *1
223  sE(0, c + 1) -= Acv; // *-1
224  }
225  for (size_t c = 0; c < numEqualityConstraintsOld; ++c) {
226  // each of these actually represents 2 inds
227  int64_t Ecv = dxy->getE(c, v);
228  if (!Ecv) continue;
229  Ecv *= step;
230  dxy->getE(c, 0) -= Ecv;
231  fE(0, c + ineqEnd) -= Ecv;
232  fE(0, c + posEqEnd) += Ecv;
233  sE(0, c + ineqEnd) -= Ecv;
234  sE(0, c + posEqEnd) += Ecv;
235  }
236  } while (++t < timeDim);
237  // dxy->truncateVars(numVar);
238  // dxy->setTimeDim(0);
239  farkasBackups[0]->truncateVars(1 + numLambda + numScheduleCoefs);
240  auto dep1 = Dependence{dxy, farkasBackups, {out, in}, !isFwd};
241  invariant(out->getNumLoops() + in->getNumLoops(),
242  dep0.getNumPhiCoefficients());
243  return {dep0, dep1};
244  }

472  [[nodiscard]] auto isSatisfied(BumpAlloc<> &alloc,
473  NotNull<const AffineSchedule> schIn,
474  NotNull<const AffineSchedule> schOut) const
475  -> bool {
476  size_t numLoopsIn = in->getNumLoops();
477  size_t numLoopsOut = out->getNumLoops();
478  size_t numLoopsCommon = std::min(numLoopsIn, numLoopsOut);
479  size_t numLoopsTotal = numLoopsIn + numLoopsOut;
480  size_t numVar = numLoopsIn + numLoopsOut + 2;
481  invariant(size_t(dependenceSatisfaction->getNumVars()), numVar);
482  auto p = alloc.scope();
483  auto schv = vector(alloc, numVar, int64_t(0));
484  const SquarePtrMatrix<int64_t> inPhi = schIn->getPhi();
485  const SquarePtrMatrix<int64_t> outPhi = schOut->getPhi();
486  auto inFusOmega = schIn->getFusionOmega();
487  auto outFusOmega = schOut->getFusionOmega();
488  auto inOffOmega = schIn->getOffsetOmega();
489  auto outOffOmega = schOut->getOffsetOmega();
490  const size_t numLambda = getNumLambda();
491  // when i == numLoopsCommon, we've passed the last loop
492  for (size_t i = 0; i <= numLoopsCommon; ++i) {
493  if (int64_t o2idiff = outFusOmega[i] - inFusOmega[i])
494  return (o2idiff > 0);
495  // we should not be able to reach `numLoopsCommon`
496  // because at the very latest, this last schedule value
497  // should be different, because either:
498  // if (numLoopsX == numLoopsY){
499  // we're at the inner most loop, where one of the instructions
500  // must have appeared before the other.
501  // } else {
502  // the loop nests differ in depth, in which case the deeper
503  // loop must appear either above or below the instructions
504  // present at that level
505  // }
506  assert(i != numLoopsCommon);
507  // forward means offset is 2nd - 1st
508  schv[0] = outOffOmega[i];
509  schv[1] = inOffOmega[i];
510  schv[_(2, 2 + numLoopsIn)] << inPhi(last - i, _);
511  schv[_(2 + numLoopsIn, 2 + numLoopsTotal)] << outPhi(last - i, _);
512  // dependenceSatisfaction is phi_t - phi_s >= 0
513  // dependenceBounding is w + u'N - (phi_t - phi_s) >= 0
514  // we implicitly 0-out `w` and `u` here,
515  if (dependenceSatisfaction->unSatisfiable(alloc, schv, numLambda) ||
516  dependenceBounding->unSatisfiable(alloc, schv, numLambda)) {
517  // if zerod-out bounding not >= 0, then that means
518  // phi_t - phi_s > 0, so the dependence is satisfied
519  return false;
520  }
521  }
522  return true;
523  }
524  [[nodiscard]] auto isSatisfied(BumpAlloc<> &alloc,
525  PtrVector<unsigned> inFusOmega,
526  PtrVector<unsigned> outFusOmega) const
527  -> bool {
528  size_t numLoopsIn = in->getNumLoops();
529  size_t numLoopsOut = out->getNumLoops();
530  size_t numLoopsCommon = std::min(numLoopsIn, numLoopsOut);
531  size_t numVar = numLoopsIn + numLoopsOut + 2;
532  invariant(dependenceSatisfaction->getNumVars() == numVar);
533  auto p = alloc.scope();

536  const size_t numLambda = getNumLambda();
537  // when i == numLoopsCommon, we've passed the last loop
538  for (size_t i = 0; i <= numLoopsCommon; ++i) {
539  if (int64_t o2idiff = outFusOmega[i] - inFusOmega[i])
540  return (o2idiff > 0);

542  // because at the very latest, this last schedule value
543  // should be different, because either:
544  // if (numLoopsX == numLoopsY){
545  // we're at the inner most loop, where one of the instructions
546  // must have appeared before the other.
547  // } else {
548  // the loop nests differ in depth, in which case the deeper
549  // loop must appear either above or below the instructions
550  // present at that level
551  // }
552  assert(i != numLoopsCommon);
553  schv[2 + i] = 1;
554  schv[2 + numLoopsIn + i] = 1;
555  // forward means offset is 2nd - 1st
556  // dependenceSatisfaction is phi_t - phi_s >= 0
557  // dependenceBounding is w + u'N - (phi_t - phi_s) >= 0
558  // we implicitly 0-out `w` and `u` here,
559  if (dependenceSatisfaction->unSatisfiable(alloc, schv, numLambda) ||
560  dependenceBounding->unSatisfiable(alloc, schv, numLambda)) {
561  // if zerod-out bounding not >= 0, then that means
562  // phi_t - phi_s > 0, so the dependence is satisfied
563  return false;
564  }
565  schv[2 + i] = 0;
566  schv[2 + numLoopsIn + i] = 0;
567  }
568  return true;
