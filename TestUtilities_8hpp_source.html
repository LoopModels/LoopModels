95  auto addLoop(PtrMatrix<int64_t> A, ptrdiff_t numLoops) -> poly::Loop * {
96  ptrdiff_t numSym = ptrdiff_t(A.numCol()) - numLoops - 1;
97  math::Vector<IR::Value *> symbols;
98  symbols.reserve(numSym);
99  if (numSym) {
100  // we assume there's some chance of recycling old symbols, so we only
101  // create new ones if we have to.
102  poly::Loop *symbolSource = nullptr;
103  ptrdiff_t numSymbolSource = 0;
104  for (poly::Loop *aln : alns) {
105  if (numSymbolSource < aln->getSyms().size()) {
106  numSymbolSource = aln->getSyms().size();
107  symbolSource = aln;
108  }
109  }
110  for (ptrdiff_t i = 0; i < std::min(numSym, numSymbolSource); ++i)
111  symbols.push_back(symbolSource->getSyms()[i]);
112  for (ptrdiff_t i = numSymbolSource; i < numSym; ++i)
113  symbols.push_back(createInt64());
114  }
115  return addLoop(A, numLoops, symbols);
116  }
