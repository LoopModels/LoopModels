21 // Classifies the type of loop dependency
22 // Possibilities:
23 // Not nested: this address is not nested inside the loop
24 // Dynamic: unknown stride
25 // Static: 0...
26 // Special values:
27 // 0: nested inside, but not dependent
28 // 1: contiguous
29 // -1: contiguous, but reversed
30 // From here, small values like `2` might mean we can efficiently shuffle.
31 // Ideally, we'd be paired with another address offset by 1.
32 // But even just `A[2*i]`, it may take less micro-ops to shuffle than to
33 // use a gather.
34 struct LoopDependency {
35  int type_;
36  static constexpr int notNested = std::numeric_limits<int>::min();
37  static constexpr int dynamic = notNested + 1;
38  // NOLINTNEXTLINE(readability-identifier-naming)
39  static constexpr auto NotNested() -> LoopDependency { return {notNested}; }
40  // NOLINTNEXTLINE(readability-identifier-naming)
41  static constexpr auto Dynamic() -> LoopDependency { return {dynamic}; }
42  // NOLINTNEXTLINE(readability-identifier-naming)
43  static constexpr auto Static(int stride) -> LoopDependency {
44  // incredibly unlikely, but we don't want to mistake "notNested"
45  // for a static stride with that value. It isn't something we can optimize
46  // anyway, so we'll just return dynamic.
47  return {stride == notNested ? dynamic : stride};
48  }

51  case notNested:
52  case dynamic: return std::nullopt;
53  default: return type_;
54  }
55  }

57  return type_ == dynamic;

72 // Lets start with the actual cost:
73 // Ca and Cb are cost of loading from A and B, respectively.
74 // Ui and Uj are the unrolling factors for i and j, respectively.
75 // Vi and Vj are the vectorization factors; only one is allowed to be !=0
76 // C = Ca * J * I/(Uj*Vj*Vi) + Cb * J/Vj
77 //
78 // Vi: C = Ca * J * I/(Uj*Vi) + Cb * J
79 // Vj: C = Ca * J * I/(Uj*Vj) + Cb * J/Vj
80 //
81 // Cost of `A` is Ca * Uj^0 * Ui^1 * I*J / (Uj * Ui)
82 // = Ca * I * J / Uj
83 // Cost of `B` is Cb * Uj^1 * Ui^0 * I^0*J / (Uj * Ui^0)
84 // = Cb * J

86 //

88 // j i

510  if (f) {
511  AddrSummary s = (*this)[f->second].setAddr(a);
512  push(s);
513  // FIXME: not okay!!! We may have different loops

515  if (f && f->second >= offset) return;
516  addOptOption(alloc, optops, s);
517  return;

519  ++numAddr;
520  AddrReference ref = addrRef(id);
521  *ref.addr = a;
522  // calc minStaticStride
523  DensePtrMatrix<int64_t> indMat = a->indexMatrix(); // dim x loop
524  // we want mapping from `indMat` index to loop index
525  uint64_t minStaticStride = std::numeric_limits<uint64_t>::max();
526  auto minStaticStrideLoops =
527  AddrSummary::minStaticStrideLoops(ref.bits, cld64(numLoops));
528  auto remainingLoops = AddrSummary::remainingLoops(ref.bits, cld64(numLoops));
529  IR::Loop *L = a->getLoop();
530  for (ptrdiff_t l = ptrdiff_t{indMat.numCol()}; l--; L = L->getLoop()) {
531  uint32_t lid = L->getID();
532  for (ptrdiff_t j = 0; j < ptrdiff_t{indMat.numRow()} - 1; ++j)
533  if (indMat(j, l)) remainingLoops.insert(lid);
534  if (int64_t x = indMat(last, l)) {
535  uint64_t absx = math::constexpr_abs(x);
536  if (minStaticStride > absx) {
537  minStaticStride = absx;
538  remainingLoops |= minStaticStrideLoops;
539  minStaticStrideLoops.clear();
540  minStaticStrideLoops.insert(lid);
541  } else if (minStaticStride == absx) {
542  minStaticStrideLoops.insert(lid);
543  } else {
544  remainingLoops.insert(lid);
545  }
546  }
547  }
548  *ref.stride = minStaticStride;
549  addOptOption(alloc, optops, {a, minStaticStride, ref.bits, numLoops});
