40  constexpr auto getLeaf() -> Child {
41  if (!first) return {nullptr, nullptr, 0};
42  for (size_t i = 0; i < std::size(children); ++i)
43  if (TrieMapNode *child = children[i])
44  if (Child leaf = child->getLeaf(); leaf.child)
45  return leaf.parent ? leaf : Child{leaf.child, this, i};
46  return {this, nullptr, 0};

48  constexpr auto getSubLeaf() -> Child {
49  Child c = getLeaf();
50  return c.child != this ? c : Child{nullptr, nullptr, 0};
51  }
52  auto findChild(const K &k) -> Child {
53  if (k == first) return {this, nullptr, 0};
54  TrieMapNode *p = this, *c = nullptr;
55  for (uint64_t h = fastHash(k);; h >>= 2) {
56  c = p->children[h & 3];
57  if (!c || (c->first == k)) return {c, p, h & 3};
58  p = c;
59  }
60  }
61  // Returns the removed node
62  auto eraseImpl(const K &k) -> TrieMapNode * {
63  Child child = findChild(k);
64  if (!child.child) return nullptr;
65  // we're erasing `child`
66  Child l = child.child->getSubLeaf();
67  if (l.child) {
68  l.parent->children[l.index] = nullptr; // leaf is moved up
69  std::swap(l.child->children, child.child->children);
70  }
71  child.parent->children[child.index] = l.child; // leaf replaces deleted
72  child.child->second = {};
73  return child.child;
74  }

78 // Future allocations will allocate from this list if possible.
79 // Thus, whenever using a pattern that involves interleaving erase and
80 // insertions, it is worth setting `EfficientErase = true`. It is common enough
81 // not to do this, that the option for `false` also exists. Don't pay for what

158  if (!child) return; // was not found

160  if (InlineTrie *descendent = child->children[index]) {

164  child->values[index] = std::move(lc->values[li]);

181  // A leaf is a key without any child keys.
182  // A leaf may have children without keys.
183  auto findLeaf() -> Pair<InlineTrie *, ptrdiff_t> {
184  InlineTrie *leaf = this;
185  bool descend[4]{false, false, false, false};
186  for (ptrdiff_t i = 0; i < std::ssize(children); ++i) {
187  if (!leaf->keys[i]) continue; // need key to be leaf
188  if (!leaf->children[i]) return {leaf, i}; // no children, no child keys
189  descend[i] = true;
190  }
191  for (ptrdiff_t i = 0; i < std::ssize(children); ++i) {
192  if (!descend[i]) continue;
193  auto ret = leaf->children[i]->findLeaf();
194  return ret.first ? ret : Pair<InlineTrie *, ptrdiff_t>{this, i};
195  };

206  for (uint64_t h = fastHash(k);;) {
207  uint64_t ind = h & 3;
208  bool noKey = !node->keys[ind];

211  if (noKey || (*node->keys[ind] == k)) return Child{node, ind, {}};
212  } else {

214  if (*node->keys[ind] == k)
215  return Pair<InlineTrie *, uint64_t>{node, ind};
216  }
217  h >>= 2;
218  if (!node->children[ind]) {
219  if constexpr (Insert) return Child{node, h & 3, ind};
