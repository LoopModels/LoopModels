57  constexpr auto registerConsumption(Order order, uint32_t depMask,
58  const AbstractMatrix auto &invunrolls)
59  const -> utils::eltype_t<decltype(invunrolls)> {
60  utils::eltype_t<decltype(invunrolls)> r{1.0};
61  // depMask bits go from [0,...,inner,...,outer]
62  uint32_t i = order.size();
63  depMask <<= (32 - i);
64  for (uint32_t indepCount = i - std::popcount(depMask); indepCount;) {
65  if (depMask & 0x80000000) r *= invunrolls[0, order[--i]];
66  else --indepCount;
67  depMask <<= 1;
68  }
69  return r;
70  }

73  constexpr auto registerPressure(const AbstractMatrix auto &invunrolls) const
74  -> utils::eltype_t<decltype(invunrolls)> {
75  // FIXME: iterate over cmpt permutations, correctly calculate use for each,
76  // and accumulate. Could try accumulating exponentials, adjusted by reg
77  // count for numerical stability, or simply storing in a vector and using
78  // multiple passes so we can calclate `exp(minimum - values)`.
79  // Latter is probably safer w/ respect to accuracy.
80  utils::eltype_t<decltype(invunrolls)> acc{0};
81  // lambda controls logsumexp (smooth maximum) sharpness
82  static constexpr double lambda = 8.0;
83  // we want the minimum register use across orders
84  // so we use a logsumexp (i.e. soft maximum) across remaining registers
85  for (Order order : perms) {
86  utils::eltype_t<decltype(invunrolls)> ao{0};
87  for (auto [c, m] : masks)
88  ao += c * registerConsumption(order, m, invunrolls);
89  acc += math::exp(lambda * register_count - lambda * ao);
90  }
91  // Using `log1p` is equivalent to adding a `0` term to our soft maximum
92  // i.e. `log(0) = 1`; `log1p` should be more numerically accuate
93  // reg_remaining = log(acc)
94  // excess_reg_consumed = max(0.0, -reg_remaining)
95  // excess_reg_consumed \approx log(exp(0.0) + exp(-reg_remaining))
96  // = log1p(exp(-log(acc)) = log1p(1/acc)
97  return log1p(1 / acc) / lambda;
98  }

103  //
104  // We scan dependencies, looking for reduction latencies
105  // can we use `TTI.getRegUsageForType()`?
106  // TargetTransformInfoImplBase defaults to `1`, and some backends like x86
107  // do not override it, so it is not something we can rely on.
108  //
109  // d.input()->reductionLatency() > 0 indicates a dependence is live across
110  // the loop otherwise, we only consider instructions within the loop?
111  // ...or...perhaps we only need to consider the loop leaf's instructions; we
112  // can see what is written and referenced.
113  // For now, will try and follow the latter approach.
114  //
115  // To support all loops, we would need to know lifetimes
116  // Plan, register use as we go for each block of the nest
117  // What data structures do we need?
118  // Where can we place spills and reloads?
119  // E.g., we don't want to count something used in an outer loop
120  // for (int m = 0; m < M; ++m){
121  // for (int n = 0; n < N; ++n){
122  // double Cmn = 0.0;
123  // double y = x[n]; // shouldn't be live across loop
124  // for (int k = 0; k < K; ++k){
125  // Cmn = Cmn + A[m,k] * B[k,n];
126  // }
127  // C[m,n] = Cmn / y;
128  // }
129  // }
130  // For now, we'll take the simple approach of only considering the loop leaf
131  // We have a data structure representing live variables, and have a
132  // remaining use count for each. Consider this example: for (int m = 0; m <
133  // M; ++m){
134  // for (int n = 0; n < N; ++n){
135  // // Consider (a)
136  // wa = w[m]*A[m,n];
137  // C[m,n] = wa + x[n];
138  // D[m,n] = wa + z[n];
139  // // vs (b)
140  // C[m,n] = w[m]*A[m,n] + x[n]; // 0
141  // D[m,n] = y[m]*B[m,n] + z[n]; // 1
142  // }
143  // }
144  // for rev-top sort; add all children, then pushfirst self
145  // for top-sort: add all parents, then push self
146  // We scan forward
147  // TODO: finish implementing Node::loopMask
148  // Here, we assemble all dependencies
149  // The basic ideas are that
150  // 1. `Load`s consume registers
151  // 2. `Instruction`s may be free, if any temporary operands have no more
152  // uses We must mark uses as temporary or not
153  // - Variables aren't temporary if they're loaded prior to or used after
154  // loop Uses can be "freed". To support this, we track of peak uses. Looking
155  // at the `B` example above, adding `1` in addition to `0` doesn't actually
156  // require us to use any extra registers. We just perform all `0`
157  // instructions (with full unrolling), and then all `1`. However, we
158  // currently don't model or support that kind of splitting. map temporary
159  // `Value*`s to number of remaining uses
160  dict::InlineTrie<IR::Value *, uint32_t> remainingUses{};
161  // `InlineTrie`'s currently can't iterate over k-v pairs
162  // and I was having trouble coming up with an actually efficient way to do
163  // it
164  dict::map<uint16_t, RegisterUse> maskCoefs{};
165  // Algorithm: we iterate forward
166  // as we encounter an instruction, we check its operands:
167  // if the operand is not defined in the loop, it is permanent w/ respect to
168  // it
169  // in this case, we must add it.
170  // if the operand is, we decrement use count
171  // Consider:
172  // double foo(std::span<double> a, double y){
173  // double x = 0.0;
174  // for (double z : a) x = x + (z - y)*(z - y);
175  // return x;
176  // }
177  // where the representation is of `x` as a memory address that we load from
178  // in front of the loop, and store to after. The constant `y` and the `x`
179  // load are marked as permanent. The `+`, stored after the loop, is also
180  // considered permanent and thus allowed to replace a permanent operand.
181  // FIXME: actually check that an operation really should be able to replace
182  // a permanent operand, i.e., that they're part of a time-cycle? NOTE: this
183  // works even if the "replaced" is not an argument, e.g. if we have `x =
184  // +(a, +(b, x))`. That's because we're just incrementing/decrementing a
185  // `RegisterUse` arg.
186  // We are relying on the topological sorting algorithm of the nodes to place
187  // instructions reasonably close to their uses, so we don't have excessive
188  // width. TODO: what kind of top sort algorithm would actually minimize
189  // width? Currently, we're just using depth-first for ease of
190  // implementation.
191  //
192  //
193  // TODO: Add PhiNode canonicalization/representation as arrays memory
194  // locations
195  // TODO: If actually representing them as memory, must guarantee we delete
196  // TODO: We want array-like representation, because const-initialization
197  // representation doesn't necessarilly capture behavior as we permute
198  // w/ resepct to interior loops
199  auto p = alloc->checkpoint();
200  for (IR::Node *n : L->nodes()) {
201  IR::Value *v;
202  uint16_t m;
203  if (auto *a = llvm::dyn_cast<IR::Addr>(n)) {
204  if (!a->isLoad()) continue;
205  m = a->loopMask();
206  v = a;
207  } else if (auto *c = llvm::dyn_cast<IR::Compute>(n)) {
208  m = c->loopMask();
209  v = c;
210  // now we check operands
211  for (IR::Value *o : c->getOperands()) {
212  uint32_t &uses = remainingUses[alloc, o];
213  uint16_t om = o->loopMask();
214  if (o->getLoop() != L) { // `v` is permanent
215  om = markPermanent(om);
216  if (!uses) {
217  ++maskCoefs[om];
218  uses = o->getUsers().size();
219  }
220  }
221  invariant(uses > 0);
222  if ((--uses) == 0) --maskCoefs[om];
223  }
224  } else continue;
225  IR::Users &users = v->getUsers();
226  remainingUses[alloc, v] = users.size();
227  if (isPermanent(users, L)) m = markPermanent(m);
228  ++maskCoefs[m];
229  }
230  alloc->rollback(p);
231  // At this point, `maskCoefs` contains uses...
232  // Next, we need to define the set of feasible unroll orders
233  // If `isPermanent`, then there isn't anything to do: register use is
234  // product of all unroll factors Otherwise, it is lifetime-based. We
235  // assemble a collection of all non-dominated permutations
236  // then final value would be the minimum of each
237  // In theory, we could have `N!` perms, where `N` is the loop depth, but
238  // that should basically never happen. Do we need anything like merging
239  // equivalence classes?
240  //
241  // The basic idea for how register use relates to nesting is
242  // unroll i
243  // unroll j
244  // unroll k
245  // that lifetimes must span the missing indices.
246  // E.g., lets say we have
247  // a[i,j,k] // m_a = 111
248  // b[j] // m_b = 010
249  // c[i,j] // m_c = 011
250  // d[j,k] // m_d = 110
251  // Here, given order `[0,1,2]` (i,j,k), because `d` is missing `i`
252  // its lifetime spans the entire unrolled body.
253  // The use is the product of all dependent indices interior to the first
254  // absent. So, `r_a=1, r_b = u_j, r_c=1, r_d=u_j*u_k`
255  //
256  // How do we build up all combinations?
257  // Whenever one dependence dominates another
258  // we always place the dominator outside.
259  // Looking at only `a`, `b`, and `c` yields the single nesting [1,0,2], i.e.
260  // unroll j; unroll i; unroll k
261  // with `r_a=1, r_b=1, r_c=1` -- perfect!
262  // `r_d = u_k`, which isn't so bad.
263  // With the addition of `d`, we would also have to consider [1,2,0]`, i.e.
264  // `unroll j; unroll k; unroll i`
265  // with `r_a=1, r_b=1, r_c=u_i, r_d=1`
266  // The chosen nesting order of `[1,0,2]` vs `[1,2,0]` would come down to
267  // whether `u_i` or `u_k` are larger.
268  //
269  // How do we build up the permutations reasonably efficiently?
270  // Assemble a graph, get SCCs
271  uint16_t depth = L->getNaturalDepth();
272  utils::IndexRelationGraph indDepGraph{int16_t(depth)};
273  masks.resize(ptrdiff_t(maskCoefs.size()));
274  ptrdiff_t i = 0;
275  for (auto [m, c] : maskCoefs) {
276  masks[i] = {m, c.maximum};
277  for (uint16_t a : utils::LoopSet::fromMask(utils::flipMask(m, depth)))
278  indDepGraph.add_edges(a, utils::LoopSet::fromMask(m));
279  }
280  graphs::stronglyConnectedComponents(cmpts, indDepGraph);
281  }
