120  size_t Align) -> std::byte * {
121  Align = Align > MinAlignment ? toPowerOf2(Align) : MinAlignment;
122  if constexpr (BumpUp) {
123  if (Ptr == SlabCur - align(OldSize)) {
124  SlabCur = Ptr + align(NewSize);
125  if (!outOfSlab(SlabCur, SlabEnd)) {
126  __asan_unpoison_memory_region(Ptr + OldSize, NewSize - OldSize);

152  [[gnu::returns_nonnull, nodiscard]] constexpr auto

154  -> void * {

157  if constexpr ((BumpDown) & (!ForOverwrite))
158  std::copy(Ptr, Ptr + OldSize, p);
159  return p;
160  }
161  if (OldSize >= NewSize) return Ptr;

163  if constexpr (BumpUp) {
164  if (Ptr == SlabCur - align(OldSize)) {
165  SlabCur = Ptr + align(NewSize);

171  }
172  } else if (Ptr == SlabCur) {

174  SlabCur -= extraSize;
