313  constexpr void initNonNegative(math::Alloc<int64_t> auto alloc,
314  PtrMatrix<int64_t> A,
315  ptrdiff_t numNonNegative) {
316  // we have an additional numNonNegative x numNonNegative identity matrix
317  // as the lower right block of `A`.
318  // numConExplicit has +1 to indicate positive.
319  // I.e., first variable (probably const offsets) is positive.
320  const ptrdiff_t numConExplicit = ptrdiff_t(A.numRow()) + 1;
321  const ptrdiff_t numConTotal = numConExplicit + numNonNegative;
322  numVar = ptrdiff_t(A.numCol());
323  Row rowV = Row{numVar + numConTotal};
324  Col colV = Col{2 * numConTotal};
328  auto B = getV(rowV, colV);
329  std::fill_n(B.begin(), B.numRow() * B.numCol(), 0);
330  B(0, 0) = 1;
331  // B = [ A_0 A_1
332  // 0 I ]
333  // V = [B' 0
334  // S I]
335  // V = [A_0' 0 0
336  // A_1' I 0
337  // S_0 S_1 I]
338  B(_(begin, numVar), _(1, numConExplicit)) << A.transpose();
339  for (ptrdiff_t j = 0; j < numNonNegative; ++j)
340  B(j + numVar - numNonNegative, numConExplicit + j) = 1;
341  for (ptrdiff_t j = 0; j < numConTotal; ++j) {
342  B(j + numVar, j) = -1;
343  B(j + numVar, j + numConTotal) = 1;
344  }
345  numEquations = numConTotal;
346  initCore(alloc);
347  }
348  constexpr void initNonNegative(math::Alloc<int64_t> auto alloc,
349  PtrMatrix<int64_t> A, PtrMatrix<int64_t> E,
350  ptrdiff_t numNonNegative) {
351  // we have an additional numNonNegative x numNonNegative identity matrix
352  // as the lower right block of `A`.
353  const ptrdiff_t numInEqConExplicit = ptrdiff_t(A.numRow()) + 1;
354  const ptrdiff_t numInEqConTotal = numInEqConExplicit + numNonNegative;
355  const ptrdiff_t numEqCon = ptrdiff_t(E.numRow());
356  numVar = ptrdiff_t(A.numCol());
357  Row rowV = Row{numVar + numInEqConTotal};
358  Col colV = Col{2 * numInEqConTotal + numEqCon};
359  auto B = getV(rowV, colV);
360  std::fill_n(B.begin(), B.numRow() * B.numCol(), 0);
361  B(0, 0) = 1;
362  // B is `A` augmented with the implicit non-negative constraints
363  // B = [ A_0 A_1
364  // 0 I ]
365  // V = [B' E' 0
366  // S 0 I]
367  // V = [A_0' 0 E_0' 0
368  // A_1' I E_1' 0
369  // S_0 S_1 0 I]
370  numEquations = numInEqConTotal + numEqCon;
371  B(_(begin, numVar), _(1, numInEqConExplicit)) << A.transpose();
372  B(_(begin, numVar), _(numInEqConTotal, numInEqConTotal + numEqCon))
373  << E.transpose();
374  if (numNonNegative)
375  B(_(numVar - numNonNegative, numVar),
376  _(numInEqConExplicit, numInEqConExplicit + numNonNegative))
377  .diag()
378  << 1;
379  for (ptrdiff_t j = 0; j < numInEqConTotal; ++j) {
380  B(j + numVar, j) = -1;
381  B(j + numVar, j + numEquations) = 1;
382  }

384  }

448  PtrMatrix<int64_t> E, bool pos0) {
449  const ptrdiff_t numInEqCon = ptrdiff_t(A.numRow()) + pos0;
450  numVar = ptrdiff_t(A.numCol());
451  const ptrdiff_t numEqCon = ptrdiff_t(E.numRow());
452  Row rowV = Row{numVar + numInEqCon};
453  Col colV = Col{2 * numInEqCon + numEqCon};
454  auto B = getV(rowV, colV);
455  B << 0;
456  // V = [A' E' 0
457  // S 0 I]
458  B(0, 0) = pos0;
459  B(_(begin, numVar), _(pos0, numInEqCon)) << A.transpose();
460  // A(_, _(pos0, end)).transpose();
461  B(_(begin, numVar), _(numInEqCon, numInEqCon + numEqCon)) << E.transpose();
462 
463  numEquations = numInEqCon + numEqCon;
464  for (ptrdiff_t j = 0; j < numInEqCon; ++j) {
465  B(j + numVar, j) = -1;
466  B(j + numVar, j + numEquations) = 1;

470  // sets U, V, and d.
471  // needs to also set their size, which is only determined here.
472  constexpr void initCore(math::Alloc<int64_t> auto alloc) {
473  // numVar + numInEq x 2*numInEq + numEq
474  MutPtrMatrix<int64_t> B = getV();
475  Row R = B.numRow();
476  MutPtrMatrix<int64_t> U = getU(); // numVar + numInEq x numVar + numInEq
477  U.diag() << 1;
478  // We will have query of the form Ax = q;
479  simplifySystemsImpl({B, U});
480  while ((R) && allZero(B(R - 1, _))) --R;
481  setURank(R);
482  ptrdiff_t numColB = ptrdiff_t(B.numCol());
483  // upper bounded by numVar + numInEq x numVar + numInEq
484  // if V is square, it is full rank and there is 1 solution
485  // if V has fewer rows, there are infinitely many solutions
486  if (R == numColB) return;
487  invariant(R < numColB);
488  // H (aliasing V and A) copied
489  // R = B.numRow() < B.numCol()
490  auto Vt{identity<int64_t>(alloc, numColB)};
491  // Ht.numRow() > Ht.numCol() = R
492  // (2*numInEq + numEq) x R
493  auto Ht = matrix<int64_t>(alloc, Row{numColB}, Col{ptrdiff_t(R)});
494  Ht << B(_(0, R), _).transpose();
495  solveSystem(Ht, Vt);
496  // upper bounded by numVar + numInEq
497  // rows/cols, but of rank R
498  // smaller based on rank
499  getD(R) << Ht.diag(); // d.size() == R
500  // upper bounded by 2*numInEq + numEq x 2*numInEq + numEq
501  getV() << Vt.transpose();
502  }

504  // Note that this is only valid when the comparator was constructed
505  // with index `0` referring to >= 0 constants (i.e., the default).
506  [[nodiscard]] constexpr auto isEmpty(Arena<> alloc) const -> bool {
507  auto V = getV();
508  auto U = getU();
509  auto d = getD();
510  StridedVector<int64_t> b{U(_, 0)};
511  if (d.empty()) {
512  if (!allZero(b[_(V.numRow(), end)])) return false;
513  Col oldn = V.numCol();
514  auto H{matrix<int64_t>(&alloc, V.numRow(), oldn + 1)};
515  // IntMatrix H{V.numRow(), oldn + 1};
516  H(_, _(0, oldn)) << V;
517  H(_, oldn) << -b;
518  solveSystem(H);
519  bool ret = true;
520  for (ptrdiff_t i = numEquations; i < H.numRow(); ++i)
521  if (auto rhs = H(i, oldn))
522  if ((rhs > 0) != (H(i, i) > 0)) {
523  ret = false;
524  break;
525  }
526  return ret;
527  }
528  // Column rank deficient case
529  Row numSlack = V.numRow() - numEquations;
530  // Vector<int64_t> dinv = d; // copy
531  // We represent D martix as a vector, and multiply the lcm to the
532  // linear equation to avoid store D^(-1) as rational type
533  int64_t lcmD = lcm(d);
534  auto b2{vector<int64_t>(&alloc, d.size())};
535  b2 << -b * lcmD / d;
536  // Vector<int64_t> b2 = -b * Dlcm / d;
537  ptrdiff_t numRowTrunc = ptrdiff_t(U.numRow());
538  auto c{vector<int64_t>(&alloc, ptrdiff_t(V.numRow() - numEquations))};
539  c << V(_(numEquations, end), _(begin, numRowTrunc)) * b2;
540  // Vector<int64_t> c = V(_(numEquations, end), _(begin, numRowTrunc)) *
541  // b2;
542  auto dimNS = V.numCol() - numRowTrunc;
543  // expand W stores [c -JV2 JV2]
544  // we use simplex to solve [-JV2 JV2][y2+ y2-]' <= JV1D^(-1)Uq
545  // where y2 = y2+ - y2-
546  auto expandW{matrix<int64_t>(&alloc, Row{numSlack}, Col{dimNS * 2 + 1})};
547  for (ptrdiff_t i = 0; i < numSlack; ++i) {
548  expandW(i, 0) = c[i];

550  for (ptrdiff_t j = 0; j < dimNS; ++j) {
551  auto val = V(i + numEquations, numRowTrunc + j) * lcmD;
552  expandW(i, j + 1) = -val;
553  expandW(i, dimNS + 1 + j) = val;
554  }

567  [[nodiscard]] constexpr auto greaterEqualFullRank(Arena<> *alloc,
568  PtrVector<int64_t> b) const
569  -> bool {
570  auto V = getV();
571  if (!allZero(b[_(V.numRow(), end)])) return false;
572  auto H = matrix<int64_t>(alloc, V.numRow(), V.numCol() + 1);
573  Col oldn = V.numCol();
574  H(_, _(0, oldn)) << V;
575  // H.numRow() == b.size(), because we're only here if dimD == 0,
576  // in which case V.numRow() == U.numRow() == b.size()
577  H(_, oldn) << b;
578  solveSystem(H);
579  for (ptrdiff_t i = numEquations; i < H.numRow(); ++i)
580  if (auto rhs = H(i, oldn))
581  if ((rhs > 0) != (H(i, i) > 0)) return false;
582  return true;
583  }
584  [[nodiscard]] constexpr auto
585  greaterEqualRankDeficient(Arena<> *alloc, MutPtrVector<int64_t> b) const
586  -> bool {
587  auto V = getV();
588  auto d = getD();
589  Row numSlack = V.numRow() - numEquations;
590  auto dinv = vector<int64_t>(alloc, d.size());
591  dinv << d; // copy
592  // We represent D martix as a vector, and multiply the lcm to the
593  // linear equation to avoid store D^(-1) as rational type
594  int64_t lcmD = lcm(dinv);
595  for (ptrdiff_t i = 0; i < dinv.size(); ++i) {
596  auto x = lcmD / dinv[i];
597  dinv[i] = x;
598  b[i] *= x;
599  }
600  ptrdiff_t numRowTrunc = getURank();
601  auto c = vector<int64_t>(alloc, unsigned(V.numRow() - numEquations));
602  c << V(_(numEquations, end), _(begin, numRowTrunc)) * b;
603  auto dimNS = V.numCol() - numRowTrunc;
604  // expand W stores [c -JV2 JV2]
605  // we use simplex to solve [-JV2 JV2][y2+ y2-]' <= JV1D^(-1)Uq
606  // where y2 = y2+ - y2-
607  auto expandW = matrix<int64_t>(alloc, numSlack, dimNS * 2 + 1);
608  for (ptrdiff_t i = 0; i < numSlack; ++i) {
609  expandW(i, 0) = c[i];
610  // expandW(i, 0) *= Dlcm;
611  for (ptrdiff_t j = 0; j < dimNS;) {
612  auto val = V(i + numEquations, numRowTrunc + j++) * lcmD;
613  expandW(i, j) = -val;
614  expandW(i, dimNS + j) = val;
615  }
616  }
617  Optional<Simplex *> optS{Simplex::positiveVariables(alloc, expandW)};
618  return optS.hasValue();
619  }

621  PtrVector<int64_t> query) const
622  -> bool {
623  auto U = getU();
624  auto b = vector<int64_t>(&alloc, unsigned(U.numRow()));
625  b << U(_, _(begin, query.size())) * query;
626  return getD().size() ? greaterEqualRankDeficient(&alloc, b)
627  : greaterEqualFullRank(&alloc, b);
628  }

842 constexpr void moveEqualities(DenseMatrix<int64_t> &A, math::IntMatrix &E,
843  const Comparator auto &C) {
844  const ptrdiff_t numVar = ptrdiff_t(E.numCol());
845  assert(A.numCol() == numVar);
846  if (A.numRow() <= 1) return;
847  for (ptrdiff_t o = ptrdiff_t(A.numRow()) - 1; o > 0;) {
848  for (ptrdiff_t i = o--; i < A.numRow(); ++i) {
849  bool isNeg = true;
850  for (ptrdiff_t v = 0; v < numVar; ++v) {
851  if (A(i, v) != -A(o, v)) {
852  isNeg = false;
853  break;
854  }
855  }
856  if (isNeg && C.equalNegative(A(i, _), A(o, _))) {
857  ptrdiff_t e = ptrdiff_t(E.numRow());
858  E.resize(e + 1, numVar);
859  for (ptrdiff_t v = 0; v < numVar; ++v) E(e, v) = A(i, v);
860  eraseConstraint(A, i, o);
861  break;
862  }
863  }
864  }
