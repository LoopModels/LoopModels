151 class LoopTree {
152  // The root of this subtree
153  NotNull<IR::Loop> loop;
154  LoopTree *parent{nullptr}; // do we need this?
155  Vec<LoopTree *> children{};
156  unsigned depth{0};

158  // the `Addr`s and instructions will determine the ordering.

160  : loop{lalloc->create<IR::Loop>(parent_->depth + 1)}, parent(parent_),
161  depth(parent_->depth + 1) {
162  // allocate the root node, and connect it to parent's node, as well as
163  // previous loop of the same level.
164  loop->setParent(parent_->loop);
165  }
166  constexpr LoopTree(Arena<> *lalloc) : loop{lalloc->create<IR::Loop>(0)} {}

169  static auto root(Arena<> *salloc, Arena<> *lalloc) -> LoopTree * {

172  // salloc: Short lived allocator, for the indexable `Vec`s
173  // Longer lived allocator, for the IR::Loop nodes
174  void addNode(Arena<> *salloc, Arena<> *lalloc, lp::ScheduledNode *node) {
175  if (node->getNumLoops() == depth) {
176  // Then it belongs here, and we add loop's dependencies.
177  // We only need to add deps to support SCC/top sort now.
178  // We also apply the rotation here.
179  // For dependencies in SCC iteration, only indvar deps get iterated.
180  auto [Pinv, denom] = math::NormalForm::scaledInv(node->getPhi());
181  NotNull<poly::Loop> affloop =
182  node->getLoopNest()->rotate(lalloc, Pinv, node->getOffset());
183  for (IR::Addr *m : node->localAddr()) {
184  m->rotate(affloop, Pinv, denom, node->getOffsetOmega(),
185  node->getOffset());
186  loop->insertAfter(m);
187  }
188  return;
189  }
190  // we need to find the sub-loop tree to which we add `node`
191  ptrdiff_t idx = node->getFusionOmega(depth);
192  invariant(idx >= 0);
193  ptrdiff_t numChildren = children.size();
194  if (idx >= children.size()) {
195  if (idx >= children.getCapacity()) {
196  // allocate extra capacity
197  children.reserve(salloc, 2 * (idx + 1));
198  }
199  // allocate new nodes and resize
200  children.resize(idx + 1);
201  for (ptrdiff_t i = numChildren; i < idx + 1; ++i)
202  children[i] = new (salloc) LoopTree{lalloc, this};
203  numChildren = idx + 1;
204  }
205  children[idx]->addNode(salloc, lalloc, node);
206  }
207  constexpr auto getChildren() -> Vec<LoopTree *> { return children; }
208  constexpr auto getLoop() -> IR::Loop * { return loop; }

227 // searches `N` and it's users for loop-independent users
228 // this exits early if it finds a dependent user; we search everything
229 // anyway, so we'll revist later anyway.
230 // We return a `IR::Node *, bool` pair, where the `bool` is true if
231 // `N` was loop independent.
232 // We do this rather than something like returning a `nullptr`, as
233 // we may have descended into instructions, found some users that are
234 // but then also found some that are not; we need to return `false`
235 // in this case, but we of course want to still return those we found.
236 inline auto searchLoopIndependentUsers(IR::Loop *L, IR::Node *N, uint8_t depth,
237  LoopDepSummary summary)
238  -> LoopIndependent {
239  if (N->dependsOnParentLoop()) return {summary, false};
240  if (llvm::isa<IR::Loop>(N)) return {summary, false};
241  if (IR::Loop *P = N->getLoop(); P != L)
242  return {summary, !(P && L->contains(P))};
243  LoopIndependent ret{summary, true};
244  IR::Addr *a = llvm::dyn_cast<IR::Addr>(N);
245  if (a) {
246  a->removeFromList();
247  if (a->indexedByInnermostLoop()) {
248  a->insertAfter(ret.summary.indexedByLoop);
249  ret.summary.indexedByLoop = a;
250  return {summary, false};
251  } else {
252  a->insertAfter(ret.summary.notIndexedByLoop);
253  ret.summary.notIndexedByLoop = a;
254  }
255  for (IR::Addr *m : a->outputAddrs(depth)) {
256  ret *= searchLoopIndependentUsers(L, m, depth, summary);
257  if (ret.independent) continue;
258  a->setDependsOnParentLoop();
259  return ret;
260  }
261  }
262  // if it isn't a Loop, must be an `Instruction`
263  IR::Value *I = llvm::cast<IR::Instruction>(N);
264  for (IR::Node *U : I->getUsers()) {
265  ret *= searchLoopIndependentUsers(L, U, depth, summary);
266  if (ret.independent) continue;
267  I->setDependsOnParentLoop();
268  return ret;
269  }
270  // then we can push it to the front of the list, meaning it is hoisted out
271  if (a) {
272  if (ret.summary.notIndexedByLoop == a)
273  ret.summary.notIndexedByLoop = llvm::cast_or_null<IR::Addr>(a->getNext());
274  }
275  I->removeFromList();
276  I->insertAfter(ret.summary.afterExit);
277  ret.summary.afterExit = I;
278  I->visit(depth);
279  return ret;
280 }
281 inline auto visitLoopDependent(IR::Loop *L, IR::Node *N, uint8_t depth,
282  IR::Node *body) -> IR::Node * {
283  invariant(N->getVisitDepth() != 254);
284  // N may have been visited as a dependent of an inner loop, which is why
285  // `visited` accepts a depth argument
286  if (N->wasVisited(depth) || !(L->contains(N))) return body;
287 #ifndef NDEBUG
288  // Our goal here is to check for cycles in debug mode.
289  // Each level of our graph is acyclic, meaning that there are no cycles at
290  // that level when traversing only edges active at that given level. However,
291  // when considering edges active at level `I`, we may have cycles at level `J`
292  // if `J>I`. In otherwords, here we are travering all edges active at
293  // `I=depth`. Within subloops, which necessarilly have depth `J>I`, we may
294  // have cycles.
295  //
296  // Thus, we need to prevent getting stuck in a cycle for these deeper loops by
297  // setting `N->visit(depth)` here, so `wasVisited` will allow them to
298  // immediately return. But, in debug mode, we'll set nodes of the same depth
299  // to `254` to check for cycles.
300  if (N->getLoop() == L) N->visit(254);
301  else N->visit(depth);
302 #else
303  N->visit(depth);
304 #endif
305  // iterate over users
306  if (IR::Addr *A = llvm::dyn_cast<IR::Addr>(N)) {
307  for (IR::Addr *m : A->outputAddrs(depth)) {
308  if (m->wasVisited(depth)) continue;
309  body = visitLoopDependent(L, m, depth, body);
310  }
311  }
312  if (IR::Instruction *I = llvm::cast<IR::Instruction>(N)) {
313  for (IR::Node *U : I->getUsers()) {
314  if (U->wasVisited(depth)) continue;
315  body = visitLoopDependent(L, U, depth, body);
316  }
317  } else if (IR::Loop *S = llvm::cast<IR::Loop>(N)) {
318  for (IR::Node *U : S->getChild()->nodes()) {
319  if (U->wasVisited(depth)) continue;
320  body = visitLoopDependent(L, U, depth, body);
321  }
322  }
323 #ifndef NDEBUG
324  if (N->getLoop() == L) N->visit(depth);
325 #endif
326  if (N->getLoop() == L) body = N->setNext(body);
327  return body;
328 }
329 inline auto topologicalSort(IR::Loop *root, unsigned depth) -> IR::Node * {
330  // basic plan for the top sort:
331  // We iterate across all users, once all of node's users have been added,
332  // we push it to the front of the list. Thus, we get a top-sorted list.
333  // We're careful about the order, so that this top sort should LICM all the
334  // addresses that it can.
335  //
336  // We must push the exit before the root (as the exit depends on the loop, and
337  // we iterate users).
338  // The exit doesn't use any in this block, so we begin by trying to push any
339  // instructions that don't depend on the loop. If we fail to push them (i.e.,
340  // because they have uses that do depend on the loop), then they get added to
341  // a revisit queue. Any instructions we are able to push-front before we push
342  // the exit, implicitly happen after the exit, i.e. they have been LICMed into
343  // the exit block. We unvisit the revisit-queue, and add them back to the main
344  // worklist. Then, we proceed with a depth-first topological sort normally
345  // (iterating over uses, pushing to the front), starting with the loop root,
346  // so that it gets pushed to the front as soon as possible. That is, so that
347  // it happens as late as possible Any instructions that get pushed to the
348  // front afterwards have been LICMed into the loop pre-header.
349  //
350  // In this first pass, we iterate over all nodes, pushing those
351  // that can be hoisted after the exit block.
352  IR::Node *C = root->getChild();
353  LoopDepSummary summary;
354  for (IR::Node *N : C->nodes())
355  summary = searchLoopIndependentUsers(root, N, depth, summary).summary;
356  // summary.afterExit will be hoisted out; every member has been marked as
357  // `visited` So, now we search all of root's users, i.e. every addr that
358  // depends on it
359  IR::Node *body;
360  for (IR::Node *N : summary.indexedByLoop->nodes())
361  body = visitLoopDependent(root, N, depth, body);
362  body = root->setNext(body); // now we can place the loop
363  for (IR::Node *N : summary.notIndexedByLoop->nodes())
364  body = visitLoopDependent(root, N, depth, body);
365  // and any remaining edges
366  return body;
367 }

369 inline auto buildGraph(IR::Loop *root, unsigned depth) -> IR::Node * {
370  // We build the instruction graph, via traversing the tree, and then
371  // top sorting as we recurse out
372  for (IR::Loop *child : root->subLoops()) buildGraph(child, depth + 1);
373  return topologicalSort(root, depth);
374 }

384 inline void eliminateAddr(IR::Addr *a, IR::Addr *b) {
385  if (a->indexMatrix() != b->indexMatrix()) return;
387  if (a->isStore()) {
388  if (b->isStore()) { // Write->Write
389  // Are there reads in between? If so, we must keep--
390  // --unless we're storing the same value twice (???)
391  // without other intervening store-edges.
392  // Without reads in between, it's safe.
393  } else { // Write->Read
394  // Can we replace the read with using the written value?
395  if (a->getLoop() != b->getLoop()) return;
396  }
397  } else if (b->isLoad()) { // Read->Read
398  // If they don't have the same parent, either...
399  // They're in different branches of loops, and load can't live
400  // in between them

404  // }

406  // A[i,j];
407  // }

410  if (a->getLoop() != b->getLoop()) return;

412  } // Read->Write, can't delete either

756  // NOLINTNEXTLINE(misc-no-recursion)
757  auto hoist(Arena<> *alloc, Addr *a, unsigned currentLoop)
758  -> Optional<unsigned> {
759  if (a->wasVisited3()) {
760  if (a->getLoopTreeSchedule() == this) return a->getBlockIdx();
761  return {};
762  }
763  a->visit3();

765  auto *L = a->getLoopTreeSchedule();

767  if ((placed && (L->getParent() != this || a->dependsOnIndVars(getDepth()))))
768  return {};

770  unsigned idxFront = 2 * currentLoop, idxBack = idxFront + 2;
771  bool legalHoistFront = true, legalHoistBack = true;
772  for (auto *p : a->inNeighbors(getDepth())) {
773  auto pIdx = getIdx(p);
774  if (pIdx <= idxFront) continue;

776  if (!fail) {

778  fail = *hIdx > currentLoop;

780  }

783  break;

788  invariant(L->try_delete(a));
789  a->setLoopTreeSchedule(this);
790  }
791  a->setBlockIdx(currentLoop);
792  push_back(alloc, a, currentLoop);
793  a->place();
794  return currentLoop;
795  }
796  for (auto *c : a->outNeighbors(getDepth())) {
797  auto cIdx = getIdx(c);
798  if (cIdx >= idxBack) continue;
799  bool fail = cIdx <= idxFront;
800  if (!fail) {
801  if (auto hIdx = hoist(alloc, c, currentLoop))
802  fail = *hIdx <= currentLoop;
803  else fail = true;
804  }
805  if (fail) {
806  legalHoistBack = false;
807  break;
808  }
809  }
810  if (legalHoistBack) {
811  if (placed) {
812  invariant(L->try_delete(a));
813  a->setLoopTreeSchedule(this);
814  }
815  a->setBlockIdx(currentLoop + 1);
816  invariant(currentLoop < currentLoop + 1);
817  push_front(alloc, a, currentLoop + 1);
818  a->place();

822  // what sort of ctrl path could produce that outcome?
823  invariant(placed);
