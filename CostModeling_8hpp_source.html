54  static auto estimateMaximumVectorWidth(llvm::LLVMContext &C,
55  llvm::TargetTransformInfo &TTI)
56  -> uint8_t {
57  uint8_t twiceMaxVectorWidth = 2;
58  auto *f32 = llvm::Type::getFloatTy(C);
59  llvm::InstructionCost prevCost = TTI.getArithmeticInstrCost(
60  llvm::Instruction::FAdd,
61  llvm::FixedVectorType::get(f32, twiceMaxVectorWidth));
62  while (true) {
63  llvm::InstructionCost nextCost = TTI.getArithmeticInstrCost(
64  llvm::Instruction::FAdd,
65  llvm::FixedVectorType::get(f32, twiceMaxVectorWidth *= 2));
66  if (nextCost > prevCost) break;
67  prevCost = nextCost;
68  }
69  return 16 * twiceMaxVectorWidth;
70  }
