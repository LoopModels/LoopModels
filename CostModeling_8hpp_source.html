87 Addr::maybeReassociableReduction(const Dependencies &deps) {
88  if (isLoad()) return;
89  // we should have a store whose first output edge is the load for
90  // the following iteration. This iter is the reverse-time
91  // edge.
92  auto edges{outputEdgeIDs(deps, getCurrentDepth())};
93  auto B = edges.begin();
94  if (B == edges.end()) return;
95  poly::Dependence::ID id{*B};
96  if (deps.revTimeEdge(id) < 0) return;
97  IR::Addr *dst = deps.output(id);
98  if (dst->isStore() || (getLoop() != dst->getLoop())) return;
99  // if we failed to hoist the `Addr` out of time-dims, then we cannot optimize.
100  if (getCurrentDepth() > deps.satLevel(id)) return;
101  if (reassociableReduction == dst) return; // multiple time dims, already found
102  auto *c = llvm::dyn_cast<IR::Compute>(getStoredVal());
103  if (!c) return;
104  if (findThroughReassociable(dst, c) != 1) return;
105  reassociableReduction = dst;
106  dst->reassociableReduction = this;
107 }

146  // Longer lived allocator, for the IR::Loop nodes
147  // NOLINTNEXTLINE(misc-no-recursion)
148  void addNode(Arena<> *salloc, Arena<> *lalloc, lp::ScheduledNode *node) {
149  if (node->getNumLoops() == depth) {
150  // Then it belongs here, and we add loop's dependencies.
151  // We only need to add deps to support SCC/top sort now.
152  // We also apply the rotation here.
153  // For dependencies in SCC iteration, only indvar deps get iterated.
154  auto [Pinv, denom] = math::NormalForm::scaledInv(node->getPhi());
155  Valid<poly::Loop> explicitLoop =
156  node->getLoopNest()->rotate(lalloc, Pinv, node->getOffset());
157  for (IR::Addr *m : node->localAddr()) {
158  m->rotate(explicitLoop, Pinv, denom, node->getOffsetOmega(),
159  node->getOffset());
160  loop->insertAfter(m);
161  }
162  return;
163  }
164  // we need to find the sub-loop tree to which we add `node`
165  ptrdiff_t idx = node->getFusionOmega(depth);
166  invariant(idx >= 0);
167  ptrdiff_t numChildren = children.size();
168  if (idx >= children.size()) {
169  if (idx >= children.getCapacity()) {
170  // allocate extra capacity
171  children.reserve(salloc, 2 * (idx + 1));
172  }
173  // allocate new nodes and resize
174  children.resize(idx + 1);
175  for (ptrdiff_t i = numChildren; i < idx + 1; ++i)
176  children[i] = new (salloc) LoopTree{lalloc, node->getLoopNest(), this};
177  numChildren = idx + 1;
178  }
179  children[idx]->addNode(salloc, lalloc, node);
180  }
181  constexpr auto getChildren() -> Vec<LoopTree *> { return children; }
182  constexpr auto getLoop() -> IR::Loop * { return loop; }
183 };

218 inline auto searchLoopIndependentUsers(const IR::Dependencies &deps,
219  IR::Loop *L, IR::Node *N, int depth,
220  LoopDepSummary summary)
221  -> LoopIndependent {
222  if (N->dependsOnParentLoop()) return {summary, false};
223  if (llvm::isa<IR::Loop>(N)) return {summary, false};
224  if (IR::Loop *P = N->getLoop(); P != L)
225  return {summary, !(P && L->contains(P))};
226  LoopIndependent ret{summary, true};
227  auto *a = llvm::dyn_cast<IR::Addr>(N);
228  if (a) {
229  a->removeFromList();
230  if (a->indexedByInnermostLoop()) {
231  a->insertAfter(ret.summary.indexedByLoop);
232  ret.summary.indexedByLoop = a;
233  return {summary, false};
234  }
235  a->insertAfter(ret.summary.notIndexedByLoop);
236  ret.summary.notIndexedByLoop = a;
237  for (IR::Addr *m : a->unhoistableOutputs(deps, depth - 1)) {
238  ret *= searchLoopIndependentUsers(deps, L, m, depth, summary);
239  if (ret.independent) continue;
240  a->setDependsOnParentLoop();
241  return ret;
242  }
243  }
244  // if it isn't a Loop or Addr, must be an `Instruction`
245  IR::Value *I = llvm::cast<IR::Instruction>(N);
246  for (IR::Node *U : I->getUsers()) {
247  ret *= searchLoopIndependentUsers(deps, L, U, depth, summary);
248  if (ret.independent) continue;
249  I->setDependsOnParentLoop();
250  return ret;
251  }
252  // then we can push it to the front of the list, meaning it is hoisted out
253  if (a && (ret.summary.notIndexedByLoop == a))
254  ret.summary.notIndexedByLoop = llvm::cast_or_null<IR::Addr>(a->getNext());
255  I->removeFromList();
256  I->insertAfter(ret.summary.afterExit);
257  ret.summary.afterExit = I;
258  I->visit(depth);
259  return ret;
260 }

264 inline auto visitLoopDependent(const IR::Dependencies &deps, IR::Loop *L,
265  IR::Node *N, int depth, IR::Node *body,
266  IR::Loop *R = nullptr) -> IR::Node * {
267  invariant(N->getVisitDepth() != 254);
268  // N may have been visited as a dependent of an inner loop, which is why
269  // `visited` accepts a depth argument
270  if (N->wasVisited(depth) || !(L->contains(N))) return body;
271 #ifndef NDEBUG
272  // Our goal here is to check for cycles in debug mode.
273  // Each level of our graph is acyclic, meaning that there are no cycles at
274  // that level when traversing only edges active at that given level.
275  // However, when considering edges active at level `I`, we may have cycles
276  // at level `J` if `J>I`. In otherwords, here we are traversing all edges
277  // active at `I=depth`. Within subloops, which necessarily have depth
278  // `J>I`, we may have cycles.
279  //
280  // Thus, we need to prevent getting stuck in a cycle for these deeper loops
281  // by setting `N->visit(depth)` here, so `wasVisited` will allow them to
282  // immediately return. But, in debug mode, we'll set nodes of the same depth
283  // to `254` to check for cycles.
284  if (N->getLoop() == L) N->visit(254);
285  else N->visit(depth);
286 #else
287  N->visit(depth);
288 #endif
289  // iterate over users
290  if (auto *A = llvm::dyn_cast<IR::Addr>(N)) {
291  // Note that `topologicalSort` calls `searchLoopIndependentUsers` which
292  // checks whether an `Addr` is `indexedByInnermostLoop`.
293  //
294  // Note that here `depth` is `0` for top-level, 1 for the outer most loop,
295  // etc. That is, loops are effectively 1-indexed here, while `satLevel`
296  // is effectively 0-indexed by loop.
297  // Example 1:
298  // for (ptrdiff_t m = 0; m < M; ++m)
299  // for (ptrdiff_t n = 0; n < N; ++n)
300  // for (ptrdiff_t k = 0; k < K; ++k) C[m,n] = C[m,n] + A[m,k]*B[k,n];
301  // we have cyclic dependencies between the load from/store to `C[m,n]`.
302  // The `C[m,n]` load -> `C[m,n]` store was not satisfied by any loop, so
303  // the sat level is 255.
304  // The `C[m,n]` store -> `C[m,n]` load has satLevel = 2.
305  // Example 2:
306  // for (ptrdiff_t m = 0; m < M; ++m)
307  // for (ptrdiff_t n = 1; n < N; ++n) C[m,n] = C[m,n] + C[m,n-1];
308  // we again have a cyple, from the load `C[m,n-1]` to the store `C[m,n]`,
309  // and from the store `C[m,n]` to the load `C[m,n-1]` on the following
310  // iteration.
311  // The former has a sat level of 255, while the latter has a sat level of
312  // `1`.
313  //
314  // isActive(depth) == satLevel() > depth
315  //
316  // a. load->store is not satisfied by any loop, instead handled by sorting
317  // of instructions in the innermost loop, i.e. sat is depth=3.
318  // b. store->load is carried by the `k` loop, i.e. sat is depth=2.
319  // Because `2 > (3-1) == false`, we do not add it here,
320  // its sorting isn't positional!
321  //
322  // TODO:
323  // - [ ] I think the current algorithm may illegally hoist certain
324  // dependencies carried on this loop. Specifically, we can hoist
325  // addresses that (a) are not indexed by this loop, but need to be
326  // repeated anyway because of some other address operation, while that
327  // combination can't be moved to registers, e.g. because their index
328  // matrices are not equal.
329  // We need to distinguish between order within the loop, for the
330  // purpose of this topsort, and placement with respect to the loop.
331  // Simply, we perhaps should simply avoid hoisting when we carry
332  // a dependence that doesn't meet the criteria of `unhoistableOutputs`
333  // - [ ] Incorporate the legality setting here?
334  for (IR::Addr *m : A->unhoistableOutputs(deps, depth - 1)) {
335  if (m->wasVisited(depth)) continue;
336  body = visitLoopDependent(deps, L, m, depth, body, R);
337  }
338  }
339  if (auto *I = llvm::dyn_cast<IR::Instruction>(N)) {
340  for (IR::Node *U : I->getUsers()) {
341  if (U->wasVisited(depth)) continue;
342  body = visitLoopDependent(deps, L, U, depth, body, R);
343  }
344  } else if (auto *S = llvm::dyn_cast<IR::Loop>(N)) {
345  for (IR::Node *U : S->getChild()->nodes()) {
346  if (U->wasVisited(depth)) continue;
347  body = visitLoopDependent(deps, L, U, depth, body, R);
348  }
349  }
350 #ifndef NDEBUG
351  if (N->getLoop() == L) N->visit(depth);
352 #endif
353  if (N->getLoop() == L) body = N->setNext(body);
354  if (R) hoist(N, R, depth - 1);
355  return body;
356 }
357 inline void addBody(const IR::Dependencies &deps, IR::Loop *root, int depth,
358  IR::Node *nodes) {
359  IR::Exit exit{}; // use to capture last node
360  IR::Node *body{&exit};
361  for (IR::Node *N : nodes->nodes())
362  body = visitLoopDependent(deps, root, N, depth, body);
363  body = root->setChild(body); // now we can place the loop
364  IR::Node *last = exit.getPrev();
365  if (last) last->setNext(nullptr);
366  root->setLast(last);
367 }
368 inline void topologicalSort(const IR::Dependencies &deps, IR::Loop *root,
369  int depth) {
370  // basic plan for the top sort:
371  // We iterate across all users, once all of node's users have been added,
372  // we push it to the front of the list. Thus, we get a top-sorted list.
373  // We're careful about the order, so that this top sort should LICM all the
374  // addresses that it can.
375  //
376  // We must push the exit before the root (as the exit depends on the loop,
377  // and we iterate users). The exit doesn't use any in this block, so we
378  // begin by trying to push any instructions that don't depend on the loop.
379  // If we fail to push them (i.e., because they have uses that do depend on
380  // the loop), then they get added to a revisit queue. Any instructions we
381  // are able to push-front before we push the exit, implicitly happen after
382  // the exit, i.e. they have been LICMed into the exit block. We unvisit the
383  // revisit-queue, and add them back to the main worklist. Then, we proceed
384  // with a depth-first topological sort normally (iterating over uses,
385  // pushing to the front), starting with the loop root, so that it gets
386  // pushed to the front as soon as possible. That is, so that it happens as
387  // late as possible Any instructions that get pushed to the front afterwards
388  // have been LICMed into the loop pre-header.
389  //
390  // In this first pass, we iterate over all nodes, pushing those
391  // that can be hoisted after the exit block.
392  //
393  IR::Node *C = root->getChild();
394  LoopDepSummary summary{};
395  for (IR::Node *N : C->nodes())
396  summary = searchLoopIndependentUsers(deps, root, N, depth, summary).summary;
397  // summary.afterExit will be hoisted out; every member has been marked as
398  // `visited` So, now we search all of root's users, i.e. every addr that
399  // depends on it
400  root->setNext(summary.afterExit);
401  IR::Loop *P = root->getLoop();
402  for (IR::Node *N : summary.afterExit->nodes()) hoist(N, P, depth - 1);
403  addBody(deps, root, depth, summary.indexedByLoop);
404  IR::Node *body{root};
405  for (IR::Node *N : summary.notIndexedByLoop->nodes())
406  body = visitLoopDependent(deps, root, N, depth, body, P);
407 }

409 inline auto buildSubGraph(const IR::Dependencies &deps, IR::Loop *root,
410  int depth, uint32_t id) -> uint32_t {
411  // We build the instruction graph, via traversing the tree, and then
412  // top sorting as we recurse out
413  for (IR::Loop *child : root->subLoops())
414  id = buildSubGraph(deps, child, depth + 1, id);
415  root->setMeta(id++);
416 
417  // The very outer `root` needs to have all instr constituents
418  // we also need to add the last instruction of each loop as `last`
419  topologicalSort(deps, root, depth);
420  return id;
421 }
422 inline auto buildGraph(const IR::Dependencies &deps, IR::Loop *root)
423  -> uint32_t {
424  // We build the instruction graph, via traversing the tree, and then
425  // top sorting as we recurse out

458  return true;

614  // part of the cost function of the linear program, i.e. we'd want to
615  // minimize the distance between loads (but allow reordering them).
616  //
617  // I think a reasonable approach is:
618  // Have a map from array pointer to Addr. Addrs form a chain.
619  // as we walk the graph, add each newly encountered addr to the front of the
620  // chain and check if we can eliminate it, or any of its predecessors.
621  //
622  // Note (bracketed means we might be able to eliminate):
623  // Read->[Read] could eliminate read
624  // Read->Write no change
625  // Write->[Read] can forward written value
626  // [Write]->Write can eliminate first write
627  // Thus, we can fuse this pass with our address cost calculation.
628  // We check if we can eliminate before calculating the new cost.
629  // The only case where we may remove an old value, write->write,
630  // we could just take the old cost and assign it to the new write.
631  // TODO: if we have only writes to a non-escaping array, we should
632  // be able to eliminate these writes too, and then also potentially
633  // remove that array temporary (e.g., if it were malloc'd).
634  // E.g. check if the array is a `llvm::isNonEscapingLocalObject` and
635  // allocated by `llvm::isRemovableAlloc`.
636  void removeRedundantAddr(IR::AddrChain addr) {
637  // outputEdges are sorted topologically from first to last.
638  // Example:
639  // for (int i = 0; i < I; ++i){
640  // acc = x[i]; // Statement: 0
641  // for (int j = 0; j < i; ++j){
642  // acc -= x[j]*U[j,i]; // Statement: 1
643  // }
644  // x[i] = acc; // Statement: 2
645  // x[i] = x[i] / U[i,i]; // Statement: 3
646  // }
647  // Here, we have a lot of redundant edges connecting the various `x[i]`s.
648  // We also have output edges between the `x[i]` and the `x[j]` load in
649  // statement 1. It is, however, satisfied at `x[i]`'s depth, and ignored.
650  // So, what would happen here:
651  // S0R->S2W, no change; break.
652  // S2W->S3R, replace read with stored value forwarding.
653  // S2W->S3W, remove S2W as it is shadowed by S3W.
654  // NOTE: we rely on the `ListRange` iterator supporting safely removing the
655  // current iter from the list.
656  for (IR::Addr *a : addr.getAddr()) eliminateAddr(a);
657  }
667  // NOLINTNEXTLINE(misc-no-recursion)
668  auto sortEdges(IR::Loop *R, int32_t pos) -> int32_t {
669  for (IR::Node *n = R->getLast(); n != R; n = n->getPrev()) {
670  if (auto *L = llvm::dyn_cast<IR::Loop>(n)) {
671  pos = sortEdges(L, pos);
672  continue;
673  }
674  auto *a = llvm::dyn_cast<IR::Addr>(n);
675  if (!a) continue;
676  a->setTopPosition(pos--);
677  // for each input edge, we push `a` to the front of the output list
678  for (int32_t id : a->inputEdgeIDs(deps)) {
679  if (deps.prevOut(Dependence::ID{id}) < 0) continue;
680  deps.removeOutEdge(id);
681  IR::Addr *b = deps.input(Dependence::ID{id});
682  int32_t oldFirst = b->getEdgeOut();
683  deps.prevOut(Dependence::ID{oldFirst}) = id;
684  deps.prevOut(Dependence::ID{id}) = -1;
685  deps.nextOut(Dependence::ID{id}) = oldFirst;
686  b->setEdgeOut(id);
687  }
688  }
689  return pos;
690  }
691  void findReductions(IR::AddrChain addr) {
692  for (IR::Addr *a : addr.getAddr()) a->maybeReassociableReduction(deps);
693  };

747 // x[i+3] -= x[j]*U[j,i+3];

750 // // j+2 iteration for i=i+1 iter goes here (but doesn't happen)

755 // }

757 // // j+3 iteration for i=i+1 iter goes here (but doesn't happen)

759 // x[i+3] /= U[i+3,i+3];
760 // }
761 // The key to legality here is that we peel off the dependence polyhedra
762 // from the loop's iteration space.
763 // We can then perform the dependent iterations in order.
764 // With masking, the above code can be vectorized in this manner.
765 // The basic approach is that we have the dependence polyhedra:
766 //
767 // 0 <= i_s < I
768 // 0 <= i_l < I
769 // 0 <= j_l < i_l
770 // i_s = j_l // dependence, yields same address in `x`
771 //
772 // Note that our schedule sets
773 // i_s = i_l
774 // Which gives:
775 // i_l = i_s = j_l < i_l
776 // a contradiction, meaning that the dependency is
777 // conditionally (on our schedule) satisfied.
778 // Excluding the `i_s = i_l` constraint from the
779 // polyhedra gives us the region of overlap.
780 //
781 // When unrolling by `U`, we get using `U=4` as an example:
782 // i^0_s + 1 = i^1_s
783 // i^0_s + 2 = i^2_s
784 // i^0_s + 3 = i^3_s

786 // 0 <= i^1_s < I
787 // 0 <= i^2_s < I
788 // 0 <= i^3_s < I
789 // 0 <= i^0_l < I
790 // 0 <= i^1_l < I
791 // 0 <= i^2_l < I
792 // 0 <= i^3_l < I
793 // 0 <= j_l < i^0_l
794 // 0 <= j_l < i^1_l
795 // 0 <= j_l < i^2_l
796 // 0 <= j_l < i^3_l
797 // i^0_s = j_l || i^1_s = j_l || i^2_s = j_l || i^3_s = j_l
798 // where the final union can be replaced with
799 // i^0_s = j_l || i^0_s+1 = j_l || i^0_s+2 = j_l || i^0_s+3 = j_l
800 // i^0_s <= j_1 <= i^0_s+3
801 //
802 // Similarly, we can compress the other inequalities...
803 // 0 <= i^0_s < I - 3
804 // 0 <= i^0_l < I - 3
805 // 0 <= j_l < i^0_l
806 // i^0_s <= j_1 <= i^0_s+3 // dependence region
807 //
808 // So, the parallel region is the union
809 // i^0_s > j_1 || j_1 > i^0_s+3
810 //
811 // In this example, note that the region `j_1 > i^0_s+3` is empty
812 // so we have one parallel region, and then one serial region.
813 //
814 // Lets consider simpler checks. We have
815 // [ 1 0 ] : x[i] -=
816 // [ 0 1 ] : x[j]
817 // [ 1 ] : x[i] /=
818 // we have a dependency when `i == j`. `i` carries the dependency, but we can
819 // peel off the independent iters from `j`, and unroll `i` for these.
820 //
821 // How to identify:
822 // [ 1 -1 ]
823 // vs, if we had two `x[i]` or two `x[j]`
824 // [ 0, 0 ]
825 // An idea: look for non-zero so we can peel?
826 // Or should we look specifically for `x[i] == x[j]` type pattern?
827 // E.g., if we had
828 // [ i, j, k, l ]
829 // [ 2, -1, 2, -1 ]
830 // we'd need a splitting algorithm.
831 // E.g., split on the 2nd loop, so we get `j == 2*i + 2*k - l`
832 // With this, we'd split iterations into groups
