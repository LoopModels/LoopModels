464  // can we hoist forward out of this loop?
465  // NOLINTNEXTLINE(misc-no-recursion)
466  auto hoist(BumpAlloc<> &alloc, Addr *a, unsigned currentLoop)
467  -> Optional<unsigned> {
468  if (a->wasVisited3()) {
469  if (a->getLoopTreeSchedule() == this) return a->getBlockIdx();
470  return {};
471  }
472  a->visit3();
473  // it isn't allowed to depend on deeper loops
474  auto *L = a->getLoopTreeSchedule();
475  bool placed = L != this;
476  if ((placed && (L->getParent() != this || a->dependsOnIndVars(getDepth()))))
477  return {};
478  // we're trying to hoist into either idxFront or idxBack
479  unsigned idxFront = 2 * currentLoop, idxBack = idxFront + 2;
480  bool legalHoistFront = true, legalHoistBack = true;
481  for (auto *p : a->inNeighbors(getDepth())) {
482  auto pIdx = getIdx(p);
483  if (pIdx <= idxFront) continue;
484  bool fail = pIdx >= idxBack;
485  if (!fail) {
486  if (auto hIdx = hoist(alloc, p, currentLoop))
487  fail = *hIdx > currentLoop;
488  else fail = true;
489  }
490  if (fail) {
491  legalHoistFront = false;
492  break;
493  }
494  }
495  if (legalHoistFront) {
496  if (placed) {
497  invariant(L->try_delete(a));
498  a->setLoopTreeSchedule(this);
499  }
500  a->setBlockIdx(currentLoop);
501  push_back(alloc, a, currentLoop);
502  a->place();
503  return currentLoop;
504  }
505  for (auto *c : a->outNeighbors(getDepth())) {
506  auto cIdx = getIdx(c);
507  if (cIdx >= idxBack) continue;
508  bool fail = cIdx <= idxFront;
509  if (!fail) {
510  if (auto hIdx = hoist(alloc, c, currentLoop))
511  fail = *hIdx <= currentLoop;
512  else fail = true;
513  }
514  if (fail) {
515  legalHoistBack = false;
516  break;
517  }
518  }
519  if (legalHoistBack) {
520  if (placed) {
521  invariant(L->try_delete(a));
522  a->setLoopTreeSchedule(this);
523  }
524  a->setBlockIdx(currentLoop + 1);
525  invariant(currentLoop < currentLoop + 1);
526  push_front(alloc, a, currentLoop + 1);
527  a->place();
528  return currentLoop + 1;
529  }
530  // FIXME: what if !placed, but hoist failed?
531  // what sort of ctrl path could produce that outcome?
532  invariant(placed);
533  return {};
534  }

536  // 1.
537  // go from a ptr to an index-based approach
538  // we want the efficient set operations that `BitSet` provides
539  // The plan will be to add `addr`s from previous and following loops
540  // as extra edges, so that we can ensure the SCC algorithm does
541  // not mix addrs from different loops.

548  // we sort via repeatedly calculating the strongly connected components
549  // of the address graph. The SCCs are in topological order.
550  // If a load or store are isolated within a SCC from a sub-loop, we hoist
551  // if it's indices do not depend on that loop.
552  //
553  // We will eventually insert all addr at this level and within sub-loops
554  // into `addr`. We process them in batches, iterating based on `omega`
555  // values. We only need to force separation here for those that are not
556  // already separated.
557  //

562 #endif

565  addrCounts.reserve(numSubTrees + 1);
566  addrCounts.emplace_back(numAddr);
567  for (auto &L : subTrees)
568  addrCounts.emplace_back(
569  numAddr += L.subTree->placeAddr(alloc, LB, addr[_(numAddr, end)]));
570 
571  addr = addr[_(0, numAddr)];
572  numAddr_ = numAddr;
573  for (unsigned i = 0; i < numAddr; ++i) {
574  addr[i]->addToSubset();
575  addr[i]->index() = i + numSubTrees;
576  }
577  for (auto *a : addr) a->calcAncestors(getDepth());
578 #ifndef NDEBUG
579  for (auto *a : addr[_(addrCounts.front(), addrCounts.back())])
580  invariant(a->wasPlaced());
581 #endif
582  Vector<std::array<BitSet, 2>> loopRelatives{numSubTrees};
583  if (numSubTrees) {
584  // iterate over loops
585  for (unsigned i = 0; i < subTrees.size(); ++i) {
586  calcLoopNestAddrs(
587  loopRelatives, addr[_(addrCounts[i], addrCounts[i + 1])], numAddr, i);
588  }
589  }
590  // auto headerAddrs = addr[_(0, addrCounts[0])];
591  llvm::SmallVector<BitSet> components;
592  {
593  AddressGraph g{addr, loopRelatives};
594  Graphs::stronglyConnectedComponents(components, g);
595  }
596  size_t currentLoop = 0;
597  InstructionBlock *B = &header;
598  LoopTreeSchedule *L;
599  bool inLoop = false;
600  // two passes, first to place the addrs that don't need hoisting
601  for (BitSet scc : components) {

603  if (sccsz == 1) {
604  size_t indRaw = scc.front(), ind = indRaw - numSubTrees;
605  if (ind < numAddr) {

607  // 1. inLoop && wasPlaced
608  // 2. inLoop && !wasPlaced
609  // 3. !inLoop && !wasPlaced
610  // 4. !inLoop && wasPlaced - scc hoisted
611  if (inLoop) continue;

616  // invariant(a->getLoopTreeSchedule()->try_delete(a));
617  // a->setLoopTreeSchedule(this);
618  // }
619  a->setBlockIdx(currentLoop);

621  a->place();
622  a->visit3();
623  } else {
624  // invariant((ind - numAddr) & 1, size_t(inLoop));
625  // invariant(currentLoop, (ind - numAddr) >> 1);
626  if (inLoop) B = &subTrees[currentLoop++].exit;
627  else L = subTrees[currentLoop].subTree;
628  inLoop = !inLoop;
629  }

647  // 1. inLoop && wasPlaced
