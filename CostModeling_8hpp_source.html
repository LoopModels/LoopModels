432  // can we hoist forward out of this loop?
433  // NOLINTNEXTLINE(misc-no-recursion)
434  auto hoist(BumpAlloc<> &alloc, Address *a, unsigned currentLoop)
435  -> Optional<unsigned> {
436  if (a->wasVisited3()) {
437  if (a->getLoopTreeSchedule() == this) return a->getBlockIdx();
438  return {};
439  }
440  a->visit3();
441  // it isn't allowed to depend on deeper loops
442  auto *L = a->getLoopTreeSchedule();
443  bool placed = L != this;
444  if ((placed && (L->getParent() != this || a->dependsOnIndVars(getDepth()))))
445  return {};
446  // we're trying to hoist into either idxFront or idxBack
447  unsigned idxFront = 2 * currentLoop, idxBack = idxFront + 2;
448  bool legalHoistFront = true, legalHoistBack = true;
449  for (auto *p : a->inNeighbors(getDepth())) {
450  auto pIdx = getIdx(p);
451  if (pIdx <= idxFront) continue;
452  bool fail = pIdx >= idxBack;
453  if (!fail) {
454  if (auto hIdx = hoist(alloc, p, currentLoop))
455  fail = *hIdx > currentLoop;
456  else fail = true;
457  }
458  if (fail) {
459  legalHoistFront = false;
460  break;
461  }
462  }
463  if (legalHoistFront) {
464  if (placed) {
465  invariant(L->try_delete(a));
466  a->setLoopTreeSchedule(this);
467  }
468  a->setBlockIdx(currentLoop);
469  push_back(alloc, a, currentLoop);
470  a->place();
471  return currentLoop;
472  }
473  for (auto *c : a->outNeighbors(getDepth())) {
474  auto cIdx = getIdx(c);
475  if (cIdx >= idxBack) continue;
476  bool fail = cIdx <= idxFront;
477  if (!fail) {
478  if (auto hIdx = hoist(alloc, c, currentLoop))
479  fail = *hIdx <= currentLoop;
480  else fail = true;
481  }
482  if (fail) {
483  legalHoistBack = false;
484  break;
485  }
486  }
487  if (legalHoistBack) {
488  if (placed) {
489  invariant(L->try_delete(a));
490  a->setLoopTreeSchedule(this);
491  }
492  a->setBlockIdx(currentLoop + 1);
493  invariant(currentLoop < currentLoop + 1);
494  push_front(alloc, a, currentLoop + 1);
495  a->place();
496  return currentLoop + 1;
497  }
498  // FIXME: what if !placed, but hoist failed?
499  // what sort of ctrl path could produce that outcome?
500  invariant(placed);
501  return {};
502  }

504  // 1.

524  // already separated.

537  numAddr += L.subTree->placeAddr(alloc, LB, addr[_(numAddr, end)]));

542  addr[i]->addToSubset();
543  addr[i]->index() = i + numSubTrees;
544  }

546 #ifndef NDEBUG

549 #endif

610  if (sccsz == 1) {

615  // 1. inLoop && wasPlaced
616  // 2. inLoop && !wasPlaced

622  // Here, we want a recursive approach

626  // visited before each search

630  invariant(hoist(alloc, a, currentLoop).hasValue() ||

632  }
