51  llvm::TargetTransformInfo &TTI)
52  -> uint8_t {
53  uint8_t twiceMaxVectorWidth = 2;
54  auto *f32 = llvm::Type::getFloatTy(C);
55  llvm::InstructionCost prevCost = TTI.getArithmeticInstrCost(
56  llvm::Instruction::FAdd,
57  llvm::FixedVectorType::get(f32, twiceMaxVectorWidth));
58  while (true) {

60  llvm::Instruction::FAdd,
61  llvm::FixedVectorType::get(f32, twiceMaxVectorWidth *= 2));
62  if (nextCost > prevCost) break;
63  prevCost = nextCost;
64  }
65  return 16 * twiceMaxVectorWidth;
