91  constexpr SmallSparseMatrix(Row numRows, Col numCols)
92  : rows{llvm::SmallVector<uint32_t>(size_t(numRows))}, col{numCols} {
93  invariant(size_t(col) <= maxElemPerRow);
94  }
95  constexpr auto get(Row i, Col j) const -> T {
96  invariant(j < col);
97  uint32_t r(rows[size_t(i)]);
98  uint32_t jshift = uint32_t(1) << uint32_t(j);
99  if (!(r & jshift)) return T{};
100  // offset from previous rows
101  uint32_t prevRowOffset = r >> maxElemPerRow;
102  uint32_t rowOffset = std::popcount(r & (jshift - 1));
103  return nonZeros[rowOffset + prevRowOffset];
104  }

106  return get(Row{i}, Col{j});
107  }
108  constexpr void insert(T x, Row i, Col j) {
109  invariant(j < col);
110  uint32_t r{rows[size_t(i)]};
111  uint32_t jshift = uint32_t(1) << size_t(j);
112  // offset from previous rows
113  uint32_t prevRowOffset = r >> maxElemPerRow;
114  uint32_t rowOffset = std::popcount(r & (jshift - 1));
115  size_t k = rowOffset + prevRowOffset;
116  if (r & jshift) {
117  nonZeros[k] = std::move(x);
118  } else {
119  nonZeros.insert(nonZeros.begin() + k, std::move(x));
120  rows[size_t(i)] = r | jshift;
121  for (size_t l = size_t(i) + 1; l < rows.size(); ++l)
122  rows[l] += uint32_t(1) << maxElemPerRow;
123  }
