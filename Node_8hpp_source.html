101  // we have a private pointer so different types can share
102  // in manner not exacctly congruent with type hiearchy
103  // in particular, `Inst` and `Load` want `User` lists
104  // while `Stow`s do not.
105  // `Addr` is the common load/store subtype
106  // So in some sense, we want both `Load` and `Store` to inherit from `Addr`,
107  // but only load to inherit 'hasUsers' and only store to inherit the operand.
108  // `Inst` would also inherit 'hasUsers', but would want a different operands
109  // type.
110  // Addr has a FAM, so multiple inheritence isn't an option for `Load`/`Stow`,
111  // and we want a common base that we can query to avoid monomorphization.

380  struct Identifier {
381  ValKind kind;
382  llvm::Type *typ;
383  union {
384  int64_t i;
385  double f;
386  const llvm::APInt *ci;
387  const llvm::APFloat *cf;
388  } payload;
389  constexpr auto operator==(const Identifier &o) const -> bool {
390  if (kind != o.kind || typ != o.typ) return false;
391  switch (kind) {
392  case VK_Cint: return payload.i == o.payload.i;
393  case VK_Cflt: return payload.f == o.payload.f;
394  case VK_Bint: return *payload.ci == *o.payload.ci;
395  default: invariant(kind == VK_Bflt); return *payload.cf == *o.payload.cf;
396  }
397  }
398  constexpr Identifier(llvm::Type *t, int64_t i)
399  : kind(VK_Cint), typ(t), payload(i){};
400  constexpr Identifier(llvm::Type *t, double f) : kind(VK_Cflt), typ(t) {
401  payload.f = f;
402  };
403  constexpr Identifier(llvm::Type *t, const llvm::APInt &i)
404  : kind(VK_Bint), typ(t) {
405  payload.ci = &i;
406  };
407  constexpr Identifier(llvm::Type *t, const llvm::APFloat &f)
408  : kind(VK_Bflt), typ(t) {
409  payload.cf = &f;
410  }
411  };
