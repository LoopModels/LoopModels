148  constexpr void pruneBoundsCore(BumpAlloc<> &alloc) {
149  auto diff = vector<int64_t>(alloc, unsigned(getA().numCol()));
150  auto p = checkpoint(alloc);
151  const size_t dyn = getNumDynamic();
152  if constexpr (HasEqualities) {
153  auto [ar, er] = removeRedundantRows(getA(), getE());
154  setNumConstraints(unsigned(ar));
155  setNumEqConstraints(unsigned(er));
156  for (size_t i = 0; i < getNumEqualityConstraints(); ++i) {
157  auto l = gcd(getE()(i, _));
158  if (l != 1) getE()(i, _) /= l;
159  }
160  }
161  auto C = initializeComparator(alloc);
162  if constexpr (CheckEmpty) {
163  if (C.isEmpty(alloc)) {
164  setNumConstraints(0);
165  if constexpr (HasEqualities) setNumEqConstraints(0);
166  return;
167  }
168  }
169  for (auto j = getNumCon(); j;) {
170  bool broke = false;
171  for (auto i = --j; i;) {
172  if (getNumCon() <= 1) return;
173  diff << getA()(--i, _) - getA()(j, _);
174  if (C.greaterEqual(alloc, diff)) {
175  eraseConstraint(i);
176  rollback(alloc, p);
177  C = initializeComparator(alloc);
178  --j; // `i < j`, and `i` has been removed
179  } else if (diff *= -1; C.greaterEqual(alloc, diff)) {
180  eraseConstraint(j);
181  rollback(alloc, p);
182  C = initializeComparator(alloc);
183  broke = true;
184  break; // `j` is gone
185  }
186  }
187  if constexpr (NonNegative) {
188  if (!broke) {
189  for (size_t i = 0; i < dyn; ++i) {
190  diff << getA()(j, _);
191  --diff[last - i];
192  if (C.greaterEqual(alloc, diff)) {
193  eraseConstraint(j);
194  rollback(alloc, p);
195  C = initializeComparator(alloc);
196  break; // `j` is gone
