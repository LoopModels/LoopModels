807  const ArrayReference &refOut = out->ref;
808  size_t numLoopsIn = refIn.getNumLoops();
809  size_t numLoopsOut = refOut.getNumLoops();
810  size_t numLoopsCommon = std::min(numLoopsIn, numLoopsOut);
811  size_t numLoopsTotal = numLoopsIn + numLoopsOut;
812  Vector<int64_t> schv;
813  schv.resizeForOverwrite(dependenceSatisfaction.getNumVar());
814  const SquarePtrMatrix<int64_t> inPhi = schIn.getPhi();
815  const SquarePtrMatrix<int64_t> outPhi = schOut.getPhi();
816  llvm::ArrayRef<int64_t> inFusOmega = schIn.getFusionOmega();
817  llvm::ArrayRef<int64_t> outFusOmega = schOut.getFusionOmega();
818  llvm::ArrayRef<int64_t> inOffOmega = schIn.getOffsetOmega();
819  llvm::ArrayRef<int64_t> outOffOmega = schOut.getOffsetOmega();
820  const size_t numLambda = getNumLambda();
821  // when i == numLoopsCommon, we've passed the last loop
822  for (size_t i = 0; i <= numLoopsCommon; ++i) {
823  if (int64_t o2idiff = outFusOmega[i] - inFusOmega[i])
824  return (o2idiff > 0);
825  // we should not be able to reach `numLoopsCommon`
826  // because at the very latest, this last schedule value
827  // should be different, because either:
828  // if (numLoopsX == numLoopsY){
829  // we're at the inner most loop, where one of the instructions
830  // must have appeared before the other.
831  // } else {
832  // the loop nests differ in depth, in which case the deeper
833  // loop must appear either above or below the instructions
834  // present at that level
835  // }
836  assert(i != numLoopsCommon);
837  schv(_(begin, numLoopsIn)) = inPhi(i, _);
838  schv(_(numLoopsIn, numLoopsTotal)) = outPhi(i, _);
839  int64_t inO = inOffOmega[i], outO = outOffOmega[i];
840  // forward means offset is 2nd - 1st

842  // dependenceSatisfaction is phi_t - phi_s >= 0
843  // dependenceBounding is w + u'N - (phi_t - phi_s) >= 0

847  // if zerod-out bounding not >= 0, then that means

850  }
