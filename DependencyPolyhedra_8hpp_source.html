137  DependencePolyhedra(const MemoryAccess &ma0, const MemoryAccess &ma1)
138  : SymbolicEqPolyhedra{} {
139 
140  assert(ma0.sizesMatch(ma1));
141  NotNull<AffineLoopNest<>> loop0 = ma0.getLoop();
142  NotNull<AffineLoopNest<>> loop1 = ma1.getLoop();
143  auto [nc0, nv0] = loop0->A.size();
144  auto [nc1, nv1] = loop1->A.size();
145  numDep0Var = loop0->getNumLoops();
146  size_t numDep1Var = loop1->getNumLoops();
147  size_t numVar = numDep0Var + numDep1Var;
148  std::array<llvm::SmallVector<unsigned int>, 2> oldToNewMaps{
149  merge(loop0->S, loop1->S)};
150  auto &oldToNewMap0 = oldToNewMaps[0];
151  auto &oldToNewMap1 = oldToNewMaps[1];
152  assert(oldToNewMap0.size() == loop0->S.size());
153  assert(oldToNewMap1.size() == loop1->S.size());
154 
155  // numDep1Var = nv1;
156  const Row nc = nc0 + nc1;
157  IntMatrix NS{nullSpace(ma0, ma1)};
158  const size_t nullDim{NS.numRow()};
159  const size_t indexDim{ma0.getArrayDim()};
160  nullStep.resize_for_overwrite(nullDim);
161  for (size_t i = 0; i < nullDim; ++i) {
162  int64_t s = 0;
163  for (size_t j = 0; j < NS.numCol(); ++j) s += NS(i, j) * NS(i, j);
164  nullStep[i] = s;
165  }
166  // column meansing in in order
167  const size_t numSymbols = getNumSymbols();
168  A.resize(nc + numVar, numSymbols + numVar + nullDim);
169  E.resize(indexDim + nullDim, A.numCol());
170  // ma0 loop
171  for (size_t i = 0; i < nc0; ++i) {
172  A(i, 0) = loop0->A(i, 0);
173  for (size_t j = 0; j < oldToNewMap0.size(); ++j)
174  A(i, 1 + oldToNewMap0[j]) = loop0->A(i, 1 + j);
175  for (size_t j = 0; j < numDep0Var; ++j)
176  A(i, j + numSymbols) = loop0->A(i, j + loop0->getNumSymbols());
177  }
178  for (size_t i = 0; i < nc1; ++i) {
179  A(nc0 + i, 0) = loop1->A(i, 0);
180  for (size_t j = 0; j < oldToNewMap1.size(); ++j)
181  A(nc0 + i, 1 + oldToNewMap1[j]) = loop1->A(i, 1 + j);
182  for (size_t j = 0; j < numDep1Var; ++j)
183  A(nc0 + i, j + numSymbols + numDep0Var) =
184  loop1->A(i, j + loop1->getNumSymbols());
185  }
186  A(_(nc, end), _(numSymbols, numSymbols + numVar)).diag() = 1;
187  // L254: Assertion `col < numCol()` failed
188  // indMats are [innerMostLoop, ..., outerMostLoop] x arrayDim
189  // offsetMats are arrayDim x numSymbols
190  PtrMatrix<int64_t> A0 = ma0.indexMatrix();
191  PtrMatrix<int64_t> A1 = ma1.indexMatrix();
192  PtrMatrix<int64_t> O0 = ma0.offsetMatrix();
193  PtrMatrix<int64_t> O1 = ma1.offsetMatrix();
194  // E(i,:)* indVars = q[i]
195  // e.g. i_0 + j_0 + off_0 = i_1 + j_1 + off_1
196  // i_0 + j_0 - i_1 - j_1 = off_1 - off_0
197  for (size_t i = 0; i < indexDim; ++i) {
198  E(i, 0) = O0(i, 0);
199  for (size_t j = 0; j < O0.numCol() - 1; ++j)
200  E(i, 1 + oldToNewMap0[j]) = O0(i, 1 + j);
201  for (size_t j = 0; j < numDep0Var; ++j) E(i, j + numSymbols) = A0(j, i);
202  E(i, 0) -= O1(i, 0);
203  for (size_t j = 0; j < O1.numCol() - 1; ++j)
204  E(i, 1 + oldToNewMap1[j]) -= O1(i, 1 + j);
205  for (size_t j = 0; j < numDep1Var; ++j)
206  E(i, j + numSymbols + numDep0Var) = -A1(j, i);
207  }
208  for (size_t i = 0; i < nullDim; ++i) {
209  for (size_t j = 0; j < NS.numCol(); ++j) {
210  int64_t nsij = NS(i, j);
211  E(indexDim + i, j + numSymbols) = nsij;
212  E(indexDim + i, j + numSymbols + numDep0Var) = -nsij;
213  }
214  E(indexDim + i, numSymbols + numDep0Var + numDep1Var + i) = 1;
215  }
216  initializeComparator();
217  pruneBounds();

236  [[nodiscard]] auto farkasPair() const -> std::array<Simplex, 2> {
237 
238  const size_t numEqualityConstraintsOld = size_t(E.numRow());
239  const size_t numInequalityConstraintsOld = size_t(A.numRow());
240 
241  const size_t numPhiCoefs = getNumPhiCoefficients();
242  const size_t numScheduleCoefs = numPhiCoefs + getNumOmegaCoefficients();
243  const size_t numBoundingCoefs = getNumSymbols();
244 
245  const size_t numConstraintsNew = size_t(A.numCol()) - getTimeDim();
246  const size_t numVarInterest = numScheduleCoefs + numBoundingCoefs;
247 
248  // lambda_0 + lambda'*A*i == psi'i
249  // we represent equal constraint as
250  // lambda_0 + lambda'*A*i - psi'i == 0
251  // lambda_0 + (lambda'*A* - psi')i == 0
252  // forward (0 -> 1, i.e. 1 >= 0):
253  // psi'i = Phi_1'i_1 - Phi_0'i_0
254  // backward (1 -> 0, i.e. 0 >= 1):
255  // psi'i = Phi_0'i_0 - Phi_1'i_1
256  // first, lambda_0:
257  const size_t ineqEnd = 1 + numInequalityConstraintsOld;
258  const size_t posEqEnd = ineqEnd + numEqualityConstraintsOld;
259  const size_t numLambda = posEqEnd + numEqualityConstraintsOld;
260  const size_t numVarNew = numVarInterest + numLambda;
261  assert(getNumLambda() == numLambda);
262  std::array<Simplex, 2> pair;
263  Simplex &fw(pair[0]);
264  fw.resize(numConstraintsNew, numVarNew + 1);
265  MutPtrMatrix<int64_t> fC{fw.getConstraints()(_, _(1, end))};
266  fC(_, 0) = 0;
267  fC(0, 0) = 1; // lambda_0
268  fC(_, _(1, 1 + numInequalityConstraintsOld)) =
269  A(_, _(begin, numConstraintsNew)).transpose();
270  // fC(_, _(ineqEnd, posEqEnd)) = E.transpose();
271  // fC(_, _(posEqEnd, numVarNew)) = -E.transpose();
272  // loading from `E` is expensive
273  // NOTE: if optimizing expression templates, should also
274  // go through and optimize loops like this
275  for (size_t j = 0; j < numConstraintsNew; ++j) {
276  for (size_t i = 0; i < numEqualityConstraintsOld; ++i) {
277  int64_t Eji = E(i, j);
278  fC(j, i + ineqEnd) = Eji;
279  fC(j, i + posEqEnd) = -Eji;
280  }
281  }
282  // schedule
283  // direction = true (aka forward=true)
284  // mean x -> y, hence schedule y - schedule x >= 0
285  //
286  // if direction==true (corresponds to forward==true),
287  // [numDep0Var...numVar) - [0...numDep0Var) + offset
288  // else
289  // [0...numDep0Var) - [numDep0Var...numVar) - offset
290  // aka, we have
291  // if direction
292  // lambda_0 + lambda' * (b - A*i) + [0...numDep0Var) -
293  // [numDep0Var...numVar) - offset == 0
294  // else
295  // lambda_0 + lambda' * (b - A*i) - [0...numDep0Var) +
296  // [numDep0Var...numVar) + offset == 0
297  //
298  // if (direction==true & boundAbove == false){
299  // sign = 1
300  // } else {
301  // sign = -1
302  // }
303  //
304  // boundAbove means we have
305  // ... == w + u'*N + psi
306  // -1 as we flip sign
307  for (size_t i = 0; i < numBoundingCoefs; ++i)
308  fC(i, i + numScheduleCoefs + numLambda) = -1;
309 
310  // so far, both have been identical
311  Simplex &bw(pair[1]);
312  bw.resize(numConstraintsNew, numVarNew + 1);
313  MutPtrMatrix<int64_t> bC{bw.getConstraints()(_, _(1, end))};
314 
315  bC(_, _(begin, numVarNew)) = PtrMatrix<int64_t>(fC(_, _(begin, numVarNew)));
316  // for (size_t i = 0; i < numConstraintsNew; ++i)
317  // for (size_t j = 0; j < numVarNew; ++j)
318  // bC(i, j) = fC(i, j);
319 
320  // equality constraints get expanded into two inequalities
321  // a == 0 ->
322  // even row: a <= 0
323  // odd row: -a <= 0
324  // fw means x'Al = x'(depVar1 - depVar0)
325  // x'Al + x'(depVar0 - depVar1) = 0
326  // so, for fw, depVar0 is positive and depVar1 is negative
327  // note that we order the coefficients inner->outer
328  // so that the ILP minimizing coefficients
329  // will tend to preserve the initial order (which is
330  // probably better than tending to reverse the initial order).
331  for (size_t i = 0; i < numPhiCoefs; ++i) {
332  int64_t s = (2 * (i < numDep0Var) - 1);
333  fC(i + numBoundingCoefs, i + numLambda) = s;
334  bC(i + numBoundingCoefs, i + numLambda) = -s;
335  }
336  // for (size_t i = 0; i < numDep0Var; ++i) {
337  // fC(numDep0Var - 1 - i + numBoundingCoefs, i + numLambda) = 1;
338  // bC(numDep0Var - 1 - i + numBoundingCoefs, i + numLambda) = -1;
339  // }
340  // for (size_t i = 0; i < numPhiCoefs - numDep0Var; ++i) {
341  // fC(numPhiCoefs - 1 - i + numBoundingCoefs,
342  // i + numDep0Var + numLambda) = -1;
343  // bC(numPhiCoefs - 1 - i + numBoundingCoefs,
344  // i + numDep0Var + numLambda) = 1;
345  // }
346  fC(0, numScheduleCoefs - 2 + numLambda) = 1;
347  fC(0, numScheduleCoefs - 1 + numLambda) = -1;
348  bC(0, numScheduleCoefs - 2 + numLambda) = -1;
349  bC(0, numScheduleCoefs - 1 + numLambda) = 1;
350  // note that delta/constant coef is handled as last `s`
351  return pair;

712  [[nodiscard]] auto isSatisfied(const Schedule &schIn,
713  const Schedule &schOut) const -> bool {
714  size_t numLoopsIn = in->getNumLoops();
715  size_t numLoopsOut = out->getNumLoops();
716  size_t numLoopsCommon = std::min(numLoopsIn, numLoopsOut);
717  size_t numLoopsTotal = numLoopsIn + numLoopsOut;
718  Vector<int64_t> schv;
719  schv.resizeForOverwrite(dependenceSatisfaction.getNumVar());
720  const SquarePtrMatrix<int64_t> inPhi = schIn.getPhi();
721  const SquarePtrMatrix<int64_t> outPhi = schOut.getPhi();
722  llvm::ArrayRef<int64_t> inFusOmega = schIn.getFusionOmega();
723  llvm::ArrayRef<int64_t> outFusOmega = schOut.getFusionOmega();
724  llvm::ArrayRef<int64_t> inOffOmega = schIn.getOffsetOmega();
725  llvm::ArrayRef<int64_t> outOffOmega = schOut.getOffsetOmega();
726  const size_t numLambda = getNumLambda();
727  // when i == numLoopsCommon, we've passed the last loop
728  for (size_t i = 0; i <= numLoopsCommon; ++i) {
729  if (int64_t o2idiff = outFusOmega[i] - inFusOmega[i])
730  return (o2idiff > 0);
731  // we should not be able to reach `numLoopsCommon`
732  // because at the very latest, this last schedule value
733  // should be different, because either:
734  // if (numLoopsX == numLoopsY){
735  // we're at the inner most loop, where one of the instructions
736  // must have appeared before the other.
737  // } else {
738  // the loop nests differ in depth, in which case the deeper
739  // loop must appear either above or below the instructions
740  // present at that level
741  // }
742  assert(i != numLoopsCommon);
743  schv[_(begin, numLoopsIn)] = inPhi(last - i, _);
744  schv[_(numLoopsIn, numLoopsTotal)] = outPhi(last - i, _);
745  int64_t inO = inOffOmega[i], outO = outOffOmega[i];
746  // forward means offset is 2nd - 1st
747  schv[numLoopsTotal] = outO - inO;
748  // dependenceSatisfaction is phi_t - phi_s >= 0
749  // dependenceBounding is w + u'N - (phi_t - phi_s) >= 0
750  // we implicitly 0-out `w` and `u` here,
751  if (dependenceSatisfaction.unSatisfiable(schv, numLambda) ||
752  dependenceBounding.unSatisfiable(schv, numLambda))
753  // if zerod-out bounding not >= 0, then that means
754  // phi_t - phi_s > 0, so the dependence is satisfied
755  return false;
756  }
757  return true;
758  }
