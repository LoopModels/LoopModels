314  S oz = this->sz;
315  this->sz = nz;
316  if constexpr (std::integral<S>) {
317  invariant(size_t(nz) <= size_t(oz));
318  } else if constexpr (std::is_same_v<S, StridedDims>) {
319  invariant(nz.row() <= oz.row());
320  invariant(nz.col() <= oz.col());
321  } else {
322  static_assert(MatrixDimension<S>, "Can only resize 1 or 2d containers.");
323  auto newX = unsigned{RowStride{nz}}, oldX = unsigned{RowStride{oz}},
324  newN = unsigned{Col{nz}}, oldN = unsigned{Col{oz}},
325  newM = unsigned{Row{nz}}, oldM = unsigned{Row{oz}};
326  invariant(newM <= oldM);
327  invariant(newN <= oldN);
328  invariant(newX <= oldX);
329  unsigned colsToCopy = newN;

332  // if we're in place, we have 1 less row to copy

338  src += oldX;
339  dst += newX;
340  std::copy_n(src, colsToCopy, dst);

342  }
