165  auto createCondition(BumpAlloc<> &alloc, Predicate::Intersection pred,
166  bool swap) -> Intr * {
167  size_t popCount = pred.popCount();
168 
169  if (popCount == 0) return getConstant(alloc, predicates[0]->getType(), 1);
170  if (popCount == 1) {
171  size_t ind = pred.getFirstIndex();
172  Intr *J = predicates[ind];
173  return swap ? J->negate(alloc, *this) : J;
174  }
175  // we have more than one instruction
176  auto And = Intrinsic(Intrinsic::OpCode{llvm::Instruction::And});
177  size_t ind = pred.getFirstIndex();
178  Intr *J = predicates[ind];
179  ind = pred.getNextIndex(ind);
180  // we keep I &= predicates[ind] until ind is invalid
181  // ind will be >= 32 when it is invalid
182  // getNextIndex will return a valid answer at least once, because
183  // popCount > 1
184  // there may be a better order than folding from the left
185  // e.g. a binary tree could allow for more out of order execution
186  // but I think a later pass should handle that sort of associativity
187  do {
188  J = getInstruction(alloc, And, J, predicates[ind], J->getType());
189  ind = pred.getNextIndex(ind);
190  } while (ind < 32);
191  return J;
192  }
