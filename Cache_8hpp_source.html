113  auto complete(Compute *I, llvm::Loop *L, TreeResult tr)
114  -> std::pair<Compute *, TreeResult> {
115  auto *i = I->getLLVMInstruction();

119  auto *op = i->getOperand(j);
120  auto [v, tret] = getValue(op, L, tr);

139  constexpr void replaceUsesByUsers(Value *oldNode, Value *newNode) {
140  invariant(oldNode->getKind() == Node::VK_Load ||
141  oldNode->getKind() >= Node::VK_Func);
142  oldNode->getUsers()->forEach([=, this](Instruction *user) {
143  // newNode may depend on old (e.g. when merging)
144  if (user == newNode) return;
145  if (auto *I = llvm::dyn_cast<Compute>(user)) {
146  for (Value *&o : I->getOperands())
147  if (o == oldNode) o = newNode;
148  user = cse(I); // operands changed, try to CSE
149  } else {
150  Addr *addr = llvm::cast<Addr>(user);
151  // could be load or store; either are predicated
152  // we might have `if (b) store(b)`, hence need both checks?
153  bool isPred = addr->getPredicate() == oldNode,
154  isStored = addr->isStore() && addr->getStoredVal() == oldNode;
155  invariant(isPred || isStored); // at least one must be true!
156  if (isPred) addr->setPredicate(newNode);
157  if (isStored) addr->setVal(newNode);
158  }
159  if (newNode->getKind() != Node::VK_Stow) newNode->addUser(alloc, user);
160  });
161  if (freeListList) freeListList->append(oldNode->getUsers());
162  else freeListList = oldNode->getUsers();
163  oldNode->setUsers(nullptr);
164  }

176  static auto blackListAllDependentLoops(const llvm::SCEV *S) -> uint64_t {
177  uint64_t flag{0};
178  if (const auto *x = llvm::dyn_cast<const llvm::SCEVNAryExpr>(S)) {
179  if (const auto *y = llvm::dyn_cast<const llvm::SCEVAddRecExpr>(x))
180  flag |= uint64_t(1) << y->getLoop()->getLoopDepth();
181  for (size_t i = 0; i < x->getNumOperands(); ++i)
182  flag |= blackListAllDependentLoops(x->getOperand(i));
183  } else if (const auto *c = llvm::dyn_cast<const llvm::SCEVCastExpr>(S)) {
184  for (size_t i = 0; i < c->getNumOperands(); ++i)
185  flag |= blackListAllDependentLoops(c->getOperand(i));
186  return flag;
187  } else if (const auto *d = llvm::dyn_cast<const llvm::SCEVUDivExpr>(S)) {
188  for (size_t i = 0; i < d->getNumOperands(); ++i)
189  flag |= blackListAllDependentLoops(d->getOperand(i));
190  return flag;
191  }
192  return flag;
193  }
194  static auto blackListAllDependentLoops(const llvm::SCEV *S, size_t numPeeled)
195  -> uint64_t {
196  return blackListAllDependentLoops(S) >> (numPeeled + 1);

199  auto // NOLINTNEXTLINE(misc-no-recursion)
200  fillAffineIndices(MutPtrVector<int64_t> v, int64_t *coffset,
201  math::Vector<int64_t> &offsets,
202  llvm::SmallVector<const llvm::SCEV *, 3> &symbolicOffsets,
203  const llvm::SCEV *S, int64_t mlt, size_t numPeeled)
204  -> uint64_t {
205  using ::poly::poly::getConstantInt;
206  uint64_t blackList{0};
207  if (const auto *x = llvm::dyn_cast<const llvm::SCEVAddRecExpr>(S)) {
208  const llvm::Loop *L = x->getLoop();
209  size_t depth = L->getLoopDepth();
210  if (depth <= numPeeled) {
211  // we effectively have an offset
212  // we'll add an
213  addSymbolic(offsets, symbolicOffsets, S, 1);
214  for (size_t i = 1; i < x->getNumOperands(); ++i)
215  blackList |= blackListAllDependentLoops(x->getOperand(i));
216 
217  return blackList;
218  }
219  // outermost loop has loopInd 0
220  ptrdiff_t loopInd = ptrdiff_t(depth) - ptrdiff_t(numPeeled + 1);
221  if (x->isAffine()) {
222  if (loopInd >= 0) {
223  if (auto c = getConstantInt(x->getOperand(1))) {
224  // we want the innermost loop to have index 0
225  v[loopInd] += *c;
226  return fillAffineIndices(v, coffset, offsets, symbolicOffsets,
227  x->getOperand(0), mlt, numPeeled);
228  }
229  blackList |= (uint64_t(1) << uint64_t(loopInd));
230  }
231  // we separate out the addition
232  // the multiplication was either peeled or involved
233  // non-const multiple
234  blackList |= fillAffineIndices(v, coffset, offsets, symbolicOffsets,
235  x->getOperand(0), mlt, numPeeled);
236  // and then add just the multiple here as a symbolic offset
237  const llvm::SCEV *addRec = SE->getAddRecExpr(
238  SE->getZero(x->getOperand(0)->getType()), x->getOperand(1),
239  x->getLoop(), x->getNoWrapFlags());
240  addSymbolic(offsets, symbolicOffsets, addRec, mlt);
241  return blackList;
242  }
243  if (loopInd >= 0) blackList |= (uint64_t(1) << uint64_t(loopInd));
244  } else if (std::optional<int64_t> c = getConstantInt(S)) {
245  *coffset += *c;
246  return 0;
247  } else if (const auto *ar = llvm::dyn_cast<const llvm::SCEVAddExpr>(S)) {
248  return fillAffineIndices(v, coffset, offsets, symbolicOffsets,
249  ar->getOperand(0), mlt, numPeeled) |
250  fillAffineIndices(v, coffset, offsets, symbolicOffsets,
251  ar->getOperand(1), mlt, numPeeled);
252  } else if (const auto *m = llvm::dyn_cast<const llvm::SCEVMulExpr>(S)) {
253  if (auto op0 = getConstantInt(m->getOperand(0))) {
254  return fillAffineIndices(v, coffset, offsets, symbolicOffsets,
255  m->getOperand(1), mlt * (*op0), numPeeled);
256  }
257  if (auto op1 = getConstantInt(m->getOperand(1))) {
258  return fillAffineIndices(v, coffset, offsets, symbolicOffsets,
259  m->getOperand(0), mlt * (*op1), numPeeled);
260  }
261  } else if (const auto *ca = llvm::dyn_cast<llvm::SCEVCastExpr>(S))
262  return fillAffineIndices(v, coffset, offsets, symbolicOffsets,
263  ca->getOperand(0), mlt, numPeeled);
264  addSymbolic(offsets, symbolicOffsets, S, mlt);
265  return blackList | blackListAllDependentLoops(S, numPeeled);
266  }

301  constexpr void replaceAllUsesWith(Instruction *oldNode, Value *newNode) {
302  invariant(oldNode->getKind() == Node::VK_Load ||
303  oldNode->getKind() >= Node::VK_Func);
304  replaceUsesByUsers(oldNode, newNode);
305  // every operand of oldNode needs their users updated
306  if (auto *I = llvm::dyn_cast<Compute>(oldNode)) {
307  for (Value *&n : I->getOperands()) n->removeFromUsers(oldNode);
308  I->setNext(freeInstList);
309  freeInstList = I;
310  } else {
311  invariant(oldNode->getKind() == Node::VK_Load);
312  if (Value *p = static_cast<Addr *>(oldNode)->getPredicate())
313  p->removeFromUsers(oldNode);
314  }

334  auto createInstruction(llvm::Instruction *I, llvm::Loop *L, TreeResult tr,
335  Value *&t) -> std::pair<Value *, TreeResult> {
336  auto *load = llvm::dyn_cast<llvm::LoadInst>(I);
337  auto *store = llvm::dyn_cast<llvm::StoreInst>(I);
338  if (!load && !store) return createCompute(I, L, tr, t);
339  auto *ptr = load ? load->getPointerOperand() : store->getPointerOperand();
340  // The loop `L` passed in is for stopping outward evaluation of deps
341  // but createArrayRef wants the loop to which the addr belongs
342  // we evaluate here, because we want to evaluate loop depth in
343  // case of failure
344  L = LI->getLoopFor(I->getParent());
345  auto ret = createArrayRef(I, L, ptr, tr);
346  t = ret.first;
347  return ret;
348  }

350  auto createCompute(llvm::Instruction *I, llvm::Loop *L, TreeResult tr,
351  Value *&t) -> std::pair<Compute *, TreeResult> {
352  auto [id, kind] = Compute::getIDKind(I);

377  auto createArrayRef(llvm::Instruction *loadOrStore, llvm::Loop *L,
378  llvm::Value *ptr, TreeResult tr)
379  -> std::pair<Value *, TreeResult> {
380  const auto *elSz = SE->getElementSize(loadOrStore);
381  const llvm::SCEV *accessFn = SE->getSCEVAtScope(ptr, L);
382  unsigned numLoops = L->getLoopDepth();
383  return createArrayRef(loadOrStore, accessFn, numLoops, elSz, tr);

385  auto createArrayRef(llvm::Instruction *loadOrStore,
386  const llvm::SCEV *accessFn, unsigned numLoops,
387  const llvm::SCEV *elSz, TreeResult tr)
388  -> std::pair<Value *, TreeResult> {
389  // https://llvm.org/doxygen/Delinearization_8cpp_source.html#l00582
390 
391  const llvm::SCEV *pb = SE->getPointerBase(accessFn);
392  const auto *arrayPtr = llvm::dyn_cast<llvm::SCEVUnknown>(pb);
393  // Do not delinearize if we cannot find the base pointer.
394  if (!arrayPtr) {
395  tr.rejectDepth = std::max(tr.rejectDepth, size_t(numLoops));
396  return {alloc.create<CVal>(loadOrStore), tr};
397  }
398  accessFn = SE->getMinusSCEV(accessFn, arrayPtr);
399  llvm::SmallVector<const llvm::SCEV *, 3> subscripts, sizes;
400  llvm::delinearize(*SE, accessFn, subscripts, sizes, elSz);
401  unsigned numDims = subscripts.size();
402  invariant(size_t(numDims), sizes.size());
403  if (numDims == 0) return {zeroDimRef(loadOrStore, arrayPtr, 0), tr};
404  unsigned numPeeled = tr.rejectDepth;
405  numLoops -= numPeeled;
406  math::IntMatrix Rt{math::StridedDims{numDims, numLoops}, 0};
407  llvm::SmallVector<const llvm::SCEV *, 3> symbolicOffsets;
408  uint64_t blackList{0};
409  math::Vector<int64_t> coffsets{unsigned(numDims), 0};
410  MutDensePtrMatrix<int64_t> offsMat{nullptr, DenseDims{numDims, 0}};
411  {
412  math::Vector<int64_t> offsets;
413  for (ptrdiff_t i = 0; i < numDims; ++i) {
414  offsets << 0;
415  blackList |=
416  fillAffineIndices(Rt(i, _), &coffsets[i], offsets, symbolicOffsets,
417  subscripts[i], 1, numPeeled);
418  if (offsets.size() > offsMat.numCol())
419  extendDensePtrMatCols(alloc, offsMat, math::Row{i},
420  math::Col{offsets.size()});
421  offsMat(i, _) << offsets;
422  }
423  }
424  size_t numExtraLoopsToPeel = 64 - std::countl_zero(blackList);
425  Addr *op = Addr::construct(alloc, arrayPtr, loadOrStore,
426  Rt(_, _(numExtraLoopsToPeel, end)),
427  {std::move(sizes), std::move(symbolicOffsets)},
428  coffsets, offsMat.data(), numLoops);
429  tr.addAddr(op);
430  tr.rejectDepth += numExtraLoopsToPeel;
431  return {op, tr};
432  }

529  auto negate(Value *I) -> Value * {
530  // first, check if its parent is a negation
531  if (auto op = Operation(I); op &&
532  op.isInstruction(llvm::Instruction::Xor) &&
533  (op.getNumOperands() == 2)) {
534  // !x where `x isa bool` is represented as `x ^ true`

545  auto createCondition(Predicate::Intersection pred, UList<Value *> *predicates,

547  size_t popCount = pred.popCount();

549  // 1: True; requires single predicate is true

551  if (popCount == 1) {
