76  constexpr void replaceUsesByUsers(Value *oldNode, Value *newNode) {
77  invariant(oldNode->getKind() == Node::VK_Load ||
78  oldNode->getKind() >= Node::VK_Func);
79  oldNode->getUsers()->forEach([=, this](Value *user) {
80  // newNode may depend on old (e.g. when merging)
81  if (user == newNode) return;
82  if (Inst *I = llvm::dyn_cast<Inst>(user)) {
83  for (Value *&o : I->getOperands())
84  if (o == oldNode) o = newNode;
85  user = cse(I); // operands changed, try to CSE
86  } else {
87  Addr *addr = llvm::cast<Addr>(user);
88  // could be load or store; either are predicated
89  // we might have `if (b) store(b)`, hence need both checks?
90  bool isPred = addr->getPredicate() == oldNode,
91  isStored = addr->isStore() && addr->getStoredVal() == oldNode;
92  invariant(isPred || isStored); // at least one must be true!
93  if (isPred) addr->setPredicate(newNode);
94  if (isStored) addr->setVal(newNode);
95  }
96  UList<Value *> *tnode = newNode->getUsers();
97  newNode->setUsers(tnode->pushUnique(alloc, user));
98  });
99  if (freeListList) freeListList->append(oldNode->getUsers());
100  else freeListList = oldNode->getUsers();
101  oldNode->setUsers(nullptr);
102  }

237  auto createCondition(Predicate::Intersection pred, UList<Value *> *predicates,
238  bool swap) -> Value * {
239  size_t popCount = pred.popCount();
240  // 0: Any; no restriction
241  // 1: True; requires single predicate is true
242  if (popCount == 0) return getConstant((*predicates)[0]->getType(), 1);
243  if (popCount == 1) {
244  ptrdiff_t ind = pred.getFirstIndex();
245  Value *J = (*predicates)[ind];
246  return swap ? negate(J) : J;
247  }
248  // we have more than one instruction
249  ptrdiff_t ind = pred.getFirstIndex();
250  Value *J = (*predicates)[ind];
251  ind = pred.getNextIndex(ind);
252  // we keep I &= predicates[ind] until ind is invalid
253  // ind will be >= 32 when it is invalid
254  // getNextIndex will return a valid answer at least once, because
255  // popCount > 1
256  // there may be a better order than folding from the left
257  // e.g. a binary tree could allow for more out of order execution
258  // but I think a later pass should handle that sort of associativity
259  do {
260  J = getOperation(llvm::Instruction::And,
261  std::array<Value *, 2>{J, (*predicates)[ind]},
262  J->getType(), J->getFastMathFlags());
263  ind = pred.getNextIndex(ind);
264  } while (ind < 32);
265  return J;
266  }
