8 [[maybe_unused]] static auto gcd(int64_t x, int64_t y) -> int64_t {
9  if (x == 0) {
10  return std::abs(y);
11  } else if (y == 0) {
12  return std::abs(x);
13  }
14  assert(x != std::numeric_limits<int64_t>::min());
15  assert(y != std::numeric_limits<int64_t>::min());
16  int64_t a = std::abs(x);
17  int64_t b = std::abs(y);
18  if ((a == 1) | (b == 1))
19  return 1;
20  int64_t az = std::countr_zero(uint64_t(x));
21  int64_t bz = std::countr_zero(uint64_t(y));
22  b >>= bz;
23  int64_t k = std::min(az, bz);
24  while (a) {
25  a >>= az;
26  int64_t d = a - b;
27  az = std::countr_zero(uint64_t(d));
28  b = std::min(a, b);
29  a = std::abs(d);
30  }
31  return b << k;
32 }

42 template <std::integral T> auto gcdx(T a, T b) -> std::tuple<T, T, T> {
43  T old_r = a;
44  T r = b;
45  T old_s = 1;
46  T s = 0;
47  T old_t = 0;
48  T t = 1;
49  while (r) {
50  T quotient = old_r / r;
51  old_r -= quotient * r;
52  old_s -= quotient * s;
53  old_t -= quotient * t;
54  std::swap(r, old_r);
55  std::swap(s, old_s);
56  std::swap(t, old_t);
57  }
58  // Solving for `t` at the end has 1 extra division, but lets us remove
59  // the `t` updates in the loop:
60  // T t = (b == 0) ? 0 : ((old_r - old_s * a) / b);
61  // For now, I'll favor forgoing the division.
62  return std::make_tuple(old_r, old_s, old_t);
63 }
