14 constexpr auto gcd(int64_t x, int64_t y) -> int64_t {
15  if (x == 0) return constexpr_abs(y);
16  if (y == 0) return constexpr_abs(x);
17  assert(x != std::numeric_limits<int64_t>::min());
18  assert(y != std::numeric_limits<int64_t>::min());
19  int64_t a = constexpr_abs(x);
20  int64_t b = constexpr_abs(y);
21  if ((a == 1) | (b == 1)) return 1;
22  int64_t az = std::countr_zero(uint64_t(x));
23  int64_t bz = std::countr_zero(uint64_t(y));
24  b >>= bz;
25  int64_t k = std::min(az, bz);

30  b = std::min(a, b);
31  a = constexpr_abs(d);
32  }
33  return b << k;
34 }

56 constexpr auto dgcdx(T a, T b)
57  -> std::array<T, 5> { // NOLINT(bugprone-easily-swappable-parameters)
58  T old_r = a;
59  T r = b;
60  T old_s = 1;
61  T s = 0;
62  T old_t = 0;
63  T t = 1;
64  while (r) {
65  T quotient = old_r / r;
66  old_r -= quotient * r;
67  old_s -= quotient * s;
68  old_t -= quotient * t;
69  std::swap(r, old_r);
70  std::swap(s, old_s);
71  std::swap(t, old_t);
72  }
73  // Solving for `t` at the end has 1 extra division, but lets us remove
74  // the `t` updates in the loop:
75  // T t = (b == 0) ? 0 : ((old_r - old_s * a) / b);
76  // For now, I'll favor forgoing the division.

78 }
