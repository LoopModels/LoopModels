246  void peelLoops(size_t numToPeel) {
247  invariant(numToPeel > 0);
248  invariant(numToPeel <= getNumLoops());
249  // we're dropping the outer-most `numToPeel` loops
250  // current memory layout:
251  // - indexMatrix (getNumLoops() x getArrayDim())
252  // - offsetMatrix (getArrayDim() x getNumSymbols())
253  // - fusionOmegas (getNumLoops()+1)
254  //
255  // indexMatrix rows are in innermost -> outermost order
256  // fusionOmegas are in outer<->inner order
257  // so we copy `offsetMatrix` `numToPeel * getArrayDim()` elements earlier
258  int64_t *p = data();
259  // When copying overlapping ranges, std::copy is appropriate when copying
260  // to the left (beginning of the destination range is outside the source
261  // range) while std::copy_backward is appropriate when copying to the
262  // right (end of the destination range is outside the source range).
263  // https://en.cppreference.com/w/cpp/algorithm/copy
264  int64_t *offOld = p + getArrayDim() * getNumLoops();
265  int64_t *fusOld = offOld + getArrayDim() * getNumSymbols();
266  int64_t *offNew = p + getArrayDim() * (getNumLoops() - numToPeel);
267  int64_t *fusNew = offNew + getArrayDim() * getNumSymbols();
268  std::copy(offOld, fusOld, offNew);
269  std::copy(fusOld + numToPeel, fusOld + getNumLoops() + 1, fusNew);
270  }
