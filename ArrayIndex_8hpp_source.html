258  void peelLoops(size_t numToPeel) {
259  invariant(numToPeel > 0);
260  invariant(numToPeel <= getNumLoops());
261  // we're dropping the outer-most `numToPeel` loops
262  // current memory layout:
263  // - indexMatrix (getNumLoops() x getArrayDim())
264  // - offsetMatrix (getArrayDim() x getNumSymbols())
265  // - fusionOmegas (getNumLoops()+1)
266  //
267  // indexMatrix rows are in innermost -> outermost order
268  // fusionOmegas are in outer<->inner order
269  // so we copy `offsetMatrix` `numToPeel * getArrayDim()` elements earlier
270  int64_t *p = data();
271  // When copying overlapping ranges, std::copy is appropriate when copying
272  // to the left (beginning of the destination range is outside the source
273  // range) while std::copy_backward is appropriate when copying to the
274  // right (end of the destination range is outside the source range).
275  // https://en.cppreference.com/w/cpp/algorithm/copy
276  int64_t *offOld = p + getArrayDim() * getNumLoops();
277  int64_t *fusOld = offOld + getArrayDim() * getNumSymbols();
278  int64_t *offNew = p + getArrayDim() * (getNumLoops() - numToPeel);
279  int64_t *fusNew = offNew + getArrayDim() * getNumSymbols();
280  std::copy(offOld, fusOld, offNew);
281  std::copy(fusOld + numToPeel, fusOld + getNumLoops() + 1, fusNew);
282  }
