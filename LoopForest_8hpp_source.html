173 [[maybe_unused]] static auto allForwardPathsReach(
174  llvm::SmallPtrSet<const llvm::BasicBlock *, 32> &visitedBBs,
175  PredicatedChain &path, llvm::ArrayRef<llvm::BasicBlock *> BBsrc,
176  llvm::BasicBlock *BBdst, llvm::Loop *L) -> bool {
177  visitedBBs.clear();
178  bool reached = false;
179  for (auto &BB : BBsrc) {
180  if (BB == BBdst) {
181  reached = true;
182  path.push_back(BB);
183  continue;
184  }
185  auto dst = allForwardPathsReach(visitedBBs, path, BB, BBdst, {}, BB, L);
186  if (dst == BBChain::reached) {
187  reached = true;
188 #ifndef NDEBUG
189  bool foundBB = false;
190  SHOWLN(BB);
191  for (auto &BBinPathFinal : path) {
192  SHOWLN(BBinPathFinal.basicBlock);
193  foundBB |= (BBinPathFinal.basicBlock == BB);
194  }
195  assert(foundBB);
196 #endif
197  // } else if (dst != BBChain::unreachable) {
198  } else if (dst == BBChain::unknown) {
199  llvm::errs() << "failed because dst was: " << dst << "\n";
200  return false;
201  }
202  }
203  path.reverse();
204  return reached;
205 }
