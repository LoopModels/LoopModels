135  llvm::SmallVector<LoopTree *> &forests,

138  llvm::ArrayRef<llvm::Loop *> subLoops,
139  llvm::BasicBlock *H, llvm::BasicBlock *E, bool anyFail)
140  -> size_t {
141  // how to avoid double counting? Probably shouldn't be an issue:
142  // can have an empty BB vector;
143  // when splitting, we're in either scenario:
144  // 1. We keep both loops but split because we don't have a direct path
145  // -- not the case here!
146  // 2. We're discarding one LoopTree; thus no duplication, give the BB to
147  // the one we don't discard.
148  //
149  // approach:
150  if (L) {
151  llvm::errs() << "Current pushBack depth = " << L->getLoopDepth()
152  << "\n";
153  SHOWLN(*L);
154  } else
155  llvm::errs() << "Current pushBack depth = toplevel\n";
156  llvm::SmallVector<LoopTree *> subForest;
157  llvm::SmallVector<PredicatedChain> paths;
158  PredicatedChain path;
159  size_t interiorDepth0 = 0;
160  llvm::SmallPtrSet<const llvm::BasicBlock *, 32> visitedBBs;
161  llvm::BasicBlock *finalStart;
162  if (size_t numSubLoops = subLoops.size()) {
163  llvm::SmallVector<llvm::BasicBlock *> exitBlocks;
164  exitBlocks.push_back(H);
165  // llvm::BasicBlock *PB = H;
166  llvm::Loop *P = nullptr;
167  for (size_t i = 0; i < numSubLoops; ++i) {
168  llvm::Loop *N = subLoops[i];
169  if (P) {
170  exitBlocks.clear();
171  // if we have a previous loop, does
172  P->getExitBlocks(exitBlocks);
173  // reach
174  // subLoops[i]->getLoopPreheader();
175  visitedBBs.clear();
176  }
177  // find back from prev exit blocks to preheader of next
178  // llvm::errs() << ""
179  SHOWLN(*N);
180  if (llvm::BranchInst *G = N->getLoopGuardBranch()) {
181  llvm::errs() << "Loop Guard:\n" << *G << "\n";
182  }
183  llvm::BasicBlock *PH = N->getLoopPreheader();
184  // exit block might == header block of next loop!
185  // equivalently, exiting block of one loop may be preheader of
186  // next! but we compare exit block with header here
187  llvm::errs() << "All BBs in *N:\n";
188  for (auto B : *N)
189  llvm::errs() << *B;
190  llvm::errs() << "\n";
191  if (((exitBlocks.size() != 1) ||
192  (N->getHeader() != exitBlocks.front())) &&
193  (!allForwardPathsReach(visitedBBs, path, exitBlocks, PH,
194  L))) {
195  llvm::errs() << "path failed for loop :" << *N << "\n";
196  P = nullptr;
197  anyFail = true;
198  split(alloc, forests, subForest, paths, subLoops, i);
199  exitBlocks.clear();
200  if (i + 1 < numSubLoops)
201  exitBlocks.push_back(
202  subLoops[i + 1]->getLoopPreheader());
203  paths.emplace_back(N->getLoopPreheader());
204  } else {
205  P = N;
206  paths.push_back(std::move(path));
207  }
208  path.clear();
209  llvm::errs()
210  << "pre-pushBack (subForest.size(),paths.size()) = ("
211  << subForest.size() << ", " << paths.size() << ")\n";
212  size_t itDepth = pushBack(alloc, forests, subForest, N, SE);
213  llvm::errs()
214  << "post-pushBack (subForest.size(),paths.size()) = ("
215  << subForest.size() << ", " << paths.size() << ")\n";
216  SHOWLN(itDepth);
217  if (itDepth == 0) {
218  llvm::errs() << "recursion failed for loop :" << *N << "\n";
219  P = nullptr;
220  anyFail = true;
221  // subForest.size() == 0 if we just hit the
222  // !allForwardPathsReach branch meaning it wouldn't need to
223  // push path However, if we didn't hit that branch, we
224  // pushed to path but not to subForest
225  assert(subForest.size() + 1 == paths.size());
226  // truncate last to drop extra blocks
227  paths.back().truncate(1);
228  split(alloc, forests, subForest, paths);
229  exitBlocks.clear();
230  if (i + 1 < numSubLoops)
231  exitBlocks.push_back(
232  subLoops[i + 1]->getLoopPreheader());
233  } else if (i == 0) {
234  interiorDepth0 = itDepth;
235  }
236  }
237  // assert(paths.size() == subForest.size());
238  // bug: anyFail == true, subForest.size() == 1, paths.size() == 2
239  if (anyFail)
240  llvm::errs()
241  << "pushBack returning 0 because anyFail == true.\n";
242  if (anyFail)
243  return invalid(alloc, forests, subForest, paths, subLoops);
244  assert(subForest.size());
245  finalStart = subLoops.back()->getExitBlock();
246  } else
247  finalStart = H;
248  llvm::errs() << "Starting second pass in pushBack\n";
249  SHOWLN(subForest.size());
250  if (subForest.size()) { // add subloops
251  AffineLoopNest<true> &subNest = subForest.front()->affineLoop;
252  SHOWLN(subNest.getNumLoops());
253  if (subNest.getNumLoops() > 1) {
254  visitedBBs.clear();
255  if (allForwardPathsReach(visitedBBs, path, finalStart, E, L)) {
256  paths.push_back(std::move(path));
257  auto *newTree = new (alloc)
258  LoopTree{L, subNest.removeInnerMost(),
259  std::move(subForest), std::move(paths)};
260  branches.push_back(newTree);
261  return ++interiorDepth0;
262  } else {
263  llvm::errs() << "No direct path from:\n"
264  << *finalStart << "\nTo:\n"
265  << *E << "\n";
266  }
267  }
268  // } else if (auto BT = SE.getBackedgeTakenCount(L)) {
269  } else if (auto BT = getBackedgeTakenCount(SE, L)) {
270  // we're at the bottom of the recursion
271  if (!llvm::isa<llvm::SCEVCouldNotCompute>(BT)) {
272  llvm::errs() << "about to add loop: " << *L
273  << "\nwith backedge taken count: " << *BT << "\n";
274  auto *BTNW = noWrapSCEV(SE, BT);
275  llvm::errs() << "after no-wrapping:\n" << *BTNW << "\n";
276  if (allForwardPathsReach(visitedBBs, path, finalStart, E, L)) {
277 
278  paths.push_back(std::move(path));
279  auto *newTree = new (alloc) LoopTree{
280  L, std::move(subForest), BTNW, SE, std::move(paths)};
281  branches.push_back(newTree);
282  return 1;
283  }
284  }
285  }
286  llvm::errs()
287  << "pushBack returning 0 because end of function reached.\nLoop: "
288  << *L << "\n";
289  SHOW(subForest.size());
290  if (subForest.size()) {
291  CSHOWLN(subForest.front()->getNumLoops());
292  } else
293  llvm::errs() << "\n";
294  return invalid(alloc, forests, subForest, paths, subLoops);
295  }
