6 // Currently, only implements top sort, and Tarjan's strongly connected
7 // components which returns the components in topological order, because we
8 // iterate over successors and push-first to components. These graphs assume IR
9 // nodes have some means of representing cycles e.g., a linked list class List{
10 // List *next;
11 // List *prev;
12 // List *nextComponent
13 // List *prevComponent
14 // public;
15 // // API methods
16 // };
17 // we could represent components
18 // A -> [B, C] -> [D, E, F] -> G -> [H, I]
19 // via this list, Let
20 // W -> (X, Y) mean W->next == X && W->nextComponent = Y
21 // and `_` means `nullptr`.
22 // `prev`s could be found by reversing the `next`s.
23 //
24 // A -> (B, _)
25 // B -> (D, C)
26 // C -> (D, B)
27 // D -> (G, E)
28 // E -> (G, F)
29 // F -> (G, D)
30 // G -> (H, _)
31 // H -> (_, I)
32 // I -> (_, H)
33 //
34 //
35 template <typename G>
36 concept AbstractPtrGraph = requires(G g, typename G::VertexType *v) {
37  { *(g.getVertices(v).begin()) } -> std::same_as<typename G::VertexType *>;
38  { g.getVertices(v) } -> std::ranges::forward_range;
39  { *(g.outNeighbors(v).begin()) } -> std::same_as<typename G::VertexType *>;
40  { g.outNeighbors(v) } -> std::ranges::forward_range;
41  { v->index() } -> std::assignable_from<unsigned>;
42  { v->lowLink() } -> std::assignable_from<unsigned>;
43  { v->onStack() } -> std::same_as<bool>;
44  { v->addToStack() };
45  { v->removeFromStack() };
46  { v->visited() } -> std::same_as<bool>;
47  { v->visit() };
48  { v->unVisit() };
49  { v->setNext(v) } -> std::same_as<typename G::VertexType *>;
50  { v->getNext() } -> std::same_as<typename G::VertexType *>;
51  { v->setNextComponent(v) } -> std::same_as<typename G::VertexType *>;
52  { v->getNextComponent() } -> std::same_as<typename G::VertexType *>;
53 };

65 template <AbstractPtrGraph G>
66 inline auto strongConnect(G g, State<vertex_t<G>> state, vertex_t<G> *v)
67  -> State<vertex_t<G>> {
68  v->index() = v->lowLink() = state.index++;
69  v->addToStack();
70  v->visit();
71 
72  state.stack = v->setNext(state.stack);
73  for (auto *w : g.outNeighbors(v))
74  if (!w->wasVisited()) {
75  state = strongConnect(g, state, w);
76  v->lowLink() = std::min(v->lowLink(), w->lowLink());
77  } else if (w->onStack()) v->lowLink() = std::min(v->lowLink(), w->index());
78  if (v->index() == v->lowLink()) {
79  vertex_t<G> *component{nullptr}, *s;
80  do {
81  s = std::exchange(state.stack, state.stack->getNext());
82  s->removeFromStack();
83  component = s->setNext(component);
84  } while (s != v);
85  state.components = component->setNextComponent(state.components);
86  }
87  return state;
88 }
