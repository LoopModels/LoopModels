100 template <typename B>
101 inline auto
102 strongConnect(AbstractIndexGraph auto &g, llvm::SmallVectorImpl<B> &components,
103  llvm::SmallVector<unsigned> &stack,
104  llvm::MutableArrayRef<std::tuple<unsigned, unsigned, bool>>
105  indexLowLinkOnStack,
106  size_t index, size_t v) -> size_t {
107  indexLowLinkOnStack[v] = std::make_tuple(index, index, true);
108  g.visit(v);
109  ++index;
110  stack.push_back(v);
111  for (auto w : g.inNeighbors(v)) {
112  if (g.wasVisited(w)) {
113  auto [wIndex, wLowLink, wOnStack] = indexLowLinkOnStack[w];
114  if (wOnStack) {
115  unsigned &vll = std::get<1>(indexLowLinkOnStack[v]);
116  vll = std::min(vll, wIndex);
117  }
118  } else { // not visited
119  strongConnect<B>(g, components, stack, indexLowLinkOnStack, index, w);
120  unsigned &vll = std::get<1>(indexLowLinkOnStack[v]);
121  vll = std::min(vll, std::get<1>(indexLowLinkOnStack[w]));
122  }
123  }
124  auto [vIndex, vLowLink, vOnStack] = indexLowLinkOnStack[v];
125  if (vIndex == vLowLink) {
126  components.emplace_back();
127  B &component = components.back();
128  unsigned w;
129  do {
130  w = stack.back();
131  stack.pop_back();
132  std::get<2>(indexLowLinkOnStack[w]) = false;
133  component.insert(w);
134  } while (w != v);
135  }
136  return index;
137 }

179 // Naive algorithm that looks like it may work to identify cycles:
180 // 0 -> 1 -> 3 -> 5
181 // \ /
182 // -> 2 -> 4 ->
183 // As we do dfs,
184 // first, we iterate down 0 -> 1, and build
185 // [0, 1, 3, 5] // all unique -> no cycle
186 // then, we iterate down 0 -> 2
187 // [0, 2, 4, 5] // all unique -> no cycle
188 // vs:
189 // 0 -> 1 -> 3 -> 0
190 // [0, 1, 3, 0] // not unique -> cycle
191 //
192 // However, it does not because dfs does not explore all possible paths, meaning
193 // it is likely to miss the cyclic paths, e.g.:
194 // 0 -> 1 -> 3 -> 5
195 // \ <-/ /
196 // -> 2 -> 4 ->
197 // [0, 1, 3, 5] // no cycle
198 // [0, 2, 4, 5] // no cycle
199 //
200 // Thus a better approach is to group a TermBundle by strongly connected
201 // components.
202 // We shall take the approach of:
203 //
204 // 1. Split graph into weakly connected components. For each wcc:
205 // 2. Prefuse these weakly connected components.
206 // 3. Group these into strongly connected components.
207 // 4. Iterate over schedules by strongly connected components.
