108 inline auto strongConnect(AbstractIndexGraph auto &g,
109  llvm::SmallVectorImpl<B> &components,
110  llvm::SmallVector<unsigned> &stack,
111  llvm::MutableArrayRef<SCC> iLLOS, unsigned index,
112  size_t v) -> unsigned {
113  iLLOS[v] = {index, index, true, true};
114  ++index;
115  stack.push_back(v);
116  for (auto w : g.inNeighbors(v)) {
117  if (iLLOS[w].visited) {
118  if (iLLOS[w].onStack)
119  iLLOS[v].lowLink = std::min(iLLOS[v].lowLink, iLLOS[w].index);
120  } else { // not visited
121  strongConnect<B>(g, components, stack, iLLOS, index, w);
122  iLLOS[v].lowLink = std::min(iLLOS[v].lowLink, iLLOS[w].lowLink);
123  }
124  }
125  if (iLLOS[v].index == iLLOS[v].lowLink) {
126  B &component = components.emplace_back();
127  unsigned w;
128  do {
129  w = stack.pop_back_val();
130  iLLOS[w].onStack = false;
131  component.insert(w);
132  } while (w != v);
133  }
134  return index;
135 }

164 strongConnect(const AbstractPtrGraph auto &g,
165  llvm::SmallVectorImpl<llvm::SmallVector<B *, 1>> &components,
166  llvm::SmallVector<B *> &stack, unsigned index, B *v) -> unsigned {
167  v->index() = v->lowLink() = index++;
168  v->setOnStack();
169  v->visit();
170  stack.push_back(v);
171  for (auto w : g.inNeighbors(v))
172  if (!w->wasVisited()) {
173  strongConnect<B>(g, components, stack, index, w);
174  v->lowLink() = std::min(v->lowLink(), w->lowLink());
175  } else if (w->onStack()) v->lowLink() = std::min(v->lowLink(), w->index());
176  if (v->index() == v->lowLink()) {
177  llvm::SmallVector<B *, 1> &component = components.emplace_back();
178  B *w;
179  do {
180  w = stack.pop_back_val();
181  w->removeFromStack();
182  component.insert(w);
183  } while (w != v);
184  }
185  return index;
186 }

226 // Naive algorithm that looks like it may work to identify cycles:
227 // 0 -> 1 -> 3 -> 5
228 // \ /
229 // -> 2 -> 4 ->
230 // As we do dfs,
231 // first, we iterate down 0 -> 1, and build
232 // [0, 1, 3, 5] // all unique -> no cycle
233 // then, we iterate down 0 -> 2
234 // [0, 2, 4, 5] // all unique -> no cycle
235 // vs:
236 // 0 -> 1 -> 3 -> 0
237 // [0, 1, 3, 0] // not unique -> cycle
238 //
239 // However, it does not because dfs does not explore all possible paths, meaning
240 // it is likely to miss the cyclic paths, e.g.:
241 // 0 -> 1 -> 3 -> 5
242 // \ <-/ /
243 // -> 2 -> 4 ->
244 // [0, 1, 3, 5] // no cycle
245 // [0, 2, 4, 5] // no cycle
246 //
247 // Thus a better approach is to group a TermBundle by strongly connected
248 // components.
249 // We shall take the approach of:
250 //
251 // 1. Split graph into weakly connected components. For each wcc:
252 // 2. Prefuse these weakly connected components.
253 // 3. Group these into strongly connected components.
254 // 4. Iterate over schedules by strongly connected components.
