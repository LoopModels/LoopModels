106 inline auto strongConnect(AbstractIndexGraph auto &g,
107  llvm::SmallVectorImpl<B> &components,
108  llvm::SmallVector<unsigned> &stack,
109  llvm::MutableArrayRef<SCC> iLLOS, unsigned index,
110  size_t v) -> unsigned {
111  iLLOS[v] = {index, index, true};
112  g.visit(v);
113  ++index;
114  stack.push_back(v);
115  for (auto w : g.inNeighbors(v)) {
116  if (g.wasVisited(w)) {
117  auto [wIndex, wLowLink, wOnStack] = iLLOS[w];
118  if (wOnStack) iLLOS[v].lowLink = std::min(iLLOS[v].lowLink, wIndex);
119  } else { // not visited
120  strongConnect<B>(g, components, stack, iLLOS, index, w);
121  iLLOS[v].lowLink = std::min(iLLOS[v].lowLink, iLLOS[w].lowLink);
122  }
123  }
124  auto [vIndex, vLowLink, vOnStack] = iLLOS[v];
125  if (vIndex == vLowLink) {
126  B &component = components.emplace_back();
127  unsigned w;
128  do {
129  w = stack.pop_back_val();
130  iLLOS[w].onStack = false;
131  component.insert(w);
132  } while (w != v);
133  }
134  return index;
135 }

162  llvm::SmallVectorImpl<llvm::SmallVector<B *, 1>> &components,
163  llvm::SmallVector<B *> &stack, unsigned index, B *v) -> unsigned {
164  v->index() = v->lowLink() = index++;
165  v->setOnStack();

167  stack.push_back(v);
168  for (auto w : g.inNeighbors(v))
169  if (!g.wasVisited(w)) {
170  strongConnect<B>(g, components, stack, index, w);
171  v->lowLink() = std::min(v->lowLink(), w->lowLink());
172  } else if (w->onStack()) v->lowLink() = std::min(v->lowLink(), w->index());
173  if (v->index() == v->lowLink()) {
174  llvm::SmallVector<B *, 1> &component = components.emplace_back();
175  B *w;
176  do {
177  w = stack.pop_back_val();
178  w->removeFromStack();
179  component.insert(w);
180  } while (w != v);
181  }
182  return index;
183 }

223 // Naive algorithm that looks like it may work to identify cycles:
224 // 0 -> 1 -> 3 -> 5
225 // \ /
226 // -> 2 -> 4 ->
227 // As we do dfs,
228 // first, we iterate down 0 -> 1, and build
229 // [0, 1, 3, 5] // all unique -> no cycle
230 // then, we iterate down 0 -> 2
231 // [0, 2, 4, 5] // all unique -> no cycle
232 // vs:
233 // 0 -> 1 -> 3 -> 0
234 // [0, 1, 3, 0] // not unique -> cycle
235 //
236 // However, it does not because dfs does not explore all possible paths, meaning
237 // it is likely to miss the cyclic paths, e.g.:
238 // 0 -> 1 -> 3 -> 5
239 // \ <-/ /
240 // -> 2 -> 4 ->
241 // [0, 1, 3, 5] // no cycle
242 // [0, 2, 4, 5] // no cycle
243 //
244 // Thus a better approach is to group a TermBundle by strongly connected
245 // components.
246 // We shall take the approach of:
247 //
248 // 1. Split graph into weakly connected components. For each wcc:
249 // 2. Prefuse these weakly connected components.
250 // 3. Group these into strongly connected components.
251 // 4. Iterate over schedules by strongly connected components.
