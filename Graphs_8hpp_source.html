76 strongConnect(AbstractGraph auto &g, llvm::SmallVector<BitSet> &components,
77  llvm::SmallVector<unsigned> &stack,
78  llvm::MutableArrayRef<std::tuple<unsigned, unsigned, bool>>
79  indexLowLinkOnStack,
80  size_t index, size_t v) {
81  indexLowLinkOnStack[v] = std::make_tuple(index, index, true);
82  g.visit(v);
83  ++index;
84  stack.push_back(v);
85  for (auto w : g.inNeighbors(v)) {
86  if (g.wasVisited(w)) {
87  auto [wIndex, wLowLink, wOnStack] = indexLowLinkOnStack[w];
88  if (wOnStack) {
89  unsigned &vll = std::get<1>(indexLowLinkOnStack[v]);
90  vll = std::min(vll, wIndex);
91  }
92  } else { // not visited
93  strongConnect(g, components, stack, indexLowLinkOnStack, index, w);
94  unsigned &vll = std::get<1>(indexLowLinkOnStack[v]);
95  vll = std::min(vll, std::get<1>(indexLowLinkOnStack[w]));
96  }
97  }
98  auto [vIndex, vLowLink, vOnStack] = indexLowLinkOnStack[v];
99  if (vIndex == vLowLink) {
100  components.emplace_back();
101  BitSet &component = components.back();
102  unsigned w;
103  do {
104  w = stack.back();
105  stack.pop_back();
106  std::get<2>(indexLowLinkOnStack[w]) = false;
107  component.insert(w);
108  } while (w != v);
109  }
110  return index;

150 // Naive algorithm that looks like it may work to identify cycles:
151 // 0 -> 1 -> 3 -> 5
152 // \ /
153 // -> 2 -> 4 ->
154 // As we do dfs,
155 // first, we iterate down 0 -> 1, and build
156 // [0, 1, 3, 5] // all unique -> no cycle
157 // then, we iterate down 0 -> 2
158 // [0, 2, 4, 5] // all unique -> no cycle
159 // vs:
160 // 0 -> 1 -> 3 -> 0
161 // [0, 1, 3, 0] // not unique -> cycle
162 //
163 // However, it does not because dfs does not explore all possible paths, meaning
164 // it is likely to miss the cyclic paths, e.g.:
165 // 0 -> 1 -> 3 -> 5
166 // \ <-/ /
167 // -> 2 -> 4 ->
168 // [0, 1, 3, 5] // no cycle
169 // [0, 2, 4, 5] // no cycle
170 //
171 // Thus a better approach is to group a TermBundle by strongly connected
172 // components.
173 // We shall take the approach of:
174 //
175 // 1. Split graph into weakly connected components. For each wcc:
176 // 2. Prefuse these weakly connected components.
177 // 3. Group these into strongly connected components.
178 // 4. Iterate over schedules by strongly connected components.
