63 // Permutation iterator using Heap's algorithm
64 // https://en.wikipedia.org/wiki/Heap%27s_algorithm
65 // This is the non-recursive variant, with the `while` loop moved
66 // into the iterator increment.
67 template <math::LinearlyIndexable V =
68  containers::TinyVector<int8_t, 15, int8_t>>
69 struct PermutationIterator {
70  V v{};
71  V c{};
72  ptrdiff_t i{1};
73  constexpr PermutationIterator(int8_t len) {
74  utils::invariant(len < 16);
75  for (int8_t j = 0; j < len; ++j) {
76  v.push_back(j);
77  c.push_back(0);
78  }
79  }
80  constexpr PermutationIterator(V v_, V c_) : v(v_), c(c_) {
81  utils::invariant(v.size() == c.size());
82  }
83  constexpr auto operator*() const -> const V & { return v; }
84  constexpr auto operator++() -> PermutationIterator & {
85  auto sz = v.size();
86  invariant(c.size() == sz);
87  while ((i < sz) && (c[i] >= i)) c[i++] = 0;
88  if (i < sz) {
89  if (i & 1) std::swap(v[c[i]], v[i]);
90  else std::swap(v[0], v[i]);
91  ++c[i];
92  i = 1;
93  }
94  return *this;
95  }
96  constexpr auto operator==(math::End) const -> bool { return i >= v.size(); }
97 };

132 struct LoopPermutations {
133  using SubPerms = containers::TinyVector<LoopSet, 15, int16_t>;
134  SubPerms subperms{};
135  // To iterate, we're imagining a nested loop, with nesting depth equal to
136  // `subperms.size()`. Each level of the loop nest uses Heap's algorithm to
137  // iterate over all permutations of the corresponding element `subperms`.
138  //
139  struct Iterator {
140  using State = containers::TinyVector<int8_t, 15, int8_t>;
141  State state{}; // `v` field in `PermutationIterator`
142  State iterator_positions{}; // `c` field in `PermutationIterator`
143  SubPerms subperms;
144  bool done{false};
145  // return `State` by value to avoid modification risk, and because it is
146  // trivially copyable
147  constexpr Iterator(SubPerms sp) : subperms(sp) {
148  for (LoopSet ls : sp) {
149  for (ptrdiff_t i : ls) {
150  invariant(i < 16);
151  state.push_back(int8_t(i));
152  iterator_positions.push_back(0);
153  }
154  }
155  }
156  constexpr auto operator*() const -> State { return state; }
157  constexpr auto operator++() -> Iterator & {
158  // lvl is the level we're incrementing. Here, 0 refers to the deepest
159  // level. If a perm is at its end, we increment to ascend.
160  if (done) return *this;
161  ptrdiff_t lvl{0}, offset{0}, nPerms = subperms.size();
162  while (true) {
163  if (++PermutationIterator<math::MutPtrVector<int8_t>>{
164  permIterator(lvl, offset)} == math::End{}) {
165  ptrdiff_t prevLvl = lvl++;
166  done = lvl == nPerms;
167  if (done) return *this;
168  offset = resetLevel(prevLvl, offset);
169  } else return *this;
170  }
171  }
172  constexpr auto operator==(math::End) const -> bool { return done; }
173 
174  private:
175  constexpr auto permIterator(ptrdiff_t lvl, ptrdiff_t offset)
176  -> PermutationIterator<math::MutPtrVector<int8_t>> {
177  ptrdiff_t L = subperms[lvl].size();
178  return {
179  math::MutPtrVector<int8_t>{state.begin() + offset, L},
180  math::MutPtrVector<int8_t>{iterator_positions.begin() + offset, L}};
181  }
182  constexpr auto resetLevel(ptrdiff_t lvl, ptrdiff_t offset) -> ptrdiff_t {
183  // when reseting the level, we don't actually need to reset the state
184  // we can use the last ending state as the initial state, iterating
185  // through its permutations from there.
186  ptrdiff_t sz = subperms[lvl].size();
187  for (ptrdiff_t i = 0; i < sz; ++i) iterator_positions[i + offset] = 0;
188  return sz + offset;
189  // for (ptrdiff_t i : subperms[lvl]) {
190  // invariant(i < 16);
191  // state[offset] = int8_t(i);
192  // iterator_positions[offset++] = 0;
193  // }
194  // return offset;
195  }
196  };
197  [[nodiscard]] constexpr auto begin() const -> Iterator { return {subperms}; }
198  static constexpr auto end() -> math::End { return {}; }
199 };
