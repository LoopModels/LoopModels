198  // remove trivially redundant constraints

202  // original number of variables
203  const size_t numVar = getNumVar();
204  MutPtrMatrix<int64_t> C{getConstraints()};
205  MutPtrVector<int64_t> basicCons{getBasicConstraints()};
206  basicCons = -2;
207  // first pass, we make sure the equalities are >= 0
208  // and we eagerly try and find columns with
209  // only a single non-0 element.
210  for (size_t c = 0; c < C.numRow(); ++c) {
211  int64_t &Ceq = C(c, 0);
212  if (Ceq >= 0) {
213  // close-open and close-close are out, open-open is in
214  for (size_t v = 1; v < numVar; ++v) {
215  if (int64_t Ccv = C(c, v)) {
216  if (((basicCons[v] == -2) && (Ccv > 0))) {
217  basicCons[v] = c;
218  } else {
219  basicCons[v] = -1;
220  }
221  }
222  }
223  } else {
224  Ceq *= -1;
225  for (size_t v = 1; v < numVar; ++v) {
226  if (int64_t Ccv = -C(c, v)) {
227  if (((basicCons[v] == -2) && (Ccv > 0))) {
228  basicCons[v] = c;
229  } else {
230  basicCons[v] = -1;
231  }
232  C(c, v) = Ccv;
233  }
234  }
235  }
236  }
237  // basicCons should now contain either `-1` or an integer >= 0
238  // indicating which row contains the only non-zero element; we'll now
239  // fill basicVars.
240  //
241  auto basicVars{getBasicVariables()};
242  basicVars = -1;
243  for (size_t v = 1; v < numVar; ++v) {
244  int64_t r = basicCons[v];
245  if (r >= 0) {
246  if (basicVars[r] == -1) {
247  basicVars[r] = v;
248  } else {
249  // this is reachable, as we could have
250  // [ 1 1 0
251  // 0 0 1 ]
252  // TODO: is their actual harm in having multiple basicCons?
253  basicCons[v] = -1;
254  }
255  }
256  }
257 #ifndef NDEBUG
258  inCanonicalForm = true;
259 #endif
260  llvm::SmallVector<unsigned> augmentVars{};
261  for (unsigned i = 0; i < basicVars.size(); ++i)
262  if (basicVars[i] == -1)
263  augmentVars.push_back(i);
264  if (augmentVars.size()) {
265  addVars(augmentVars.size()); // NOTE: invalidates all refs
266  MutPtrMatrix<int64_t> C{getConstraints()};
267  MutStridedVector<int64_t> basicVars{getBasicVariables()};
268  MutPtrVector<int64_t> basicCons{getBasicConstraints()};
269  MutPtrVector<int64_t> costs{getCost()};
270  tableau(1, _) = 0;
271  for (size_t i = 0; i < augmentVars.size(); ++i) {
272  size_t a = augmentVars[i];
273  basicVars[a] = i + numVar;
274  basicCons[i + numVar] = a;
275  C(a, numVar + i) = 1;
276  // we now zero out the implicit cost of `1`
277  costs(_(begin, numVar)) -= C(a, _(begin, numVar));
278  }
279  // false/0 means feasible
280  // true/non-zero infeasible
281  if (runCore() != 0)
282  return true;
283  for (size_t c = 0; c < C.numRow(); ++c) {
284  if (size_t(basicVars(c)) >= numVar) {
285  assert(C(c, 0) == 0);
286  assert(c == size_t(basicCons(basicVars(c))));
287  assert(C(c, basicVars(c)) >= 0);
288  // find var to make basic in its place
289  for (size_t v = numVar; v != 0;) {
290  // search for a non-basic variable (basicConstraints<0)
291  assert(v > 1);
292  if ((basicCons(--v) >= 0) || (C(c, v) == 0))
293  continue;
294  if (C(c, v) < 0)
295  C(c, _) *= -1;
296  for (size_t i = 0; i < C.numRow(); ++i)
297  if (i != size_t(c))
298  NormalForm::zeroWithRowOperation(C, i, c, v, 0);
299  basicVars[c] = v;
300  basicCons[v] = c;
301  break;
302  }
303  }
304  }
305  // all augment vars are now 0
306  truncateVars(numVar);
307  }
308  assertCanonical();
309  return false;
310  }

319  static int getLeavingVariable(MutPtrMatrix<int64_t> C,
320  size_t enteringVariable) {
321  // inits guarantee first valid is selected
322  // we need
323  int64_t n = -1;
324  int64_t d = 0;
325  int j = 0;
326  for (size_t i = 1; i < C.numRow(); ++i) {
327  int64_t Civ = C(i, enteringVariable);
328  if (Civ > 0) {
329  int64_t Ci0 = C(i, 0);
330  if (Ci0 == 0)
331  return --i;
332  assert(Ci0 > 0);
333  if ((n * Ci0) < (Civ * d)) {
334  n = Civ;
335  d = Ci0;
336  j = i;
337  }
338  }
339  }
340  return --j;
341  }
342  int64_t makeBasic(MutPtrMatrix<int64_t> C, int64_t f,
343  int enteringVariable) {
344  int leavingVariable = getLeavingVariable(C, enteringVariable);
345  if (leavingVariable == -1)
346  return 0; // unbounded
347  for (size_t i = 0; i < C.numRow(); ++i)
348  if (i != size_t(leavingVariable + 1)) {
349  int64_t m = NormalForm::zeroWithRowOperation(
350  C, i, leavingVariable + 1, enteringVariable,
351  i == 0 ? f : 0);

353  f = m;

358  basicVars[leavingVariable] = enteringVariable;

360  basicConstraints[oldBasicVar] = -1;
361  basicConstraints[enteringVariable] = leavingVariable;

642  // is it a valid solution to set the first `x.size()` variables to `x`?
643  // first, check that >= 0 constraint is satisfied
