228  initiateFeasible() -> bool {
229  tableau(0, 0) = 0;
230  // remove trivially redundant constraints
231  hermiteNormalForm();
232  // [ I; X ; b ]
233  //
234  // original number of variables
235  const auto numVar = ptrdiff_t(getNumVar());
236  MutPtrMatrix<int64_t> C{getConstraints()};
237  MutPtrVector<int64_t> basicCons{getBasicConstraints()};
238  basicCons << -2;
239  // first pass, we make sure the equalities are >= 0
240  // and we eagerly try and find columns with
241  // only a single non-0 element.
242  for (ptrdiff_t c = 0; c < C.numRow(); ++c) {
243  int64_t &Ceq = C(c, 0);
244  if (Ceq >= 0) {
245  // close-open and close-close are out, open-open is in
246  for (ptrdiff_t v = 1; v < numVar; ++v) {
247  if (int64_t Ccv = C(c, v)) {
248  if (((basicCons[v] == -2) && (Ccv > 0))) basicCons[v] = c;
249  else basicCons[v] = -1;
250  }
251  }
252  } else {
253  Ceq *= -1;
254  for (ptrdiff_t v = 1; v < numVar; ++v) {
255  if (int64_t Ccv = -C(c, v)) {
256  if (((basicCons[v] == -2) && (Ccv > 0))) basicCons[v] = c;
257  else basicCons[v] = -1;
258  C(c, v) = Ccv;
259  }
260  }
261  }
262  }
263  // basicCons should now contain either `-1` or an integer >= 0
264  // indicating which row contains the only non-zero element; we'll
265  // now fill basicVars.
266  //
267  auto basicVars{getBasicVariables()};
268  basicVars << -1;
269  for (ptrdiff_t v = 1; v < numVar; ++v) {
270  int64_t r = basicCons[v];
271  if (r >= 0) {
272  if (basicVars[r] == -1) {
273  basicVars[r] = v;
274  } else {
275  // this is reachable, as we could have
276  // [ 1 1 0
277  // 0 0 1 ]
278  // TODO: is their actual harm in having multiple
279  // basicCons?
280  basicCons[v] = -1;
281  }
282  }
283  }
284 #ifndef NDEBUG
285  inCanonicalForm = true;
286 #endif
287  llvm::SmallVector<unsigned> augmentVars{};
288  for (unsigned i = 0; i < basicVars.size(); ++i)
289  if (basicVars[i] == -1) augmentVars.push_back(i);
290  if (augmentVars.size())
291  if (removeAugmentVars(augmentVars, numVar)) return true;

297  auto removeAugmentVars(llvm::ArrayRef<unsigned> augmentVars, ptrdiff_t numVar)
298  -> bool {
299  addVars(augmentVars.size()); // NOTE: invalidates all refs
300  MutPtrMatrix<int64_t> C{getConstraints()};
301  MutStridedVector<int64_t> basicVars{getBasicVariables()};
302  MutPtrVector<int64_t> basicCons{getBasicConstraints()};
303  MutPtrVector<int64_t> costs{getCost()};
304  tableau(1, _) << 0;
305  for (ptrdiff_t i = 0; i < ptrdiff_t(augmentVars.size()); ++i) {
306  ptrdiff_t a = augmentVars[i];
307  basicVars[a] = i + numVar;
308  basicCons[i + numVar] = a;
309  C(a, numVar + i) = 1;
310  // we now zero out the implicit cost of `1`
311  costs[_(begin, numVar)] -= C(a, _(begin, numVar));
312  }
313  // false/0 means feasible
314  // true/non-zero infeasible
315  if (runCore() != 0) return true;
316  for (ptrdiff_t c = 0; c < C.numRow(); ++c) {
317  if (basicVars[c] >= numVar) {
318  assert(C(c, 0) == 0);
319  assert(c == basicCons[basicVars[c]]);
320  assert(C(c, basicVars[c]) >= 0);
321  // find var to make basic in its place
322  for (ptrdiff_t v = numVar; v != 0;) {
323  // search for a non-basic variable
324  // (basicConstraints<0)
325  assert(v > 1);
326  if ((basicCons[--v] >= 0) || (C(c, v) == 0)) continue;
327  if (C(c, v) < 0) C(c, _) *= -1;
328  for (size_t i = 0; i < C.numRow(); ++i)
329  if (i != size_t(c)) NormalForm::zeroWithRowOperation(C, i, c, v, 0);
330  basicVars[c] = v;
331  basicCons[v] = c;
332  break;
333  }
334  }
335  }
336  // all augment vars are now 0
337  truncateVars(numVar);
338  return false;
339  }

341  [[nodiscard]] static auto getEnteringVariable(PtrVector<int64_t> costs)
342  -> Optional<unsigned int> {
343  // Bland's algorithm; guaranteed to terminate
344  auto f = costs.begin();
345  auto l = costs.end();

350  [[nodiscard]] static auto getLeavingVariable(MutPtrMatrix<int64_t> C,
351  size_t enteringVariable)
352  -> Optional<unsigned int> {
353  // inits guarantee first valid is selected
354  // we need
355  int64_t n = -1;
356  int64_t d = 0;
357  unsigned int j = 0;
358  for (size_t i = 1; i < C.numRow(); ++i) {
359  int64_t Civ = C(i, enteringVariable);
360  if (Civ > 0) {
361  int64_t Ci0 = C(i, 0);
362  if (Ci0 == 0) return --i;
363  assert(Ci0 > 0);
364  if ((n * Ci0) < (Civ * d)) {
365  n = Civ;
366  d = Ci0;
367  j = i;
368  }
369  }
370  }
371  // note, if we fail to find a leaving variable, then `j = 0`,
372  // and it will unsigned wrap to `size_t(-1)`, which indicates
373  // an empty `Optional<unsigned int>`
374  return --j;
375  }

377  unsigned int enteringVariable) -> int64_t {

382  if (i != leavingVariable + 1) {
383  int64_t m = NormalForm::zeroWithRowOperation(
384  C, i, leavingVariable + 1, enteringVariable, i == 0 ? f : 0);
385  if (i == 0) f = m;

391  MutPtrVector<int64_t> basicConstraints{getBasicConstraints()};
392  basicConstraints[oldBasicVar] = -1;
393  basicConstraints[enteringVariable] = leavingVariable;
394  return f;
