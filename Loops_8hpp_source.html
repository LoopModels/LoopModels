41 [[nodiscard]] inline auto getBackedgeTakenCount(llvm::ScalarEvolution &SE,
42  llvm::Loop *L)
43  -> const llvm::SCEV * {
44  auto b = L->getBounds(SE);
45  if (!b || (!isKnownOne(SE, b->getStepValue())))
46  return SE.getBackedgeTakenCount(L);
47  const llvm::SCEV *LB = SE.getSCEV(&b->getInitialIVValue());
48  const llvm::SCEV *UB = SE.getSCEV(&b->getFinalIVValue());
49  if (const auto *umm = llvm::dyn_cast<llvm::SCEVUMaxExpr>(UB)) {
50  const llvm::SCEV *m0 =
51  SE.getMinusSCEV(umm->getOperand(0), LB, llvm::SCEV::NoWrapFlags::FlagNUW);
52  const llvm::SCEV *m1 =
53  SE.getMinusSCEV(umm->getOperand(1), LB, llvm::SCEV::NoWrapFlags::FlagNUW);
54  // Does checking known negative make sense if we have NUW?
55  if (SE.isKnownNegative(m0)) return m1;
56  if (SE.isKnownNegative(m1)) return m0;
57  } else if (const auto *smm = llvm::dyn_cast<llvm::SCEVSMaxExpr>(UB)) {
58  const llvm::SCEV *m0 =
59  SE.getMinusSCEV(smm->getOperand(0), LB, llvm::SCEV::NoWrapFlags::FlagNSW);
60  const llvm::SCEV *m1 =
61  SE.getMinusSCEV(smm->getOperand(1), LB, llvm::SCEV::NoWrapFlags::FlagNSW);
62  if (SE.isKnownNegative(m0)) return m1;
63  if (SE.isKnownNegative(m1)) return m0;
64  }
65  return SE.getMinusSCEV(UB, LB, llvm::SCEV::NoWrapMask);
66 }

134 [[nodiscard]] inline auto simplifyMinMax(llvm::ScalarEvolution &SE,
135  const llvm::SCEVMinMaxExpr *S)
136  -> const llvm::SCEV * {
137  // FIXME: This is probably a bit aggressive...
138  bool isMin =
139  llvm::isa<llvm::SCEVSMinExpr>(S) || llvm::isa<llvm::SCEVUMinExpr>(S);
140  bool isSigned =
141  llvm::isa<llvm::SCEVSMinExpr>(S) || llvm::isa<llvm::SCEVSMaxExpr>(S);
142  auto GE = isSigned ? llvm::ICmpInst::Predicate::ICMP_SGE
143  : llvm::ICmpInst::Predicate::ICMP_UGE;
144 
145  const llvm::SCEV *op0 = S->getOperand(0);
146  const llvm::SCEV *op1 = S->getOperand(1);
147  auto [LB0, UB0] = getMinMaxValueSCEV(SE, op0);
148  auto [LB1, UB1] = getMinMaxValueSCEV(SE, op1);
149  // op0 >= op1
150  if (SE.isKnownPredicate(GE, LB0, UB1)) return isMin ? op1 : op0;
151  // op1 >= op0
152  if (SE.isKnownPredicate(GE, LB1, UB0)) return isMin ? op0 : op1;
153  return S;
154 }

182 [[nodiscard]] inline auto
183 addSymbol(std::array<IntMatrix, 2> &AB, // NOLINT(misc-no-recursion)
184  llvm::SmallVectorImpl<const llvm::SCEV *> &symbols, llvm::Loop *L,
185  const llvm::SCEV *v, llvm::ScalarEvolution &SE,
186  Range<size_t, size_t> lu, int64_t mlt, size_t minDepth) -> size_t {
187  auto &[A, B] = AB;
188  // first, we check if `v` in `Symbols`
189  if (size_t i = findSymbolicIndex(symbols, v)) {
190  A(lu, i) += mlt;
191  return minDepth;
192  }
193  if (std::optional<int64_t> c = getConstantInt(v)) {
194  A(lu, 0) += mlt * (*c);
195  return minDepth;
196  }
197  if (const auto *ar = llvm::dyn_cast<const llvm::SCEVAddExpr>(v)) {
198  const llvm::SCEV *op0 = ar->getOperand(0);
199  const llvm::SCEV *op1 = ar->getOperand(1);
200  Row M = A.numRow();
201  minDepth = addSymbol(AB, symbols, L, op0, SE, lu, mlt, minDepth);
202  if (M != A.numRow())
203  minDepth =
204  addSymbol(AB, symbols, L, op1, SE, _(M, A.numRow()), mlt, minDepth);
205  return addSymbol(AB, symbols, L, op1, SE, lu, mlt, minDepth);
206  }
207  if (const auto *m = llvm::dyn_cast<const llvm::SCEVMulExpr>(v)) {
208  if (auto op0 = getConstantInt(m->getOperand(0)))
209  return addSymbol(AB, symbols, L, m->getOperand(1), SE, lu, mlt * (*op0),
210  minDepth);
211  if (auto op1 = getConstantInt(m->getOperand(1)))
212  return addSymbol(AB, symbols, L, m->getOperand(0), SE, lu, mlt * (*op1),
213  minDepth);
214  } else if (const auto *x = llvm::dyn_cast<const llvm::SCEVAddRecExpr>(v)) {
215  size_t recDepth = x->getLoop()->getLoopDepth();
216  if (x->isAffine()) {
217  minDepth =
218  addSymbol(AB, symbols, L, x->getOperand(0), SE, lu, mlt, minDepth);
219  if (auto opc = getConstantInt(x->getOperand(1))) {
220  // swap order vs recDepth to go inner<->outer
221  B(lu, recDepth - 1) << mlt * (*opc);
222  return minDepth;
223  }
224  v = SE.getAddRecExpr(SE.getZero(x->getOperand(0)->getType()),
225  x->getOperand(1), x->getLoop(), x->getNoWrapFlags());
226  }
227  // we only support affine SCEVAddRecExpr with constant steps
228  // we use a flag "minSupported", which defaults to 0
229  // 0 means we support all loops, as the outer most depth is 1
230  // Depth of 0 means toplevel.
231  minDepth = std::max(minDepth, recDepth);
232  } else if (const auto *mm = llvm::dyn_cast<const llvm::SCEVMinMaxExpr>(v)) {
233  const auto *Sm = simplifyMinMax(SE, mm);
234  if (Sm != v) return addSymbol(AB, symbols, L, Sm, SE, lu, mlt, minDepth);
235  bool isMin =
236  llvm::isa<llvm::SCEVSMinExpr>(mm) || llvm::isa<llvm::SCEVUMinExpr>(mm);
237  const llvm::SCEV *op0 = mm->getOperand(0);
238  const llvm::SCEV *op1 = mm->getOperand(1);
239  if (isMin ^ (mlt < 0)) { // we can represent this as additional constraints
240  Row M = A.numRow();
241  Row Mp = M + lu.size();
242  A.resize(Mp);
243  B.resize(Mp);
244  A(_(M, Mp), _) = A(lu, _);
245  B(_(M, Mp), _) = B(lu, _);
246  minDepth = addSymbol(AB, symbols, L, op0, SE, lu, mlt, minDepth);
247  minDepth = addSymbol(AB, symbols, L, op1, SE, _(M, Mp), mlt, minDepth);
248  } else if (addRecMatchesLoop(op0, L)) {
249  return addSymbol(AB, symbols, L, op1, SE, lu, mlt, minDepth);
250  } else if (addRecMatchesLoop(op1, L)) {
251  return addSymbol(AB, symbols, L, op0, SE, lu, mlt, minDepth);
252  }
253  } else if (const auto *ex = llvm::dyn_cast<llvm::SCEVCastExpr>(v))
254  return addSymbol(AB, symbols, L, ex->getOperand(0), SE, lu, mlt, minDepth);
255  addSymbol(A, symbols, v, lu, mlt);
256  return minDepth;
257 }

267 inline auto // NOLINTNEXTLINE(misc-no-recursion)
268 addBackedgeTakenCount(std::array<IntMatrix, 2> &AB,
269  llvm::SmallVectorImpl<const llvm::SCEV *> &symbols,
270  llvm::Loop *L, const llvm::SCEV *BT,
271  llvm::ScalarEvolution &SE, size_t minDepth,
272  llvm::OptimizationRemarkEmitter *ORE) -> size_t {
273  // A contains syms
274  auto &[A, B] = AB;
275  Row M = A.numRow();
276  A.resize(M + 1);
277  B.resize(M + 1);
278  minDepth = addSymbol(AB, symbols, L, BT, SE, _(M, M + 1), 1, minDepth);
279  assert(A.numRow() == B.numRow());
280  size_t depth = L->getLoopDepth() - 1;
281  for (auto m = size_t(M); m < A.numRow(); ++m) B(m, depth) = -1; // indvar
282  // recurse, if possible to add an outer layer
283  if (llvm::Loop *P = L->getParentLoop()) {
284  if (areSymbolsLoopInvariant(A, symbols, P, SE)) {
285  // llvm::SmallVector<const llvm::SCEVPredicate *, 4> predicates;
286  // auto *BTI = SE.getPredicatedBackedgeTakenCount(L,
287  // predicates);
288  if (const llvm::SCEV *BTP = getBackedgeTakenCount(SE, P)) {
289  if (!llvm::isa<llvm::SCEVCouldNotCompute>(BTP))
290  return addBackedgeTakenCount(AB, symbols, P, BTP, SE, minDepth, ORE);
291  if (ORE) [[unlikely]] {
292  llvm::SmallVector<char, 128> msg;
293  llvm::raw_svector_ostream os(msg);
294  os << "SCEVCouldNotCompute from loop: " << *P << "\n";
295  llvm::OptimizationRemarkAnalysis analysis{
296  remarkAnalysis("AffineLoopConstruction", L)};
297  ORE->emit(analysis << os.str());
298  }
299  }
300  } else if (ORE) [[unlikely]] {
301  llvm::SmallVector<char, 256> msg;
302  llvm::raw_svector_ostream os(msg);
303  os << "Fail because symbols are not loop invariant in loop:\n"
304  << *P << "\n";
305  if (auto b = L->getBounds(SE))
306  os << "Loop Bounds:\nInitial: " << b->getInitialIVValue()
307  << "\nStep: " << *b->getStepValue()
308  << "\nFinal: " << b->getFinalIVValue() << "\n";
309  for (const auto *s : symbols) os << *s << "\n";
310  llvm::OptimizationRemarkAnalysis analysis{
311  remarkAnalysis("AffineLoopConstruction", L)};
312  ORE->emit(analysis << os.str());
313  }
314  }
315  return std::max(depth, minDepth);
316 }

331  static inline auto construct(BumpAlloc<> &alloc, llvm::Loop *L,
332  const llvm::SCEV *BT, llvm::ScalarEvolution &SE,
333  llvm::OptimizationRemarkEmitter *ORE = nullptr)
334  -> NotNull<AffineLoopNest> {
335  // A holds symbols
336  // B holds loop bounds
337  // they're separate so we can grow them independently
338  std::array<IntMatrix, 2> AB;
339  auto &[A, B] = AB;
340  // once we're done assembling these, we'll concatenate A and B
341  size_t maxDepth = L->getLoopDepth();
342  // size_t maxNumSymbols = BT->getExpressionSize();
343  A.resizeForOverwrite(
344  StridedDims{0, 1, unsigned(1) + BT->getExpressionSize()});
345  B.resizeForOverwrite(StridedDims{0, maxDepth, maxDepth});
346  llvm::SmallVector<const llvm::SCEV *, 3> symbols;
347  size_t minDepth =
348  loopNestCtor::addBackedgeTakenCount(AB, symbols, L, BT, SE, 0, ORE);
349  // We first check for loops in B that are shallower than minDepth
350  // we include all loops such that L->getLoopDepth() > minDepth
351  // note that the outer-most loop has a depth of 1.
352  // We turn these loops into `getAddRecExprs`s, so that we can
353  // add them as variables to `A`.
354  for (size_t d = 0; d < minDepth; ++d) {
355  // loop at depth d+1
356  llvm::Loop *P = nullptr;
357  // search B(_,d) for references
358  for (size_t i = 0; i < B.numRow(); ++i) {
359  // TODO; confirm `last` vs `end`
360  if (int64_t Bid = B(i, d)) {
361  if (!P) { // find P
362  P = L;
363  for (size_t r = d + 1; r < maxDepth; ++r) P = P->getParentLoop();
364  }
365  // TODO: find a more efficient way to get IntTyp
366  llvm::Type *intTyp = P->getInductionVariable(SE)->getType();
367  loopNestCtor::addSymbol(A, symbols,
368  SE.getAddRecExpr(SE.getZero(intTyp),
369  SE.getOne(intTyp), P,
370  llvm::SCEV::NoWrapMask),
371  _(i, i + 1), Bid);
372  }
373  }
374  }
375  invariant(1 + symbols.size(), size_t(A.numCol()));
376  size_t depth = maxDepth - minDepth;
377  unsigned numConstraints = unsigned(A.numRow()), N = unsigned(A.numCol());
378  NotNull<AffineLoopNest> aln{
379  AffineLoopNest::allocate(alloc, numConstraints, depth, symbols)};
380  aln->getA()(_, _(0, N)) << A;
381  // copy the included loops from B
382  // we use outer <-> inner order, so we skip unsupported outer loops.
383  aln->getA()(_, _(N, N + depth)) << B(_, _(end - depth, end));
384  return aln;
385  // addZeroLowerBounds();
386  // NOTE: pruneBounds() is not legal here if we wish to use
387  // removeInnerMost later.
388  // pruneBounds();
389  }

404  [[nodiscard]] constexpr auto rotate(BumpAlloc<> &alloc, PtrMatrix<int64_t> R,
405  const int64_t *offsets) const
406  -> NotNull<AffineLoopNest<false>> {
407  // if offsets is not null, we have the equivalent of
408  // A * O * [I 0; 0 R]
409  // where O = I - [0 0; offsets 0],
410  // where offsets is a vector of length getNumLoops() and O is square
411  size_t numExtraVar = 0, numConst = this->getNumSymbols();
412  if constexpr (NonNegative) numExtraVar = getNumLoops();
413  invariant(unsigned(R.numCol()), getNumLoops());
414  invariant(unsigned(R.numRow()), getNumLoops());
415  auto A{getA()};
416  const auto [M, N] = A.size();
417  auto syms{getSyms()};
418  NotNull<AffineLoopNest<false>> aln{AffineLoopNest<false>::allocate(
419  alloc, size_t(M) + numExtraVar, numLoops, syms)};
420  auto B{aln->getA()};
421  invariant(B.numRow(), M + numExtraVar);
422  invariant(B.numCol(), N);
423  B(_(0, M), _(0, numConst)) << A(_, _(0, numConst));
424  B(_(0, M), _(numConst, end)) << A(_, _(numConst, end)) * R;
425  if constexpr (NonNegative) {
426  B(_(M, end), _(0, numConst)) << 0;
427  B(_(M, end), _(numConst, end)) << R;
428  }
429  // A * O * [I 0; 0 R] = A * [I 0; 0 R] - A * [0 0; offs 0] * [I 0; 0 R]
430  // above, we computed `A * [I 0; 0 R]`, now if offsets != nullptr,
431  // we subtract A * [0 0; offs 0] * [I 0; 0 R].
432  // note that we have (s = number of dynamic symbols, l = number of loops)
433  // 1 s l 1 s l 1 s l
434  // 1 [ 0 0 0 [ 1 0 0 [ 0 0 0
435  // s 0 0 0 * 0 I 0 = 0 0 0
436  // l offs 0 0 ] 0 0 R ] offs 0 0 ]
437  // thus, we can ignore R here, and simply update the result using A.
438  if (offsets) {
439  for (size_t l = 0, L = getNumLoops(); l < L; ++l) {
440  if (int64_t mlt = offsets[l]) {
441  B(_(0, M), 0) -= mlt * A(_, numConst + l);
442  if constexpr (NonNegative) B(M + l, 0) = -mlt;
443  }
444  }
445  }
446  // aln->pruneBounds(alloc);
447  return aln;
448  }

474  [[nodiscard]] auto removeInnerMost(BumpAlloc<> &alloc) const
475  -> NotNull<AffineLoopNest<NonNegative>> {
476  // order is outer<->inner
477  auto A{getA()};
478  auto ret = AffineLoopNest<NonNegative>::allocate(
479  alloc, unsigned(A.numRow()), getNumLoops() - 1, getSyms());
480  MutPtrMatrix<int64_t> B{ret->getA()};
481  B << A(_, _(0, last));
482  // no loop may be conditioned on the innermost loop, so we should be able to
483  // safely remove all constraints that reference it
484  for (Row m = B.numRow(); m--;) {
485  if (A(m, last)) {
486  if (m != B.numRow() - 1) B(m, _) << B(last, _);
487  B.truncate(B.numRow() - 1);
488  }
489  }
490  ret->truncateConstraints(unsigned(B.numRow()));
491  return ret;
492  }

646  [[nodiscard]] auto
647  zeroExtraItersUponExtending(LinAlg::Alloc<int64_t> auto &alloc, size_t _i,
648  bool extendLower) const -> bool {
649  auto p = alloc.scope();
650  AffineLoopNest<NonNegative> *tmp = copy(alloc);
651  // question is, does the inner most loop have 0 extra iterations?
652  const size_t numPrevLoops = getNumLoops() - 1;
653  // we changed the behavior of removeLoop to actually drop loops that are
654  // no longer present.
655  for (size_t i = 0; i < numPrevLoops - 1; ++i)
656  tmp = tmp->removeLoop(alloc, i >= _i);
657  // for (size_t i = 0; i < numPrevLoops; ++i)
658  // if (i != _i) tmp = tmp->removeLoop(alloc, i);
659  // loop _i is now loop 0
660  // innermost loop is now loop 1
661  bool indep = true;
662  const size_t numConst = getNumSymbols();
663  auto A{tmp->getA()};
664  for (size_t n = 0; n < A.numRow(); ++n)
665  if ((A(n, numConst) != 0) && (A(n, 1 + numConst) != 0)) indep = false;
666  if (indep) return false;
667  AffineLoopNest<NonNegative> *margi = tmp->removeLoop(alloc, 1);
668  AffineLoopNest<NonNegative> *tmp2;
669  invariant(margi->getNumLoops(), unsigned(1));
670  invariant(tmp->getNumLoops(), unsigned(2));
671  invariant(margi->getA().numCol() + 1, tmp->getA().numCol());
672  // margi contains extrema for `_i`
673  // we can substitute extended for value of `_i`
674  // in `tmp`
675  auto p2 = alloc.checkpoint();
676  int64_t sign = 2 * extendLower - 1; // extendLower ? 1 : -1
677  for (size_t c = 0; c < margi->getNumInequalityConstraints(); ++c) {
678  int64_t b = sign * margi->getA()(c, numConst);
679  if (b <= 0) continue;
680  alloc.rollback(p2);
681  tmp2 = tmp->copy(alloc);
682  invariant(tmp2->getNumLoops(), unsigned(2));
683  invariant(margi->getNumLoops() + 1, tmp2->getNumLoops());
684  // increment to increase bound
685  // this is correct for both extending lower and extending upper
686  // lower: a'x + i + b >= 0 -> i >= -a'x - b
687  // upper: a'x - i + b >= 0 -> i <= a'x + b
688  // to decrease the lower bound or increase the upper, we increment
689  // `b`
690  ++(margi->getA())(c, 0);
691  // our approach here is to set `_i` equal to the extended bound
692  // and then check if the resulting polyhedra is empty.
693  // if not, then we may have >0 iterations.
694  for (size_t cc = 0; cc < tmp2->getNumCon(); ++cc) {
695  if (int64_t d = tmp2->getA()(cc, numConst)) {
696  tmp2->getA()(cc, _(0, last)) << b * tmp2->getA()(cc, _(0, last)) -
697  (d * sign) * margi->getA()(c, _);
698  }
699  }
700  for (size_t cc = size_t(tmp2->getNumCon()); cc;)
701  if (tmp2->getA()(--cc, 1 + numConst) == 0) tmp2->eraseConstraint(cc);
702  if (!(tmp2->calcIsEmpty(alloc))) return false;
703  }
704  if constexpr (NonNegative) {
705  if (extendLower) {
706  // increment to increase bound
707  // this is correct for both extending lower and extending upper
708  // lower: a'x + i + b >= 0 -> i >= -a'x - b
709  // upper: a'x - i + b >= 0 -> i <= a'x + b
710  // to decrease the lower bound or increase the upper, we
711  // increment `b` our approach here is to set `_i` equal to the
712  // extended bound and then check if the resulting polyhedra is
713  // empty. if not, then we may have >0 iterations.
714  for (size_t cc = 0; cc < tmp->getNumCon(); ++cc) {
715  if (int64_t d = tmp->getA()(cc, numConst)) {
716  // lower bound is i >= 0
717  // so setting equal to the extended lower bound now
718  // means that i = -1 so we decrement `d` from the column
719  tmp->getA()(cc, 0) -= d;
720  tmp->getA()(cc, numConst) = 0;
721  }
722  }
723  for (size_t cc = size_t(tmp->getNumCon()); cc;)
724  if (tmp->getA()(--cc, 1 + numConst) == 0) tmp->eraseConstraint(cc);
725  if (!(tmp->calcIsEmpty(alloc))) return false;
726  }
727  }
728  return true;
729  }
