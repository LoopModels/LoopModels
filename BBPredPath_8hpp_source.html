102  descendBlock(BumpAlloc<> &alloc, IR::Cache &cache,
103  aset<llvm::BasicBlock *> &visited, Predicate::Map &predMap,
104  llvm::BasicBlock *BBsrc, llvm::BasicBlock *BBdst,
105  Predicate::Intersection predicate, llvm::BasicBlock *BBhead,
106  llvm::Loop *L) -> Destination {
107  if (BBsrc == BBdst) {
108  assert(!predMap.contains(BBsrc));
109  predMap.insert({BBsrc, Set{predicate}});
110  return Destination::Reached;
111  }
112  if (L && (!(L->contains(BBsrc)))) {
113  // oops, we seem to have skipped the preheader and escaped the
114  // loop.
115  return Destination::Returned;
116  }
117  if (visited.contains(BBsrc)) {
118  // FIXME: This is terribly hacky.
119  // if `BBsrc == BBhead`, then we assume we hit a path that
120  // bypasses the following loop, e.g. there was a loop guard.
121  //
122  // Thus, we return `Returned`, indicating that it was a
123  // non-fatal dead-end. Otherwise, we check if it seems to have
124  // led to a live, non-empty path.
125  // TODO: should we union the predicates in case of returned?
126  if ((BBsrc != BBhead) && predMap.find(BBsrc) != predMap.rend())
127  return Destination::Reached;
128  return Destination::Returned;
129  }
130  // Inserts a tombstone to indicate that we have visited BBsrc, but
131  // not actually reached a destination.
132  visited.insert(BBsrc);
133  const llvm::Instruction *I = BBsrc->getTerminator();
134  if (!I) return Destination::Unknown;
135  if (llvm::isa<llvm::ReturnInst>(I)) return Destination::Returned;
136  if (llvm::isa<llvm::UnreachableInst>(I)) return Destination::Unreachable;
137  const auto *BI = llvm::dyn_cast<llvm::BranchInst>(I);
138  if (!BI) return Destination::Unknown;
139  if (BI->isUnconditional()) {
140  auto rc = descendBlock(alloc, cache, visited, predMap,
141  BI->getSuccessor(0), BBdst, predicate, BBhead, L);
142  if (rc == Destination::Reached) predMap.reach(alloc, BBsrc, predicate);
143  return rc;
144  }
145  // We have a conditional branch.
146  llvm::Value *cond = BI->getCondition();
147  // We need to check both sides of the branch and add a predicate.
148  size_t predInd = predMap.addPredicate(alloc, cache, cond);
149  auto rc0 = descendBlock(
150  alloc, cache, visited, predMap, BI->getSuccessor(0), BBdst,
151  predicate.intersect(predInd, Predicate::Relation::True), BBhead, L);
152  if (rc0 == Destination::Unknown) // bail
153  return Destination::Unknown;
154  auto rc1 = descendBlock(
155  alloc, cache, visited, predMap, BI->getSuccessor(1), BBdst,
156  predicate.intersect(predInd, Predicate::Relation::False), BBhead, L);
157  if ((rc0 == Destination::Returned) || (rc0 == Destination::Unreachable)) {
158  if (rc1 == Destination::Reached) {
159  // we're now assuming that !cond
160  predMap.assume(
161  Predicate::Intersection(predInd, Predicate::Relation::False));
162  predMap.reach(alloc, BBsrc, predicate);
163  }
164  return rc1;
165  }
166  if ((rc1 == Destination::Returned) || (rc1 == Destination::Unreachable)) {
167  if (rc0 == Destination::Reached) {
168  // we're now assuming that cond
169  predMap.assume(
170  Predicate::Intersection(predInd, Predicate::Relation::True));
171  predMap.reach(alloc, BBsrc, predicate);
172  }
173  return rc0;
174  }
175  if (rc0 != rc1) return Destination::Unknown;
176  if (rc0 == Destination::Reached) predMap.reach(alloc, BBsrc, predicate);
177  return rc0;
178  }
