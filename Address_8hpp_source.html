147  constexpr auto calcOrthAxes(ptrdiff_t depth) -> OrthogonalAxes {
148  invariant((depth <= 24) && (depth >= 0));
149  invariant(depth >= naturalDepth);
150  invariant(currentDepth >= depth);
151  currentDepth = depth;
152  bool indepAxes = true;
153  uint32_t contig{0}, indep{(uint32_t(1) << depth) - 1};
155  DensePtrMatrix<int64_t> inds{indexMatrix()};
156  for (ptrdiff_t l = 0; l < inds.numCol(); ++l) {
157  if (!inds[0, l]) continue;
158  contig |= uint32_t(1) << l;
159  indep &= ~(uint32_t(1) << l);
160  }
161  for (ptrdiff_t d = 1; d < inds.numRow(); ++d) {
162  for (ptrdiff_t l = 0; l < inds.numCol(); ++l) {
163  if (!inds[d, l]) continue;
164  if (!(indep & (uint32_t(1) << l))) indepAxes = false;
165  indep &= ~(uint32_t(1) << l);
166  }
167  }
168  axes = {indepAxes, contig, indep};
169  return axes;
170  }

190  constexpr void rotate(Valid<poly::Loop> explicitLoop,
191  SquarePtrMatrix<int64_t> Pinv, int64_t denom,
192  PtrVector<int64_t> omega, int64_t *offsets) {
193  loop = explicitLoop;
194  // we are updating in place; we may now have more loops than we did before
195  unsigned oldNatDepth = getNaturalDepth();
196  DensePtrMatrix<int64_t> M{indexMatrix()}; // aD x nLma
197  MutPtrVector<int64_t> offsetOmega{getOffsetOmega()};
198  unsigned depth = uint8_t(ptrdiff_t(Pinv.numCol()));
199  MutDensePtrMatrix<int64_t> mStar{indexMatrix()};
200  // M is implicitly padded with zeros, newNumLoops >= oldNumLoops
201  invariant(maxDepth >= naturalDepth);
202  invariant(oldNatDepth <= naturalDepth);
203  invariant(ptrdiff_t(oldNatDepth), ptrdiff_t(M.numRow()));
204  getDenominator() = denom;
205  // layout goes offsetOmega, indexMatrix, fusionOmega
206  // When we call `rotate`, we don't need fusionOmega anymore, because
207  // placement represented via the `ScheduledNode` and then IR graph
208  // Thus, we only need to update indexMatrix and offsetOmega
209  // offsetOmegas exactly alias, so we have no worries there.
210  // For `indexMatrix`, we use the unused `fusionOmega` space
211  // as a temporary, to avoid the aliasing problem.
212  //
213  // Use `M` before updating it, to update `offsetOmega`
214  if (offsets)
215  offsetOmega -= PtrVector<int64_t>{offsets, oldNatDepth} * M.t();
216  // update `M` into `mStar`
217  // mStar << M * Pinv(_(0, oldNumLoops), _);
218  MutPtrVector<int64_t> buff{getFusionOmega()[_(0, math::last)]};
219  invariant(buff.size(), ptrdiff_t(depth));
220  unsigned newNatDepth = 0;
221  for (ptrdiff_t d = getArrayDim(); d--;) {
222  buff << 0;
223  for (ptrdiff_t k = 0; k < oldNatDepth; ++k) buff += M[d, k] * Pinv[k, _];
224  mStar[d, _] << buff;
225  if (newNatDepth == depth) continue;
226  // find last, as buf goes outer<->inner
227  auto range = std::ranges::reverse_view{buff[_(newNatDepth, depth)]};
228  auto m = std::ranges::find_if(range, [](int64_t i) { return i != 0; });
229  if (m == range.end()) continue;
230  newNatDepth = depth - std::distance(range.begin(), m);
231  }
232  // use `mStar` to update offsetOmega`
233  offsetOmega -= omega * mStar.t();
234  this->naturalDepth = newNatDepth;
235  if (newNatDepth == depth) return;
236  invariant(newNatDepth < depth);
237  MutDensePtrMatrix<int64_t> indMat{this->indexMatrix()};
238  for (ptrdiff_t d = 1; d < getArrayDim(); ++d)
239  indMat[d, _] << mStar[d, _(0, newNatDepth)];
240  }

274  [[nodiscard]] static constexpr auto intMemNeeded(size_t numLoops, size_t dim)
275  -> size_t {
276  // d = dim, l = numLoops
277  // Memory layout: offset, size
278  // 0,1 for denom
279  // 1,d for offsetOmega
280  // 1 + d, d*l for indexMatrix
281  // 1 + d + d*l, l+1 for fusionOmega
282  // 1 + d + d*l + l + 1 == 1 + (d + 1)*(l + 1)
283  return 1 + (numLoops + 1) * (dim + 1);
284  }

335  construct(Arena<> *alloc, const llvm::SCEVUnknown *arrayPtr,
336  llvm::Instruction *user, PtrMatrix<int64_t> indMat,
337  std::array<llvm::SmallVector<const llvm::SCEV *, 3>, 2> szOff,
338  PtrVector<int64_t> coffsets, int64_t *offsets, unsigned numLoops,
339  unsigned maxNumLoops) -> Valid<Addr> {
340  // we don't want to hold any other pointers that may need freeing
341  unsigned arrayDim = szOff[0].size(), nOff = szOff[1].size();
342  size_t memNeeded = intMemNeeded(maxNumLoops, arrayDim);
343  auto *mem = static_cast<Addr *>(
344  alloc->allocate(sizeof(Addr) + memNeeded * sizeof(int64_t)));
345  const auto **syms = // over alloc by numLoops - 1, in case we remove
346  alloc->allocate<const llvm::SCEV *>(arrayDim + nOff + numLoops - 1);
347  unsigned natDepth = numLoops;
348  for (; natDepth; --natDepth)
349  if (math::anyNEZero(indMat[_, natDepth - 1])) break;
350  auto *ma = new (mem) Addr(arrayPtr, user, offsets, syms,
351  std::array<unsigned, 2>{arrayDim, nOff}, numLoops,
352  natDepth, maxNumLoops);
353  std::copy_n(szOff[0].begin(), arrayDim, syms);
354  std::copy_n(szOff[1].begin(), nOff, syms + arrayDim);
355  ma->indexMatrix() << indMat[_, _(0, natDepth)]; // naturalDepth
356  ma->getOffsetOmega() << coffsets;

443  // L is the inner most loop being removed
444  void updateOffsMat(Arena<> *alloc, size_t numToPeel,
445  llvm::ScalarEvolution *SE) {
446  invariant(numToPeel > 0);
447  // need to condition on loop
448  // remove the numExtraLoopsToPeel from Rt
449  // that is, we want to move Rt(_,_(end-numExtraLoopsToPeel,end))
450  // would this code below actually be expected to boost
451  // performance? if
452  // (Bt.numCol()+numExtraLoopsToPeel>Bt.rowStride())
453  // Bt.resize(Bt.numRow(),Bt.numCol(),Bt.numCol()+numExtraLoopsToPeel);
454  // order of loops in Rt is outermost->innermost
455  DensePtrMatrix<int64_t> oldOffsMat{offsetMatrix()}, Rt{indexMatrix()};
456  size_t dynSymInd = numDynSym;
457  numDynSym += numToPeel;
458  MutPtrVector<const llvm::SCEV *> sym{getSymbolicOffsets()};
459  offSym = alloc->allocate<int64_t>(size_t(numDynSym) * numDim);
460  MutDensePtrMatrix<int64_t> offsMat{offsetMatrix()};
461  if (dynSymInd) offsMat[_, _(0, dynSymInd)] << oldOffsMat;
462  llvm::Loop *L = loop->getLLVMLoop();
463  for (unsigned d = loop->getNumLoops() - numToPeel; d--;)
464  L = L->getParentLoop();
465  for (size_t i = numToPeel; i;) {
466  L = L->getParentLoop();
467  if (allZero(Rt[_, --i])) continue;
468  // push the SCEV
469  auto *iTyp = L->getInductionVariable(*SE)->getType();
470  const llvm::SCEV *S = SE->getAddRecExpr(
471  SE->getZero(iTyp), SE->getOne(iTyp), L, llvm::SCEV::NoWrapMask);
472  if (const llvm::SCEV **j = std::ranges::find(sym, S); j != sym.end()) {
473  --numDynSym;
474  offsMat[_, std::distance(sym.begin(), j)] += Rt[_, i];
475  } else {
476  offsMat[_, dynSymInd] << Rt[_, i];
477  sym[dynSymInd++] = S;
478  }
479  }
480  }
481  // we peel off the outer `numToPeel` loops
482  // however, we may have already peeled off some number of loops
483  // we check how many have already been peeled via...
484  void peelLoops(Arena<> *alloc, ptrdiff_t numToPeel,
485  llvm::ScalarEvolution *SE) {
486  invariant(numToPeel > 0);
487  loop->removeOuterMost(numToPeel, SE);
488  ptrdiff_t numLoops = getCurrentDepth();
489  invariant(numToPeel <= maxDepth);
490  // we need to compare numToPeel with actual depth
491  // because we might have peeled some loops already
492  invariant(numLoops <= maxDepth);
493  numToPeel -= maxDepth - numLoops;
494  if (numToPeel == 0) return;
495  // we're dropping the outer-most `numToPeel` loops
496  // first, we update offsMat
497  updateOffsMat(alloc, numToPeel, SE);
498  // current memory layout (outer <-> inner):
499  // - denom (1)
500  // - offsetOmega (arrayDim)
501  // - indexMatrix (arrayDim x numLoops)
502  // - fusionOmegas (numLoops+1)
503  int64_t *dst = indMatPtr(), *src = dst + numToPeel;
504  ptrdiff_t dim = getArrayDim(), oldNatDepth = this->naturalDepth;
505  this->currentDepth = numLoops - numToPeel;
506  this->naturalDepth = oldNatDepth - numToPeel;
507  invariant(currentDepth < numLoops);
508  // we want d < dim for indexMatrix, and then == dim for fusion omega
509  for (ptrdiff_t d = dim;;) {
510  std::copy_n(src, (d) ? naturalDepth : (currentDepth + 1), dst);
511  if (!d--) break;
512  src += oldNatDepth;
513  dst += naturalDepth;
514  }
515  }

573  auto calculateCostContiguousLoadStore(const llvm::TargetTransformInfo &TTI,
574  unsigned int vectorWidth)
575  -> cost::RecipThroughputLatency {
576  constexpr unsigned int addrSpace = 0;
577  llvm::Type *T = cost::getType(getType(), vectorWidth);
578  llvm::Align alignment = getAlign();
579  if (!predicate) {
580  llvm::Intrinsic::ID id =
581  isLoad() ? llvm::Instruction::Load : llvm::Instruction::Store;
582  return {
583  TTI.getMemoryOpCost(id, T, alignment, addrSpace,
584  llvm::TargetTransformInfo::TCK_RecipThroughput),
585  TTI.getMemoryOpCost(id, T, alignment, addrSpace,
586  llvm::TargetTransformInfo::TCK_Latency)};
587  }
588  llvm::Intrinsic::ID id =
589  isLoad() ? llvm::Intrinsic::masked_load : llvm::Intrinsic::masked_store;
