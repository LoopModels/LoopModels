311  void updateOffsMat(BumpAlloc<> &alloc, size_t numToPeel, llvm::Loop *L,
312  llvm::ScalarEvolution *SE) {
313  invariant(numToPeel > 0);
314  // need to condition on loop
315  // remove the numExtraLoopsToPeel from Rt
316  // that is, we want to move Rt(_,_(end-numExtraLoopsToPeel,end))
317  // would this code below actually be expected to boost
318  // performance? if
319  // (Bt.numCol()+numExtraLoopsToPeel>Bt.rowStride())
320  // Bt.resize(Bt.numRow(),Bt.numCol(),Bt.numCol()+numExtraLoopsToPeel);
321  // order of loops in Rt is outermost->innermost
322  DensePtrMatrix<int64_t> oldOffsMat{offsetMatrix()}, Rt{indexMatrix()};
323  size_t dynSymInd = numDynSym;
324  numDynSym += numToPeel;
325  MutPtrVector<const llvm::SCEV *> sym{getSymbolicOffsets()};
326  offSym = alloc.allocate<int64_t>(size_t(numDynSym) * numDim);
327  MutDensePtrMatrix<int64_t> offsMat{offsetMatrix()};
328  if (dynSymInd) offsMat(_, _(0, dynSymInd)) << oldOffsMat;
329  for (size_t i = numToPeel; i;) {
330  L = L->getParentLoop();
331  if (allZero(Rt(_, --i))) continue;
332  // push the SCEV
333  auto *iTyp = L->getInductionVariable(*SE)->getType();
334  const llvm::SCEV *S = SE->getAddRecExpr(
335  SE->getZero(iTyp), SE->getOne(iTyp), L, llvm::SCEV::NoWrapMask);
336  if (const llvm::SCEV **j = std::ranges::find(sym, S); j != sym.end()) {
337  --numDynSym;
338  offsMat(_, std::distance(sym.begin(), j)) += Rt(_, i);
339  } else {
340  offsMat(_, dynSymInd) << Rt(_, i);
341  sym[dynSymInd++] = S;
342  }
343  }
344  }
345  void peelLoops(BumpAlloc<> &alloc, size_t numToPeel, llvm::Loop *L,
346  llvm::ScalarEvolution *SE) {
347  invariant(numToPeel > 0);
348  size_t maxDepth = nodeOrDepth.maxDepth, numLoops = getNumLoops();
349  invariant(numToPeel <= maxDepth);
350  // we need to compare numToPeel with actual depth
351  // because we might have peeled some loops already
352  invariant(numLoops <= maxDepth);
353  numToPeel -= maxDepth - numLoops;
354  if (numToPeel == 0) return;
355  // we're dropping the outer-most `numToPeel` loops
356  // first, we update offsMat
357  updateOffsMat(alloc, numToPeel, L, SE);
358  // current memory layout (outer <-> inner):
359  // - denom (1)
360  // - offsetOmega (arrayDim)
361  // - indexMatrix (arrayDim x numLoops)
362  // - fusionOmegas (numLoops+1)
363  int64_t *dst = indMatPtr(), *src = dst + numToPeel;
364  depth = numLoops - numToPeel;
365  size_t dim = getArrayDim();
366  invariant(depth < numLoops);
367  // we want d < dim for indexMatrix, and then == dim for fusion omega
368  for (size_t d = 0; d <= dim; ++d) {
369  std::copy_n(src, depth + (d == dim), dst);
370  src += numLoops;
371  dst += depth;
372  }
373  }
